53 static JsonWebKeyECCurve curve_id_to_jwk_curve(CurveID curve_id)
54 {
55 switch (curve_id)
56 {
57 case CurveID::SECP384R1:
58 return JsonWebKeyECCurve::P384;
59 case CurveID::SECP256R1:
60 return JsonWebKeyECCurve::P256;
61 case CurveID::SECP256K1:
62 return JsonWebKeyECCurve::P256K1;
63 default:
64 throw std::logic_error(fmt::format("Unknown curve {}", curve_id));
65 }
66 }
