35 {
36 LOG_DEBUG_FMT("Configurations: submit ORC for #{} from {}", rid, from);

38 ObservedReconfigurationCommit::In ps = {from, rid};

40 http::Request request(fmt::format(

43 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);

45 auto body = serdes::pack(ps, serdes::Pack::Text);
46 request.set_body(&body);
