40 // This is a container for a write-set + dependencies. It can be applied to
41 // a given state, or used to track a set of operations on a state
42 struct ChangeSet : public AbstractChangeSet
43 {
44 protected:
45 ChangeSet() {}
46
47 public:
48 const size_t rollback_counter = {};
49 const kv::untyped::State state = {};
50 const kv::untyped::State committed = {};
51 const Version start_version = {};
52
53 Version read_version = NoVersion;
54 kv::untyped::Read reads = {};
55 kv::untyped::Write writes = {};
56
57 ChangeSet(
58 size_t rollbacks,
59 kv::untyped::State& current_state,
60 kv::untyped::State& committed_state,
61 Version current_version) :
62 rollback_counter(rollbacks),
63 state(current_state),
64 committed(committed_state),
65 start_version(current_version)
66 {}
67
68 ChangeSet(ChangeSet&) = delete;
69
70 bool has_writes() const override
71 {
72 return !writes.empty();
73 }
74 };
75
76 using ChangeSetPtr = std::unique_ptr<ChangeSet>;
77
78 // This is a container for a snapshot. It has no dependencies as the snapshot
79 // obliterates the current state.
80 struct SnapshotChangeSet : public ChangeSet
81 {
82 const kv::untyped::State state;
83 const Version version;
84
85 SnapshotChangeSet(kv::untyped::State&& snapshot_state, Version version_) :
86 state(std::move(snapshot_state)),
87 version(version_)
88 {}
89
90 SnapshotChangeSet(SnapshotChangeSet&) = delete;
91
92 bool has_writes() const override
93 {
94 return true;
95 }
96 };
97}
