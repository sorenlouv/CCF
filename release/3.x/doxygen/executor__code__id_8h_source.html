37inline std::pair<grpc_status, std::string> verification_error(
38 ccf::QuoteVerificationResult result)
39{
40 switch (result)
41 {
42 case ccf::QuoteVerificationResult::Failed:
43 return std::make_pair(
44 GRPC_STATUS_UNAUTHENTICATED, "Quote could not be verified");
45 case ccf::QuoteVerificationResult::FailedCodeIdNotFound:
46 return std::make_pair(
47 GRPC_STATUS_UNAUTHENTICATED,
48 "Quote does not contain known enclave measurement");
49 case ccf::QuoteVerificationResult::FailedInvalidQuotedPublicKey:
50 return std::make_pair(
51 GRPC_STATUS_UNAUTHENTICATED,
52 "Quote report data does not contain node's public key hash");
53 default:
54 return std::make_pair(
55 GRPC_STATUS_INTERNAL, "Unknown quote verification error");
56 }
