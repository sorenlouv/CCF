15 // A Serializer is generally used as a template argument. It should provide:
16 // - serialize(...): returning a tuple of PartialSerializations representing
17 // all passed args. It may be templated and variadic or accept only precise
18 // argument types. Ideally it should give a useful error message at compile
19 // time when called with argument types it doesn't support.
20 // - deserialize(const uint8_t* data, size_t size): returning a tuple of
21 // elements parsed from byte buffer, potentially throwing logic errors if
22 // anything is missing/malformed. May need to be templated on expected
23 // types.

27 const uint8_t* data;
28 const size_t size;

35 template <size_t I = 0, typename F, typename... Ts>
36 static auto tuple_apply(const std::tuple<Ts...>& t, const F& f)
37 {
38 if constexpr (sizeof...(Ts) == 1)
39 {
40 return std::make_tuple();
41 }
42 else if constexpr (I == sizeof...(Ts) - 1)
43 {
44 return f(std::get<I>(t));
45 }
46 else
47 {
48 return std::tuple_cat(f(std::get<I>(t)), tuple_apply<I + 1>(t, f));
49 }
50 }

66 template <size_t I, typename T>
67 struct close_enough_at
68 {
69 using CanonTarget =
70 std::remove_cvref_t<typename std::tuple_element_t<I, Tup>>;
71 using CanonArgument = std::remove_cvref_t<T>;
72
73 // This determines what types a Serializer will accept as arguments to
74 // serialize(...), relative to the declared param types.
75 // The main feature is the removal of const, volatile, and references
76 // from the type so that a Serializer<int> will accept an argument of
77 // type const int&.
78 // Additionally, this will accept ByteRange arguments for parameters
79 // declared as std::vector<uint8_t> and vice versa - when serializing we
80 // can copy bytes from either in the same way, but there is a
81 // distinction in the deserialized type of whether we are copying (to a
82 // byte vector) or referring to an existing byte range. It may be
83 // possible to generalise this further and replace with
84 // std::is_constructible, but these restrictions are sufficient for the
85 // current uses.
86 // Additionally again, this will accept an array-like of ByteRanges for
87 // a single ByteRange parameter. These will be serialised in-order, and
88 // produce a single ByteRange in deserialisation.
89 static constexpr bool value =
90 std::is_same_v<CanonTarget, CanonArgument> ||
91 (std::is_same_v<CanonTarget, std::vector<uint8_t>> &&
92 std::is_same_v<CanonArgument, ByteRange>) ||
93 (std::is_same_v<CanonTarget, ByteRange> &&
94 std::is_same_v<CanonArgument, std::vector<uint8_t>>) ||
95 (std::is_same_v<CanonTarget, ByteRange> &&
96 (std::is_array_v<CanonArgument> &&
97 std::is_same_v<std::remove_extent_t<CanonArgument>, ByteRange>));
98 };

381 template <typename T, typename... Ts>
382 static auto deserialize_impl(const uint8_t* data, size_t size)
383 {
384 using StrippedT = std::remove_cvref_t<T>;
385
386 if constexpr (sizeof...(Ts) == 0)
387 {
388 return std::make_tuple(
389 deserialize_value_final(data, size, Tag<StrippedT>{}));
390 }
391 else
392 {
393 const auto next =
394 std::make_tuple(deserialize_value(data, size, Tag<StrippedT>{}));
395 return std::tuple_cat(next, deserialize_impl<Ts...>(data, size));
396 }
397 }

401 using EmptySerializer::serialize;

405 template <typename T, typename... Ts>
406 static auto serialize(T&& t, Ts&&... ts)
407 {
408 if constexpr (sizeof...(Ts) == 0)
409 {
410 return serialize_value_final(std::forward<T>(t));
411 }
412 else
413 {
414 const auto next = serialize_value(std::forward<T>(t));
415 return std::tuple_cat(next, serialize(std::forward<Ts>(ts)...));
416 }
417 }

419 template <typename... Ts>
420 static auto deserialize(const uint8_t* data, size_t size)
421 {
422 if constexpr (sizeof...(Ts) == 0)
423 {
424 return EmptySerializer::deserialize(data, size);
425 }
426 else
427 {
428 return deserialize_impl<Ts...>(data, size);
