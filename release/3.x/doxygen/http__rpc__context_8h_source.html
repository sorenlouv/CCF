40 void serialise()
41 {
42 if (!serialised)
43 {
44 const auto request_prefix = fmt::format(
45 "{} {} HTTP/1.1\r\n"
46 "{}"
47 "\r\n",
48 verb.c_str(),
49 url,
50 http::get_header_string(request_headers));
51
52 serialised_request.resize(request_prefix.size() + request_body.size());
53 ::memcpy(
54 serialised_request.data(),
55 request_prefix.data(),
56 request_prefix.size());
57 if (!request_body.empty())
58 {
59 ::memcpy(
60 serialised_request.data() + request_prefix.size(),
61 request_body.data(),
62 request_body.size());
63 }
64 }
65
66 serialised = true;
67 }

264 inline static std::optional<std::string> extract_actor(HttpRpcContext& ctx)
265 {
266 const auto path = ctx.get_method();
267 const auto first_slash = path.find_first_of('/');
268 const auto second_slash = path.find_first_of('/', first_slash + 1);
269
270 if (first_slash != 0 || second_slash == std::string::npos)
271 {
272 return std::nullopt;
273 }
274
275 auto actor = path.substr(first_slash + 1, second_slash - first_slash - 1);
276 auto remaining_path = path.substr(second_slash);
277
278 // The "actor" is generally just a single path component, eg. `gov` or
279 // `node`. We make an exception for .well-known paths however, which use
280 // two components, ie. `.well-known/acme-challenge`. This restricts the
281 // ACME frontend to just that particular sub-directory, and allows the
282 // application to handle other .well-known paths.
283 if (actor == ".well-known")
284 {
285 const auto third_slash = path.find_first_of('/', second_slash + 1);
286 if (third_slash == std::string::npos)
287 {
288 return std::nullopt;
289 }
290
291 actor = path.substr(first_slash + 1, third_slash - first_slash - 1);
292 remaining_path = path.substr(third_slash);
293 }
294
295 if (actor.empty() || remaining_path.empty())
296 {
297 return std::nullopt;
298 }
299
300 // if the extracted actor is a known type, set the remaining path
301 if (ccf::is_valid_actor(actor))
302 {
303 ctx.set_method(remaining_path);
304 }
305 return actor;
306 }

311 inline std::shared_ptr<http::HttpRpcContext> make_rpc_context(
312 std::shared_ptr<ccf::SessionContext> s, const std::vector<uint8_t>& packed)
313 {
314 http::SimpleRequestProcessor processor;
315 http::RequestParser parser(processor);
316
317 parser.execute(packed.data(), packed.size());
318
319 if (processor.received.size() != 1)
320 {
321 throw std::logic_error(fmt::format(
322 "Expected packed to contain a single complete HTTP message. Actually "
323 "parsed {} messages",
324 processor.received.size()));
325 }
326
327 const auto& msg = processor.received.front();
328
329 return std::make_shared<http::HttpRpcContext>(
330 s, msg.method, msg.url, msg.headers, msg.body, packed);
331 }

333 inline std::shared_ptr<http::HttpRpcContext> make_fwd_rpc_context(
334 std::shared_ptr<ccf::SessionContext> s,
335 const std::vector<uint8_t>& packed,
336 ccf::FrameFormat frame_format)
337 {
338 switch (frame_format)
339 {
340 case ccf::FrameFormat::http:
341 {
342 return make_rpc_context(s, packed);
343 }
344 default:
345 throw std::logic_error("Unknown Frame Format");
346 }
