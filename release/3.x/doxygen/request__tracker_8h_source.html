34 struct RequestComp

36 bool operator()(const Request* lhs, const Request* rhs) const
37 {
38 const std::array<uint64_t, 4>& lhs_hash =
39 (std::array<uint64_t, 4>&)lhs->hash;
40 const std::array<uint64_t, 4>& rhs_hash =
41 (std::array<uint64_t, 4>&)rhs->hash;
42
43 for (uint32_t i = 0; i < 4; ++i)
44 {
45 if (lhs_hash[i] == rhs_hash[i])
46 {
47 continue;
48 }
49 return lhs_hash[i] > rhs_hash[i];
50 }
51 return false;
52 }
53 };

185 static void insert(
186 const crypto::Sha256Hash& hash,
187 std::chrono::milliseconds time,
188 std::multiset<Request*, RequestComp>& requests_,
189 snmalloc::DLList<Request, std::nullptr_t, true>& requests_list_)
190 {
191 if (
192 requests_list_.get_tail() != nullptr &&
193 requests_list_.get_tail()->time > time)
194 {
195 // Time is not a precise measurement and can be different
196 // on different threads. To ensure that the time values are
197 // correctly ordered reuse the time value of the last
198 // inserted item if the new item's time is not the highest known value.
199 time = requests_list_.get_tail()->time;
200 }
201 auto r = std::make_unique<Request>(hash, time);
202 requests_.insert(r.get());
203 requests_list_.insert_back(r.release());
204 }
205
206 bool remove(
207 const crypto::Sha256Hash& hash,
208 std::multiset<Request*, RequestComp>& requests_,
209 snmalloc::DLList<Request, std::nullptr_t, true>& requests_list_)
210 {
211 Request r(hash);
212 auto ret = requests_.equal_range(&r);
213 if (ret.first == ret.second)
214 {
215 return false;
216 }
217
218 auto it = ret.first;
219 for (auto c = ret.first; c != ret.second; ++c)
220 {
221 if ((*it)->time > (*c)->time)
222 {
223 it = c;
224 }
225 }
226
227 std::unique_ptr<Request> req(*it);
228 requests_.erase(it);
229 requests_list_.remove(req.get());
230 return true;
231 }
