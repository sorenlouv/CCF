126 void snapshot_(
127 std::unique_ptr<kv::AbstractStore::AbstractSnapshot> snapshot)
128 {
129 auto snapshot_version = snapshot->get_version();
130
131 auto serialised_snapshot = store->serialise_snapshot(std::move(snapshot));
132
133 auto tx = store->create_tx();
134 auto evidence = tx.rw<SnapshotEvidence>(Tables::SNAPSHOT_EVIDENCE);
135 auto snapshot_hash = crypto::Sha256Hash(serialised_snapshot);
136 evidence->put({snapshot_hash, snapshot_version});
137
138 ccf::ClaimsDigest cd;
139 cd.set(std::move(snapshot_hash));
140
141 crypto::Sha256Hash ws_digest;
142 std::string commit_evidence;
143 auto capture_ws_digest_and_commit_evidence =
144 [&ws_digest, &commit_evidence](
145 const std::vector<uint8_t>& write_set,
146 const std::string& commit_evidence_) {
147 new (&ws_digest)
148 crypto::Sha256Hash({write_set.data(), write_set.size()});
149 commit_evidence = commit_evidence_;
150 };
151
152 auto rc =
153 tx.commit(cd, false, nullptr, capture_ws_digest_and_commit_evidence);
154 if (rc != kv::CommitResult::SUCCESS)
155 {
156 LOG_FAIL_FMT(
157 "Could not commit snapshot evidence for seqno {}: {}",
158 snapshot_version,
159 rc);
160 return;
161 }
162
163 auto evidence_version = tx.commit_version();
164
165 record_snapshot(snapshot_version, evidence_version, serialised_snapshot);
166 consensus::Index snapshot_idx =
167 static_cast<consensus::Index>(snapshot_version);
168 consensus::Index snapshot_evidence_idx =
169 static_cast<consensus::Index>(evidence_version);
170 pending_snapshots.emplace_back(
171 snapshot_idx,
172 snapshot_evidence_idx,
173 ws_digest,
174 commit_evidence,
175 cd.value());
176
177 LOG_DEBUG_FMT(
178 "Snapshot successfully generated for seqno {}, with evidence seqno "
179 "{}: "
180 "{}, ws digest: {}",
181 snapshot_idx,
182 snapshot_evidence_idx,
183 cd.value(),
184 ws_digest);
185 }

187 void update_indices(consensus::Index idx)
188 {
189 while ((next_snapshot_indices.size() > 1) &&
190 (std::next(next_snapshot_indices.begin())->idx <= idx))
191 {
192 next_snapshot_indices.pop_front();
193 }
194
195 for (auto it = pending_snapshots.begin(); it != pending_snapshots.end();)
196 {
197 if (idx > it->evidence_idx)
198 {
199 auto serialised_receipt = build_and_serialise_receipt(
200 it->sig.value(),
201 it->tree.value(),
202 it->node_id.value(),
203 it->node_cert.value(),
204 it->evidence_idx,
205 it->write_set_digest,
206 it->commit_evidence,
207 std::move(it->snapshot_digest));
208 commit_snapshot(it->idx, serialised_receipt);
209 auto it_ = it;
210 ++it;
211 pending_snapshots.erase(it_);
212 }
213 else
214 {
215 ++it;
216 }
217 }
218 }

266 bool record_committable(consensus::Index idx) override
267 {
268 // Returns true if the committable idx will require the generation of a
269 // snapshot, and thus a new ledger chunk
270 std::lock_guard<ccf::pal::Mutex> guard(lock);
271
272 CCF_ASSERT_FMT(
273 idx >= next_snapshot_indices.back().idx,
274 "Committable seqno {} < next snapshot seqno {}",
275 idx,
276 next_snapshot_indices.back().idx);
277
278 bool forced = store->flag_enabled_unsafe(
279 kv::AbstractStore::Flag::SNAPSHOT_AT_NEXT_SIGNATURE);
280
281 consensus::Index last_unforced_idx = last_snapshot_idx;
282 for (auto it = next_snapshot_indices.rbegin();
283 it != next_snapshot_indices.rend();
284 it++)
285 {
286 if (!it->forced)
287 {
288 last_unforced_idx = it->idx;
289 break;
290 }
291 }
292
293 auto due = (idx - last_unforced_idx) >= snapshot_tx_interval;
294 if (due || forced)
295 {
296 next_snapshot_indices.push_back({idx, !due, false});
297 LOG_TRACE_FMT(
298 "{} {} as snapshot index", !due ? "Forced" : "Recorded", idx);
299 store->unset_flag_unsafe(
300 kv::AbstractStore::Flag::SNAPSHOT_AT_NEXT_SIGNATURE);
301 return due;
302 }
303
304 return false;
305 }

345 {
346 auto msg = std::make_unique<threading::Tmsg<SnapshotMsg>>(&snapshot_cb);
347 msg->data.self = shared_from_this();
348 msg->data.snapshot = store->snapshot(idx);
349 static uint32_t generation_count = 0;
350 threading::ThreadMessaging::thread_messaging.add_task(
351 threading::ThreadMessaging::get_execution_thread(generation_count++),

355 void commit(consensus::Index idx, bool generate_snapshot) override
356 {
357 // If generate_snapshot is true, takes a snapshot of the key value store
358 // at the last snapshottable index before idx, and schedule snapshot
359 // serialisation on another thread (round-robin). Otherwise, only record
360 // that a snapshot was generated.
361 std::lock_guard<ccf::pal::Mutex> guard(lock);
362
363 update_indices(idx);
364
365 if (idx < last_snapshot_idx)
366 {
367 throw std::logic_error(fmt::format(
368 "Cannot snapshot at seqno {} which is earlier than last snapshot "
369 "seqno {}",
370 idx,
371 last_snapshot_idx));
372 }
373
374 CCF_ASSERT_FMT(
375 idx >= next_snapshot_indices.front().idx,
376 "Cannot commit snapshotter at {}, which is before last snapshottable "
377 "idx {}",
378 idx,
379 next_snapshot_indices.front().idx);
380
381 auto& next = next_snapshot_indices.front();
382 auto due = next.idx - last_snapshot_idx >= snapshot_tx_interval;
383 if (due || (next.forced && !next.done))
384 {
385 if (snapshot_generation_enabled && generate_snapshot && next.idx)
386 {
387 schedule_snapshot(next.idx);
388 next.done = true;
389 }
390
391 if (due && !next.forced)
392 {
393 // last_snapshot_idx records the last normally scheduled, i.e.
394 // unforced, snapshot index, so that backups (which don't know forced
395 // indices) continue the snapshot interval normally.
396 last_snapshot_idx = next.idx;
397 LOG_TRACE_FMT(
398 "Recorded {} as last snapshot index", last_snapshot_idx);
399 }
400 }
401 }

403 void rollback(consensus::Index idx) override
404 {
405 std::lock_guard<ccf::pal::Mutex> guard(lock);
406
407 while (!next_snapshot_indices.empty() &&
408 (next_snapshot_indices.back().idx > idx))
409 {
410 next_snapshot_indices.pop_back();
411 }
412
413 if (next_snapshot_indices.empty())
414 {
415 next_snapshot_indices.push_back({last_snapshot_idx, false, true});
416 }
417
418 LOG_TRACE_FMT(
419 "Rolled back snapshotter: last snapshottable idx is now {}",
420 next_snapshot_indices.front().idx);
421
422 while (!pending_snapshots.empty() &&
423 (pending_snapshots.back().evidence_idx > idx))
424 {
425 pending_snapshots.pop_back();
426 }
427 }
