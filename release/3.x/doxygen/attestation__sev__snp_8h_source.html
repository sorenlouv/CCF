190 static EndorsementEndpointsConfiguration
191 make_endorsement_endpoint_configuration(
192 const Attestation& quote,
193 const snp::EndorsementsServers& endorsements_servers = {})
194 {
195 EndorsementEndpointsConfiguration config;
196
197 auto chip_id_hex = fmt::format("{:02x}", fmt::join(quote.chip_id, ""));
198 auto reported_tcb =
199 fmt::format("{:0x}", *(uint64_t*)(&quote.reported_tcb));
200
201 if (endorsements_servers.empty())
202 {
203 // Default to Azure server if no servers are specified
204 config.servers.emplace_back(make_azure_endorsements_server(
205 default_azure_endorsements_endpoint_host, chip_id_hex, reported_tcb));
206 return config;
207 }
208
209 for (auto const& server : endorsements_servers)
210 {
211 switch (server.type)
212 {
213 case EndorsementsEndpointType::Azure:
214 {
215 auto url =
216 server.url.value_or(default_azure_endorsements_endpoint_host);
217 config.servers.emplace_back(
218 make_azure_endorsements_server(url, chip_id_hex, reported_tcb));
219 break;
220 }
221 case EndorsementsEndpointType::AMD:
222 {
223 auto boot_loader =
224 fmt::format("{}", quote.reported_tcb.boot_loader);
225 auto tee = fmt::format("{}", quote.reported_tcb.tee);
226 auto snp = fmt::format("{}", quote.reported_tcb.snp);
227 auto microcode = fmt::format("{}", quote.reported_tcb.microcode);
228
229 auto url =
230 server.url.value_or(default_azure_endorsements_endpoint_host);
231 config.servers.emplace_back(make_amd_endorsements_server(
232 url, chip_id_hex, boot_loader, tee, snp, microcode));
233 break;
234 }
235 default:
236 {
237 throw std::logic_error(fmt::format(
238 "Unsupported endorsements server type: {}", server.type));
239 }
240 }
241 }
242
243 return config;
244 }
