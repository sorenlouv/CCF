48 static void generate_quote(
49 attestation_report_data& report_data,
50 RetrieveEndorsementCallback endorsement_cb,
51 const snp::EndorsementsServers& endorsements_servers = {})
52 {
53 QuoteInfo node_quote_info = {};
54
55 node_quote_info.format = QuoteFormat::amd_sev_snp_v1;
56 int fd = open(snp::DEVICE, O_RDWR | O_CLOEXEC);
57 if (fd < 0)
58 {
59 throw std::logic_error(fmt::format("Failed to open \"{}\"", snp::DEVICE));
60 }
61
62 snp::AttestationReq req = {};
63 snp::AttestationResp resp = {};
64
65 // Arbitrary report data
66 memcpy(
67 req.report_data, report_data.data(), snp_attestation_report_data_size);
68
69 // Documented at
70 // https://www.kernel.org/doc/html/latest/virt/coco/sev-guest.html
71 snp::GuestRequest payload = {
72 .req_msg_type = snp::MSG_REPORT_REQ,
73 .rsp_msg_type = snp::MSG_REPORT_RSP,
74 .msg_version = 1,
75 .request_len = sizeof(req),
76 .request_uaddr = reinterpret_cast<uint64_t>(&req),
77 .response_len = sizeof(resp),
78 .response_uaddr = reinterpret_cast<uint64_t>(&resp),
79 .error = 0};
80
81 int rc = ioctl(fd, SEV_SNP_GUEST_MSG_REPORT, &payload);
82 if (rc < 0)
83 {
84 CCF_APP_FAIL("IOCTL call failed: {}", strerror(errno));
85 CCF_APP_FAIL("Payload error: {}", payload.error);
86 throw std::logic_error("Failed to issue ioctl SEV_SNP_GUEST_MSG_REPORT");
87 }
88
89 auto quote = &resp.report;
90 auto quote_bytes = reinterpret_cast<uint8_t*>(&resp.report);
91 node_quote_info.quote.assign(quote_bytes, quote_bytes + resp.report_size);
92
93 if (endorsement_cb != nullptr)
94 {
95 endorsement_cb(
96 node_quote_info,
97 snp::make_endorsement_endpoint_configuration(
98 *quote, endorsements_servers));
99 }
100 }

105 static void verify_quote(
106 const QuoteInfo& quote_info,
107 attestation_measurement& unique_id,
108 attestation_report_data& report_data)
109 {
110 auto is_sev_snp = access(snp::DEVICE, F_OK) == 0;
111
112 if (quote_info.format == QuoteFormat::insecure_virtual)
113 {
114 if (is_sev_snp)
115 {
116 throw std::logic_error(
117 "Cannot verify virtual quote if node is SEV-SNP");
118 }
119 unique_id = {};
120 report_data = {};
121 }
122 else if (quote_info.format == QuoteFormat::amd_sev_snp_v1)
123 {
124 if (!is_sev_snp)
125 {
126 throw std::logic_error(
127 "Cannot verify SEV-SNP quote if node is virtual");
128 }
129
130 auto quote =
131 *reinterpret_cast<const snp::Attestation*>(quote_info.quote.data());
132
133 std::copy(
134 std::begin(quote.report_data),
135 std::end(quote.report_data),
136 report_data.begin());
137 std::copy(
138 std::begin(quote.measurement),
139 std::end(quote.measurement),
140 unique_id.begin());
141
142 auto certificates = crypto::split_x509_cert_bundle(std::string(
143 quote_info.endorsements.begin(), quote_info.endorsements.end()));
144 auto chip_certificate = certificates[0];
145 auto sev_version_certificate = certificates[1];
146 auto root_certificate = certificates[2];
147
148 auto root_cert_verifier = crypto::make_verifier(root_certificate);
149
150 if (
151 root_cert_verifier->public_key_pem().str() !=
152 snp::amd_milan_root_signing_public_key)
153 {
154 throw std::logic_error(fmt::format(
155 "The root of trust public key for this attestation was not "
156 "the expected one {}",
157 root_cert_verifier->public_key_pem().str()));
158 }
159
160 if (!root_cert_verifier->verify_certificate({&root_certificate}))
161 {
162 throw std::logic_error(
163 "The root of trust public key for this attestation was not "
164 "self signed as expected");
165 }
166
167 auto chip_cert_verifier = crypto::make_verifier(chip_certificate);
168 if (!chip_cert_verifier->verify_certificate(
169 {&root_certificate, &sev_version_certificate}))
170 {
171 throw std::logic_error(
172 "The chain of signatures from the root of trust to this "
173 "attestation is broken");
174 }
175
176 if (quote.signature_algo != snp::SignatureAlgorithm::ecdsa_p384_sha384)
177 {
178 throw std::logic_error(fmt::format(
179 "Unsupported signature algorithm: {} (supported: {})",
180 quote.signature_algo,
181 snp::SignatureAlgorithm::ecdsa_p384_sha384));
182 }
183
184 // Make ASN1 DER signature
185 auto quote_signature = crypto::ecdsa_sig_from_r_s(
186 quote.signature.r,
187 sizeof(quote.signature.r),
188 quote.signature.s,
189 sizeof(quote.signature.s),
190 false /* little endian */
191 );
192
193 std::span quote_without_signature{
194 quote_info.quote.data(),
195 quote_info.quote.size() - sizeof(quote.signature)};
196 if (!chip_cert_verifier->verify(quote_without_signature, quote_signature))
197 {
198 throw std::logic_error(
199 "Chip certificate (VCEK) did not sign this attestation");
200 }
201
202 if (quote.policy.debug == 1)
203 {
204 throw std::logic_error(
205 "SNP attestation report guest policy debugging must not be "
206 "enabled");
207 }
208
209 if (quote.policy.migrate_ma == 1)
210 {
211 throw std::logic_error("Migration agents must not be enabled");
212 }
213 }
214 else
215 {
216 if (is_sev_snp)
217 {
218 throw std::logic_error(fmt::format(
219 "Cannot verify non SEV-SNP attestation report: {}",
220 quote_info.format));
221 }
222 else
223 {
224 throw std::logic_error(
225 "Cannot verify real attestation report on virtual build");
226 }
227 }
228 }

232 static void generate_quote(
233 attestation_report_data& report_data,
234 RetrieveEndorsementCallback endorsement_cb,
235 const snp::EndorsementsServers& endorsements_servers = {})
236 {
237 QuoteInfo node_quote_info = {};
238 node_quote_info.format = QuoteFormat::oe_sgx_v1;
239
240 sgx::Evidence evidence;
241 sgx::Endorsements endorsements;
242 sgx::SerialisedClaims serialised_custom_claims;
243
244 // Serialise hash of node's public key as a custom claim
245 const size_t custom_claim_length = 1;
246 oe_claim_t custom_claim;
247 custom_claim.name = const_cast<char*>(sgx::report_data_claim_name);
248 custom_claim.value = report_data.data();
249 custom_claim.value_size = report_data.size();
250
251 auto rc = oe_serialize_custom_claims(
252 &custom_claim,
253 custom_claim_length,
254 &serialised_custom_claims.buffer,
255 &serialised_custom_claims.size);
256 if (rc != OE_OK)
257 {
258 throw std::logic_error(fmt::format(
259 "Could not serialise node's public key as quote custom claim: {}",
260 oe_result_str(rc)));
261 }
262
263 rc = oe_get_evidence(
264 &sgx::oe_quote_format,
265 0,
266 serialised_custom_claims.buffer,
267 serialised_custom_claims.size,
268 nullptr,
269 0,
270 &evidence.buffer,
271 &evidence.size,
272 &endorsements.buffer,
273 &endorsements.size);
274 if (rc != OE_OK)
275 {
276 throw std::logic_error(
277 fmt::format("Failed to get evidence: {}", oe_result_str(rc)));
278 }
279
280 node_quote_info.quote.assign(
281 evidence.buffer, evidence.buffer + evidence.size);
282 node_quote_info.endorsements.assign(
283 endorsements.buffer, endorsements.buffer + endorsements.size);
284
285 if (endorsement_cb != nullptr)
286 {
287 endorsement_cb(node_quote_info, {});
288 }
289 }

291 static void verify_quote(
292 const QuoteInfo& quote_info,
293 attestation_measurement& unique_id,
294 attestation_report_data& report_data)
295 {
296 if (quote_info.format != QuoteFormat::oe_sgx_v1)
297 {
298 throw std::logic_error(
299 fmt::format("Cannot verify non OE SGX report: {}", quote_info.format));
300 }
301
302 sgx::Claims claims;
303
304 auto rc = oe_verify_evidence(
305 &sgx::oe_quote_format,
306 quote_info.quote.data(),
307 quote_info.quote.size(),
308 quote_info.endorsements.data(),
309 quote_info.endorsements.size(),
310 nullptr,
311 0,
312 &claims.data,
313 &claims.length);
314 if (rc != OE_OK)
315 {
316 throw std::logic_error(
317 fmt::format("Failed to verify evidence: {}", oe_result_str(rc)));
318 }
319
320 bool unique_id_found = false;
321 bool sgx_report_data_found = false;
322 for (size_t i = 0; i < claims.length; i++)
323 {
324 auto& claim = claims.data[i];
325 auto claim_name = std::string(claim.name);
326 if (claim_name == OE_CLAIM_UNIQUE_ID)
327 {
328 std::copy(
329 claim.value, claim.value + claim.value_size, unique_id.begin());
330 unique_id_found = true;
331 }
332 else if (claim_name == OE_CLAIM_CUSTOM_CLAIMS_BUFFER)
333 {
334 // Find sgx report data in custom claims
335 sgx::CustomClaims custom_claims;
336 rc = oe_deserialize_custom_claims(
337 claim.value,
338 claim.value_size,
339 &custom_claims.data,
340 &custom_claims.length);
341 if (rc != OE_OK)
342 {
343 throw std::logic_error(fmt::format(
344 "Failed to deserialise custom claims", oe_result_str(rc)));
345 }
346
347 for (size_t j = 0; j < custom_claims.length; j++)
348 {
349 auto& custom_claim = custom_claims.data[j];
350 if (std::string(custom_claim.name) == sgx::report_data_claim_name)
351 {
352 if (custom_claim.value_size != report_data.size())
353 {
354 throw std::logic_error(fmt::format(
355 "Expected {} of size {}, had size {}",
356 sgx::report_data_claim_name,
357 report_data.size(),
358 custom_claim.value_size));
359 }
360
361 std::copy(
362 custom_claim.value,
363 custom_claim.value + custom_claim.value_size,
364 report_data.begin());
365 sgx_report_data_found = true;
366 break;
367 }
368 }
369 }
370 }
371
372 if (!unique_id_found)
373 {
374 throw std::logic_error("Could not find measurement");
375 }
376
377 if (!sgx_report_data_found)
378 {
379 throw std::logic_error("Could not find report data");
380 }
381 }
