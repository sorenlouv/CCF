44 void handle_incoming_data_thread(std::vector<uint8_t>&& data) override
45 {
46 tls_io->recv_buffered(data.data(), data.size());
47
48 LOG_TRACE_FMT("recv called with {} bytes", data.size());
49
50 // Try to parse all incoming data, reusing the vector we were just passed
51 // for storage. Increase the size if the received vector was too small
52 // (for the case where this chunk is very small, but we had some previous
53 // data to continue reading).
54 constexpr auto min_read_block_size = 4096;
55 if (data.size() < min_read_block_size)
56 {
57 data.resize(min_read_block_size);
58 }
59
60 auto n_read = tls_io->read(data.data(), data.size(), false);
61
62 while (true)
63 {
64 if (n_read == 0)
65 {
66 return;
67 }
68
69 LOG_TRACE_FMT("Going to parse {} bytes", n_read);
70
71 bool cont = parse({data.data(), n_read});
72 if (!cont)
73 {
74 return;
75 }
76
77 // Used all provided bytes - check if more are available
78 n_read = tls_io->read(data.data(), data.size(), false);
79 }
80 }

197 bool parse(std::span<const uint8_t> data) override
198 {
199 try
200 {
201 server_parser.execute(data.data(), data.size());
202 return true;
203 }
204 catch (const std::exception& e)
205 {
206 LOG_DEBUG_FMT("Error parsing HTTP request: {}", e.what());
207
208 http::HeaderMap headers;
209 headers[http::headers::CONTENT_TYPE] =
210 http::headervalues::contenttype::TEXT;
211
212 auto body = fmt::format(
213 "Unable to parse data as a HTTP request. Error details are "
214 "below.\n\n{}",
215 e.what());
216
217 // NB: If we have an error where we don't know a stream ID, we respond
218 // on the default stream
219 send_response(
220 HTTP_STATUS_BAD_REQUEST,
221 std::move(headers),
222 {},
223 {(const uint8_t*)body.data(), body.size()});
224
225 tls_io->close();
226 }
227 return false;

230 void handle_request(
231 llhttp_method verb,
232 const std::string_view& url,
233 http::HeaderMap&& headers,
234 std::vector<uint8_t>&& body,
235 int32_t stream_id) override
236 {
237 LOG_TRACE_FMT(
238 "Processing msg({}, {} [{} bytes])",
239 llhttp_method_name(verb),
240 url,
241 body.size());
242
243 auto responder = get_stream_responder(stream_id);
244 auto session_ctx = get_session_ctx(stream_id);
245
246 try
247 {
248 std::shared_ptr<http::HttpRpcContext> rpc_ctx = nullptr;
249 try
250 {
251 rpc_ctx = std::make_shared<HttpRpcContext>(
252 session_ctx, verb, url, std::move(headers), std::move(body));
253 }
254 catch (std::exception& e)
255 {
256 send_odata_error_response(ccf::ErrorDetails{
257 HTTP_STATUS_INTERNAL_SERVER_ERROR,
258 ccf::errors::InternalError,
259 fmt::format("Error constructing RpcContext: {}", e.what())});
260 }
261
262 const auto actor_opt = http::extract_actor(*rpc_ctx);
263 std::optional<std::shared_ptr<ccf::RpcHandler>> search;
264 ccf::ActorsType actor = ccf::ActorsType::unknown;
265 if (actor_opt.has_value())
266 {
267 const auto& actor_s = actor_opt.value();
268 actor = rpc_map->resolve(actor_s);
269 search = rpc_map->find(actor);
270 }
271 if (
272 !actor_opt.has_value() || actor == ccf::ActorsType::unknown ||
273 !search.has_value())
274 {
275 // if there is no actor, proceed with the "app" as the ActorType and
276 // process the request
277 search = rpc_map->find(ccf::ActorsType::users);
278 }
279
280 search.value()->process(rpc_ctx);
281
282 if (rpc_ctx->response_is_pending)
283 {
284 // If the RPC is pending, hold the connection.
285 LOG_TRACE_FMT("Pending");
286 return;
287 }
288 else
289 {
290 responder->send_response(
291 rpc_ctx->get_response_http_status(),
292 rpc_ctx->get_response_headers(),
293 rpc_ctx->get_response_trailers(),
294 std::move(rpc_ctx->get_response_body()));
295 }
296 }
297 catch (const std::exception& e)
298 {
299 responder->send_odata_error_response(ccf::ErrorDetails{
300 HTTP_STATUS_INTERNAL_SERVER_ERROR,
301 ccf::errors::InternalError,
302 fmt::format("Exception: {}", e.what())});
303
304 // On any exception, close the connection.
305 LOG_FAIL_FMT("Closing connection");
306 LOG_DEBUG_FMT("Closing connection due to exception: {}", e.what());
307 tls_io->close();
308 throw;
309 }
310 }
