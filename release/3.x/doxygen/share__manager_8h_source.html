52 std::vector<uint8_t> wrap(const LedgerSecretPtr& ledger_secret)
53 {
54 if (has_wrapped)
55 {
56 throw std::logic_error(
57 "Ledger secret wrapping key has already wrapped once");
58 }
59
60 crypto::GcmCipher encrypted_ls(ledger_secret->raw_key.size());
61
62 crypto::make_key_aes_gcm(data)->encrypt(
63 encrypted_ls.hdr.get_iv(), // iv is always 0 here as the share wrapping
64 // key is never re-used for encryption
65 ledger_secret->raw_key,
66 {},
67 encrypted_ls.cipher,
68 encrypted_ls.hdr.tag);
69
70 has_wrapped = true;
71
72 return encrypted_ls.serialise();
73 }

75 LedgerSecretPtr unwrap(
76 const std::vector<uint8_t>& wrapped_latest_ledger_secret)
77 {
78 crypto::GcmCipher encrypted_ls;
79 encrypted_ls.deserialise(wrapped_latest_ledger_secret);
80 std::vector<uint8_t> decrypted_ls;
81
82 if (!crypto::make_key_aes_gcm(data)->decrypt(
83 encrypted_ls.hdr.get_iv(),
84 encrypted_ls.hdr.tag,
85 encrypted_ls.cipher,
86 {},
87 decrypted_ls))
88 {
89 throw std::logic_error("Unwrapping latest ledger secret failed");
90 }
91
92 return std::make_shared<LedgerSecret>(std::move(decrypted_ls));
93 }

100 // The ShareManager class provides the interface between the ledger secrets
101 // object and the shares, ledger secrets and submitted shares KV tables. In
102 // particular, it is used to:
103 // - Issue new recovery shares whenever required (e.g. on startup, rekey and
104 // membership updates)
105 // - Re-assemble the ledger secrets on recovery, once a threshold of members
106 // have successfully submitted their shares

112 EncryptedSharesMap compute_encrypted_shares(
113 kv::Tx& tx, const LedgerSecretWrappingKey& ls_wrapping_key)
114 {
115 EncryptedSharesMap encrypted_shares;
116
117 auto secret_to_split =
118 ls_wrapping_key.get_raw_data<SecretSharing::SplitSecret>();
119
120 GenesisGenerator g(network, tx);
121 auto active_recovery_members_info = g.get_active_recovery_members();
122 size_t recovery_threshold = g.get_recovery_threshold();
123
124 if (active_recovery_members_info.empty())
125 {
126 throw std::logic_error(
127 "There should be at least one active recovery member to issue "
128 "recovery shares");
129 }
130
131 if (recovery_threshold == 0)
132 {
133 throw std::logic_error(
134 "Recovery threshold should be set before recovery "
135 "shares are computed");
136 }
137
138 if (recovery_threshold > active_recovery_members_info.size())
139 {
140 throw std::logic_error(fmt::format(
141 "Recovery threshold {} should be equal to or less than the number of "
142 "active recovery members {}",
143 recovery_threshold,
144 active_recovery_members_info.size()));
145 }
146
147 auto shares = SecretSharing::split(
148 secret_to_split,
149 active_recovery_members_info.size(),
150 recovery_threshold);
151
152 size_t share_index = 0;
153 for (auto const& [member_id, enc_pub_key] : active_recovery_members_info)
154 {
155 auto member_enc_pubk = crypto::make_rsa_public_key(enc_pub_key);
156 auto raw_share = std::vector<uint8_t>(
157 shares[share_index].begin(), shares[share_index].end());
158 encrypted_shares[member_id] = member_enc_pubk->rsa_oaep_wrap(raw_share);
159 OPENSSL_cleanse(raw_share.data(), raw_share.size());
160 OPENSSL_cleanse(shares[share_index].data(), shares[share_index].size());
161 share_index++;
162 }
163
164 OPENSSL_cleanse(secret_to_split.data(), secret_to_split.size());
165
166 return encrypted_shares;
167 }

169 void shuffle_recovery_shares(
170 kv::Tx& tx, const LedgerSecretPtr& latest_ledger_secret)
171 {
172 auto ls_wrapping_key = LedgerSecretWrappingKey();
173 auto wrapped_latest_ls = ls_wrapping_key.wrap(latest_ledger_secret);
174 auto recovery_shares = tx.rw(network.shares);
175 recovery_shares->put(
176 {wrapped_latest_ls,
177 compute_encrypted_shares(tx, ls_wrapping_key),
178 latest_ledger_secret->previous_secret_stored_version});
179 }

181 void set_recovery_shares_info(
182 kv::Tx& tx,
183 const LedgerSecretPtr& latest_ledger_secret,
184 const std::optional<VersionedLedgerSecret>& previous_ledger_secret =
185 std::nullopt,
186 std::optional<kv::Version> latest_ls_version = std::nullopt)
187 {
188 // First, generate a fresh ledger secrets wrapping key and wrap the
189 // latest ledger secret with it. Then, encrypt the penultimate ledger
190 // secret with the latest ledger secret and split the ledger secret
191 // wrapping key, allocating a new share for each active recovery member.
192 // Finally, encrypt each share with the public key of each member and
193 // record it in the shares table.
194
195 shuffle_recovery_shares(tx, latest_ledger_secret);
196
197 auto encrypted_ls = tx.rw(network.encrypted_ledger_secrets);
198
199 std::vector<uint8_t> encrypted_previous_secret = {};
200 kv::Version version_previous_secret = kv::NoVersion;
201 if (previous_ledger_secret.has_value())
202 {
203 version_previous_secret = previous_ledger_secret->first;
204
205 crypto::GcmCipher encrypted_previous_ls(
206 previous_ledger_secret->second->raw_key.size());
207 encrypted_previous_ls.hdr.set_random_iv();
208
209 latest_ledger_secret->key->encrypt(
210 encrypted_previous_ls.hdr.get_iv(),
211 previous_ledger_secret->second->raw_key,
212 {},
213 encrypted_previous_ls.cipher,
214 encrypted_previous_ls.hdr.tag);
215
216 encrypted_previous_secret = encrypted_previous_ls.serialise();
217 encrypted_ls->put(
218 {PreviousLedgerSecretInfo(
219 std::move(encrypted_previous_secret),
220 version_previous_secret,
221 encrypted_ls->get_version_of_previous_write()),
222 latest_ls_version});
223 }
224 else
225 {
226 encrypted_ls->put({std::nullopt, latest_ls_version});
227 }
228 }

267 LedgerSecretWrappingKey combine_from_encrypted_submitted_shares(kv::Tx& tx)
268 {
269 auto encrypted_submitted_shares =
270 tx.rw(network.encrypted_submitted_shares);
271 auto config = tx.rw(network.config);
272
273 std::vector<SecretSharing::Share> shares = {};
274 encrypted_submitted_shares->foreach(
275 [&shares, &tx, this](
276 const MemberId, const EncryptedSubmittedShare& encrypted_share) {
277 SecretSharing::Share share;
278 auto decrypted_share = decrypt_submitted_share(
279 encrypted_share, network.ledger_secrets->get_latest(tx).second);
280 std::copy_n(
281 decrypted_share.begin(),
282 SecretSharing::SHARE_LENGTH,
283 share.begin());
284 OPENSSL_cleanse(decrypted_share.data(), decrypted_share.size());
285 shares.emplace_back(share);
286 return true;
287 });
288
289 auto recovery_threshold = config->get()->recovery_threshold;
290 if (recovery_threshold > shares.size())
291 {
292 throw std::logic_error(fmt::format(
293 "Error combining recovery shares: only {} recovery shares were "
294 "submitted but recovery threshold is {}",
295 shares.size(),
296 recovery_threshold));
297 }
298
299 return LedgerSecretWrappingKey(
300 SecretSharing::combine(shares, shares.size()));
301 }

368 LedgerSecretsMap restore_recovery_shares_info(
369 kv::Tx& tx,
370 const RecoveredEncryptedLedgerSecrets& recovery_ledger_secrets)
371 {
372 // First, re-assemble the ledger secret wrapping key from the submitted
373 // encrypted shares. Then, unwrap the latest ledger secret and use it to
374 // decrypt the sequence of recovered ledger secrets, from the last one.
375
376 if (recovery_ledger_secrets.empty())
377 {
378 throw std::logic_error("No recovery ledger secrets");
379 }
380
381 auto recovery_shares_info = tx.ro(network.shares)->get();
382 if (!recovery_shares_info.has_value())
383 {
384 throw std::logic_error(
385 "Failed to retrieve current recovery shares info");
386 }
387
388 auto restored_ls = combine_from_encrypted_submitted_shares(tx).unwrap(
389 recovery_shares_info->wrapped_latest_ledger_secret);
390 auto decryption_key = restored_ls->raw_key;
391
392 LOG_DEBUG_FMT(
393 "Recovering {} encrypted ledger secrets",
394 recovery_ledger_secrets.size());
395
396 auto& current_ledger_secret_version =
397 recovery_ledger_secrets.back().next_version;
398 if (!current_ledger_secret_version.has_value())
399 {
400 // This should always be set by the recovery hook, which sets this to
401 // the version at which it is called if unset in the store
402 throw std::logic_error("Current ledger secret version should be set");
403 }
404
405 auto encrypted_previous_ledger_secret =
406 tx.ro(network.encrypted_ledger_secrets);
407
408 LedgerSecretsMap restored_ledger_secrets = {};
409 auto s = restored_ledger_secrets.emplace(
410 current_ledger_secret_version.value(),
411 std::make_shared<LedgerSecret>(
412 std::move(restored_ls->raw_key),
413 encrypted_previous_ledger_secret->get_version_of_previous_write()));
414 auto latest_ls = s.first->second;
415
416 for (auto it = recovery_ledger_secrets.rbegin();
417 it != recovery_ledger_secrets.rend();
418 it++)
419 {
420 if (!it->previous_ledger_secret.has_value())
421 {
422 // Very first entry does not encrypt any other ledger secret
423 break;
424 }
425
426 auto decrypted_ls_raw = decrypt_previous_ledger_secret_raw(
427 latest_ls, it->previous_ledger_secret->encrypted_data);
428
429 auto s = restored_ledger_secrets.emplace(
430 it->previous_ledger_secret->version,
431 std::make_shared<LedgerSecret>(
432 std::move(decrypted_ls_raw),
433 it->previous_ledger_secret->previous_secret_stored_version));
434 latest_ls = s.first->second;
435 }
436
437 return restored_ledger_secrets;
438 }
