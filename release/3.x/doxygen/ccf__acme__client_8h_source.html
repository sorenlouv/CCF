106 virtual void check_expiry(
107 std::shared_ptr<kv::Store> tables,
108 std::unique_ptr<NetworkIdentity>& identity)
109 {
110 auto now = std::chrono::system_clock::now();
111 bool renew = false;
112 auto tx = tables->create_read_only_tx();
113 auto certs = tx.ro<ACMECertificates>(Tables::ACME_CERTIFICATES);
114 auto cert = certs->get(config_name);
115 if (cert)
116 {
117 auto v = crypto::make_verifier(*cert);
118 double rem_pct = v->remaining_percentage(now);
119 LOG_TRACE_FMT(
120 "ACME: remaining certificate validity for '{}': {}%, {} seconds",
121 config_name,
122 100.0 * rem_pct,
123 v->remaining_seconds(now));
124 renew = rem_pct < 0.33;
125 }
126
127 if (renew || !cert)
128 {
129 get_certificate(make_key_pair(identity->priv_key));
130 }
131 }

197 virtual void on_challenge(
198 const std::string& token, const std::string& response) override
199 {
200 if (!challenge_handler)
201 {
202 throw std::runtime_error("No ACME challenge handler");
203 }
204
205 using namespace threading;
206
207 challenge_handler->token_responses[token] = response;
208
209 auto msg = std::make_unique<Tmsg<ACMEClientMsg>>(
210 [](std::unique_ptr<Tmsg<ACMEClientMsg>> msg) {
211 auto& client = msg->data.client;
212 auto& handler = msg->data.handler;
213 auto& token = msg->data.token;
214 if (handler->ready(token))
215 {
216 client.start_challenge(token);
217 }
218 else
219 {
220 ThreadMessaging::thread_messaging.add_task_after(
221 std::move(msg), std::chrono::seconds(1));
222 }
223 },
224 *this,
225 challenge_handler,
226 token);
227
228 ThreadMessaging::thread_messaging.add_task_after(
229 std::move(msg), std::chrono::seconds(1));
230 }
231
232 virtual void on_challenge_finished(const std::string& token) override
233 {
234 if (challenge_handler)
235 {
236 challenge_handler->remove(token);
237 }
238 }

240 virtual void on_certificate(const std::string& certificate) override
241 {
242 // Write the endorsed certificate to the certificate table; all nodes
243 // will install it later, in the global hook on that table.
244 auto tx = store->create_tx();
245 auto certs = tx.rw<ACMECertificates>(Tables::ACME_CERTIFICATES);
246 certs->put(config_name, crypto::Pem(certificate));
247 tx.commit();
248 }
