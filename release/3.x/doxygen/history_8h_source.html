225 class Proof
226 {
227 private:
228 HistoryTree::Hash root;
229 std::shared_ptr<HistoryTree::Path> path = nullptr;
230
231 public:
232 Proof() {}
233
234 Proof(const std::vector<uint8_t>& v)
235 {
236 size_t position = 0;
237 root.deserialise(v, position);
238 path = std::make_shared<HistoryTree::Path>(v, position);
239 }
240
241 const HistoryTree::Hash& get_root() const
242 {
243 return root;
244 }
245
246 std::shared_ptr<HistoryTree::Path> get_path()
247 {
248 return path;
249 }
250
251 Proof(HistoryTree* tree, uint64_t index)
252 {
253 root = tree->root();
254 path = tree->path(index);
255 }
256
257 Proof(const Proof&) = delete;
258
259 bool verify(HistoryTree* tree) const
260 {
261 if (path->max_index() > tree->max_index())
262 {
263 return false;
264 }
265 else if (tree->max_index() == path->max_index())
266 {
267 return tree->root() == root && path->verify(root);
268 }
269 else
270 {
271 auto past_root = tree->past_root(path->max_index());
272 return path->verify(*past_root);
273 }
274 }
275
276 std::vector<uint8_t> to_v() const
277 {
278 std::vector<uint8_t> v;
279 root.serialise(v);
280 path->serialise(v);
281 return v;
282 }
283 };

512 void start_signature_emit_timer() override
513 {
514 struct EmitSigMsg
515 {
516 EmitSigMsg(HashedTxHistory<T>* self_) : self(self_) {}
517 HashedTxHistory<T>* self;
518 };
519
520 auto emit_sig_msg = std::make_unique<threading::Tmsg<EmitSigMsg>>(
521 [](std::unique_ptr<threading::Tmsg<EmitSigMsg>> msg) {
522 auto self = msg->data.self;
523
524 std::unique_lock<ccf::pal::Mutex> mguard(
525 self->signature_lock, std::defer_lock);
526
527 bool should_emit_signature = false;
528
529 if (mguard.try_lock())
530 {
531 auto consensus = self->store.get_consensus();
532 if (consensus != nullptr)
533 {
534 auto sig_disp = consensus->get_signature_disposition();
535 switch (sig_disp)
536 {
537 case kv::Consensus::SignatureDisposition::CANT_REPLICATE:
538 {
539 break;
540 }
541 case kv::Consensus::SignatureDisposition::CAN_SIGN:
542 {
543 if (self->store.committable_gap() > 0)
544 {
545 should_emit_signature = true;
546 }
547 break;
548 }
549 case kv::Consensus::SignatureDisposition::SHOULD_SIGN:
550 {
551 should_emit_signature = true;
552 break;
553 }
554 }
555 }
556 }
557
558 if (should_emit_signature)
559 {
560 msg->data.self->emit_signature();
561 }
562
563 self->emit_signature_timer_entry =
564 threading::ThreadMessaging::thread_messaging.add_task_after(
565 std::move(msg), std::chrono::milliseconds(self->sig_ms_interval));
566 },
567 this);
568
569 emit_signature_timer_entry =
570 threading::ThreadMessaging::thread_messaging.add_task_after(
571 std::move(emit_sig_msg), std::chrono::milliseconds(sig_ms_interval));
572 }
573
574 ~HashedTxHistory()
575 {
576 threading::ThreadMessaging::thread_messaging.cancel_timer_task(
577 emit_signature_timer_entry);
578 }

585 bool init_from_snapshot(
586 const std::vector<uint8_t>& hash_at_snapshot) override
587 {
588 std::lock_guard<ccf::pal::Mutex> guard(state_lock);
589 // The history can be initialised after a snapshot has been applied by
590 // deserialising the tree in the signatures table and then applying the
591 // hash of the transaction at which the snapshot was taken
592 auto tx = store.create_read_only_tx();
593 auto tree_h = tx.template ro<ccf::SerialisedMerkleTree>(
594 ccf::Tables::SERIALISED_MERKLE_TREE);
595 auto tree = tree_h->get();
596 if (!tree.has_value())
597 {
598 LOG_FAIL_FMT("No tree found in serialised tree map");
599 return false;
600 }
601
602 CCF_ASSERT_FMT(
603 !replicated_state_tree.in_range(1),
604 "Tree is not empty before initialising from snapshot");
605
606 replicated_state_tree.deserialise(tree.value());
607
608 crypto::Sha256Hash hash;
609 std::copy_n(
610 hash_at_snapshot.begin(), crypto::Sha256Hash::SIZE, hash.h.begin());
611 replicated_state_tree.append(hash);
612 return true;
613 }

650 bool verify(
651 kv::Term* term = nullptr, PrimarySignature* signature = nullptr) override
652 {
653 auto tx = store.create_read_only_tx();
654 auto signatures =
655 tx.template ro<ccf::Signatures>(ccf::Tables::SIGNATURES);
656 auto sig = signatures->get();
657 if (!sig.has_value())
658 {
659 LOG_FAIL_FMT("No signature found in signatures map");
660 return false;
661 }
662 auto& sig_value = sig.value();
663 if (term)
664 {
665 *term = sig_value.view;
666 }
667
668 if (signature)
669 {
670 *signature = sig_value;
671 }
672
673 auto root = get_replicated_state_root();
674 log_hash(root, VERIFY);
675 return verify_node_signature(tx, sig_value.node, sig_value.sig, root);
676 }

716 kv::Version last_signed_tx = 0;
717 std::chrono::milliseconds time_of_last_signature =

720 ccf::pal::Mutex signature_lock;

722 void try_emit_signature() override
723 {
724 std::unique_lock<ccf::pal::Mutex> mguard(signature_lock, std::defer_lock);
725 if (store.committable_gap() < sig_tx_interval || !mguard.try_lock())
726 {
727 return;
728 }
729
730 if (store.committable_gap() >= sig_tx_interval)
731 {
732 mguard.unlock();
733 emit_signature();
734 }
735 }
736
737 void emit_signature() override
738 {
739 // Signatures are only emitted when there is a consensus
740 auto consensus = store.get_consensus();
741 if (!consensus)
742 {
743 return;
744 }
745
746 if (!endorsed_cert.has_value())
747 {
748 throw std::logic_error(
749 fmt::format("No endorsed certificate set to emit signature"));
750 }
751
752 // NB: This call will also reset consensus' should_sign, so we should
753 // always proceed to producing a signature from here.
754 auto commit_txid = consensus->get_signable_txid();
755 auto txid = store.next_txid();
756
757 last_signed_tx = commit_txid.version;
758 time_of_last_signature =
759 threading::ThreadMessaging::thread_messaging.get_current_time_offset();
760
761 LOG_DEBUG_FMT(
762 "Signed at {} in view: {} commit was: {}.{} (previous .{})",
763 txid.version,
764 txid.term,
765 commit_txid.term,
766 commit_txid.version,
767 commit_txid.previous_version);
768
769 store.commit(
770 txid,
771 std::make_unique<MerkleTreeHistoryPendingTx<T>>(
772 txid, commit_txid, store, *this, id, kp, endorsed_cert.value()),
773 true);
774 }
