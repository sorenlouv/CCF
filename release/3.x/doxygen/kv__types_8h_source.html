49 struct TxID
50 {
51 Term term = 0;
52 Version version = 0;
53
54 TxID() = default;
55 TxID(Term t, Version v) : term(t), version(v) {}
56
57 // Would like to remove these duplicate types, but for now we just do free
58 // conversion
59 TxID(const ccf::TxID& other) : term(other.view), version(other.seqno) {}
60
61 operator ccf::TxID() const
62 {
63 return {term, version};
64 }
65
66 bool operator==(const TxID& other) const
67 {
68 return term == other.term && version == other.version;
69 }
70
71 std::string str() const
72 {
73 return fmt::format("{}.{}", term, version);
74 }
75 };

303 constexpr auto public_domain_prefix = "public:";
304
305 static inline SecurityDomain get_security_domain(const std::string& name)
306 {
307 if (name.starts_with(public_domain_prefix))
308 {
309 return SecurityDomain::PUBLIC;
310 }
311
312 return SecurityDomain::PRIVATE;
313 }
314
315 static inline std::pair<SecurityDomain, AccessCategory> parse_map_name(
316 std::string_view name)
317 {
318 constexpr auto internal_category_prefix = "ccf.internal.";
319 constexpr auto governance_category_prefix = "ccf.gov.";
320 constexpr auto reserved_category_prefix = "ccf.";
321
322 auto security_domain = SecurityDomain::PRIVATE;
323 if (name.starts_with(public_domain_prefix))
324 {
325 name.remove_prefix(strlen(public_domain_prefix));
326 security_domain = SecurityDomain::PUBLIC;
327 }
328
329 auto access_category = AccessCategory::APPLICATION;
330 if (name.starts_with(internal_category_prefix))
331 {
332 access_category = AccessCategory::INTERNAL;
333 }
334 else if (name.starts_with(governance_category_prefix))
335 {
336 access_category = AccessCategory::GOVERNANCE;
337 }
338 else if (name.starts_with(reserved_category_prefix))
339 {
340 throw std::logic_error(fmt::format(
341 "Map name '{}' includes disallowed reserved prefix '{}'",
342 name,
343 reserved_category_prefix));
344 }
345
346 return {security_domain, access_category};
347 }

682 // Setting a short rollback is a work around that should be fixed
683 // shortly. In BFT mode when we deserialize and realize we need to
684 // create a new map we remember this. If we need to create the same
685 // map multiple times (for tx in the same group of append entries) the
686 // first create successes but the second fails because the map is
687 // already there. This works around the problem by stopping just
688 // before the 2nd create (which failed at this point) and when the
689 // primary resends the append entries we will succeed as the map is
690 // already there. This will only occur on BFT startup so not a perf
691 // problem but still need to be resolved.

693 // Thus, a large rollback is one which did not result from the map creating
