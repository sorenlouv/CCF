284 void launch_node()
285 {
286 auto code_id = AttestationProvider::get_code_id(quote_info);
287 if (code_id.has_value())
288 {
289 node_code_id = code_id.value();
290 }
291 else
292 {
293 throw std::logic_error("Failed to extract code id from quote");
294 }
295
296 // Verify that the security policy matches the quoted digest of the policy
297 if (quote_info.format == QuoteFormat::amd_sev_snp_v1)
298 {
299 auto quoted_digest = AttestationProvider::get_host_data(quote_info);
300 if (!quoted_digest.has_value())
301 {
302 throw std::logic_error("Unable to find security policy");
303 }
304
305 if (!config.security_policy.has_value())
306 {
307 LOG_INFO_FMT(
308 "Security Policy environment variable not set, skipping check "
309 "against digest");
310 }
311 else
312 {
313 auto digest = crypto::Sha256Hash(config.security_policy.value());
314 if (digest != quoted_digest.value())
315 {
316 throw std::logic_error(fmt::format(
317 "Raw security policy {} digested to {} doesn't match digest {} "
318 "provided in attestation",
319 config.security_policy.value(),
320 digest.hex_str(),
321 quoted_digest.value().hex_str()));
322 }
323 LOG_INFO_FMT("Digest matches raw security policy");
324 }
325 }
326
327 switch (start_type)
328 {
329 case StartType::Start:
330 {
331 create_and_send_boot_request(
332 aft::starting_view_change, true /* Create new consortium */);
333 return;
334 }
335 case StartType::Join:
336 {
337 if (!startup_snapshot.empty())
338 {
339 initialise_startup_snapshot();
340 }
341
342 sm.advance(NodeStartupState::pending);
343 start_join_timer();
344 return;
345 }
346 case StartType::Recover:
347 {
348 setup_recovery_hook();
349 if (!startup_snapshot.empty())
350 {
351 initialise_startup_snapshot(true);
352 snapshotter->set_last_snapshot_idx(last_recovered_idx);
353 }
354
355 sm.advance(NodeStartupState::readingPublicLedger);
356 start_ledger_recovery_unsafe();
357 return;
358 }
359 default:
360 {
361 throw std::logic_error(
362 fmt::format("Node was launched in unknown mode {}", start_type));
363 }
364 }

367 void initiate_quote_generation()
368 {
369 auto fetch_endorsements =
370 [this](
371 const QuoteInfo& quote_info_,
372 const pal::snp::EndorsementEndpointsConfiguration& endpoint_config) {
373 if (quote_info_.format != QuoteFormat::amd_sev_snp_v1)
374 {
375 // Note: Node lock is already taken here as this is called back
376 // synchronously with the call to pal::generate_quote
377 CCF_ASSERT_FMT(
378 quote_info_.format == QuoteFormat::insecure_virtual ||
379 !quote_info_.endorsements.empty(),
380 "SGX quote generation should have already fetched endorsements");
381 quote_info = quote_info_;
382 launch_node();
383 return;
384 }
385
386 quote_endorsements_client = std::make_shared<QuoteEndorsementsClient>(
387 rpcsessions,
388 endpoint_config,
389 [this, quote_info_](std::vector<uint8_t>&& endorsements) {
390 // Note: Only called for SEV-SNP
391 std::lock_guard<pal::Mutex> guard(lock);
392 quote_info = quote_info_;
393 quote_info.endorsements = std::move(endorsements);
394 try
395 {
396 launch_node();
397 }
398 catch (const std::exception& e)
399 {
400 LOG_FAIL_FMT("{}", e.what());
401 throw;
402 }
403 quote_endorsements_client.reset();
404 });
405
406 quote_endorsements_client->fetch_endorsements();
407 };
408
409 pal::attestation_report_data report_data = {};
410 crypto::Sha256Hash node_pub_key_hash((node_sign_kp->public_key_der()));
411 std::copy(
412 node_pub_key_hash.h.begin(),
413 node_pub_key_hash.h.end(),
414 report_data.begin());
415 pal::generate_quote(
416 report_data,
417 fetch_endorsements,
418 config.attestation.snp_endorsements_servers);
419 }

421 NodeCreateInfo create(
422 StartType start_type_,
423 StartupConfig&& config_,
424 std::vector<uint8_t>&& startup_snapshot_)
425 {
426 std::lock_guard<pal::Mutex> guard(lock);

428 start_type = start_type_;
429
430 config = std::move(config_);
431 startup_snapshot = std::move(startup_snapshot_);

436 node_sign_kp,

438 subject_alt_names,
439 config.startup_host_time,
440 config.node_certificate.initial_validity_days);
441
442 accept_node_tls_connections();
443 open_frontend(ActorsType::nodes);
444

446 // recovered
447 setup_history();
448 setup_snapshotter();
449 setup_encryptor();
450
451 setup_acme_clients();
452
453 initiate_quote_generation();
454
455 switch (start_type)
456 {

458 {
459 network.identity = std::make_unique<ReplicatedNetworkIdentity>(
460 curve_id,
461 config.startup_host_time,
462 config.initial_service_certificate_validity_days);
463
464 network.ledger_secrets->init();
465

467 ServiceStatus::OPENING,
468 config.start.service_configuration.reconfiguration_type.value_or(
469 ReconfigurationType::ONE_TRANSACTION),
470 false,

472

476 LOG_INFO_FMT("Created new node {}", self);

479 case StartType::Join:
480 {
481 LOG_INFO_FMT("Created join node {}", self);
482 return {self_signed_node_cert, {}};
483 }
484 case StartType::Recover:
485 {
486 if (!config.recover.previous_service_identity)
487 {
488 throw std::logic_error(
489 "Recovery requires the certificate of the previous service "
490 "identity");
491 }
492
493 network.identity = std::make_unique<ReplicatedNetworkIdentity>(
494 curve_id,
495 config.startup_host_time,
496 config.initial_service_certificate_validity_days);
497
498 LOG_INFO_FMT("Created recovery node {}", self);
499 return {self_signed_node_cert, network.identity->cert};
500 }
501 default:
502 {
503 throw std::logic_error(
504 fmt::format("Node was started in unknown mode {}", start_type));
505 }
506 }

512
513 void initiate_join_unsafe()
514 {
515 sm.expect(NodeStartupState::pending);
516
517 auto network_ca = std::make_shared<tls::CA>(std::string(
518 config.join.service_cert.begin(), config.join.service_cert.end()));
519
520 auto join_client_cert = std::make_unique<tls::Cert>(
521 network_ca,
522 self_signed_node_cert,
523 node_sign_kp->private_key_pem(),
524 config.join.target_rpc_address);
525
526 // Create RPC client and connect to remote node
527 // Note: For now, assume that target node accepts same application
528 // protocol as this node's main RPC interface
529 auto join_client = rpcsessions->create_client(
530 std::move(join_client_cert),
531 rpcsessions->get_app_protocol_main_interface());
532
533 auto [target_host, target_port] =
534 split_net_address(config.join.target_rpc_address);
535
536 join_client->connect(
537 target_host,
538 target_port,
539 [this](
540 http_status status,
541 http::HeaderMap&& headers,
542 std::vector<uint8_t>&& data) {
543 std::lock_guard<pal::Mutex> guard(lock);
544 if (!sm.check(NodeStartupState::pending))
545 {
546 return;
547 }
548
549 if (status != HTTP_STATUS_OK)
550 {
551 const auto& location = headers.find(http::headers::LOCATION);
552 if (
553 status == HTTP_STATUS_PERMANENT_REDIRECT &&
554 location != headers.end())
555 {
556 const auto& url = http::parse_url_full(location->second);
557 config.join.target_rpc_address =
558 make_net_address(url.host, url.port);
559 LOG_INFO_FMT("Target node redirected to {}", location->second);
560 }
561 else
562 {
563 LOG_FAIL_FMT(
564 "An error occurred while joining the network: {} {}{}",
565 status,
566 http_status_str(status),
567 data.empty() ?
568 "" :
569 fmt::format(" '{}'", std::string(data.begin(), data.end())));
570 }
571 return;
572 }
573
574 auto j = serdes::unpack(data, serdes::Pack::Text);
575
576 JoinNetworkNodeToNode::Out resp;
577 try
578 {
579 resp = j.get<JoinNetworkNodeToNode::Out>();
580 }
581 catch (const std::exception& e)
582 {
583 LOG_FAIL_FMT(
584 "An error occurred while parsing the join network response");
585 LOG_DEBUG_FMT(
586 "An error occurred while parsing the join network response: {}",
587 j.dump());
588 return;
589 }
590
591 // Set network secrets, node id and become part of network.
592 if (
593 resp.node_status == NodeStatus::TRUSTED ||
594 resp.node_status == NodeStatus::LEARNER)
595 {
596 if (resp.network_info->consensus_type != network.consensus_type)
597 {
598 throw std::logic_error(fmt::format(
599 "Enclave initiated with consensus type {} but target node "
600 "responded with consensus {}",
601 network.consensus_type,
602 resp.network_info->consensus_type));
603 }
604
605 network.identity = std::make_unique<ReplicatedNetworkIdentity>(
606 resp.network_info->identity);
607 network.ledger_secrets->init_from_map(
608 std::move(resp.network_info->ledger_secrets));
609
610 crypto::Pem n2n_channels_cert;
611 if (!resp.network_info->endorsed_certificate.has_value())
612 {
613 // Endorsed node certificate is included in join response
614 // from 2.x (CFT only). When joining an existing 1.x service,
615 // self-sign own certificate and use it to endorse TLS
616 // connections.
617 endorsed_node_cert = create_endorsed_node_cert(
618 default_node_cert_validity_period_days);
619 history->set_endorsed_certificate(endorsed_node_cert.value());
620 n2n_channels_cert = endorsed_node_cert.value();
621 open_frontend(ActorsType::members);
622 open_user_frontend();
623 accept_network_tls_connections();
624 }
625 else
626 {
627 n2n_channels_cert =
628 resp.network_info->endorsed_certificate.value();
629 }
630
631 setup_consensus(
632 resp.network_info->service_status.value_or(
633 ServiceStatus::OPENING),
634 resp.network_info->reconfiguration_type.value_or(
635 ReconfigurationType::ONE_TRANSACTION),
636 resp.network_info->public_only,
637 n2n_channels_cert);
638 auto_refresh_jwt_keys();
639
640 if (resp.network_info->public_only)
641 {
642 last_recovered_signed_idx =
643 resp.network_info->last_recovered_signed_idx;
644 setup_recovery_hook();
645 snapshotter->set_snapshot_generation(false);
646 }
647
648 View view = VIEW_UNKNOWN;
649 std::vector<kv::Version> view_history = {};
650 if (startup_snapshot_info)
651 {
652 // It is only possible to deserialise the entire snapshot then,
653 // once the ledger secrets have been passed in by the network
654 kv::ConsensusHookPtrs hooks;
655 deserialise_snapshot(
656 network.tables,
657 startup_snapshot_info->raw,
658 hooks,
659 &view_history,
660 resp.network_info->public_only,
661 config.recover.previous_service_identity);
662
663 for (auto& hook : hooks)
664 {
665 hook->call(consensus.get());
666 }
667
668 auto tx = network.tables->create_read_only_tx();
669 auto signatures = tx.ro(network.signatures);
670 auto sig = signatures->get();
671 if (!sig.has_value())
672 {
673 throw std::logic_error(
674 fmt::format("No signatures found after applying snapshot"));
675 }
676 view = sig->view;
677
678 if (!resp.network_info->public_only)
679 {
680 // Only clear snapshot if not recovering. When joining the
681 // public network the snapshot is used later to initialise the
682 // recovery store
683 startup_snapshot_info.reset();
684 }
685
686 LOG_INFO_FMT(
687 "Joiner successfully resumed from snapshot at seqno {} and "
688 "view {}",
689 network.tables->current_version(),
690 view);
691 }
692
693 consensus->init_as_backup(
694 network.tables->current_version(),
695 view,
696 view_history,
697 last_recovered_signed_idx);
698
699 snapshotter->set_last_snapshot_idx(
700 network.tables->current_version());
701 history->start_signature_emit_timer();
702
703 if (resp.network_info->public_only)
704 {
705 sm.advance(NodeStartupState::partOfPublicNetwork);
706 }
707 else
708 {
709 reset_data(quote_info.quote);
710 reset_data(quote_info.endorsements);
711 sm.advance(NodeStartupState::partOfNetwork);
712 }
713
714 LOG_INFO_FMT(
715 "Node has now joined the network as node {}: {}",
716 self,
717 (resp.network_info->public_only ? "public only" : "all domains"));
718 }
719 else if (resp.node_status == NodeStatus::PENDING)
720 {
721 LOG_INFO_FMT(
722 "Node {} is waiting for votes of members to be trusted", self);
723 }
724 },
725 [this](const std::string& error_msg) {
726 std::lock_guard<pal::Mutex> guard(lock);
727 auto long_error_msg = fmt::format(
728 "Early error when joining existing network at {}: {}. Shutting "
729 "down node gracefully...",
730 config.join.target_rpc_address,
731 error_msg);
732 LOG_FAIL_FMT("{}", long_error_msg);
733 RINGBUFFER_WRITE_MESSAGE(
734 AdminMessage::fatal_error_msg, to_host, long_error_msg);
735 });
736
737 // Send RPC request to remote node to join the network.
738 JoinNetworkNodeToNode::In join_params;
739
740 join_params.node_info_network = config.network;
741 join_params.public_encryption_key = node_encrypt_kp->public_key_pem();
742 join_params.quote_info = quote_info;
743 join_params.consensus_type = network.consensus_type;
744 join_params.startup_seqno = startup_seqno;
745 join_params.certificate_signing_request = node_sign_kp->create_csr(
746 config.node_certificate.subject_name, subject_alt_names);
747 join_params.node_data = config.node_data;
748
749 LOG_DEBUG_FMT(
750 "Sending join request to {}", config.join.target_rpc_address);
751
752 const auto body = serdes::pack(join_params, serdes::Pack::Text);
753
754 http::Request r(
755 fmt::format("/{}/{}", get_actor_prefix(ActorsType::nodes), "join"));
756 r.set_header(
757 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
758 r.set_body(&body);
759
760 join_client->send_request(std::move(r));
761 }

841 void recover_public_ledger_entries(const std::vector<uint8_t>& entries)
842 {
843 std::lock_guard<pal::Mutex> guard(lock);
844
845 sm.expect(NodeStartupState::readingPublicLedger);
846
847 auto data = entries.data();
848 auto size = entries.size();
849
850 if (size == 0)
851 {
852 recover_public_ledger_end_unsafe();
853 return;
854 }
855
856 while (size > 0)
857 {
858 auto entry = consensus::LedgerEnclave::get_entry(data, size);
859
860 LOG_INFO_FMT("Deserialising public ledger entry [{}]", entry.size());
861
862 // When reading the private ledger, deserialise in the recovery store
863
864 kv::ApplyResult result = kv::ApplyResult::FAIL;
865 try
866 {
867 auto r = network.tables->deserialize(entry, ConsensusType::CFT, true);
868 result = r->apply();
869 if (result == kv::ApplyResult::FAIL)
870 {
871 LOG_FAIL_FMT(
872 "Failed to deserialise public ledger entry: {}", result);
873 recover_public_ledger_end_unsafe();
874 return;
875 }
876 ++last_recovered_idx;
877
878 // Not synchronised because consensus isn't effectively running then
879 for (auto& hook : r->get_hooks())
880 {
881 hook->call(consensus.get());
882 }
883 }
884 catch (const std::exception& e)
885 {
886 LOG_FAIL_FMT(
887 "Failed to deserialise public ledger entry: {}", e.what());
888 recover_public_ledger_end_unsafe();
889 return;
890 }
891
892 // If the ledger entry is a signature, it is safe to compact the store
893 if (result == kv::ApplyResult::PASS_SIGNATURE)
894 {
895 // If the ledger entry is a signature, it is safe to compact the store
896 network.tables->compact(last_recovered_idx);
897 auto tx = network.tables->create_read_only_tx();
898 auto last_sig = tx.ro(network.signatures)->get();
899
900 if (!last_sig.has_value())
901 {
902 throw std::logic_error("Signature missing");
903 }
904
905 LOG_DEBUG_FMT(
906 "Read signature at {} for view {}",
907 last_recovered_idx,
908 last_sig->view);
909 // Initial transactions, before the first signature, must have
910 // happened in the first signature's view (eg - if the first
911 // signature is at seqno 20 in view 4, then transactions 1->19 must
912 // also have been in view 4). The brief justification is that while
913 // the first node may start in an arbitrarily high view (it does not
914 // necessarily start in view 1), it cannot _change_ view before a
915 // valid signature.
916 const auto view_start_idx =
917 view_history.empty() ? 1 : last_recovered_signed_idx + 1;
918 CCF_ASSERT_FMT(
919 last_sig->view >= 0,
920 "last_sig->view is invalid, {}",
921 last_sig->view);
922 for (auto i = view_history.size();
923 i < static_cast<size_t>(last_sig->view);
924 ++i)
925 {
926 view_history.push_back(view_start_idx);
927 }
928 last_recovered_signed_idx = last_recovered_idx;
929 }
930 }
931
932 read_ledger_entries(
933 last_recovered_idx + 1, last_recovered_idx + recovery_batch_size);
934 }

955 void recover_public_ledger_end_unsafe()
956 {
957 sm.expect(NodeStartupState::readingPublicLedger);
958
959 // When reaching the end of the public ledger, truncate to last signed
960 // index
961 const auto last_recovered_term = view_history.size();
962 auto new_term = last_recovered_term + aft::starting_view_change;
963 LOG_INFO_FMT("Setting term on public recovery store to {}", new_term);
964
965 // Note: KV term must be set before the first Tx is committed
966 network.tables->rollback(
967 {last_recovered_term, last_recovered_signed_idx}, new_term);
968 ledger_truncate(last_recovered_signed_idx, true);
969 snapshotter->rollback(last_recovered_signed_idx);
970
971 LOG_INFO_FMT(
972 "End of public ledger recovery - Truncating ledger to last signed "
973 "TxID: {}.{}",
974 last_recovered_term,
975 last_recovered_signed_idx);
976
977 auto tx = network.tables->create_read_only_tx();
978 if (network.consensus_type == ConsensusType::BFT)
979 {
980 endorsed_node_cert = create_endorsed_node_cert(
981 config.node_certificate.initial_validity_days);
982 history->set_endorsed_certificate(endorsed_node_cert.value());
983 accept_network_tls_connections();
984 open_frontend(ActorsType::members);
985 }
986
987 network.ledger_secrets->init(last_recovered_signed_idx + 1);
988
989 // Initialise snapshotter after public recovery
990 snapshotter->init_after_public_recovery();
991 snapshotter->set_snapshot_generation(false);
992
993 kv::Version index = 0;
994 kv::Term view = 0;
995 kv::Version global_commit = 0;
996
997 auto ls = tx.ro(network.signatures)->get();
998 if (ls.has_value())
999 {
1000 auto s = ls.value();
1001 index = s.seqno;
1002 view = s.view;
1003 global_commit = s.commit_seqno;
1004 }
1005
1006 auto h = dynamic_cast<MerkleTxHistory*>(history.get());
1007 if (h)
1008 {
1009 h->set_node_id(self);
1010 }
1011
1012 auto service_config = tx.ro(network.config)->get();
1013 auto reconfiguration_type = service_config->reconfiguration_type.value_or(
1014 ReconfigurationType::ONE_TRANSACTION);
1015
1016 setup_consensus(ServiceStatus::OPENING, reconfiguration_type, true);
1017 auto_refresh_jwt_keys();
1018
1019 LOG_DEBUG_FMT(
1020 "Restarting consensus at view: {} seqno: {} commit_seqno {}",
1021 view,
1022 index,
1023 global_commit);
1024
1025 consensus->force_become_primary(index, view, view_history, index);
1026
1027 create_and_send_boot_request(
1028 new_term, false /* Restore consortium from ledger */);
1029 }

1031 //
1032 // funcs in state "readingPrivateLedger"
1033 //
1034 void recover_private_ledger_entries(const std::vector<uint8_t>& entries)
1035 {
1036 std::lock_guard<pal::Mutex> guard(lock);
1037 if (!sm.check(NodeStartupState::readingPrivateLedger))
1038 {
1039 LOG_FAIL_FMT(
1040 "Node in state {} cannot recover private ledger entries", sm.value());
1041 return;
1042 }
1043
1044 auto data = entries.data();
1045 auto size = entries.size();
1046
1047 if (size == 0)
1048 {
1049 recover_private_ledger_end_unsafe();
1050 return;
1051 }
1052
1053 while (size > 0)
1054 {
1055 auto entry = consensus::LedgerEnclave::get_entry(data, size);
1056
1057 LOG_INFO_FMT("Deserialising private ledger entry [{}]", entry.size());
1058
1059 // When reading the private ledger, deserialise in the recovery store
1060 kv::ApplyResult result = kv::ApplyResult::FAIL;
1061 try
1062 {
1063 result =
1064 recovery_store->deserialize(entry, ConsensusType::CFT)->apply();
1065 if (result == kv::ApplyResult::FAIL)
1066 {
1067 LOG_FAIL_FMT(
1068 "Failed to deserialise private ledger entry: {}", result);
1069 // Note: rollback terms do not matter here as recovery store is
1070 // about to be discarded
1071 recovery_store->rollback({0, last_recovered_idx}, 0);
1072 recover_private_ledger_end_unsafe();
1073 return;
1074 }
1075 ++last_recovered_idx;
1076 }
1077 catch (const std::exception& e)
1078 {
1079 LOG_FAIL_FMT(
1080 "Failed to deserialise private ledger entry: {}", e.what());
1081 recover_private_ledger_end_unsafe();
1082 return;
1083 }
1084
1085 if (result == kv::ApplyResult::PASS_SIGNATURE)
1086 {
1087 recovery_store->compact(last_recovered_idx);
1088 }
1089 }
1090
1091 if (recovery_store->current_version() == recovery_v)
1092 {
1093 LOG_INFO_FMT("Reached recovery final version at {}", recovery_v);
1094 recover_private_ledger_end_unsafe();
1095 }
1096 else
1097 {
1098 read_ledger_entries(
1099 last_recovered_idx + 1,
1100 std::min(last_recovered_idx + recovery_batch_size, recovery_v));
1101 }
1102 }

1104 void recover_private_ledger_end_unsafe()
1105 {
1106 // When reaching the end of the private ledger, make sure the same
1107 // ledger has been read and swap in private state
1108
1109 sm.expect(NodeStartupState::readingPrivateLedger);
1110
1111 if (recovery_v != recovery_store->current_version())
1112 {
1113 throw std::logic_error(fmt::format(
1114 "Private recovery did not reach public ledger seqno: {}/{}",
1115 recovery_store->current_version(),
1116 recovery_v));
1117 }
1118
1119 auto h =
1120 dynamic_cast<MerkleTxHistory*>(recovery_store->get_history().get());
1121 if (h->get_replicated_state_root() != recovery_root)
1122 {
1123 throw std::logic_error(fmt::format(
1124 "Root of public store does not match root of private store at {}",
1125 recovery_v));
1126 }
1127
1128 network.tables->swap_private_maps(*recovery_store.get());
1129 recovery_store.reset();
1130
1131 // Raft should deserialise all security domains when network is opened
1132 consensus->enable_all_domains();
1133
1134 // Snapshots are only generated after recovery is complete
1135 snapshotter->set_snapshot_generation(true);
1136
1137 // Open the service
1138 if (consensus->can_replicate())
1139 {
1140 setup_one_off_secret_hook();
1141 auto tx = network.tables->create_tx();
1142
1143 // Clear recovery shares that were submitted to initiate the recovery
1144 // procedure
1145 share_manager.clear_submitted_recovery_shares(tx);
1146
1147 // Shares for the new ledger secret can only be issued now, once the
1148 // previous ledger secrets have been recovered
1149 share_manager.issue_recovery_shares(tx);
1150
1151 GenesisGenerator g(network, tx);
1152 if (!g.open_service())
1153 {
1154 throw std::logic_error("Service could not be opened");
1155 }
1156
1157 // Trigger a snapshot (at next signature) to ensure we have a working
1158 // snapshot signed by the current (now new) service identity, in case
1159 // we need to recover soon again.
1160 trigger_snapshot(tx);
1161
1162 if (tx.commit() != kv::CommitResult::SUCCESS)
1163 {
1164 throw std::logic_error(
1165 "Could not commit transaction when finishing network recovery");
1166 }
1167 }
1168 recovered_encrypted_ledger_secrets.clear();
1169 reset_data(quote_info.quote);
1170 reset_data(quote_info.endorsements);
1171 sm.advance(NodeStartupState::partOfNetwork);
1172 }

1177 // secret before recovery is recorded in the store. This can only be
1178 // fired once, after the recovery shares for the post-recovery ledger

1180 network.tables->set_map_hook(
1181 network.encrypted_ledger_secrets.get_name(),

1189 "Unexpected removal from {} table",
1190 network.encrypted_ledger_secrets.get_name()));

1194 version);
1195
1196 network.tables->unset_map_hook(
1197 network.encrypted_ledger_secrets.get_name());
1198
1199 return kv::ConsensusHookPtr(nullptr);
1200 }));

1383 void transition_service_to_open(
1384 kv::Tx& tx,
1385 AbstractGovernanceEffects::ServiceIdentities identities) override
1386 {
1387 std::lock_guard<pal::Mutex> guard(lock);
1388
1389 auto service = tx.rw<Service>(Tables::SERVICE);
1390 auto service_info = service->get();
1391 if (!service_info.has_value())
1392 {
1393 throw std::logic_error(
1394 "Service information cannot be found to transition service to "
1395 "open");
1396 }
1397
1398 // Idempotence: if the service is already open or waiting for recovery
1399 // shares, this function should succeed with no effect
1400 if (
1401 service_info->status == ServiceStatus::WAITING_FOR_RECOVERY_SHARES ||
1402 service_info->status == ServiceStatus::OPEN)
1403 {
1404 LOG_DEBUG_FMT(
1405 "Service in state {} is already open", service_info->status);
1406 return;
1407 }
1408
1409 if (service_info->status == ServiceStatus::RECOVERING)
1410 {
1411 const auto prev_ident =
1412 tx.ro<PreviousServiceIdentity>(Tables::PREVIOUS_SERVICE_IDENTITY)
1413 ->get();
1414 if (!prev_ident.has_value() || !identities.previous.has_value())
1415 {
1416 throw std::logic_error(
1417 "Recovery with service certificates requires both, a previous "
1418 "service identity written to the KV during recovery genesis and a "
1419 "transition_service_to_open proposal that contains previous and "
1420 "next service certificates");
1421 }
1422
1423 const crypto::Pem from_proposal(
1424 identities.previous->data(), identities.previous->size());
1425 if (prev_ident.value() != from_proposal)
1426 {
1427 throw std::logic_error(fmt::format(
1428 "Previous service identity does not match.\nActual:\n{}\nIn "
1429 "proposal:\n{}",
1430 prev_ident->str(),
1431 from_proposal.str()));
1432 }
1433 }
1434
1435 if (identities.next != service_info->cert)
1436 {
1437 throw std::logic_error(fmt::format(
1438 "Service identity mismatch: the next service identity in the "
1439 "transition_service_to_open proposal does not match the current "
1440 "service identity:\nNext:\n{}\nCurrent:\n{}",
1441 identities.next.str(),
1442 service_info->cert.str()));
1443 }
1444
1445 service_info->previous_service_identity_version =
1446 service->get_version_of_previous_write();
1447
1448 if (is_part_of_public_network())
1449 {
1450 // If the node is in public mode, start accepting member recovery
1451 // shares
1452 share_manager.clear_submitted_recovery_shares(tx);
1453 service_info->status = ServiceStatus::WAITING_FOR_RECOVERY_SHARES;
1454 service->put(service_info.value());
1455 return;
1456 }
1457 else if (is_part_of_network())
1458 {
1459 // Otherwise, if the node is part of the network. Open the network
1460 // straight away. Recovery shares are allocated to each recovery
1461 // member.
1462 try
1463 {
1464 share_manager.issue_recovery_shares(tx);
1465 }
1466 catch (const std::logic_error& e)
1467 {
1468 throw std::logic_error(
1469 fmt::format("Failed to issue recovery shares: {}", e.what()));
1470 }
1471
1472 GenesisGenerator g(network, tx);
1473 g.open_service();
1474 trigger_snapshot(tx);
1475 return;
1476 }
1477 else
1478 {
1479 throw std::logic_error(
1480 fmt::format("Node in state {} cannot open service", sm.value()));
1481 }
1482 }

1696 bool is_ip(const std::string_view& hostname)
1697 {
1698 // IP address components are purely numeric. DNS names may be largely
1699 // numeric, but at least the final component (TLD) must not be
1700 // all-numeric. So this distinguishes "1.2.3.4" (an IP address) from
1701 // "1.2.3.c4m" (a DNS name). "1.2.3." is invalid for either, and will
1702 // throw. Attempts to handle IPv6 by also splitting on ':', but this is
1703 // untested.
1704 const auto final_component =
1705 nonstd::split(nonstd::split(hostname, ".").back(), ":").back();
1706 if (final_component.empty())
1707 {
1708 throw std::runtime_error(fmt::format(
1709 "{} has a trailing period, is not a valid hostname", hostname));
1710 }
1711 for (const auto c : final_component)
1712 {
1713 if (c < '0' || c > '9')
1714 {
1715 return false;
1716 }
1717 }
1718
1719 return true;
1720 }
