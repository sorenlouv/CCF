14 class ACMERpcEndpoints : public CommonEndpointRegistry
15 {
16 public:
17 ACMERpcEndpoints(
18 NetworkState& network, ccfapp::AbstractNodeContext& context) :
19 CommonEndpointRegistry(
20 get_actor_prefix(ActorsType::acme_challenge), context)
21 {
22 auto handler = [this](auto& ctx) {
23 http_status response_status = HTTP_STATUS_INTERNAL_SERVER_ERROR;
24 std::string response_body;
25
26 const auto& path_params = ctx.rpc_ctx->get_request_path_params();
27 const auto url_token_it = path_params.find("token");
28
29 if (url_token_it == path_params.end())
30 {
31 ctx.rpc_ctx->set_response_status(HTTP_STATUS_NOT_FOUND);
32 ctx.rpc_ctx->set_response_body("no token in URL");
33 }
34
35 std::string token = url_token_it->second;
36 LOG_DEBUG_FMT("ACME: challenge request for token '{}'", token);
37
38 auto tit = prepared_responses.find(token);
39 if (tit == prepared_responses.end())
40 {
41 auto prit = prepared_responses.find("");
42 if (prit != prepared_responses.end())
43 {
44 response_status = HTTP_STATUS_OK;
45 response_body = token + "." + prit->second;
46 }
47 else
48 {
49 response_status = HTTP_STATUS_NOT_FOUND;
50 response_body =
51 fmt::format("Challenge response for token '{}' not found", token);
52 }
53 }
54 else
55 {
56 response_status = HTTP_STATUS_OK;
57 response_body = token + "." + tit->second;
58 }
59
60 ctx.rpc_ctx->set_response_status(response_status);
61 ctx.rpc_ctx->set_response_body(std::move(response_body));
62 };
63
64 make_endpoint("/{token}", HTTP_GET, handler, no_auth_required)
65 .set_forwarding_required(endpoints::ForwardingRequired::Never)
66 .set_auto_schema<void, std::string>()
67 .install();
68 }
69
70 virtual ~ACMERpcEndpoints() = default;
71
72 void add(const std::string& token, const std::string& response)
73 {
74 LOG_TRACE_FMT(
75 "ACME: challenge server received response for token '{}' ({})",
76 token,
77 response);
78
79 prepared_responses.emplace(token, response);
80 }
81
82 void remove(const std::string& token)
83 {
84 prepared_responses.erase(token);
85 }
86
87 protected:
88 std::map<std::string, std::string> prepared_responses;
89 };
