72 void get_certificate(
73 std::shared_ptr<crypto::KeyPair> service_key, bool override_time = false)
74 {
75 using namespace std::chrono_literals;
76 using namespace std::chrono;
77
78 bool ok = true;
79 system_clock::duration delta(0);
80
81 if (last_request && !override_time)
82 {
83 // Let's encrypt recommends this retry strategy in their integration
84 // guide, see https://letsencrypt.org/docs/integration-guide/
85
86 delta = system_clock::now() - *last_request;
87 ok = false;
88 switch (num_failed_attempts)
89 {
90 case 0:
91 ok = true;
92 break;
93 case 1:
94 ok = delta >= 1min;
95 break;
96 case 2:
97 ok = delta >= 10min;
98 break;
99 case 3:
100 ok = delta >= 100min;
101 break;
102 default:
103 ok = delta >= 24h;
104 break;
105 }
106 }
107
108 if (ok)
109 {
110 this->service_key = service_key;
111 last_request = system_clock::now();
112 num_failed_attempts++;
113 request_directory();
114 }
115 else
116 {
117 LOG_INFO_FMT(
118 "ACME: Ignoring certificate request due to {} recent failed "
119 "attempt(s) within {} seconds",
120 num_failed_attempts,
121 duration_cast<seconds>(delta).count());
122 }
123 }

174 void make_request(
175 llhttp_method method,
176 const http::URL& url,
177 const std::vector<uint8_t>& body,
178 http_status expected_status,
179 std::function<bool(const http::HeaderMap&, const std::vector<uint8_t>&)>
180 ok_callback)
181 {
182 std::unique_lock<ccf::pal::Mutex> guard(req_lock);
183
184 try
185 {
186 auto port = url.port.empty() ? "443" : url.port;
187 LOG_INFO_FMT(
188 "ACME: Requesting https://{}:{}{}", url.host, port, url.path);
189
190 http::Request r(url.path, method);
191 r.set_header(http::headers::ACCEPT, "*/*");
192 r.set_header(
193 http::headers::HOST, fmt::format("{}:{}", url.host, url.port));
194 if (!body.empty())
195 {
196 r.set_header(http::headers::CONTENT_TYPE, "application/jose+json");
197 r.set_body(&body);
198 }
199 auto req = r.build_request();
200 std::string reqs(req.begin(), req.end());
201 LOG_TRACE_FMT("ACME: request:\n{}", reqs);
202
203 on_http_request(
204 url,
205 std::move(r),
206 [this, expected_status, ok_callback](
207 http_status status,
208 http::HeaderMap&& headers,
209 std::vector<uint8_t>&& data) {
210 for (auto& [k, v] : headers)
211 {
212 LOG_TRACE_FMT("ACME: H: {}: {}", k, v);
213 }
214
215 if (status != expected_status && status != HTTP_STATUS_OK)
216 {
217 LOG_INFO_FMT(
218 "ACME: request failed with status={} and body={}",
219 (int)status,
220 std::string(data.begin(), data.end()));
221 return false;
222 }
223 else
224 {
225 LOG_TRACE_FMT(
226 "ACME: data: {}", std::string(data.begin(), data.end()));
227 }
228
229 auto nonce_opt = get_header_value(headers, "replay-nonce");
230 if (nonce_opt)
231 {
232 nonces.push_back(*nonce_opt);
233 }
234
235 try
236 {
237 ok_callback(headers, data);
238 }
239 catch (const std::exception& ex)
240 {
241 LOG_FAIL_FMT("ACME: response callback failed: {}", ex.what());
242 return false;
243 }
244 return true;
245 });
246 }
247 catch (const std::exception& ex)
248 {
249 LOG_FAIL_FMT("ACME: failed to connect to ACME server: {}", ex.what());
250 }
251 }

425 static void convert_signature_to_ieee_p1363(
426 std::vector<uint8_t>& sig, const crypto::KeyPair& signer)
427 {
428 // Convert signature from ASN.1 format to IEEE P1363
429 const unsigned char* pp = sig.data();
430 ECDSA_SIG* sig_r_s = d2i_ECDSA_SIG(NULL, &pp, sig.size());
431 const BIGNUM* r = ECDSA_SIG_get0_r(sig_r_s);
432 const BIGNUM* s = ECDSA_SIG_get0_s(sig_r_s);
433 size_t sz = signer.coordinates().x.size();
434 sig = std::vector<uint8_t>(2 * sz, 0);
435 BN_bn2binpad(r, sig.data(), sz);
436 BN_bn2binpad(s, sig.data() + sz, sz);
437 ECDSA_SIG_free(sig_r_s);
438 }

440 class JWS : public nlohmann::json::object_t
441 {
442 public:
443 JWS(
444 const nlohmann::json& header_,
445 const nlohmann::json& payload_,
446 crypto::KeyPair& signer_,
447 bool empty_payload = false)
448 {
449 LOG_TRACE_FMT("ACME: JWS header: {}", header_.dump());
450 LOG_TRACE_FMT("ACME: JWS payload: {}", payload_.dump());
451 auto header_b64 = json_to_b64url(header_, false);
452 auto payload_b64 = empty_payload ? "" : json_to_b64url(payload_, false);
453 set(header_b64, payload_b64, signer_);
454 }
455
456 JWS(const nlohmann::json& header_, crypto::KeyPair& signer_) :
457 JWS(header_, nlohmann::json::object_t(), signer_, true)
458 {}
459
460 virtual ~JWS() {}
461
462 protected:
463 void set(
464 const std::string& header_b64,
465 const std::string& payload_b64,
466 crypto::KeyPair& signer)
467 {
468 auto msg = header_b64 + "." + payload_b64;
469 auto sig = signer.sign(s2v(msg));
470 convert_signature_to_ieee_p1363(sig, signer);
471 auto sig_b64 = crypto::b64url_from_raw(sig);
472
473 (*this)["protected"] = header_b64;
474 (*this)["payload"] = payload_b64;
475 (*this)["signature"] = sig_b64;
476 }
477 };

542 static std::pair<std::string, std::string> get_crv_alg(
543 const std::shared_ptr<crypto::KeyPair>& key_pair)
544 {
545 std::string crv, alg;
546 if (key_pair->get_curve_id() == crypto::CurveID::SECP256R1)
547 {
548 crv = "P-256";
549 alg = "ES256";
550 }
551 else if (key_pair->get_curve_id() == crypto::CurveID::SECP384R1)
552 {
553 crv = "P-384";
554 alg = "ES384";
555 }
556 else
557 throw std::runtime_error("Unsupported curve");
558
559 return std::make_pair(crv, alg);
560 }

581 void remove_order(const std::string& order_url)
582 {
583 LOG_TRACE_FMT("ACME: removing order {}", order_url);
584
585 std::unique_lock<ccf::pal::Mutex> guard(orders_lock);
586 for (auto it = active_orders.begin(); it != active_orders.end();)
587 {
588 if (it->order_url == order_url)
589 {
590 for (const auto& [_, challenge] : it->challenges)
591 {
592 on_challenge_finished(challenge.token);
593 }
594 it = active_orders.erase(it);
595 break;
596 }
597 else
598 {
599 it++;
600 }
601 }

604 nlohmann::json mk_kid_header(
605 const std::string& account_url,
606 const std::string& nonce,
607 const std::string& resource_url)
608 {
609 // For all other requests, the request is signed using an existing
610 // account, and there MUST be a "kid" field. This field MUST contain the
611 // account URL received by POSTing to the newAccount resource.
612
613 auto crv_alg = get_crv_alg(account_key_pair);
614
615 nlohmann::json r = {
616 {"alg", crv_alg.second},
617 {"kid", account_url},
618 {"nonce", nonce},
619 {"url", resource_url}};
620
621 return r;
622 }

653 void request_new_account()
654 {
655 std::string new_account_url =
656 directory.at("newAccount").get<std::string>();
657
658 if (nonces.empty())
659 {
660 request_new_nonce([this]() { request_new_account(); });
661 }
662 else
663 {
664 auto nonce = nonces.front();
665 nonces.pop_front();
666
667 auto crv_alg = get_crv_alg(account_key_pair);
668 auto key_coords = account_key_pair->coordinates();
669
670 JWK jwk(
671 "EC",
672 crv_alg.first,
673 crypto::b64url_from_raw(key_coords.x, false),
674 crypto::b64url_from_raw(key_coords.y, false));
675
676 nlohmann::json header = {
677 {"alg", crv_alg.second},
678 {"jwk", jwk},
679 {"nonce", nonce},
680 {"url", new_account_url}};
681
682 nlohmann::json payload = {
683 {"termsOfServiceAgreed", config.terms_of_service_agreed},
684 {"contact", config.contact}};
685
686 JWS jws(header, payload, *account_key_pair);
687
688 http::URL url = with_default_port(new_account_url);
689 make_json_request(
690 HTTP_POST,
691 url,
692 json_to_bytes(jws),
693 HTTP_STATUS_CREATED,
694 [this](const http::HeaderMap& headers, const nlohmann::json& j) {
695 expect_string(j, "status", "valid");
696 account = j;
697 auto loc_opt = get_header_value(headers, "location");
698 request_new_order(loc_opt.value_or(""));
699 });
700 }

719 void request_new_order(const std::string& account_url)
720 {
721 if (nonces.empty())
722 {
723 request_new_nonce(
724 [this, account_url]() { request_new_order(account_url); });
725 }
726 else
727 {
728 auto nonce = nonces.front();
729 nonces.pop_front();
730
731 auto header =
732 mk_kid_header(account_url, nonce, directory.at("newOrder"));
733
734 nlohmann::json payload = {
735 {"identifiers",
736 nlohmann::json::array({
737 {{"type", "dns"}, {"value", config.service_dns_name}},
738 })}};
739
740 if (config.not_before)
741 {
742 payload["notBefore"] = *config.not_before;
743 }
744 if (config.not_after)
745 {
746 payload["notAfter"] = *config.not_after;
747 }
748
749 JWS jws(header, payload, *account_key_pair);
750
751 http::URL url = with_default_port(directory.at("newOrder"));
752 make_json_request(
753 HTTP_POST,
754 url,
755 json_to_bytes(jws),
756 HTTP_STATUS_CREATED,
757 [this, account_url](
758 const http::HeaderMap& headers, const nlohmann::json& j) {
759 expect(j, "status");
760 expect(j, "finalize");
761
762 auto order_url_opt = get_header_value(headers, "location");
763 if (!order_url_opt)
764 {
765 throw std::runtime_error("Missing order location");
766 }
767
768 std::unique_lock<ccf::pal::Mutex> guard(orders_lock);
769 active_orders.emplace_back(Order{
770 ACTIVE, account_url, *order_url_opt, j["finalize"], "", {}, {}});
771
772 Order& order = active_orders.back();
773
774 if (j["status"] == "pending" && j.contains("authorizations"))
775 {
776 expect(j, "authorizations");
777 order.authorizations =
778 j["authorizations"].get<std::unordered_set<std::string>>();
779 guard.unlock();
780 authorize_next_challenge(*order_url_opt);
781 }
782 else if (j["status"] == "ready")
783 {
784 expect(j, "finalize");
785 guard.unlock();
786 request_finalization(*order_url_opt);
787 }
788 else if (j["status"] == "valid")
789 {
790 expect(j, "certificate");
791 order.certificate_url = j["certificate"];
792 guard.unlock();
793 request_certificate(*order_url_opt);
794 }
795 else
796 {
797 LOG_FATAL_FMT(
798 "ACME: unknown order status '{}', aborting", j["status"]);
799 guard.unlock();
800 remove_order(*order_url_opt);
801 }
802 });
803 }

806 void request_authorization(Order& order, const std::string& authz_url)
807 {
808 post_as_get_json(
809 order.account_url,
810 authz_url,
811 [this, order_url = order.order_url, authz_url](

943 post_as_get(
944 order->account_url,

966 }

968 {

971 }
972 else
973 {

975 "ACME: challenge for token '{}' failed with status '{}' ",
976 challenge_token,
