48 void handle_incoming_data_thread(std::vector<uint8_t>&& data) override
49 {
50 tls_io->recv_buffered(data.data(), data.size());
51
52 LOG_TRACE_FMT("recv called with {} bytes", data.size());
53
54 // Try to parse all incoming data, reusing the vector we were just passed
55 // for storage. Increase the size if the received vector was too small
56 // (for the case where this chunk is very small, but we had some previous
57 // data to continue reading).
58 constexpr auto min_read_block_size = 4096;
59 if (data.size() < min_read_block_size)
60 {
61 data.resize(min_read_block_size);
62 }
63
64 auto n_read = tls_io->read(data.data(), data.size(), false);
65
66 while (true)
67 {
68 if (n_read == 0)
69 {
70 return;
71 }
72
73 LOG_TRACE_FMT("Going to parse {} bytes", n_read);
74
75 bool cont = parse({data.data(), n_read});
76 if (!cont)
77 {
78 return;
79 }
80
81 // Used all provided bytes - check if more are available
82 n_read = tls_io->read(data.data(), data.size(), false);
83 }
84 }

114 bool parse(std::span<const uint8_t> data) override
115 {
116 // Catch request parsing errors and convert them to error responses
117 try
118 {
119 request_parser.execute(data.data(), data.size());
120
121 return true;
122 }
123 catch (RequestPayloadTooLarge& e)
124 {
125 if (error_reporter)
126 {
127 error_reporter->report_request_payload_too_large_error(session_id);
128 }
129
130 LOG_DEBUG_FMT("Request is too large: {}", e.what());
131
132 send_odata_error_response(ccf::ErrorDetails{
133 HTTP_STATUS_PAYLOAD_TOO_LARGE,
134 ccf::errors::RequestBodyTooLarge,
135 e.what()});
136
137 tls_io->close();
138 }
139 catch (RequestHeaderTooLarge& e)
140 {
141 if (error_reporter)
142 {
143 error_reporter->report_request_header_too_large_error(session_id);
144 }
145
146 LOG_DEBUG_FMT("Request header is too large: {}", e.what());
147
148 send_odata_error_response(ccf::ErrorDetails{
149 HTTP_STATUS_REQUEST_HEADER_FIELDS_TOO_LARGE,
150 ccf::errors::RequestHeaderTooLarge,
151 e.what()});
152
153 tls_io->close();
154 }
155 catch (const std::exception& e)
156 {
157 if (error_reporter)
158 {
159 error_reporter->report_parsing_error(session_id);
160 }
161 LOG_DEBUG_FMT("Error parsing HTTP request: {}", e.what());
162
163 http::HeaderMap headers;
164 headers[http::headers::CONTENT_TYPE] =
165 http::headervalues::contenttype::TEXT;
166
167 // NB: Avoid formatting input data a string, as it may contain null
168 // bytes. Instead insert it at the end of this message, verbatim
169 auto body_s = fmt::format(
170 "Unable to parse data as a HTTP request. Error message is: {}\n"
171 "Error occurred while parsing fragment:\n",
172 e.what());
173 std::vector<uint8_t> response_body(
174 std::begin(body_s), std::end(body_s));
175 response_body.insert(response_body.end(), data.begin(), data.end());
176
177 send_response(
178 HTTP_STATUS_BAD_REQUEST, std::move(headers), {}, response_body);
179
180 tls_io->close();
181 }
182
183 return false;
184 }

186 void handle_request(
187 llhttp_method verb,
188 const std::string_view& url,
189 http::HeaderMap&& headers,
190 std::vector<uint8_t>&& body,
191 int32_t) override
192 {
193 LOG_TRACE_FMT(
194 "Processing msg({}, {} [{} bytes])",
195 llhttp_method_name(verb),
196 url,
197 body.size());
198
199 try
200 {
201 if (session_ctx == nullptr)
202 {
203 session_ctx = std::make_shared<ccf::SessionContext>(
204 session_id, tls_io->peer_cert(), interface_id);
205 }
206
207 std::shared_ptr<http::HttpRpcContext> rpc_ctx = nullptr;
208 try
209 {
210 rpc_ctx = std::make_shared<HttpRpcContext>(
211 session_ctx, verb, url, std::move(headers), std::move(body));
212 }
213 catch (std::exception& e)
214 {
215 send_odata_error_response(ccf::ErrorDetails{
216 HTTP_STATUS_INTERNAL_SERVER_ERROR,
217 ccf::errors::InternalError,
218 fmt::format("Error constructing RpcContext: {}", e.what())});
219 tls_io->close();
220 }
221
222 const auto actor_opt = http::extract_actor(*rpc_ctx);
223 std::optional<std::shared_ptr<ccf::RpcHandler>> search;
224 ccf::ActorsType actor = ccf::ActorsType::unknown;
225 if (actor_opt.has_value())
226 {
227 const auto& actor_s = actor_opt.value();
228 actor = rpc_map->resolve(actor_s);
229 search = rpc_map->find(actor);
230 }
231 if (
232 !actor_opt.has_value() || actor == ccf::ActorsType::unknown ||
233 !search.has_value())
234 {
235 // if there is no actor, proceed with the "app" as the ActorType and
236 // process the request
237 search = rpc_map->find(ccf::ActorsType::users);
238 }
239
240 search.value()->process(rpc_ctx);
241
242 if (rpc_ctx->response_is_pending)
243 {
244 // If the RPC is pending, hold the connection.
245 LOG_TRACE_FMT("Pending");
246 return;
247 }
248 else
249 {
250 send_response(
251 rpc_ctx->get_response_http_status(),
252 rpc_ctx->get_response_headers(),
253 rpc_ctx->get_response_trailers(),
254 std::move(rpc_ctx->get_response_body()));
255
256 if (rpc_ctx->terminate_session)
257 {
258 tls_io->close();
259 }
260 }
261 }
262 catch (const std::exception& e)
263 {
264 send_odata_error_response(ccf::ErrorDetails{
265 HTTP_STATUS_INTERNAL_SERVER_ERROR,
266 ccf::errors::InternalError,
267 fmt::format("Exception: {}", e.what())});
268
269 // On any exception, close the connection.
270 LOG_FAIL_FMT("Closing connection");
271 LOG_DEBUG_FMT("Closing connection due to exception: {}", e.what());
272 tls_io->close();
273 throw;
274 }
275 }

277 void send_response(
278 http_status status_code,
279 http::HeaderMap&& headers,
280 http::HeaderMap&& trailers,
281 std::span<const uint8_t> body) override
282 {
283 if (!trailers.empty())
284 {
285 throw std::logic_error("Cannot return trailers over HTTP/1");
286 }
287
288 auto response = http::Response(status_code);
289 for (const auto& [k, v] : headers)
290 {
291 response.set_header(k, v);
292 }
293 response.set_body(body.data(), body.size());
294
295 auto data = response.build_response();
296 tls_io->send_raw(data.data(), data.size());
297 }
