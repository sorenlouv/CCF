22 static std::vector<uint8_t> ecdsa_sig_from_r_s(
23 const uint8_t* r,
24 size_t r_size,
25 const uint8_t* s,
26 size_t s_size,
27 bool big_endian = true)
28 {
29 OpenSSL::Unique_BIGNUM r_bn;
30 OpenSSL::Unique_BIGNUM s_bn;
31 if (big_endian)
32 {
33 OpenSSL::CHECKNULL(BN_bin2bn(r, r_size, r_bn));
34 OpenSSL::CHECKNULL(BN_bin2bn(s, s_size, s_bn));
35 }
36 else
37 {
38 OpenSSL::CHECKNULL(BN_lebin2bn(r, r_size, r_bn));
39 OpenSSL::CHECKNULL(BN_lebin2bn(s, s_size, s_bn));
40 }
41 OpenSSL::Unique_ECDSA_SIG sig;
42 OpenSSL::CHECK1(ECDSA_SIG_set0(sig, r_bn, s_bn));
43 // Ignore previous pointers, as they're now managed by ECDSA_SIG_set0
44 // https://www.openssl.org/docs/man1.1.1/man3/ECDSA_SIG_get0.html
45 (void)r_bn.release();
46 (void)s_bn.release();
47 auto der_size = i2d_ECDSA_SIG(sig, nullptr);
48 OpenSSL::CHECK0(der_size);
49 std::vector<uint8_t> der_sig(der_size);
50 auto der_sig_buf = der_sig.data();
51 OpenSSL::CHECK0(i2d_ECDSA_SIG(sig, &der_sig_buf));
52 return der_sig;
53 }

59 static std::vector<uint8_t> ecdsa_sig_p1363_to_der(
60 const std::vector<uint8_t>& signature)
61 {
62 auto half_size = signature.size() / 2;

67 static std::vector<uint8_t> ecdsa_sig_der_to_p1363(
68 const std::vector<uint8_t>& signature, CurveID curveId)
69 {
70 auto sig_ptr = signature.data();
71 OpenSSL::Unique_ECDSA_SIG ecdsa_sig(
72 d2i_ECDSA_SIG(NULL, &sig_ptr, signature.size()));
73 // r and s are managed by Unique_ECDSA_SIG object, so we shouldn't use
74 // Unique_BIGNUM for them
75 const BIGNUM* r = ECDSA_SIG_get0_r(ecdsa_sig);
76 const BIGNUM* s = ECDSA_SIG_get0_s(ecdsa_sig);
77 const int nid = PublicKey_OpenSSL::get_openssl_group_id(curveId);
78 OpenSSL::Unique_EC_GROUP ec_group(nid);
79 const int group_order_bits = EC_GROUP_order_bits(ec_group);
80 OpenSSL::CHECKPOSITIVE(group_order_bits);
81 const size_t n = (group_order_bits + 7) / 8;
82 std::vector<uint8_t> sig_p1363(n * 2);
83 OpenSSL::CHECKEQUAL(n, BN_bn2binpad(r, sig_p1363.data(), n));
84 OpenSSL::CHECKEQUAL(n, BN_bn2binpad(s, sig_p1363.data() + n, n));
85 return sig_p1363;
86 }
