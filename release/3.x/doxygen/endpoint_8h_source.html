220 template <typename In, typename Out>
221 Endpoint& set_auto_schema(std::optional<http_status> status = std::nullopt)
222 {
223 if constexpr (!std::is_same_v<In, void>)
224 {
225 params_schema = ds::json::build_schema<In>();
226
227 schema_builders.push_back(
228 [](nlohmann::json& document, const Endpoint& endpoint) {
229 const auto http_verb = endpoint.dispatch.verb.get_http_method();
230 if (!http_verb.has_value())
231 {
232 // Non-HTTP endpoints are not documented
233 return;
234 }
235
236 ds::openapi::add_request_body_schema<In>(
237 document, endpoint.full_uri_path, http_verb.value());
238 });
239 }
240 else
241 {
242 params_schema = nullptr;
243 }
244
245 if constexpr (!std::is_same_v<Out, void>)
246 {
247 success_status = status.value_or(HTTP_STATUS_OK);
248
249 result_schema = ds::json::build_schema<Out>();
250
251 schema_builders.push_back(
252 [](nlohmann::json& document, const Endpoint& endpoint) {
253 const auto http_verb = endpoint.dispatch.verb.get_http_method();
254 if (!http_verb.has_value())
255 {
256 return;
257 }
258
259 ds::openapi::add_response_schema<Out>(
260 document,
261 endpoint.full_uri_path,
262 http_verb.value(),
263 endpoint.success_status);
264 });
265 }
266 else
267 {
268 success_status = status.value_or(HTTP_STATUS_NO_CONTENT);
269 result_schema = nullptr;
270 }
271
272 return *this;
273 }

306 template <typename T>
307 Endpoint& add_query_parameter(

309 QueryParamPresence presence = RequiredParameter)
310 {
311 schema_builders.push_back(
312 [param_name,
313 presence](nlohmann::json& document, const Endpoint& endpoint) {
314 const auto http_verb = endpoint.dispatch.verb.get_http_method();
315 if (!http_verb.has_value())
316 {
317 // Non-HTTP endpoints are not documented
318 return;
319 }
320
321 const auto schema_name = ds::json::schema_name<T>();
322 const auto query_schema = ds::json::build_schema<T>();
323
324 auto parameter = nlohmann::json::object();
325 parameter["name"] = param_name;
326 parameter["in"] = "query";
327 parameter["required"] = presence == RequiredParameter;
328 parameter["schema"] = ds::openapi::add_schema_to_components(
329 document, schema_name, query_schema);
330 ds::openapi::add_request_parameter_schema(
331 document, endpoint.full_uri_path, http_verb.value(), parameter);
332 });
333
334 return *this;
335 }
