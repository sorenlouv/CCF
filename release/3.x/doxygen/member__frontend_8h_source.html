119 ccf::jsgov::ProposalInfoSummary resolve_proposal(
120 kv::Tx& tx,
121 const ProposalId& proposal_id,
122 const std::vector<uint8_t>& proposal,
123 const std::string& constitution)
124 {
125 auto pi =
126 tx.rw<ccf::jsgov::ProposalInfoMap>(jsgov::Tables::PROPOSALS_INFO);
127 auto pi_ = pi->get(proposal_id);
128
129 std::vector<std::pair<MemberId, bool>> votes;
130 std::optional<ccf::jsgov::Votes> final_votes = std::nullopt;
131 std::optional<ccf::jsgov::VoteFailures> vote_failures = std::nullopt;
132 for (const auto& [mid, mb] : pi_->ballots)
133 {
134 js::Runtime rt(&tx);
135 js::Context context(rt, js::TxAccess::GOV_RO);
136 rt.add_ccf_classdefs();
137 js::TxContext txctx{&tx};
138 js::populate_global(
139 &txctx,
140 nullptr,
141 nullptr,
142 std::nullopt,
143 nullptr,
144 nullptr,
145 nullptr,
146 nullptr,
147 nullptr,
148 nullptr,
149 context);
150 auto ballot_func = context.function(
151 mb,
152 "vote",
153 fmt::format(
154 "public:ccf.gov.proposal_info[{}].ballots[{}]", proposal_id, mid));
155
156 std::vector<js::JSWrappedValue> argv = {
157 context.new_string_len((const char*)proposal.data(), proposal.size()),
158 context.new_string_len(
159 pi_->proposer_id.data(), pi_->proposer_id.size())};
160
161 auto val = context.call(ballot_func, argv);
162 if (!JS_IsException(val))
163 {
164 votes.emplace_back(mid, JS_ToBool(context, val));
165 }
166 else
167 {
168 if (!vote_failures.has_value())
169 {
170 vote_failures = ccf::jsgov::VoteFailures();
171 }
172
173 auto [reason, trace] = js::js_error_message(context);
174
175 if (context.host_time.request_timed_out)
176 {
177 reason = "Operation took too long to complete.";
178 }
179 vote_failures.value()[mid] = ccf::jsgov::Failure{reason, trace};
180 }
181 }
182
183 {
184 js::Runtime rt(&tx);
185 js::Context js_context(rt, js::TxAccess::GOV_RO);
186 rt.add_ccf_classdefs();
187 js::TxContext txctx{&tx};
188 js::populate_global(
189 &txctx,
190 nullptr,
191 nullptr,
192 std::nullopt,
193 nullptr,
194 nullptr,
195 nullptr,
196 nullptr,
197 nullptr,
198 nullptr,
199 js_context);
200 auto resolve_func = js_context.function(
201 constitution, "resolve", "public:ccf.gov.constitution[0]");
202
203 std::vector<js::JSWrappedValue> argv;
204 argv.push_back(js_context.new_string_len(
205 (const char*)proposal.data(), proposal.size()));
206
207 argv.push_back(js_context.new_string_len(
208 pi_->proposer_id.data(), pi_->proposer_id.size()));
209
210 auto vs = js_context.new_array();
211 size_t index = 0;
212 for (auto& [mid, vote] : votes)
213 {
214 auto v = JS_NewObject(js_context);
215 auto member_id = JS_NewStringLen(js_context, mid.data(), mid.size());
216 JS_DefinePropertyValueStr(
217 js_context, v, "member_id", member_id, JS_PROP_C_W_E);
218 auto vote_status = JS_NewBool(js_context, vote);
219 JS_DefinePropertyValueStr(
220 js_context, v, "vote", vote_status, JS_PROP_C_W_E);
221 JS_DefinePropertyValueUint32(
222 js_context, vs, index++, v, JS_PROP_C_W_E);
223 }
224 argv.push_back(vs);
225
226 auto val = js_context.call(resolve_func, argv);
227
228 std::optional<jsgov::Failure> failure = std::nullopt;
229 if (JS_IsException(val))
230 {
231 pi_.value().state = ProposalState::FAILED;
232 auto [reason, trace] = js::js_error_message(js_context);
233 if (js_context.host_time.request_timed_out)
234 {
235 reason = "Operation took too long to complete.";
236 }
237 failure = ccf::jsgov::Failure{
238 fmt::format("Failed to resolve(): {}", reason), trace};
239 }
240 else if (JS_IsString(val))
241 {
242 auto status = js_context.to_str(val).value_or("");
243 if (status == "Open")
244 {
245 pi_.value().state = ProposalState::OPEN;
246 }
247 else if (status == "Accepted")
248 {
249 pi_.value().state = ProposalState::ACCEPTED;
250 }
251 else if (status == "Withdrawn")
252 {
253 pi_.value().state = ProposalState::FAILED;
254 }
255 else if (status == "Rejected")
256 {
257 pi_.value().state = ProposalState::REJECTED;
258 }
259 else if (status == "Failed")
260 {
261 pi_.value().state = ProposalState::FAILED;
262 }
263 else if (status == "Dropped")
264 {
265 pi_.value().state = ProposalState::DROPPED;
266 }
267 else
268 {
269 pi_.value().state = ProposalState::FAILED;
270 failure = ccf::jsgov::Failure{
271 fmt::format(
272 "resolve() returned invalid status value: \"{}\"", status),
273 std::nullopt};
274 }
275 }
276 else
277 {
278 pi_.value().state = ProposalState::FAILED;
279 failure = ccf::jsgov::Failure{
280 "resolve() returned invalid status value", std::nullopt};
281 }
282
283 if (pi_.value().state != ProposalState::OPEN)
284 {
285 remove_all_other_non_open_proposals(tx, proposal_id);
286 final_votes = std::unordered_map<ccf::MemberId, bool>();
287 for (auto& [mid, vote] : votes)
288 {
289 final_votes.value()[mid] = vote;
290 }
291 if (pi_.value().state == ProposalState::ACCEPTED)
292 {
293 js::Runtime rt(&tx);
294 js::Context js_context(rt, js::TxAccess::GOV_RW);
295 rt.add_ccf_classdefs();
296 js::TxContext txctx{&tx};
297
298 auto gov_effects =
299 context.get_subsystem<AbstractGovernanceEffects>();
300 if (gov_effects == nullptr)
301 {
302 throw std::logic_error(
303 "Unexpected: Could not access GovEffects subsytem");
304 }
305
306 js::populate_global(
307 &txctx,
308 nullptr,
309 nullptr,
310 std::nullopt,
311 nullptr,
312 gov_effects.get(),
313 nullptr,
314 &network,
315 nullptr,
316 this,
317 js_context);
318 auto apply_func = js_context.function(
319 constitution, "apply", "public:ccf.gov.constitution[0]");
320
321 std::vector<js::JSWrappedValue> argv = {
322 js_context.new_string_len(
323 (const char*)proposal.data(), proposal.size()),
324 js_context.new_string_len(
325 proposal_id.c_str(), proposal_id.size())};
326
327 auto val = js_context.call(apply_func, argv);
328
329 if (JS_IsException(val))
330 {
331 pi_.value().state = ProposalState::FAILED;
332 auto [reason, trace] = js::js_error_message(js_context);
333 if (js_context.host_time.request_timed_out)
334 {
335 reason = "Operation took too long to complete.";
336 }
337 failure = ccf::jsgov::Failure{
338 fmt::format("Failed to apply(): {}", reason), trace};
339 }
340 }
341 }
342
343 return jsgov::ProposalInfoSummary{
344 proposal_id,
345 pi_->proposer_id,
346 pi_.value().state,
347 pi_.value().ballots.size(),
348 final_votes,
349 vote_failures,
350 failure};
351 }
352 }

398 ProposalSubmissionStatus is_proposal_submission_acceptable(
399 kv::Tx& tx,
400 const std::string& created_at,
401 const std::vector<uint8_t>& request_digest,
402 const ccf::ProposalId& proposal_id,
403 ccf::ProposalId& colliding_proposal_id,
404 std::string& min_created_at)
405 {
406 auto cose_recent_proposals = tx.rw(network.cose_recent_proposals);
407 auto key = fmt::format("{}:{}", created_at, ds::to_hex(request_digest));
408
409 std::vector<std::string> replay_keys;
410 cose_recent_proposals->foreach_key(
411 [&replay_keys](const std::string& replay_key) {
412 replay_keys.push_back(replay_key);
413 return true;
414 });
415
416 std::sort(replay_keys.begin(), replay_keys.end());
417
418 // New proposal must be more recent than median proposal kept
419 if (!replay_keys.empty())
420 {
421 min_created_at = std::get<0>(
422 nonstd::split_1(replay_keys[replay_keys.size() / 2], ":"));
423 auto [key_ts, __] = nonstd::split_1(key, ":");
424 if (key_ts < min_created_at)
425 {
426 return ProposalSubmissionStatus::TooOld;
427 }
428 }
429
430 if (cose_recent_proposals->has(key))
431 {
432 colliding_proposal_id = cose_recent_proposals->get(key).value();
433 return ProposalSubmissionStatus::DuplicateInWindow;
434 }
435 else
436 {
437 size_t window_size = ccf::default_recent_cose_proposals_window_size;
438 auto service = tx.ro(network.config);
439 auto service_config = service->get();
440 if (
441 service_config.has_value() &&
442 service_config->recent_cose_proposals_window_size.has_value())
443 {
444 window_size =
445 service_config->recent_cose_proposals_window_size.value();
446 }
447 cose_recent_proposals->put(key, proposal_id);
448 // Only keep the most recent window_size proposals, to avoid
449 // unbounded memory usage
450 if (replay_keys.size() >= window_size)
451 {
452 for (size_t i = 0; i < (replay_keys.size() - window_size); i++)
453 {
454 cose_recent_proposals->remove(replay_keys[i]);
455 }
456 }
457 return ProposalSubmissionStatus::Acceptable;
458 }

475 template <typename T>
476 void add_kv_wrapper_endpoint(T table)
477 {
478 constexpr bool is_map = nonstd::is_specialization<T, kv::TypedMap>::value;
479 constexpr bool is_value =
480 nonstd::is_specialization<T, kv::TypedValue>::value;
481
482 if constexpr (!(is_map || is_value))
483 {
484 static_assert(nonstd::dependent_false_v<T>, "Unsupported table type");
485 }
486
487 auto getter =
488 [&, table](endpoints::ReadOnlyEndpointContext& ctx, nlohmann::json&&) {
489 LOG_TRACE_FMT("Called getter for {}", table.get_name());
490 auto response_body = nlohmann::json::object();
491
492 auto handle = ctx.tx.template ro(table);
493 if constexpr (is_map)
494 {
495 handle->foreach([&response_body](const auto& k, const auto& v) {
496 if constexpr (
497 std::is_same_v<typename T::Key, ccf::CodeDigest> ||
498 std::is_same_v<typename T::Key, crypto::Sha256Hash>)
499 {
500 response_body[k.hex_str()] = v;
501 }
502 else
503 {
504 response_body[k] = v;
505 }
506 return true;
507 });
508 }
509 else if constexpr (is_value)
510 {
511 response_body = handle->get();
512 }
513
514 return ccf::make_success(response_body);
515 };
516
517 std::string uri = table.get_name();
518 constexpr auto gov_prefix = "public:ccf.gov.";
519 if (uri.starts_with(gov_prefix))
520 {
521 uri.erase(0, strlen(gov_prefix));
522 }
523 else
524 {
525 throw std::logic_error(fmt::format(
526 "Should only be used to wrap governance tables. '{}' is not "
527 "supported",
528 uri));
529 }
530
531 // Replace . separators with /
532 {
533 auto idx = uri.find('.');
534 while (idx != std::string::npos)
535 {
536 uri[idx] = '/';
537 idx = uri.find('.', idx);
538 }
539 }
540
541 auto endpoint = make_read_only_endpoint(
542 fmt::format("/kv/{}", uri),
543 HTTP_GET,
544 json_read_only_adapter(getter),
545 ccf::no_auth_required);
546
547 if constexpr (is_map)
548 {
549 endpoint.template set_auto_schema<
550 void,
551 std::map<typename T::Key, typename T::Value>>();
552 }
553 else if constexpr (is_value)
554 {
555 endpoint.template set_auto_schema<void, typename T::Value>();
556 }
557
558 endpoint.install();
559 }

671 void init_handlers() override
672 {
673 CommonEndpointRegistry::init_handlers();
674
676 auto ack = [this](ccf::endpoints::EndpointContext& ctx) {
677 std::optional<ccf::MemberSignatureAuthnIdentity> sig_auth_id =
678 std::nullopt;
679 std::optional<ccf::MemberCOSESign1AuthnIdentity> cose_auth_id =
680 std::nullopt;
681 std::optional<MemberId> member_id = std::nullopt;
682 if (!authnz_active_member(
683 ctx, member_id, sig_auth_id, cose_auth_id, false))
684 {
685 return;
686 }
687
688 auto params = nlohmann::json::parse(
689 cose_auth_id.has_value() ? cose_auth_id->content :
690 ctx.rpc_ctx->get_request_body());
691
692 auto mas = ctx.tx.rw(this->network.member_acks);
693 const auto ma = mas->get(member_id.value());
694 if (!ma)
695 {
696 ctx.rpc_ctx->set_error(
697 HTTP_STATUS_FORBIDDEN,
698 ccf::errors::AuthorizationFailed,
699 fmt::format(
700 "No ACK record exists for caller {}.", member_id.value()));
701 return;
702 }
703
704 const auto digest = params.get<StateDigest>();
705 if (ma->state_digest != digest.state_digest)
706 {
707 ctx.rpc_ctx->set_error(
708 HTTP_STATUS_BAD_REQUEST,
709 ccf::errors::StateDigestMismatch,
710 "Submitted state digest is not valid.");
711 return;
712 }
713
714 auto sig = ctx.tx.rw(this->network.signatures);
715 const auto s = sig->get();
716 if (sig_auth_id.has_value())
717 {
718 if (!s)
719 {
720 mas->put(
721 member_id.value(), MemberAck({}, sig_auth_id->signed_request));
722 }
723 else
724 {
725 mas->put(
726 member_id.value(),
727 MemberAck(s->root, sig_auth_id->signed_request));
728 }
729 }
730
731 if (cose_auth_id.has_value())
732 {
733 std::vector<uint8_t> cose_sign1 = {
734 cose_auth_id->envelope.begin(), cose_auth_id->envelope.end()};
735 if (!s)
736 {
737 mas->put(member_id.value(), MemberAck({}, cose_sign1));
738 }
739 else
740 {
741 mas->put(member_id.value(), MemberAck(s->root, cose_sign1));
742 }
743 }
744
745 // update member status to ACTIVE
746 GenesisGenerator g(this->network, ctx.tx);
747 try
748 {
749 g.activate_member(member_id.value());
750 }
751 catch (const std::logic_error& e)
752 {
753 ctx.rpc_ctx->set_error(
754 HTTP_STATUS_FORBIDDEN,
755 ccf::errors::AuthorizationFailed,
756 fmt::format("Error activating new member: {}", e.what()));
757 return;
758 }
759
760 auto service_status = g.get_service_status();
761 if (!service_status.has_value())
762 {
763 ctx.rpc_ctx->set_error(
764 HTTP_STATUS_INTERNAL_SERVER_ERROR,
765 ccf::errors::InternalError,
766 "No service currently available.");
767 return;

895 auto submit_recovery_share = [this](
896 ccf::endpoints::EndpointContext& ctx) {
897 // Only active members can submit their shares for recovery
898 const auto member_id = get_caller_member_id(ctx);
899 if (!member_id.has_value())
900 {
901 ctx.rpc_ctx->set_error(
902 HTTP_STATUS_FORBIDDEN,
903 ccf::errors::AuthorizationFailed,
904 "Member is unknown.");
905 return;
906 }
907 if (!check_member_active(ctx.tx, member_id.value()))
908 {
909 ctx.rpc_ctx->set_error(
910 HTTP_STATUS_FORBIDDEN,
911 errors::AuthorizationFailed,
912 "Member is not active.");
913 return;
914 }
915
916 const auto* cose_auth_id =
917 ctx.try_get_caller<ccf::MemberCOSESign1AuthnIdentity>();
918 auto params = nlohmann::json::parse(
919 cose_auth_id ? cose_auth_id->content :
920 ctx.rpc_ctx->get_request_body());
921
922 GenesisGenerator g(this->network, ctx.tx);
923 if (
924 g.get_service_status() != ServiceStatus::WAITING_FOR_RECOVERY_SHARES)
925 {
926 ctx.rpc_ctx->set_error(
927 HTTP_STATUS_FORBIDDEN,
928 errors::ServiceNotWaitingForRecoveryShares,
929 "Service is not waiting for recovery shares.");
930 return;
931 }
932
933 auto node_operation = context.get_subsystem<AbstractNodeOperation>();
934 if (node_operation == nullptr)
935 {
936 throw std::logic_error(
937 "Unexpected: Could not access NodeOperation subsystem");
938 }
939
940 if (node_operation->is_reading_private_ledger())
941 {
942 ctx.rpc_ctx->set_error(
943 HTTP_STATUS_FORBIDDEN,
944 errors::NodeAlreadyRecovering,
945 "Node is already recovering private ledger.");
946 return;
947 }
948
949 std::string share = params["share"];
950 auto raw_recovery_share = crypto::raw_from_b64(share);
951 OPENSSL_cleanse(const_cast<char*>(share.data()), share.size());
952
953 size_t submitted_shares_count = 0;
954 try
955 {
956 submitted_shares_count = share_manager.submit_recovery_share(
957 ctx.tx, member_id.value(), raw_recovery_share);
958 }
959 catch (const std::exception& e)
960 {
961 constexpr auto error_msg = "Error submitting recovery shares.";
962 LOG_FAIL_FMT(error_msg);
963 LOG_DEBUG_FMT("Error: {}", e.what());
964 ctx.rpc_ctx->set_error(
965 HTTP_STATUS_INTERNAL_SERVER_ERROR,
966 errors::InternalError,
967 error_msg);
968 return;
969 }
970 OPENSSL_cleanse(raw_recovery_share.data(), raw_recovery_share.size());
971
972 if (submitted_shares_count < g.get_recovery_threshold())
973 {
974 // The number of shares required to re-assemble the secret has not yet
975 // been reached
976 auto recovery_share = SubmitRecoveryShare::Out{fmt::format(
977 "{}/{} recovery shares successfully submitted.",
978 submitted_shares_count,
979 g.get_recovery_threshold())};
980 ctx.rpc_ctx->set_response_header(
981 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
982 ctx.rpc_ctx->set_response_body(nlohmann::json(recovery_share).dump());
983 ctx.rpc_ctx->set_response_status(HTTP_STATUS_OK);
984 return;
985 }
986
987 LOG_DEBUG_FMT(
988 "Reached recovery threshold {}", g.get_recovery_threshold());
989
990 try
991 {
992 node_operation->initiate_private_recovery(ctx.tx);
993 }
994 catch (const std::exception& e)
995 {
996 // Clear the submitted shares if combination fails so that members can
997 // start over.
998 constexpr auto error_msg = "Failed to initiate private recovery.";
999 LOG_FAIL_FMT(error_msg);
1000 LOG_DEBUG_FMT("Error: {}", e.what());
1001 share_manager.clear_submitted_recovery_shares(ctx.tx);
1002 ctx.rpc_ctx->set_apply_writes(true);
1003 ctx.rpc_ctx->set_error(
1004 HTTP_STATUS_INTERNAL_SERVER_ERROR,
1005 errors::InternalError,
1006 error_msg);
1007 return;
1008 }
1009
1010 auto recovery_share = SubmitRecoveryShare::Out{fmt::format(
1011 "{}/{} recovery shares successfully submitted. End of recovery "
1012 "procedure initiated.",
1013 submitted_shares_count,
1014 g.get_recovery_threshold())};
1015 ctx.rpc_ctx->set_response_header(
1016 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
1017 ctx.rpc_ctx->set_response_body(nlohmann::json(recovery_share).dump());
1018 ctx.rpc_ctx->set_response_status(HTTP_STATUS_OK);
1019 };
