19 inline std::vector<uint8_t> construct_raw_signed_string(
20 std::string verb,
21 const std::string_view& url,
22 const http::HeaderMap& headers,
23 const std::vector<std::string_view>& headers_to_sign)
24 {
25 std::string signed_string = {};
26 std::string value = {};
27 bool first = true;
28
29 for (const auto f : headers_to_sign)
30 {
31 if (f == auth::SIGN_HEADER_REQUEST_TARGET)
32 {
33 // Store verb as lowercase
34 nonstd::to_lower(verb);
35 value = fmt::format("{} {}", verb, url);
36 }
37 else
38 {
39 const auto h = headers.find(f);
40 if (h == headers.end())
41 {
42 throw std::logic_error(
43 fmt::format("Signed header '{}' does not exist", f));
44 }
45
46 value = h->second;
47 }
48
49 if (!first)
50 {
51 signed_string.append("\n");
52 }
53 first = false;
54
55 signed_string.append(f);
56 signed_string.append(": ");
57 signed_string.append(value);
58 }
59
60 return std::vector<uint8_t>({signed_string.begin(), signed_string.end()});
61 }

63 inline void add_digest_header(http::Request& request)
64 {
65 // Ensure digest is present and up-to-date
66 crypto::Sha256Hash body_digest(
67 request.get_content_data(), request.get_content_length());
68 request.set_header(
69 headers::DIGEST,
70 fmt::format(
71 "{}={}",
72 "SHA-256",
73 crypto::b64_from_raw(body_digest.h.data(), body_digest.SIZE)));

76 inline void sign_request(
77 http::Request& request,
78 const crypto::KeyPairPtr& kp,
79 const std::string& key_id,
80 const std::vector<std::string_view>& headers_to_sign)
81 {
82 add_digest_header(request);
83
84 const auto to_sign = construct_raw_signed_string(
85 llhttp_method_name(request.get_method()),
86 fmt::format("{}{}", request.get_path(), request.get_formatted_query()),
87 request.get_headers(),
88 headers_to_sign);
89
90 const auto signature = kp->sign(to_sign);
91
92 auto auth_value = fmt::format(
93 "Signature "
94 "keyId=\"{}\",algorithm=\"{}\",headers=\"{}\",signature="
95 "\"{}\"",
96 key_id,
97 auth::SIGN_ALGORITHM_HS_2019,
98 fmt::format("{}", fmt::join(headers_to_sign, " ")),
99 crypto::b64_from_raw(signature.data(), signature.size()));
100
101 request.set_header(headers::AUTHORIZATION, auth_value);
102 }

117 // Implements verification of "Signature" scheme from
118 // https://tools.ietf.org/html/draft-cavage-http-signatures-12
119 //
120 // Notes:
121 // - Only supports public key crytography (i.e. no HMAC)
122 // - Only supports SHA-256 as request digest algorithm
123 // - Only supports ecdsa-sha256 and hs2019 as signature algorithms
124 // - keyId can be set to a SHA-256 digest of a cert against which the
125 // signature verifies

137 static void parse_auth_scheme(std::string_view& auth_header_value)
138 {
139 auto next_space = auth_header_value.find(" ");
140 if (next_space == std::string::npos)
141 {
142 throw std::logic_error(fmt::format(
143 "'{}' header only contains one field", headers::AUTHORIZATION));
144 }
145 auto auth_scheme = auth_header_value.substr(0, next_space);
146 if (auth_scheme != auth::SIGN_AUTH_SCHEME)
147 {
148 throw std::logic_error(fmt::format(
149 "'{}' scheme for signature should be '{}'",
150 headers::AUTHORIZATION,
151 auth::SIGN_AUTH_SCHEME));
152 }
153 auth_header_value = auth_header_value.substr(next_space + 1);
154 }

156 static bool verify_digest(
157 const http::HeaderMap& headers,
158 const std::vector<uint8_t>& body,
159 std::string& error_reason)
160 {
161 // First, retrieve digest from header
162 auto digest = headers.find(headers::DIGEST);
163 if (digest == headers.end())
164 {
165 error_reason = fmt::format("Missing '{}' header", headers::DIGEST);
166 return false;
167 }
168
169 auto equal_pos = digest->second.find("=");
170 if (equal_pos == std::string::npos)
171 {
172 error_reason = fmt::format(
173 "'{}' header does not contain key=value", headers::DIGEST);
174 return false;
175 }
176
177 auto sha_key = digest->second.substr(0, equal_pos);
178 if (sha_key != auth::DIGEST_SHA256)
179 {
180 error_reason = fmt::format(
181 "'{}' for request digest is not supported, allowed: '{}'",
182 sha_key,
183 auth::DIGEST_SHA256);
184 return false;
185 }
186
187 auto raw_digest =
188 crypto::raw_from_b64(digest->second.substr(equal_pos + 1));
189
190 // Then, hash the request body
191 crypto::Sha256Hash body_digest({body.data(), body.size()});
192
193 if (!std::equal(
194 raw_digest.begin(),
195 raw_digest.end(),
196 body_digest.h.begin(),
197 body_digest.h.end()))
198 {
199 error_reason = fmt::format(
200 "Request body does not match '{}' header, calculated body "
201 "digest = {:02x}",
202 headers::DIGEST,
203 fmt::join(body_digest.h, ""));
204 return false;
205 }
206
207 return true;
208 }

210 // Parses a delimited string with no delimiter at the end
211 // (e.g. "foo,bar,baz") and returns a vector parsed string views (e.g.
212 // ["foo", "bar", "baz"])
213 static std::vector<std::string_view> parse_delimited_string(
214 std::string_view& s, const std::string& delimiter)
215 {
216 std::vector<std::string_view> strings;
217 bool last_string = false;
218
219 auto next_delimiter = s.find(delimiter);
220 while (next_delimiter != std::string::npos || !last_string)
221 {
222 auto token = s.substr(0, next_delimiter);
223 if (next_delimiter == std::string::npos)
224 {
225 last_string = true;
226 }
227
228 strings.emplace_back(token);
229
230 if (!last_string)
231 {
232 s = s.substr(next_delimiter + 1);
233 next_delimiter = s.find(delimiter);
234 }
235 }
236
237 return strings;
238 }

240 static SignatureParams parse_signature_params(
241 std::string_view& auth_header_value)
242 {
243 SignatureParams sig_params = {};
244
245 auto parsed_params =
246 parse_delimited_string(auth_header_value, auth::SIGN_PARAMS_DELIMITER);
247
248 for (auto& p : parsed_params)
249 {
250 auto eq_pos = p.find("=");
251 if (eq_pos != std::string::npos)
252 {
253 auto k = p.substr(0, eq_pos);
254 auto v = p.substr(eq_pos + 1);
255
256 // Remove quotes around value, if present
257 const bool begins_with_quote = v.front() == '"';
258 const bool ends_with_quote = v.back() == '"';
259 if (v.size() >= 2 && (begins_with_quote || ends_with_quote))
260 {
261 if (!(begins_with_quote && ends_with_quote))
262 {
263 throw std::logic_error(fmt::format(
264 "Unbalanced quotes in '{}' header: {}",
265 headers::AUTHORIZATION,
266 p));
267 }
268
269 v = v.substr(1, v.size() - 2);
270 }
271
272 if (k == auth::SIGN_PARAMS_KEYID)
273 {
274 sig_params.key_id = v;
275 }
276 else if (k == auth::SIGN_PARAMS_ALGORITHM)
277 {
278 sig_params.signature_algorithm = v;
279 if (
280 v != auth::SIGN_ALGORITHM_ECDSA_SHA256 &&
281 v != auth::SIGN_ALGORITHM_HS_2019)
282 {
283 throw std::logic_error(
284 fmt::format("Signature algorithm '{}' is not supported", v));
285 }
286 }
287 else if (k == auth::SIGN_PARAMS_SIGNATURE)
288 {
289 sig_params.signature = v;
290 }
291 else if (k == auth::SIGN_PARAMS_HEADERS)
292 {
293 auto parsed_signed_headers =
294 parse_delimited_string(v, auth::SIGN_PARAMS_HEADERS_DELIMITER);
295
296 if (parsed_signed_headers.size() == 0)
297 {
298 throw std::logic_error(fmt::format(
299 "No headers specified in '{}' field",
300 auth::SIGN_PARAMS_HEADERS));
301 }
302
303 for (const auto& h : parsed_signed_headers)
304 {
305 sig_params.signed_headers.emplace_back(h);
306 }
307 }
308 }
309 else
310 {
311 throw std::logic_error(fmt::format(
312 "authorization parameter '{}' does not contain \"=\"", p));
313 }
314 }
315
316 // If any sig params were not found, this is invalid
317 if (sig_params.key_id.empty())
318 {
319 throw std::logic_error(fmt::format(
320 "Signature params: Missing '{}'", auth::SIGN_PARAMS_KEYID));
321 }
322 if (sig_params.signature_algorithm.empty())
323 {
324 throw std::logic_error(fmt::format(
325 "Signature params: Missing '{}'", auth::SIGN_PARAMS_ALGORITHM));
326 }
327 if (sig_params.signature.empty())
328 {
329 throw std::logic_error(fmt::format(
330 "Signature params: Missing '{}'", auth::SIGN_PARAMS_SIGNATURE));
331 }
332 if (sig_params.signed_headers.empty())
333 {
334 throw std::logic_error(fmt::format(
335 "Signature params: Missing '{}'", auth::SIGN_PARAMS_HEADERS));
336 }
337
338 return sig_params;
339 }

341 static std::optional<ccf::SignedReq> parse(
342 const std::string& verb,
343 const std::string_view& url,
344 const http::HeaderMap& headers,
345 const std::vector<uint8_t>& body)
346 {
347 auto auth = headers.find(headers::AUTHORIZATION);
348 if (auth != headers.end())
349 {
350 std::string_view authz_header = auth->second;
351
352 parse_auth_scheme(authz_header);
353
354 std::string verify_error_reason;
355 if (!verify_digest(headers, body, verify_error_reason))
356 {
357 throw std::logic_error(fmt::format(
358 "Error verifying HTTP '{}' header: {}",
359 headers::DIGEST,
360 verify_error_reason));
361 }
362
363 auto parsed_sign_params = parse_signature_params(authz_header);
364
365 const auto& signed_headers = parsed_sign_params.signed_headers;
366 std::vector<std::string> missing_required_headers;
367 for (const auto& required_header : http::required_signature_headers)
368 {
369 const auto it = std::find(
370 signed_headers.begin(), signed_headers.end(), required_header);
371 if (it == signed_headers.end())
372 {
373 missing_required_headers.push_back(required_header);
374 }
375 }
376
377 if (!missing_required_headers.empty())
378 {
379 throw std::logic_error(fmt::format(
380 "HTTP signature does not cover required fields: '{}'",
381 fmt::join(missing_required_headers, ", ")));
382 }
383
384 auto signed_raw =
385 construct_raw_signed_string(verb, url, headers, signed_headers);
386
387 auto sig_raw = crypto::raw_from_b64(parsed_sign_params.signature);
388
389 crypto::MDType md_type = crypto::MDType::NONE;
390 if (
391 parsed_sign_params.signature_algorithm ==
392 auth::SIGN_ALGORITHM_ECDSA_SHA256)
393 {
394 md_type = crypto::MDType::SHA256;
395 }
396
397 return ccf::SignedReq{
398 sig_raw, signed_raw, body, md_type, parsed_sign_params.key_id};
399 }
400
401 // Request does not contain authorization header
402 return std::nullopt;
403 }
