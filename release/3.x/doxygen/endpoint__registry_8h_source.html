80 bool get_path_param(
81 const ccf::PathParams& params,
82 const std::string& param_name,
83 T& value,
84 std::string& error)
85 {
86 const auto it = params.find(param_name);
87 if (it == params.end())
88 {
89 error = fmt::format("No parameter named '{}' in path", param_name);
90 return false;
91 }
92
93 const auto param_s = it->second;
94 const auto [p, ec] =
95 std::from_chars(param_s.data(), param_s.data() + param_s.size(), value);
96 if (ec != std::errc())
97 {
98 error = fmt::format(
99 "Unable to parse path parameter '{}' as a {}", param_s, param_name);
100 return false;
101 }
102
103 return true;
104 }
