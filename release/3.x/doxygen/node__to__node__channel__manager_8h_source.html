32 std::shared_ptr<Channel> get_channel(const NodeId& peer_id)
33 {
34 CCF_ASSERT_FMT(
35 this_node == nullptr || this_node->node_id != peer_id,
36 "Requested channel with self {}",
37 peer_id);
38
39 std::lock_guard<ccf::pal::Mutex> guard(lock);
40 CCF_ASSERT_FMT(
41 this_node != nullptr && this_node->endorsed_node_cert.has_value(),
42 "Endorsed node certificate has not yet been set");
43
44 auto search = channels.find(peer_id);
45 if (search != channels.end())
46 {
47 return search->second;
48 }
49
50 // Create channel
51 channels.try_emplace(
52 peer_id,
53 std::make_shared<Channel>(
54 writer_factory,
55 this_node->service_cert,
56 this_node->node_kp,
57 this_node->endorsed_node_cert.value(),
58 this_node->node_id,
59 peer_id,
60 message_limit));
61 return channels.at(peer_id);
62 }
