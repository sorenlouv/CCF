148 Version next_version_unsafe()
149 {
150 // Get the next global version

152

154 // deletions. Maintain this restriction for compatibility with old code.

268 void add_dynamic_map(
269 kv::Version v, const std::shared_ptr<AbstractMap>& map_) override
270 {
271 auto map = std::dynamic_pointer_cast<kv::untyped::Map>(map_);
272 if (map == nullptr)
273 {
274 throw std::logic_error(fmt::format(
275 "Can't add dynamic map - {} is not of expected type",
276 map_->get_name()));
277 }
278
279 const auto map_name = map->get_name();
280 if (get_map(v, map_name) != nullptr)
281 {
282 throw std::logic_error(fmt::format(
283 "Can't add dynamic map - already have a map named {}", map_name));
284 }
285
286 LOG_DEBUG_FMT("Adding newly created map '{}' at version {}", map_name, v);
287 maps[map_name] = std::make_pair(v, map);
288
289 {
290 // If we have any hooks for the given map name, set them on this new map
291 const auto global_it = global_hooks.find(map_name);
292 if (global_it != global_hooks.end())
293 {
294 map->set_global_hook(global_it->second);
295 }
296
297 const auto map_it = map_hooks.find(map_name);
298 if (map_it != map_hooks.end())
299 {
300 map->set_map_hook(map_it->second);
301 }
302 }

336 std::unique_ptr<AbstractSnapshot> snapshot(Version v) override
337 {
338 auto cv = compacted_version();
339 if (v < cv)
340 {
341 throw std::logic_error(fmt::format(
342 "Cannot snapshot at version {} which is earlier than last "
343 "compacted version {} ",
344 v,
345 cv));
346 }
347
348 if (v > current_version())
349 {
350 throw std::logic_error(fmt::format(
351 "Cannot snapshot at version {} which is later than current "
352 "version {} ",
353 v,
354 current_version()));
355 }
356
357 auto snapshot = std::make_unique<StoreSnapshot>(v);
358
359 {
360 std::lock_guard<ccf::pal::Mutex> mguard(maps_lock);
361
362 for (auto& it : maps)
363 {
364 auto& [_, map] = it.second;
365 map->lock();
366 }
367
368 for (auto& it : maps)
369 {
370 auto& [_, map] = it.second;
371 snapshot->add_map_snapshot(map->snapshot(v));
372 }
373
374 auto h = get_history();
375 if (h)
376 {
377 snapshot->add_hash_at_snapshot(h->get_raw_leaf(v));
378 }
379
380 auto c = get_consensus();
381 if (c)
382 {
383 snapshot->add_view_history(c->get_view_history(v));
384 }
385
386 for (auto& it : maps)
387 {
388 auto& [_, map] = it.second;
389 map->unlock();
390 }
391 }
392
393 return snapshot;
394 }

403 ApplyResult deserialise_snapshot(
404 const uint8_t* data,
405 size_t size,
406 kv::ConsensusHookPtrs& hooks,
407 std::vector<Version>* view_history = nullptr,
408 bool public_only = false) override
409 {
410 auto e = get_encryptor();
411 auto d = KvStoreDeserialiser(
412 e,
413 public_only ? kv::SecurityDomain::PUBLIC :
414 std::optional<kv::SecurityDomain>());
415
416 kv::Term term;
417 auto v_ = d.init(data, size, term, is_historical);
418 if (!v_.has_value())
419 {
420 LOG_FAIL_FMT("Initialisation of deserialise object failed");
421 return ApplyResult::FAIL;
422 }
423 auto v = v_.value();
424
425 std::lock_guard<ccf::pal::Mutex> mguard(maps_lock);
426
427 for (auto& it : maps)
428 {
429 auto& [_, map] = it.second;
430 map->lock();
431 }
432
433 std::vector<uint8_t> hash_at_snapshot;
434 auto h = get_history();
435 if (h)
436 {
437 hash_at_snapshot = d.deserialise_raw();
438 }
439
440 std::vector<Version> view_history_;
441 if (view_history)
442 {
443 view_history_ = d.deserialise_view_history();
444 }
445
446 OrderedChanges changes;
447 MapCollection new_maps;
448
449 for (auto r = d.start_map(); r.has_value(); r = d.start_map())
450 {
451 const auto map_name = r.value();
452
453 std::shared_ptr<kv::untyped::Map> map = nullptr;
454
455 auto search = maps.find(map_name);
456 if (search == maps.end())
457 {
458 map = std::make_shared<kv::untyped::Map>(
459 this,
460 map_name,
461 get_security_domain(map_name),
462 is_map_replicated(map_name),
463 should_track_dependencies(map_name));
464 new_maps[map_name] = map;
465 LOG_DEBUG_FMT(
466 "Creating map {} while deserialising snapshot at version {}",
467 map_name,
468 v);
469 }
470 else
471 {
472 map = search->second.second;
473 }
474
475 auto changes_search = changes.find(map_name);
476 if (changes_search != changes.end())
477 {
478 LOG_FAIL_FMT("Failed to deserialise snapshot at version {}", v);
479 LOG_DEBUG_FMT("Multiple writes on map {}", map_name);
480 return ApplyResult::FAIL;
481 }
482
483 auto deserialised_snapshot_changes =
484 map->deserialise_snapshot_changes(d);
485
486 // Take ownership of the produced change set, store it to be committed
487 // later
488 changes.emplace_hint(
489 changes_search,
490 std::piecewise_construct,
491 std::forward_as_tuple(map_name),
492 std::forward_as_tuple(map, std::move(deserialised_snapshot_changes)));
493 }
494
495 for (auto& it : maps)
496 {
497 auto& [_, map] = it.second;
498 map->unlock();
499 }
500
501 if (!d.end())
502 {
503 LOG_FAIL_FMT("Unexpected content in snapshot at version {}", v);
504 return ApplyResult::FAIL;
505 }
506
507 // Each map is committed at a different version, independently of the
508 // overall snapshot version. The commit versions for each map are
509 // contained in the snapshot and applied when the snapshot is committed.
510 auto r = apply_changes(
511 changes,
512 [](bool) { return std::make_tuple(NoVersion, NoVersion); },
513 hooks,
514 new_maps);
515 if (!r.has_value())
516 {
517 LOG_FAIL_FMT("Failed to commit deserialised snapshot at version {}", v);
518 return ApplyResult::FAIL;
519 }
520
521 {
522 std::lock_guard<ccf::pal::Mutex> vguard(version_lock);
523 version = v;
524 last_replicated = v;
525 }
526
527 if (h)
528 {
529 if (!h->init_from_snapshot(hash_at_snapshot))
530 {
531 return ApplyResult::FAIL;
532 }
533 }
534
535 if (view_history)
536 {
537 *view_history = std::move(view_history_);
538 }
539
540 return ApplyResult::PASS;
541 }

543 void compact(Version v) override
544 {
545 // This is called when the store will never be rolled back to any
546 // state before the specified version.
547 // No transactions can be prepared or committed during compaction.
548
549 if (snapshotter)
550 {
551 auto c = get_consensus();
552 bool generate_snapshot = c && c->is_primary();
553 snapshotter->commit(v, generate_snapshot);
554 }
555
556 std::lock_guard<ccf::pal::Mutex> mguard(maps_lock);
557
558 if (v > current_version())
559 {
560 return;
561 }
562
563 for (auto& it : maps)
564 {
565 auto& [_, map] = it.second;
566 map->lock();
567 }
568
569 for (auto& it : maps)
570 {
571 auto& [_, map] = it.second;
572 map->compact(v);
573 }
574
575 for (auto& it : maps)
576 {
577 auto& [_, map] = it.second;
578 map->unlock();
579 }
580
581 {
582 std::lock_guard<ccf::pal::Mutex> vguard(version_lock);
583 compacted = v;
584
585 auto h = get_history();
586 if (h)
587 {
588 h->compact(v);
589 }
590 }
591
592 for (auto& it : maps)
593 {
594 auto& [_, map] = it.second;
595 map->post_compact();
596 }
597 }

599 void rollback(const TxID& tx_id, Term term_of_next_version_) override
600 {
601 // This is called to roll the store back to the state it was in
602 // at the specified version.
603 // No transactions can be prepared or committed during rollback.
604
605 if (snapshotter)
606 {
607 snapshotter->rollback(tx_id.version);
608 }
609
610 std::lock_guard<ccf::pal::Mutex> mguard(maps_lock);
611
612 {
613 std::lock_guard<ccf::pal::Mutex> vguard(version_lock);
614 if (tx_id.version < compacted)
615 {
616 throw std::logic_error(fmt::format(
617 "Attempting rollback to {}, earlier than commit version {}",
618 tx_id.version,
619 compacted));
620 }
621
622 // The term should always be updated on rollback() when passed
623 // regardless of whether version needs to be updated or not
624 term_of_next_version = term_of_next_version_;
625 term_of_last_version = tx_id.term;
626
627 // History must be informed of the term_of_last_version change, even if
628 // no actual rollback is required
629 auto h = get_history();
630 if (h)
631 {
632 h->rollback(tx_id, term_of_next_version);
633 }
634
635 if (tx_id.version >= version)
636 {
637 return;
638 }
639
640 version = tx_id.version;
641 last_replicated = tx_id.version;
642 unset_flag_unsafe(Flag::LEDGER_CHUNK_AT_NEXT_SIGNATURE);
643 unset_flag_unsafe(Flag::SNAPSHOT_AT_NEXT_SIGNATURE);
644 rollback_count++;
645 pending_txs.clear();
646 auto e = get_encryptor();
647 if (e)
648 {
649 e->rollback(tx_id.version);
650 }
651 }
652
653 for (auto& it : maps)
654 {
655 auto& [_, map] = it.second;
656 map->lock();
657 }
658
659 auto it = maps.begin();
660 while (it != maps.end())
661 {
662 auto& [map_creation_version, map] = it->second;
663 // Rollback this map whether we're forgetting about it or not. Anyone
664 // else still holding it should see it has rolled back
665 map->rollback(tx_id.version);
666 if (map_creation_version > tx_id.version)
667 {
668 // Map was created more recently; its creation is being forgotten.
669 // Erase our knowledge of it
670 map->unlock();
671 it = maps.erase(it);
672 }
673 else
674 {
675 ++it;
676 }
677 }
678
679 for (auto& it : maps)
680 {
681 auto& [_, map] = it.second;
682 map->unlock();
683 }
684 }

686 void initialise_term(Term t) override
687 {
688 // Note: This should only be called once, when the store is first
689 // initialised. term_of_next_version is later updated via rollback.
690 std::lock_guard<ccf::pal::Mutex> vguard(version_lock);
691 if (term_of_next_version != 0)
692 {
693 throw std::logic_error("term_of_next_version is already initialised");
694 }
695
696 term_of_next_version = t;
697 auto h = get_history();
698 if (h)
699 {
700 h->set_term(term_of_next_version);
701 }

704 bool fill_maps(
705 const std::vector<uint8_t>& data,
706 bool public_only,
707 kv::Version& v,
708 kv::Term& view,
709 OrderedChanges& changes,
710 MapCollection& new_maps,
711 ccf::ClaimsDigest& claims_digest,
712 std::optional<crypto::Sha256Hash>& commit_evidence_digest,
713 bool ignore_strict_versions = false) override
714 {
715 // This will return FAILED if the serialised transaction is being
716 // applied out of order.
717 // Processing transactions locally and also deserialising to the
718 // same store will result in a store version mismatch and
719 // deserialisation will then fail.
720 auto e = get_encryptor();
721
722 auto d = KvStoreDeserialiser(
723 e,
724 public_only ? kv::SecurityDomain::PUBLIC :
725 std::optional<kv::SecurityDomain>());
726
727 auto v_ = d.init(data.data(), data.size(), view, is_historical);
728 if (!v_.has_value())
729 {
730 LOG_FAIL_FMT("Initialisation of deserialise object failed");
731 return false;
732 }
733 v = v_.value();
734
735 claims_digest = std::move(d.consume_claims_digest());
736 LOG_TRACE_FMT(
737 "Deserialised claim digest {} {}",
738 claims_digest.value(),
739 claims_digest.empty());
740
741 commit_evidence_digest = std::move(d.consume_commit_evidence_digest());
742 if (commit_evidence_digest.has_value())
743 LOG_TRACE_FMT(
744 "Deserialised commit evidence digest {}",
745 commit_evidence_digest.value());
746
747 // Throw away any local commits that have not propagated via the
748 // consensus.
749 rollback({term_of_last_version, v - 1}, term_of_next_version);
750
751 if (strict_versions && !ignore_strict_versions)
752 {
753 // Make sure this is the next transaction.
754 auto cv = current_version();
755 if (cv != (v - 1))
756 {
757 LOG_FAIL_FMT(
758 "Tried to deserialise {} but current_version is {}", v, cv);
759 return false;
760 }
761 }
762
763 // Deserialised transactions express read dependencies as versions,
764 // rather than with the actual value read. As a result, they don't
765 // need snapshot isolation on the map state, and so do not need to
766 // lock each of the maps before creating the transaction.
767 std::lock_guard<ccf::pal::Mutex> mguard(maps_lock);
768
769 for (auto r = d.start_map(); r.has_value(); r = d.start_map())
770 {
771 const auto map_name = r.value();
772
773 auto map = get_map_internal(v, map_name);
774 if (map == nullptr)
775 {
776 auto new_map = std::make_shared<kv::untyped::Map>(
777 this,
778 map_name,
779 get_security_domain(map_name),
780 is_map_replicated(map_name),
781 should_track_dependencies(map_name));
782 map = new_map;
783 new_maps[map_name] = new_map;
784 LOG_DEBUG_FMT(
785 "Creating map '{}' while deserialising transaction at version {}",
786 map_name,
787 v);
788 }
789
790 auto change_search = changes.find(map_name);
791 if (change_search != changes.end())
792 {
793 LOG_FAIL_FMT("Failed to deserialise transaction at version {}", v);
794 LOG_DEBUG_FMT("Multiple writes on map {}", map_name);
795 return false;
796 }
797
798 auto deserialised_changes = map->deserialise_changes(d, v);
799
800 // Take ownership of the produced change set, store it to be applied
801 // later
802 changes.emplace_hint(
803 change_search,
804 std::piecewise_construct,
805 std::forward_as_tuple(map_name),
806 std::forward_as_tuple(map, std::move(deserialised_changes)));
807 }
808
809 if (!d.end())
810 {
811 LOG_FAIL_FMT("Unexpected content in transaction at version {}", v);
812 return false;
813 }
814
815 return true;
816 }

837 bool operator==(const Store& that) const
838 {
839 // Only used for debugging, not thread safe.
840 if (version != that.version)
841 return false;
842
843 if (maps.size() != that.maps.size())
844 return false;
845
846 for (auto it = maps.begin(); it != maps.end(); ++it)
847 {
848 auto search = that.maps.find(it->first);
849
850 if (search == that.maps.end())
851 return false;
852
853 auto& [this_v, this_map] = it->second;
854 auto& [that_v, that_map] = search->second;
855
856 if (this_v != that_v)
857 return false;
858
859 if (*this_map != *that_map)
860 return false;
861 }
862
863 return true;
864 }

873 kv::TxID current_txid() override
874 {

907 CommitResult commit(
908 const TxID& txid,
909 std::unique_ptr<PendingTx> pending_tx,
910 bool globally_committable) override
911 {
912 auto c = get_consensus();
913 if (!c)
914 {
915 return CommitResult::SUCCESS;
916 }
917
918 std::lock_guard<ccf::pal::Mutex> cguard(commit_lock);
919
920 LOG_DEBUG_FMT(
921 "Store::commit {}{}",
922 txid.version,
923 (globally_committable ? " globally_committable" : ""));
924
925 BatchVector batch;
926 Version previous_last_replicated = 0;
927 Version next_last_replicated = 0;
928 Version previous_rollback_count = 0;
929 ccf::View replication_view = 0;
930
931 {
932 std::lock_guard<ccf::pal::Mutex> vguard(version_lock);
933 if (txid.term != term_of_next_version && consensus->is_primary())
934 {
935 // This can happen when a transaction started before a view change,
936 // but tries to commit after the view change is complete.
937 LOG_DEBUG_FMT(
938 "Want to commit for term {} but term is {}",
939 txid.term,
940 term_of_next_version);
941
942 return CommitResult::FAIL_NO_REPLICATE;
943 }
944
945 if (globally_committable && txid.version > last_committable)
946 {
947 last_committable = txid.version;
948 }
949
950 pending_txs.insert(
951 {txid.version,
952 std::make_tuple(std::move(pending_tx), globally_committable)});
953
954 LOG_TRACE_FMT("Inserting pending tx at {}", txid.version);
955
956 auto h = get_history();
957 auto c = get_consensus();
958
959 for (Version offset = 1; true; ++offset)
960 {
961 auto search = pending_txs.find(last_replicated + offset);
962 if (search == pending_txs.end())
963 {
964 LOG_TRACE_FMT(
965 "Couldn't find {} = {} + {}, giving up on batch while committing "
966 "{}.{}",
967 last_replicated + offset,
968 last_replicated,
969 offset,
970 txid.term,
971 txid.version);
972 break;
973 }
974
975 auto& [pending_tx_, committable_] = search->second;
976 auto
977 [success_, data_, claims_digest_, commit_evidence_digest_, hooks_] =
978 pending_tx_->call();
979 auto data_shared =
980 std::make_shared<std::vector<uint8_t>>(std::move(data_));
981 auto hooks_shared =
982 std::make_shared<kv::ConsensusHookPtrs>(std::move(hooks_));
983
984 // NB: this cannot happen currently. Regular Tx only make it here if
985 // they did succeed, and signatures cannot conflict because they
986 // execute in order with a read_version that's version - 1, so even
987 // two contiguous signatures are fine
988 if (success_ != CommitResult::SUCCESS)
989 {
990 LOG_DEBUG_FMT("Failed Tx commit {}", last_replicated + offset);
991 }
992
993 if (h)
994 {
995 h->append_entry(ccf::entry_leaf(
996 *data_shared, commit_evidence_digest_, claims_digest_));
997 }
998
999 LOG_DEBUG_FMT(
1000 "Batching {} ({}) during commit of {}.{}",
1001 last_replicated + offset,
1002 data_shared->size(),
1003 txid.term,
1004 txid.version);
1005
1006 batch.emplace_back(
1007 last_replicated + offset, data_shared, committable_, hooks_shared);
1008 pending_txs.erase(search);
1009 }
1010
1011 if (batch.size() == 0)
1012 {
1013 return CommitResult::SUCCESS;
1014 }
1015
1016 previous_rollback_count = rollback_count;
1017 previous_last_replicated = last_replicated;
1018 next_last_replicated = last_replicated + batch.size();
1019
1020 replication_view = term_of_next_version;
1021
1022 if (consensus->type() == ConsensusType::BFT && consensus->is_backup())
1023 {
1024 last_replicated = next_last_replicated;
1025 }
1026 }
1027
1028 if (c->replicate(batch, replication_view))
1029 {
1030 std::lock_guard<ccf::pal::Mutex> vguard(version_lock);
1031 if (
1032 last_replicated == previous_last_replicated &&
1033 previous_rollback_count == rollback_count &&
1034 !(consensus->type() == ConsensusType::BFT && consensus->is_backup()))
1035 {
1036 last_replicated = next_last_replicated;
1037 }
1038 return CommitResult::SUCCESS;
1039 }
1040 else
1041 {
1042 LOG_DEBUG_FMT("Failed to replicate");
1043 return CommitResult::FAIL_NO_REPLICATE;
1044 }
1045 }

1126 void swap_private_maps(Store& store)
1127 {
1128 {
1129 const auto source_version = store.current_version();
1130 const auto target_version = current_version();
1131 if (source_version > target_version)
1132 {
1133 throw std::runtime_error(fmt::format(
1134 "Invalid call to swap_private_maps. Source is at version {} while "
1135 "target is at {}",
1136 source_version,
1137 target_version));
1138 }
1139 }
1140
1141 std::lock_guard<ccf::pal::Mutex> this_maps_guard(maps_lock);
1142 std::lock_guard<ccf::pal::Mutex> other_maps_guard(store.maps_lock);
1143
1144 // Each entry is (Name, MyMap, TheirMap)
1145 using MapEntry = std::tuple<std::string, AbstractMap*, AbstractMap*>;
1146 std::vector<MapEntry> entries;
1147
1148 // Get the list of private maps from the source store
1149 for (auto& [name, pair] : store.maps)
1150 {
1151 auto& [_, map] = pair;
1152 if (map->get_security_domain() == SecurityDomain::PRIVATE)
1153 {
1154 map->lock();
1155 entries.emplace_back(name, nullptr, map.get());
1156 }
1157 }
1158
1159 // For each source map, either create it or, where it already exists,
1160 // confirm it is PRIVATE. Lock it and store it in entries
1161 auto entry = entries.begin();
1162 while (entry != entries.end())
1163 {
1164 const auto& [name, _, their_map] = *entry;
1165 std::shared_ptr<AbstractMap> map = nullptr;
1166 const auto it = maps.find(name);
1167 if (it == maps.end())
1168 {
1169 // NB: We lose the creation version from the original map, but assume
1170 // it is irrelevant - its creation should no longer be at risk of
1171 // rollback
1172 auto new_map = std::make_pair(
1173 NoVersion,
1174 std::make_shared<kv::untyped::Map>(
1175 this,
1176 name,
1177 SecurityDomain::PRIVATE,
1178 is_map_replicated(name),
1179 should_track_dependencies(name)));
1180 maps[name] = new_map;
1181 map = new_map.second;
1182 }
1183 else
1184 {
1185 map = it->second.second;
1186 if (map->get_security_domain() != SecurityDomain::PRIVATE)
1187 {
1188 throw std::logic_error(fmt::format(
1189 "Swap mismatch - map {} is private in source but not in target",
1190 name));
1191 }
1192 }
1193
1194 std::get<1>(*entry) = map.get();
1195 map->lock();
1196 ++entry;
1197 }
1198
1199 for (auto& [name, lhs, rhs] : entries)
1200 {
1201 lhs->swap(rhs);
1202 }
1203
1204 for (auto& [name, lhs, rhs] : entries)
1205 {
1206 lhs->unlock();
1207 rhs->unlock();
1208 }
1209 }
