42 void expect_enclave_file_suffix(
43 const std::string& file,
44 char const* expected_suffix,
45 host::EnclaveType type)
46 {
47 if (!file.ends_with(expected_suffix))
48 {
49 // Remove possible suffixes to try and get root of filename, to build
50 // suggested filename
51 auto basename = file;
52 for (const char* suffix :
53 {".signed", ".debuggable", ".so", ".enclave", ".virtual", ".snp"})
54 {
55 if (basename.ends_with(suffix))
56 {
57 basename = basename.substr(0, basename.size() - strlen(suffix));
58 }
59 }
60 const auto suggested = fmt::format("{}{}", basename, expected_suffix);
61 throw std::logic_error(fmt::format(
62 "Given enclave file '{}' does not have suffix expected for enclave "
63 "type "
64 "{}. Did you mean '{}'?",
65 file,
66 nlohmann::json(type).dump(),
67 suggested));
68 }
69 }

71 static std::pair<uint8_t*, size_t> allocate_8_aligned(size_t size)
72 {
73 const auto aligned_size = (size + 7) & ~(7ull);
74 auto data = static_cast<uint8_t*>(std::aligned_alloc(8u, aligned_size));
75 if (data == nullptr)
76 {
77 throw std::runtime_error(fmt::format(
78 "Unable to allocate {} bytes for aligned data", aligned_size));
79 }
80 return std::make_pair(data, aligned_size);
81 }

107 Enclave(const std::string& path, EnclaveType type, EnclavePlatform platform)
108 {
109 if (!std::filesystem::exists(path))
110 {
111 throw std::logic_error(
112 fmt::format("No enclave file found at {}", path));
113 }
114
115 switch (platform)
116 {
117 case host::EnclavePlatform::SGX:
118 {
119#ifdef PLATFORM_SGX
120 uint32_t oe_flags = 0;
121 if (type == host::EnclaveType::DEBUG)
122 {
123 expect_enclave_file_suffix(path, ".enclave.so.debuggable", type);
124 oe_flags |= OE_ENCLAVE_FLAG_DEBUG;
125 }
126 else
127 {
128 expect_enclave_file_suffix(path, ".enclave.so.signed", type);
129 }
130
131# ifndef VERBOSE_LOGGING
132 oe_log_set_callback(nullptr, nop_oe_logger);
133# endif
134
135 auto err = oe_create_ccf_enclave(
136 path.c_str(),
137 OE_ENCLAVE_TYPE_SGX,
138 oe_flags,
139 nullptr,
140 0,
141 &sgx_handle);
142
143 if (err != OE_OK)
144 {
145 throw std::logic_error(
146 fmt::format("Could not create enclave: {}", oe_result_str(err)));
147 }
148#else
149 throw std::logic_error(fmt::format(
150 "SGX enclaves are not supported in current build - cannot launch "
151 "{}",
152 path));
153#endif // defined(PLATFORM_SGX)
154 break;
155 }
156
157 case host::EnclavePlatform::SNP:
158 {
159#if defined(PLATFORM_SNP)
160 expect_enclave_file_suffix(path, ".snp.so", type);
161 virtual_handle = load_virtual_enclave(path.c_str());
162#else
163 throw std::logic_error(fmt::format(
164 "SNP enclaves are not supported in current build - cannot launch "
165 "{}",
166 path));
167#endif // defined(PLATFORM_SNP)
168 break;
169 }
170
171 case host::EnclavePlatform::VIRTUAL:
172 {
173#if defined(PLATFORM_VIRTUAL)
174 expect_enclave_file_suffix(path, ".virtual.so", type);
175 virtual_handle = load_virtual_enclave(path.c_str());
176#else
177 throw std::logic_error(fmt::format(
178 "Virtual enclaves are not supported in current build - cannot "
179 "launch {}",
180 path));
181#endif // defined(PLATFORM_VIRTUAL)
182 break;
183 }
184
185 default:
186 {
187 throw std::logic_error(fmt::format(
188 "Unsupported enclave type: {}", nlohmann::json(type).dump()));
189 }
190 }
191 }

193 CreateNodeStatus create_node(
194 const EnclaveConfig& enclave_config,
195 const StartupConfig& ccf_config,
196 std::vector<uint8_t>&& startup_snapshot,
197 std::vector<uint8_t>& node_cert,
198 std::vector<uint8_t>& service_cert,
199 StartType start_type,
200 size_t num_worker_thread,
201 void* time_location)
202 {
203 CreateNodeStatus status = CreateNodeStatus::InternalError;
204 constexpr size_t enclave_version_size = 256;
205 std::vector<uint8_t> enclave_version_buf(enclave_version_size);
206
207 size_t node_cert_len = 0;
208 size_t service_cert_len = 0;
209 size_t enclave_version_len = 0;
210
211 // Pad config and startup snapshot with NULLs to a multiple of 8, in an
212 // 8-byte aligned allocation
213 auto config_s = nlohmann::json(ccf_config).dump();
214 auto [config, config_aligned_size] = allocate_8_aligned(config_s.size());
215 LOG_DEBUG_FMT(
216 "Padding config of size {} to {} bytes",
217 config_s.size(),
218 config_aligned_size);
219 auto copy_end = std::copy(config_s.begin(), config_s.end(), config);
220 std::fill(copy_end, config + config_aligned_size, 0);
221
222 auto [snapshot, snapshot_aligned_size] =
223 allocate_8_aligned(startup_snapshot.size());
224 LOG_DEBUG_FMT(
225 "Padding startup snapshot of size {} to {} bytes",
226 startup_snapshot.size(),
227 snapshot_aligned_size);
228
229 auto snapshot_copy_end =
230 std::copy(startup_snapshot.begin(), startup_snapshot.end(), snapshot);
231 std::fill(snapshot_copy_end, snapshot + snapshot_aligned_size, 0);
232
233#define CREATE_NODE_ARGS \
234 &status, (void*)&enclave_config, config, config_aligned_size, snapshot, \
235 snapshot_aligned_size, node_cert.data(), node_cert.size(), &node_cert_len, \
236 service_cert.data(), service_cert.size(), &service_cert_len, \
237 enclave_version_buf.data(), enclave_version_buf.size(), \
238 &enclave_version_len, start_type, num_worker_thread, time_location
239
240 oe_result_t err = OE_FAILURE;
241
242// Assume that constructor correctly set the appropriate field, and call
243// appropriate function
244#if defined(PLATFORM_VIRTUAL) || defined(PLATFORM_SNP)
245 if (virtual_handle != nullptr)
246 {
247 err = virtual_create_node(virtual_handle, CREATE_NODE_ARGS);
248 }
249#endif
250#ifdef PLATFORM_SGX
251 if (sgx_handle != nullptr)
252 {
253 err = enclave_create_node(sgx_handle, CREATE_NODE_ARGS);
254 }
255#endif
256
257 std::free(config);
258 std::free(snapshot);
259
260 if (err != OE_OK || status != CreateNodeStatus::OK)
261 {
262 // Logs have described the errors already, we just need to allow the
263 // host to read them (via read_all()).
264 return status;
265 }
266
267 // Host and enclave versions must match. Otherwise the node may crash much
268 // later (e.g. unhandled ring buffer message on either end)
269 auto enclave_version = std::string(
270 enclave_version_buf.begin(),
271 enclave_version_buf.begin() + enclave_version_len);
272 if (ccf::ccf_version != enclave_version)
273 {
274 LOG_FAIL_FMT(
275 "Host/Enclave versions mismatch: {} != {}",
276 ccf::ccf_version,
277 enclave_version);
278 return CreateNodeStatus::VersionMismatch;
279 }
280
281 node_cert.resize(node_cert_len);
282 service_cert.resize(service_cert_len);
283
284 return CreateNodeStatus::OK;
285 }
286
287 // Run a processor over this circuit inside the enclave - should be called
288 // from a thread
289 bool run()
290 {
291 bool ret = true;
292 oe_result_t err = OE_FAILURE;
293
294#if defined(PLATFORM_VIRTUAL) || defined(PLATFORM_SNP)
295 if (virtual_handle != nullptr)
296 {
297 err = virtual_run(virtual_handle, &ret);
298 }
299#endif
300#ifdef PLATFORM_SGX
301 if (sgx_handle != nullptr)
302 {
303 err = enclave_run(sgx_handle, &ret);
304 }
305#endif
306
307 if (err != OE_OK)
308 {
309 throw std::logic_error(
310 fmt::format("Failed to call in enclave_run: {}", oe_result_str(err)));
311 }
312
313 return ret;
314 }
315 };
