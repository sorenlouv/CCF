13 /*
14 * For simple app methods which expect a JSON request, potentially msgpack'd,
15 * these functions do the common decoding of the input and setting of response
16 * fields, to reduce handler complexity and repetition.
17 *
18 * Rather than:
19 * auto foo = [](auto& ctx) {
20 * nlohmann::json params;
21 * serdes::Pack pack_type;
22 * if (<content-type is JSON>)
23 * {
24 * params = unpack(ctx.rpc_ctx->get_request_body());
25 * pack_type = Text;
26 * }
27 * else
28 * {
29 * ...
30 * }
31 * auto result = fn(params);
32 * if (is_error(result))
33 * {
34 * ctx.rpc_ctx->set_response_status(SOME_ERROR);
35 * ctx.rpc_ctx->set_response_header(content_type, Text);
36 * ctx.rpc_ctx->set_response_body(error_msg(result));
37 * }
38 * if (pack_type == Text)
39 * {
40 * ctx.rpc_ctx->set_response_header(content_type, JSON);
41 * ctx.rpc_ctx->set_response_body(pack(result, Text));
42 * }
43 * else
44 * {
45 * ...
46 * }
47 * };
48 *
49 * it is possible to write the shorter, clearer, return-based lambda:
50 * auto foo = json_adapter([](auto& ctx, nlohmann::json&& params)
51 * {
52 * auto result = fn(params);
53 * if (is_error(result))
54 * {
55 * return make_error(SOME_ERROR, error_msg(result));
56 * }
57 * else
58 * {
59 * return make_success(result);
60 * }
61 * });
62 */
