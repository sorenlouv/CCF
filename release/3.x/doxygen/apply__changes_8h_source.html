24 // Atomically checks for conflicts then applies the writes in the given change
25 // sets to their underlying Maps. Calls f() at most once, iff the writes are
26 // applied, to retrieve a unique Version for the write set and return the max
27 // version which can have a conflict with the transaction.
28 //
29 // The track_read_versions parameter tells the store if it needs to track the
30 // last read version for every key. This is required for backup execution as
31 // described at the top of tx.h
32
33 using VersionLastNewMap = Version;
34 using VersionResolver = std::function<std::tuple<Version, VersionLastNewMap>(
35 bool tx_contains_new_map)>;
36
37 static inline std::optional<Version> apply_changes(
38 OrderedChanges& changes,
39 VersionResolver version_resolver_fn,
40 kv::ConsensusHookPtrs& hooks,
41 const MapCollection& new_maps = {},
42 const std::optional<Version>& new_maps_conflict_version = std::nullopt,
43 bool track_read_versions = false)
44 {
45 // All maps with pending writes are locked, transactions are prepared
46 // and possibly committed, and then all maps with pending writes are
47 // unlocked. This is to prevent transactions from being committed in an
48 // interleaved fashion.
49 Version version = NoVersion;
50 bool has_writes = false;
51
52 std::map<std::string, std::unique_ptr<AbstractCommitter>> views;
53 for (const auto& [map_name, mc] : changes)
54 {
55 views[map_name] = mc.map->create_committer(mc.changeset.get());
56 }
57
58 for (auto it = changes.begin(); it != changes.end(); ++it)
59 {
60 bool changeset_has_writes = it->second.changeset->has_writes();
61 if (changeset_has_writes)
62 {
63 has_writes = true;
64 }
65 if (changeset_has_writes || track_read_versions)
66 {
67 it->second.map->lock();
68 }
69 }
70
71 bool ok = true;
72 if (has_writes)
73 {
74 for (auto it = views.begin(); it != views.end(); ++it)
75 {
76 if (!it->second->prepare(track_read_versions))
77 {
78 ok = false;
79 break;
80 }
81 }
82 }
83
84 for (const auto& [map_name, map_ptr] : new_maps)
85 {
86 // Check that none of these pending maps have already been created.
87 // It is possible for non-conflicting other transactions to commit here
88 // and increment the version, so we may ask this question at different
89 // versions. This is fine - none can create maps (ie - change their
90 // conflict set with this operation) while we hold the store lock. Assume
91 // that the caller is currently holding store->lock()
92 auto store = map_ptr->get_store();
93
94 // This is to avoid recursively locking version_lock by calling
95 // current_version() in the commit_reserved case.
96 kv::Version current_v;
97 if (new_maps_conflict_version.has_value())
98 {
99 current_v = *new_maps_conflict_version;
100 }
101 else
102 {
103 current_v = store->current_version();
104 }
105
106 if (store->get_map(current_v, map_name) != nullptr)
107 {
108 ok = false;
109 break;
110 }
111 }
112
113 if (ok && has_writes)
114 {
115 // Get the version number to be used for this commit.
116 kv::Version version_last_new_map;
117 std::tie(version, version_last_new_map) =
118 version_resolver_fn(!new_maps.empty());
119
120 if (!track_read_versions)
121 {
122 // Transfer ownership of these new maps to their target stores, iff we
123 // have writes to them
124 for (const auto& [map_name, map_ptr] : new_maps)
125 {
126 const auto it = views.find(map_name);
127 if (it != views.end() && it->second->has_writes())
128 {
129 map_ptr->get_store()->add_dynamic_map(version, map_ptr);
130 }
131 }
132
133 for (auto it = views.begin(); it != views.end(); ++it)
134 {
135 it->second->commit(version, track_read_versions);
136 }
137
138 // Collect ConsensusHooks
139 for (auto it = views.begin(); it != views.end(); ++it)
140 {
141 auto hook_ptr = it->second->post_commit();
142 if (hook_ptr != nullptr)
143 {
144 hooks.push_back(std::move(hook_ptr));
145 }
146 }
147 }
148 else
149 {
150 // A linearizability violation was detected
151 ok = false;
152 }
153 }
154
155 for (auto it = changes.begin(); it != changes.end(); ++it)
156 {
157 if (it->second.changeset->has_writes() || track_read_versions)
158 {
159 it->second.map->unlock();
160 }
161 }
162
163 if (!ok)
164 {
165 return std::nullopt;
166 }
167
168 return version;
169 }
170}
