50 static Write deserialise_write(const kv::untyped::Write& w)
51 {
52 Write typed_writes;
53 for (const auto& [uk, opt_uv] : w)
54 {
55 if (!opt_uv.has_value())
56 {
57 // Deletions are indicated by nullopt. We cannot deserialise them,
58 // they are deletions here as well
59 typed_writes[KSerialiser::from_serialised(uk)] = std::nullopt;
60 }
61 else
62 {
63 typed_writes[KSerialiser::from_serialised(uk)] =
64 VSerialiser::from_serialised(opt_uv.value());
65 }
66 }
67 return typed_writes;
68 }
