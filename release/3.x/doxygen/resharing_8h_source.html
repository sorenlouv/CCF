30 ResharingsHook(kv::Version version_, const Resharings::Write& w) :
31 version(version_)
32 {
33 for (const auto& [rid, opt_rr] : w)
34 {
35 if (opt_rr.has_value())
36 {
37 LOG_DEBUG_FMT(

45 {

92 virtual void reshare(const kv::Configuration& config) override
93 {
94 auto rid = config.rid;
95 LOG_DEBUG_FMT("Resharings: start resharing for configuration #{}", rid);
96 sessions.emplace(rid, ResharingSession(config));
97
98 const auto& node_cert = endorsed_node_cert.has_value() ?
99 endorsed_node_cert.value() :
100 self_signed_node_cert;
101 auto msg = std::make_unique<threading::Tmsg<UpdateResharingTaskMsg>>(
102 update_resharing_cb, rid, rpc_map, node_sign_kp, node_cert);
103
104 threading::ThreadMessaging::thread_messaging.add_task(
105 threading::ThreadMessaging::get_execution_thread(
106 threading::MAIN_THREAD_ID),
107 std::move(msg));

110 virtual std::optional<kv::ReconfigurationId> find_reconfiguration(
111 const kv::Configuration::Nodes& nodes) const override
112 {
113 // We're searching for a configuration with the same set of nodes as
114 // `nodes`. We currently don't have an easy way to look up the
115 // reconfiguration ID, which would make this unnecessary.
116 for (auto& [rid, config] : configs)
117 {
118 bool have_all = true;
119 for (auto& [nid, _] : nodes)
120 {
121 if (config.nodes.find(nid) == config.nodes.end())
122 {
123 have_all = false;
124 break;
125 }
126 }
127 if (have_all)
128 {
129 return rid;
130 }
131 }
132 return std::nullopt;
133 }

135 virtual bool have_resharing_result_for(

138 auto idt = results.find(rid);

147 LOG_DEBUG_FMT(

153 virtual void compact(Index idx) override
154 {
155 for (auto it = sessions.begin(); it != sessions.end();)
156 {
157 if (it->first <= idx)
158 {
159 it = sessions.erase(it);
160 }
161 else
162 {
163 it++;
164 }
165 }
166 }
167
168 virtual ResharingResult get_resharing_result(
169 kv::ReconfigurationId rid) const override
170 {
171 auto iit = results.find(rid);
172 if (iit == results.end())
173 {
174 throw std::runtime_error("missing resharing result");
175 }
176 return iit->second;
177 }
178
179 virtual void rollback(Index idx) override
180 {
181 for (auto it = results.begin(); it != results.end();)
182 {
183 if (it->second.seqno > idx)
184 {
185 assert(sessions.find(it->first) == sessions.end());
186 it = results.erase(it);
187 }
188 else
189 {
190 it++;
191 }
192 }
193 }

227 static inline bool make_request(
228 http::Request& request,
229 std::shared_ptr<ccf::RPCMap> rpc_map,
230 crypto::KeyPairPtr node_sign_kp,
231 const crypto::Pem& node_cert)
232 {
233 auto node_cert_der = crypto::cert_pem_to_der(node_cert);
234 const auto key_id = crypto::Sha256Hash(node_cert_der).hex_str();
235
236 http::sign_request(request, node_sign_kp, key_id);
237 std::vector<uint8_t> packed = request.build_request();
238 auto node_session = std::make_shared<ccf::SessionContext>(
239 ccf::InvalidSessionId, node_cert.raw());
240 auto ctx = ccf::make_rpc_context(node_session, packed);
241
242 const auto actor_opt = http::extract_actor(*ctx);
243 if (!actor_opt.has_value())
244 {
245 throw std::logic_error("Unable to get actor");
246 }
247
248 const auto actor = rpc_map->resolve(actor_opt.value());
249 auto frontend_opt = rpc_map->find(actor);
250 if (!frontend_opt.has_value())
251 {
252 throw std::logic_error(
253 "RpcMap::find returned invalid (empty) frontend");
254 }
255
256 auto frontend = frontend_opt.value();
257 frontend->process(ctx);
258
259 auto rs = ctx->get_response_status();
260
261 if (rs != HTTP_STATUS_OK)
262 {
263 auto ser_res = ctx->serialise_response();
264 std::string str((char*)ser_res.data(), ser_res.size());
265 LOG_FAIL_FMT("request failed: {}", str);
266 }
267
268 return rs == HTTP_STATUS_OK;
269 }

271 static inline bool request_update_resharing(
272 kv::ReconfigurationId rid,
273 std::shared_ptr<ccf::RPCMap> rpc_map,
274 crypto::KeyPairPtr node_sign_kp,
275 const crypto::Pem& node_cert)
276 {
277 ccf::UpdateResharing::In ps = {rid};
278
279 http::Request request(fmt::format(
280 "/{}/{}",
281 ccf::get_actor_prefix(ccf::ActorsType::nodes),
282 "update-resharing"));
283 request.set_header(
284 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
285
286 auto body = serdes::pack(ps, serdes::Pack::Text);
287 request.set_body(&body);
288 return make_request(request, rpc_map, node_sign_kp, node_cert);
289 }
290
291 static void update_resharing_cb(
292 std::unique_ptr<threading::Tmsg<UpdateResharingTaskMsg>> msg)
293 {
294 if (!request_update_resharing(
295 msg->data.rid,
296 msg->data.rpc_map,
297 msg->data.node_sign_kp,
298 msg->data.node_cert))
299 {
300 if (--msg->data.retries > 0)
301 {
302 threading::ThreadMessaging::thread_messaging.add_task(
303 threading::ThreadMessaging::get_execution_thread(
304 threading::MAIN_THREAD_ID),
305 std::move(msg));
306 }
307 else
308 {
309 LOG_DEBUG_FMT(
310 "Failed request, giving up as there are no more retries left");
311 }
312 }
313 }
