19 using CompressedFlag = uint8_t;
20 using MessageLength = uint32_t;
21
22 static constexpr size_t message_frame_length =
23 sizeof(CompressedFlag) + sizeof(MessageLength);
24
25 MessageLength read_message_frame(const uint8_t*& data, size_t& size)
26 {
27 auto compressed_flag = serialized::read<CompressedFlag>(data, size);
28 if (compressed_flag >= 1)
29 {
30 throw std::logic_error(fmt::format(
31 "gRPC compressed flag has unexpected value {} - currently only "
32 "support "
33 "unencoded gRPC payloads",
34 compressed_flag));
35 }
36 return ntohl(serialized::read<MessageLength>(data, size));
37 }
38
39 void write_message_frame(uint8_t*& data, size_t& size, size_t message_size)
40 {
41 CompressedFlag compressed_flag = 0;
42 serialized::write(data, size, compressed_flag);
43 serialized::write(data, size, htonl(message_size));
44 }

96 template <typename In>
97 In get_grpc_payload(const std::shared_ptr<ccf::RpcContext>& ctx)
98 {
99 auto& request_body = ctx->get_request_body();
100 auto request_content_type =
101 ctx->get_request_header(http::headers::CONTENT_TYPE);
102
103 auto data = request_body.data();
104 auto size = request_body.size();
105
106 if (request_content_type != http::headervalues::contenttype::GRPC)
107 {
108 throw RpcException(
109 HTTP_STATUS_UNSUPPORTED_MEDIA_TYPE,
110 ccf::errors::UnsupportedContentType,
111 fmt::format(
112 "Unsupported content type. Only {} is supported ",
113 http::headervalues::contenttype::GRPC));
114 }
115
116 if constexpr (nonstd::is_std_vector<In>::value)
117 {
118 using Message = typename In::value_type;
119 In messages;
120 while (size != 0)
121 {
122 const auto message_length = impl::read_message_frame(data, size);
123 if (message_length > size)
124 {
125 throw std::logic_error(fmt::format(
126 "Error in gRPC frame: only {} bytes remaining but message header "
127 "says messages is {} bytes",
128 size,
129 message_length));
130 }
131
132 Message& msg = messages.emplace_back();
133 if (!msg.ParseFromArray(data, message_length))
134 {
135 throw std::logic_error(fmt::format(
136 "Error deserialising protobuf payload of type {}, size {} (message "
137 "{} in "
138 "stream)",
139 msg.GetTypeName(),
140 size,
141 messages.size()));
142 }
143 data += message_length;
144 size -= message_length;
145 }
146 return messages;
147 }
148 else
149 {
150 const auto message_length = impl::read_message_frame(data, size);
151 if (size != message_length)
152 {
153 throw std::logic_error(fmt::format(
154 "Error in gRPC frame: frame size is {} but messages is {} bytes",
155 size,
156 message_length));
157 }
158
159 In in;
160 if (!in.ParseFromArray(data, message_length))
161 {
162 throw std::logic_error(fmt::format(
163 "Error deserialising protobuf payload of type {}, size {}",
164 in.GetTypeName(),
165 size));
166 }
167 return in;
168 }

171 template <typename Out>
172 void set_grpc_response(
173 const GrpcAdapterResponse<Out>& r,
174 const std::shared_ptr<ccf::RpcContext>& ctx)
175 {
176 auto success_response = std::get_if<SuccessResponse<Out>>(&r);
177 if (success_response != nullptr)
178 {
179 std::vector<uint8_t> r;
180
181 if constexpr (nonstd::is_std_vector<Out>::value)
182 {
183 using Message = typename Out::value_type;
184 const Out& messages = success_response->body;
185 size_t r_size = std::accumulate(
186 messages.begin(),
187 messages.end(),
188 0,
189 [](size_t current, const Message& msg) {
190 return current + impl::message_frame_length + msg.ByteSizeLong();
191 });
192 r.resize(r_size);
193 auto r_data = r.data();
194
195 for (const Message& msg : messages)
196 {
197 const auto message_length = msg.ByteSizeLong();
198 impl::write_message_frame(r_data, r_size, message_length);
199
200 if (!msg.SerializeToArray(r_data, r_size))
201 {
202 throw std::logic_error(fmt::format(
203 "Error serialising protobuf response of type {}, size {}",
204 msg.GetTypeName(),
205 message_length));
206 }
207
208 r_data += message_length;
209 r_size += message_length;
210 }
211 }
212 else
213 {
214 const Out& resp = success_response->body;
215 const auto message_length = resp.ByteSizeLong();
216 size_t r_size = impl::message_frame_length + message_length;
217 r.resize(r_size);
218 auto r_data = r.data();
219
220 impl::write_message_frame(r_data, r_size, message_length);
221
222 if (!resp.SerializeToArray(r_data, r_size))
223 {
224 throw std::logic_error(fmt::format(
225 "Error serialising protobuf response of type {}, size {}",
226 resp.GetTypeName(),
227 message_length));
228 }
229 }
230
231 ctx->set_response_body(r);
232 ctx->set_response_header(
233 http::headers::CONTENT_TYPE, http::headervalues::contenttype::GRPC);
234
235 ctx->set_response_trailer("grpc-status", success_response->status.code());
236 ctx->set_response_trailer(
237 "grpc-message", success_response->status.message());
238 }
239 else
240 {
241 auto error_response = std::get<ErrorResponse>(r);
242 ctx->set_response_trailer("grpc-status", error_response.status.code());
243 ctx->set_response_trailer(
244 "grpc-message", error_response.status.message());
245 }
246 }
