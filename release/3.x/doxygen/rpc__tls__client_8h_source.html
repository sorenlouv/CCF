47 std::vector<uint8_t> gen_http_request_internal(
48 const std::string& method,
49 const std::span<const uint8_t> params,
50 const std::string& content_type,
51 llhttp_method verb,
52 const char* auth_token = nullptr)
53 {
54 auto path = method;
55 if (prefix.has_value())
56 {
57 path = fmt::format("/{}/{}", prefix.value(), path);
58 }
59
60 auto r = http::Request(path, verb);
61 r.set_body(params.data(), params.size());
62 r.set_header(http::headers::CONTENT_TYPE, content_type);
63 r.set_header("Host", host);
64 if (auth_token != nullptr)
65 {
66 r.set_header(
67 http::headers::AUTHORIZATION, fmt::format("Bearer {}", auth_token));
68 }
69
70 if (key_pair != nullptr)
71 {
72 http::sign_request(r, key_pair, key_id);
73 }
74
75 return r.build_request();
76 }

180 Response get(
181 const std::string& method, const nlohmann::json& params = nullptr)
182 {
183 // GET body is ignored, so params must be placed in query
184 auto full_path = method;
185 if (!params.is_null())
186 {
187 for (auto it = params.begin(); it != params.end(); ++it)
188 {
189 full_path += fmt::format(
190 "{}{}={}",
191 it == params.begin() ? "?" : "&",
192 it.key(),
193 it.value().is_string() ? it.value().get<std::string>() :
194 it.value().dump());
195 }
196 }
197 return call(full_path, nullptr, HTTP_GET);
198 }
