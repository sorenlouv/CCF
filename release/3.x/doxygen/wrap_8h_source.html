550 class JSWrappedPropertyEnum
551 {
552 public:
553 JSWrappedPropertyEnum(JSContext* ctx, const JSWrappedValue& value)
554 {
555 if (!JS_IsObject(value))
556 {
557 throw std::logic_error(
558 fmt::format("object value required for property enum"));
559 }
560
561 JSPropertyEnum* prop_enum;
562 uint32_t prop_count;
563
564 if (
565 JS_GetOwnPropertyNames(
566 ctx,
567 &prop_enum,
568 &prop_count,
569 value,
570 JS_GPN_STRING_MASK | JS_GPN_ENUM_ONLY) == -1)
571 {
572 throw std::logic_error(
573 fmt::format("Could not extract property names of enum"));
574 }
575 for (size_t i = 0; i < prop_count; i++)
576 properties.push_back(JSWrappedAtom(ctx, prop_enum[i].atom));
577 for (uint32_t i = 0; i < prop_count; i++)
578 JS_FreeAtom(ctx, prop_enum[i].atom);
579 js_free(ctx, prop_enum);
580 }
581 ~JSWrappedPropertyEnum() {}
582
583 JSWrappedAtom operator[](size_t i) const
584 {
585 return properties[i];
586 }
587
588 size_t size() const
589 {
590 return properties.size();
591 }
592
593 JSContext* ctx;
594 std::vector<JSWrappedAtom> properties;
595 };
