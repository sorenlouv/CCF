27 std::vector<uint8_t> pending;

29 NodeConnectionBehaviour(
30 const char* name,
31 NodeConnections& parent,
32 std::optional<ccf::NodeId> node = std::nullopt) :
33 SocketBehaviour<TCP>(name, "TCP"),
34 parent(parent),
35 node(node)
36 {}
37
38 void on_read(size_t len, uint8_t*& incoming, sockaddr)
39 {
40 LOG_DEBUG_FMT(
41 "from node {} received {} bytes",
42 node.value_or(UnassociatedNode),
43 len);
44
45 pending.insert(pending.end(), incoming, incoming + len);
46
47 const uint8_t* data = pending.data();
48 size_t size = pending.size();
49 const auto size_before = size;
50
51 while (true)
52 {
53 if (!msg_size.has_value())
54 {
55 if (size < sizeof(uint32_t))
56 {
57 break;
58 }
59
60 msg_size = serialized::read<uint32_t>(data, size);
61 }
62
63 if (size < msg_size.value())
64 {
65 LOG_DEBUG_FMT(
66 "from node {} have {}/{} bytes",
67 node.value_or(UnassociatedNode),
68 size,
69 msg_size.value());
70 break;
71 }
72
73 const auto size_pre_headers = size;
74 auto msg_type = serialized::read<ccf::NodeMsgType>(data, size);
75 ccf::NodeId from = serialized::read<ccf::NodeId::Value>(data, size);
76 const auto size_post_headers = size;
77 const size_t payload_size =
78 msg_size.value() - (size_pre_headers - size_post_headers);
79
80 if (!node.has_value())
81 {
82 associate_incoming(from);
83 node = from;
84 }
85
86 LOG_DEBUG_FMT(
87 "node in: from node {}, size {}, type {}",
88 node.value(),
89 msg_size.value(),
90 msg_type);
91
92 RINGBUFFER_WRITE_MESSAGE(
93 ccf::node_inbound,
94 parent.to_enclave,
95 msg_type,
96 from.value(),
97 serializer::ByteRange{data, payload_size});
98
99 data += payload_size;
100 size -= payload_size;
101 msg_size.reset();
102 }
103
104 const auto size_after = size;
105 const auto used = size_before - size_after;
106 if (used > 0)
107 {
108 pending.erase(pending.begin(), pending.begin() + used);
109 }
110 }
111
112 virtual void associate_incoming(const ccf::NodeId&) {}
113 };

115 class NodeIncomingBehaviour : public NodeConnectionBehaviour
116 {
117 public:
118 size_t id;
119 std::optional<ccf::NodeId> node_id;
120
121 NodeIncomingBehaviour(NodeConnections& parent, size_t id_) :
122 NodeConnectionBehaviour("Node Incoming", parent),
123 id(id_)
124 {}
125
126 void on_disconnect() override
127 {
128 LOG_DEBUG_FMT("Disconnecting incoming connection {}", id);
129 parent.unassociated_incoming.erase(id);
130
131 if (node_id.has_value())
132 {
133 parent.remove_connection(node_id.value());
134 }
135 }
136
137 void associate_incoming(const ccf::NodeId& n) override
138 {
139 node_id = n;
140
141 const auto unassociated = parent.unassociated_incoming.find(id);
142 CCF_ASSERT_FMT(
143 unassociated != parent.unassociated_incoming.end(),
144 "Associating node {} with incoming ID {}, but have already forgotten "
145 "the incoming connection",
146 n,
147 id);
148
149 // Always prefer this (probably) newer connection. Pathological case is
150 // where both nodes open outgoings to each other at the same time, both
151 // see the corresponding incoming connections and _drop_ their outgoing
152 // connections. Both have a useless incoming connection they think they
153 // can use. Assumption is that they progress at different rates, and one
154 // of them eventually spots the dead connection and opens a new one
155 // which succeeds.
156 parent.connections[n] = unassociated->second;
157 parent.unassociated_incoming.erase(unassociated);
158
159 LOG_DEBUG_FMT(
160 "Node incoming connection ({}) associated with {}", id, n);
161 }
162 };

287 DISPATCHER_SET_MESSAGE_HANDLER(
288 disp, ccf::node_outbound, [this](const uint8_t* data, size_t size) {
289 // Read piece-by-piece rather than all at once
290 ccf::NodeId to = serialized::read<ccf::NodeId::Value>(data, size);
291
292 TCP outbound_connection = nullptr;
293 {
294 const auto connection_it = connections.find(to);
295 if (connection_it == connections.end())
296 {
297 const auto address_it = node_addresses.find(to);
298 if (address_it == node_addresses.end())
299 {
300 LOG_FAIL_FMT("Ignoring node_outbound to unknown node {}", to);
301 return;
302 }
303
304 const auto& [host, port] = address_it->second;
305 outbound_connection = create_connection(to, host, port);
306 if (outbound_connection.is_null())
307 {
308 LOG_FAIL_FMT(
309 "Unable to connect to {}, dropping outbound message message",
310 to);
311 return;
312 }
313 }
314 else
315 {
316 outbound_connection = connection_it->second;
317 }
318 }
319
320 // Rather than reading and reserialising, use the msg_type and from_id
321 // that are already serialised on the ringbuffer
322 auto data_to_send = data;
323 auto size_to_send = size;
324
325 // If the message is a consensus append entries message, affix the
326 // corresponding ledger entries
327 auto msg_type = serialized::read<ccf::NodeMsgType>(data, size);
328 serialized::read<ccf::NodeId::Value>(data, size); // Ignore from_id
329 if (
330 msg_type == ccf::NodeMsgType::consensus_msg &&
331 (serialized::read<aft::RaftMsgType>(data, size) ==
332 aft::raft_append_entries))
333 {
334 // Parse the indices to be sent to the recipient.
335 const auto& ae =
336 serialized::overlay<consensus::AppendEntriesIndex>(data, size);
337
338 // Find the total frame size, and write it along with the header.
339 uint32_t frame = (uint32_t)size_to_send;
340
341 if (ae.idx > ae.prev_idx)
342 {
343 std::optional<asynchost::LedgerReadResult> read_result =
344 ledger.read_entries(ae.prev_idx + 1, ae.idx);
345
346 if (!read_result.has_value())
347 {
348 LOG_FAIL_FMT(
349 "Unable to send AppendEntries ({}, {}]: Ledger read failed",
350 ae.prev_idx,
351 ae.idx);
352 return;
353 }
354 else if (ae.idx != read_result->end_idx)
355 {
356 // NB: This should never happen since we do not pass a max_size
357 // to read_entries
358 LOG_FAIL_FMT(
359 "Unable to send AppendEntries ({}, {}]: Ledger read returned "
360 "entries to {}",
361 ae.prev_idx,
362 ae.idx,
363 read_result->end_idx);
364 return;
365 }
366 else
367 {
368 const auto& framed_entries = read_result->data;
369 frame += (uint32_t)framed_entries.size();
370 outbound_connection->write(sizeof(uint32_t), (uint8_t*)&frame);
371 outbound_connection->write(size_to_send, data_to_send);
372
373 outbound_connection->write(
374 framed_entries.size(), framed_entries.data());
375 }
376 }
377 else
378 {
379 // Header-only AE
380 outbound_connection->write(sizeof(uint32_t), (uint8_t*)&frame);
381 outbound_connection->write(size_to_send, data_to_send);
382 }
383
384 LOG_DEBUG_FMT(
385 "send AE to node {} [{}]: {}, {}",
386 to,
387 frame,
388 ae.idx,
389 ae.prev_idx);
390 }
391 else
392 {
393 // Write as framed data to the recipient.
394 uint32_t frame = (uint32_t)size_to_send;
395
396 LOG_DEBUG_FMT("node send to {} [{}]", to, frame);
397
398 outbound_connection->write(sizeof(uint32_t), (uint8_t*)&frame);
399 outbound_connection->write(size_to_send, data_to_send);
400 }
401 });
402 }

405 TCP create_connection(
406 const ccf::NodeId& node_id,
407 const std::string& host,
408 const std::string& port)
409 {
410 auto s = TCP(true, client_connection_timeout);
411 s->set_behaviour(std::make_unique<NodeOutgoingBehaviour>(*this, node_id));
412
413 if (!s->connect(host, port, client_interface))
414 {
415 LOG_FAIL_FMT("Failed to connect to {} on {}:{}", node_id, host, port);
416 return nullptr;
417 }
418
419 connections.emplace(node_id, s);
420 LOG_DEBUG_FMT(
421 "Added node connection with {} ({}:{})", node_id, host, port);
422
423 return s;
424 }
425
426 bool remove_connection(const ccf::NodeId& node)
427 {
428 if (connections.erase(node) < 1)
429 {
430 LOG_DEBUG_FMT("Cannot remove node connection {}: does not exist", node);
431 return false;
432 }
433
434 LOG_DEBUG_FMT("Removed node connection with {}", node);
435 return true;
436 }

438 size_t get_next_id()
439 {
440 auto id = next_id++;
441
442 while (unassociated_incoming.find(id) != unassociated_incoming.end())
443 {
444 id = next_id++;
