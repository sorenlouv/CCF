21 class FragmentReconstructor
22 {
23 messaging::RingbufferDispatcher& dispatcher;
24
25 struct PartialMessage
26 {
27 const ringbuffer::Message m;
28 const size_t total_size;
29
30 size_t received;
31 uint8_t* data;
32 };
33
34 std::unordered_map<size_t, PartialMessage> partial_messages;
35
36 public:
37 FragmentReconstructor(messaging::RingbufferDispatcher& d) : dispatcher(d)
38 {
39 DISPATCHER_SET_MESSAGE_HANDLER(
40 d,
41 OversizedMessage::fragment,
42 [this](const uint8_t* data, size_t size) {
43 auto message_id = serialized::read<size_t>(data, size);
44
45 auto it = partial_messages.find(message_id);
46 if (it == partial_messages.end())
47 {
48 // First reference to this oversized message - should contain a
49 // header. Read its type, size, then allocate space for it
50 auto m = serialized::read<ringbuffer::Message>(data, size);
51 auto total_size = serialized::read<size_t>(data, size);
52
53 // No safety checks on the size - trust that in normal operation the
54 // Writer has set sensible limits, don't duplicate here
55 uint8_t* dest = new uint8_t[total_size];
56
57 auto ib =
58 partial_messages.insert({message_id, {m, total_size, 0, dest}});
59
60 it = ib.first;
61 }
62
63 auto& partial = it->second;
64 if (size + partial.received > partial.total_size)
65 {
66 throw ringbuffer::message_error(
67 message_id,
68 fmt::format(
69 "Too much data for oversized fragmented message. Message {} "
70 "asked for {} bytes, has already written {}, but has sent a "
71 "further {}",
72 message_id,
73 partial.total_size,
74 partial.received,
75 size));
76 }
77
78 ::memcpy(partial.data + partial.received, data, size);
79 partial.received += size;
80 data += size;
81 size -= size;
82
83 if (partial.received == partial.total_size)
84 {
85 // Entire message received - dispatch it then free buffer
86 dispatcher.dispatch(partial.m, partial.data, partial.total_size);
87
88 delete[] partial.data;
89
90 // Erase by key - dispatch may have invalidated previous iterator
91 // (nested fragmented messages - odd, but no reason to disallow)
92 partial_messages.erase(message_id);
93 }
94 });
95 }
96
97 ~FragmentReconstructor()
98 {
99 dispatcher.remove_message_handler(OversizedMessage::fragment);
100
101 for (const auto& [_, partial] : partial_messages)
102 {
103 delete[] partial.data;
104 }
105 }
106 };

117 class Writer : public ringbuffer::AbstractWriter
118 {
119 private:
120 ringbuffer::WriterPtr underlying_writer;
121
122 const size_t max_fragment_size;
123 const size_t max_total_size;
124
125 struct FragmentProgress
126 {
127 WriteMarker marker; // Track this so a later call can finish this fragment
128 size_t identifier; // Identifier for all fragments of oversized message
129 size_t remainder; // Remaining space in currently prepared fragment buffer
130 };
131
132 // None iff the message is small enough to fit in a single fragment, or
133 // we're not currently within a [prepare, write_bytes*, finish] loop
134 std::optional<FragmentProgress> fragment_progress;
135
136 public:
137 Writer(const ringbuffer::WriterPtr& writer, size_t f, size_t t = -1) :
138 underlying_writer(writer),
139 max_fragment_size(f),
140 max_total_size(t),
141 fragment_progress({})
142 {
143 if (max_fragment_size >= max_total_size)
144 throw std::logic_error(fmt::format(
145 "Fragment sizes must be smaller than total max: {} >= {}",
146 max_fragment_size,
147 max_total_size));
148
149 constexpr auto header_size = sizeof(InitialFragmentHeader);
150 if (max_fragment_size <= header_size)
151 throw std::logic_error(fmt::format(
152 "Fragment size must be large enough to contain the header for the "
153 "initial fragment, and some additional payload data: {} <= {}",
154 max_fragment_size,
155 header_size));
156 }
157
158 virtual WriteMarker prepare(
159 ringbuffer::Message m,
160 size_t total_size,
161 bool wait = true,
162 size_t* identifier = nullptr) override
163 {
164 // Ensure this is not called out of order
165 if (fragment_progress.has_value())
166 {
167 throw std::logic_error("This Writer is already preparing a message");
168 }
169
170 // Small enough to be handled directly by underlying writer
171 if (total_size <= max_fragment_size)
172 {
173 return underlying_writer->prepare(m, total_size, wait, identifier);
174 }
175
176 if (total_size > max_total_size)
177 {
178 throw std::logic_error(fmt::format(
179 "Requested a write of {} bytes, max allowed is {}",
180 total_size,
181 max_total_size));
182 }
183
184 // Need to split this message into multiple fragments
185
186 if (!wait)
187 {
188 throw std::logic_error(fmt::format(
189 "Requested write of {} bytes will be split into multiple fragments: "
190 "caller must wait for these to complete as fragment writes will be "
191 "blocking",
192 total_size));
193 }
194
195 // Prepare space for the first fragment, getting an id for all related
196 // fragments
197 size_t outer_id;
198 const auto marker = underlying_writer->prepare(
199 OversizedMessage::fragment, max_fragment_size, wait, &outer_id);
200 if (!marker.has_value())
201 {
202 return {};
203 }
204
205 // Write the header
206 InitialFragmentHeader header = {outer_id, m, total_size};
207 auto next = underlying_writer->write_bytes(
208 marker, (const uint8_t*)&header, sizeof(header));
209
210 // Track progress in current oversized message
211 fragment_progress = {
212 marker, outer_id, max_fragment_size - sizeof(header)};
213
214 if (identifier != nullptr)
215 *identifier = outer_id;
216
217 // Don't need to store next - it will be an argument of the next call to
218 // write_bytes
219 return next;
220 }
221
222 virtual void finish(const WriteMarker& marker) override
223 {
224 if (fragment_progress.has_value())
225 {
226 // We were writing an oversized message, the given marker means nothing
227 // to us
228 if (fragment_progress->remainder != 0)
229 {
230 throw std::logic_error(
231 "Attempting to finish an oversized message before the entire "
232 "requested payload has been written");
233 }
234
235 // Finish the final fragment message
236 underlying_writer->finish(fragment_progress->marker);
237
238 // Clean up, ready for next call to prepare
239 fragment_progress = {};
240 }
241 else
242 {
243 // We were writing a small message - get underlying writer to finish it
244 underlying_writer->finish(marker);
245 }
246 }
247
248 virtual WriteMarker write_bytes(
249 const WriteMarker& marker, const uint8_t* bytes, size_t size) override
250 {
251 if (!marker.has_value())
252 {
253 return {};
254 }
255
256 if (!fragment_progress.has_value())
257 {
258 // Writing a small message - nothing to do here
259 return underlying_writer->write_bytes(marker, bytes, size);
260 }
261
262 // Append as much as possible into the current prepared buffer
263 auto write_size = std::min(size, fragment_progress->remainder);
264 auto next = underlying_writer->write_bytes(marker, bytes, write_size);
265 bytes += write_size;
266 size -= write_size;
267 fragment_progress->remainder -= write_size;
268
269 // While there is more to write...
270 while (size > 0)
271 {
272 // Prepare a new fragment
273 const auto id = fragment_progress->identifier;
274 const auto frag_size = std::min(size + sizeof(id), max_fragment_size);
275 next = underlying_writer->prepare(
276 OversizedMessage::fragment, frag_size, true);
277
278 if (!next.has_value())
279 {
280 // Intermediate fragment failed - this is unexpected
281 throw std::logic_error(
282 "Failed to create fragment for oversized message");
283
284 // If this path is hit it is likely because we have allowed oversized
285 // writes to write without waiting. Some initial fragments were
286 // written, but there is insufficient space to write this fragment.
287 // In this case we can either cancel the entire oversized message, or
288 // retry. In either case we should send a message to inform the
289 // reader.
290 fragment_progress->remainder = 0;
291 break;
292 }
293
294 // Finish the previous fragment
295 underlying_writer->finish(fragment_progress->marker);
296
297 // Update progress tracking to reference the new fragment
298 write_size = frag_size - sizeof(id);
299 fragment_progress->marker = next;
300 fragment_progress->remainder = write_size;
301
302 // Write the id of the oversized message
303 next =
304 underlying_writer->write_bytes(next, (const uint8_t*)&id, sizeof(id));
305
306 // Write some fragment payload
307 next = underlying_writer->write_bytes(next, bytes, write_size);
308 bytes += write_size;
309 size -= write_size;
310 fragment_progress->remainder -= write_size;
311 }
312
313 return next;
314 }

318 return max_total_size;
319 }
