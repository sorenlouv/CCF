225 void adjust_ranges(
226 const SeqNoCollection& new_seqnos,
227 bool should_include_receipts,
228 SeqNo earliest_ledger_secret_seqno)
229 {
230 bool any_diff = false;
231
232 // If a seqno is earlier than the earliest known ledger secret, we will
233 // store that it was requested with a nullptr in `my_stores`, but not
234 // add it to `all_stores` to begin fetching until a sufficiently early
235 // secret has been retrieved. To avoid awkwardly sharding requests (and
236 // delaying the secret-fetch with a large request for a later range), we
237 // extend that to say that if _any_ seqno is too early, then _all_
238 // subsequent seqnos will be pending. This bool tracks that behaviour.
239 bool any_too_early = false;
240
241 {
242 auto prev_it = my_stores.begin();
243 auto new_it = new_seqnos.begin();
244 while (new_it != new_seqnos.end())
245 {
246 if (*new_it == prev_it->first)
247 {
248 // Asking for a seqno which was also requested previously - do
249 // nothing and advance to compare next entries
250 ++new_it;
251 ++prev_it;
252 }
253 else if (*new_it > prev_it->first)
254 {
255 // No longer looking for a seqno which was previously requested.
256 // Remove it from my_stores
257 prev_it = my_stores.erase(prev_it);
258 any_diff |= true;
259 }
260 else
261 {
262 // *new_it < prev_it->first
263 // Asking for a seqno which was not previously being fetched =>
264 // check if another request was fetching it, else create new
265 // details to track it
266 if (*new_it < earliest_ledger_secret_seqno || any_too_early)
267 {
268 // If this is too early for known secrets, just record that it
269 // was requested but don't add it to all_stores yet
270 prev_it = my_stores.insert_or_assign(prev_it, *new_it, nullptr);
271 any_too_early = true;
272 }
273 else
274 {
275 auto all_it = all_stores.find(*new_it);
276 auto details =
277 all_it == all_stores.end() ? nullptr : all_it->second.lock();
278 if (details == nullptr)
279 {
280 HISTORICAL_LOG("{} is newly requested", *new_it);
281 details = std::make_shared<StoreDetails>();
282 all_stores.insert_or_assign(all_it, *new_it, details);
283 }
284 prev_it = my_stores.insert_or_assign(prev_it, *new_it, details);
285 }
286 any_diff |= true;
287 }
288 }
289
290 if (prev_it != my_stores.end())
291 {
292 // If we have a suffix of seqnos previously requested, now
293 // unrequested, purge them
294 my_stores.erase(prev_it, my_stores.end());
295 any_diff |= true;
296 }
297 }
298
299 if (!any_diff && (should_include_receipts == include_receipts))
300 {
301 HISTORICAL_LOG("Identical to previous request");
302 return;
303 }
304
305 // If the range has changed, forget what ledger secrets we may have been
306 // fetching - the caller can begin asking for them again
307 ledger_secret_recovery_info = nullptr;
308
309 const auto newly_requested_receipts =
310 should_include_receipts && !include_receipts;
311
312 include_receipts = should_include_receipts;
313
314 HISTORICAL_LOG(
315 "Clearing {} supporting signatures", supporting_signatures.size());
316 supporting_signatures.clear();
317 if (newly_requested_receipts)
318 {
319 // If requesting signatures, populate receipts for each entry that we
320 // already have. Normally this would be done when each entry was
321 // received, but in the case that we have the entries already and only
322 // request signatures now, we delay that work to now.
323
324 for (auto seqno : new_seqnos)
325 {
326 populate_receipts(seqno);
327 }
328 }
329 }
330
331 void populate_receipts(ccf::SeqNo new_seqno)
332 {
333 HISTORICAL_LOG(
334 "Looking at {}, and populating receipts from it", new_seqno);
335 auto new_details = get_store_details(new_seqno);
336 if (new_details != nullptr && new_details->store != nullptr)
337 {
338 if (new_details->is_signature)
339 {
340 HISTORICAL_LOG("{} is a signature", new_seqno);
341
342 fill_receipts_from_signature(new_details);
343 }
344 else
345 {
346 // This isn't a signature. To find the signature for this, we look
347 // through every subsequent transaction, until we find either a gap
348 // (a seqno that hasn't been fetched yet), or a signature. If it is
349 // a signature, and we've found a contiguous range of seqnos to it,
350 // then it must be a signature over this seqno. Else we find a gap
351 // first, and fetch it in case it is the signature. It's possible
352 // that we already have the later signature, and wastefully fill in
353 // the gaps, but this reduces the cases we have to consider so makes
354 // the code much simpler.
355
356 HISTORICAL_LOG("{} is not a signature", new_seqno);
357 supporting_signatures.erase(new_seqno);
358
359 auto next_seqno = new_seqno + 1;
360 while (true)
361 {
362 auto all_it = all_stores.find(next_seqno);
363 auto details =
364 all_it == all_stores.end() ? nullptr : all_it->second.lock();
365 if (details == nullptr)
366 {
367 HISTORICAL_LOG(
368 "Looking for new supporting signature at {}", next_seqno);
369 details = std::make_shared<StoreDetails>();
370 all_stores.insert_or_assign(all_it, next_seqno, details);
371 }
372
373 if (details->store == nullptr)
374 {
375 // Whether we just started fetching or someone else was already
376 // looking for this, it's the first gap we've found so _may_ be
377 // our signature
378 HISTORICAL_LOG(
379 "Assigning {} as potential signature for {}",
380 next_seqno,
381 new_seqno);
382 supporting_signatures[next_seqno] = details;
383 return;
384 }
385 else if (details->is_signature)
386 {
387 const auto filled_this =
388 fill_receipts_from_signature(details, new_seqno);
389
390 if (
391 !filled_this && my_stores.find(new_seqno) != my_stores.end())
392 {
393 throw std::logic_error(fmt::format(
394 "Unexpected: Found a signature at {}, and contiguous range "
395 "of transactions from {}, yet signature does not cover "
396 "this seqno!",
397 next_seqno,
398 new_seqno));
399 }
400
401 return;
402 }
403 else
404 {
405 // This is a normal transaction, and its already fetched.
406 // Nothing to do, consider the next.
407 ++next_seqno;
408 }
409 }
410 }
411 }
412 }

414 private:
415 bool fill_receipts_from_signature(
416 const std::shared_ptr<StoreDetails>& sig_details,
417 std::optional<ccf::SeqNo> should_fill = std::nullopt)
418 {
419 // Iterate through earlier indices. If this signature covers them
420 // then create a receipt for them
421 const auto sig = get_signature(sig_details->store);
422 ccf::MerkleTreeHistory tree(get_tree(sig_details->store).value());
423
424 // This is either pointing at the sig itself, or the closest larger
425 // seqno we're holding
426 auto sig_lower_bound_it =
427 my_stores.lower_bound(sig_details->transaction_id.seqno);
428
429 if (sig_lower_bound_it != my_stores.begin()) // Skip empty map edge case
430 {
431 // Construct reverse iterator to search backwards from here
432 auto search_rit = std::reverse_iterator(sig_lower_bound_it);
433 while (search_rit != my_stores.rend())
434 {
435 auto seqno = search_rit->first;
436 if (tree.in_range(seqno))
437 {
438 auto details = search_rit->second;
439 if (details != nullptr && details->store != nullptr)
440 {
441 auto proof = tree.get_proof(seqno);
442 details->transaction_id = {sig->view, seqno};
443 details->receipt = std::make_shared<TxReceiptImpl>(
444 sig->sig,
445 proof.get_root(),
446 proof.get_path(),
447 sig->node,
448 sig->cert,
449 details->entry_digest,
450 details->get_commit_evidence(),
451 details->claims_digest);
452 HISTORICAL_LOG(
453 "Assigned a sig for {} after given signature at {}",
454 seqno,
455 sig_details->transaction_id.to_str());
456
457 if (should_fill.has_value() && seqno == *should_fill)
458 {
459 should_fill.reset();
460 }
461 }
462
463 ++search_rit;
464 }
465 else
466 {
467 // Found a seqno which this signature doesn't cover. It can't
468 // cover anything else, so break here
469 break;
470 }
471 }
472 }
473
474 return !should_fill.has_value();
475 }

508 std::unique_ptr<LedgerSecretRecoveryInfo> fetch_supporting_secret_if_needed(
509 ccf::SeqNo seqno)
510 {
511 auto [earliest_ledger_secret_seqno, earliest_ledger_secret] =
512 get_earliest_known_ledger_secret();
513 if (seqno < earliest_ledger_secret_seqno)
514 {
515 // Still need more secrets, fetch the next
516 auto previous_secret_stored_version =
517 earliest_ledger_secret->previous_secret_stored_version;
518 if (!previous_secret_stored_version.has_value())
519 {
520 throw std::logic_error(fmt::format(
521 "Earliest known ledger secret at {} has no earlier secret stored "
522 "version ({})",
523 earliest_ledger_secret_seqno,
524 seqno));
525 }
526
527 const auto seqno_to_fetch = previous_secret_stored_version.value();
528 LOG_TRACE_FMT(
529 "Requesting historical entry at {} but first known ledger "
530 "secret is applicable from {}",
531 seqno,
532 earliest_ledger_secret_seqno);
533
534 auto it = all_stores.find(seqno_to_fetch);
535 auto details = it == all_stores.end() ? nullptr : it->second.lock();
536 if (details == nullptr)
537 {
538 LOG_TRACE_FMT("Requesting older secret at {} now", seqno_to_fetch);
539 details = std::make_shared<StoreDetails>();
540 all_stores.insert_or_assign(it, seqno_to_fetch, details);
541 fetch_entry_at(seqno_to_fetch);
542 }
543
544 return std::make_unique<LedgerSecretRecoveryInfo>(
545 seqno_to_fetch, earliest_ledger_secret, details);
546 }
547
548 return nullptr;
549 }

551 void process_deserialised_store(
552 const StoreDetailsPtr& details,
553 const kv::StorePtr& store,
554 const crypto::Sha256Hash& entry_digest,
555 ccf::SeqNo seqno,
556 bool is_signature,
557 ccf::ClaimsDigest&& claims_digest,
558 bool has_commit_evidence)
559 {
560 // Deserialisation includes a GCM integrity check, so all entries
561 // have been verified by the time we get here.
562 details->current_stage = RequestStage::Trusted;
563 details->has_commit_evidence = has_commit_evidence;
564
565 details->entry_digest = entry_digest;
566 if (!claims_digest.empty())
567 details->claims_digest = std::move(claims_digest);
568
569 CCF_ASSERT_FMT(
570 details->store == nullptr,
571 "Cache already has store for seqno {}",
572 seqno);
573 details->store = store;
574
575 details->is_signature = is_signature;
576 if (is_signature)
577 {
578 // Construct a signature receipt.
579 // We do this whether it was requested or not, because we have all
580 // the state to do so already, and it's simpler than constructing
581 // the receipt _later_ for an already-fetched signature
582 // transaction.
583 const auto sig = get_signature(details->store);
584 assert(sig.has_value());
585 details->transaction_id = {sig->view, sig->seqno};
586 details->receipt = std::make_shared<TxReceiptImpl>(
587 sig->sig, sig->root.h, nullptr, sig->node, sig->cert);
588 }
589
590 auto request_it = requests.begin();
591 while (request_it != requests.end())
592 {
593 auto& [handle, request] = *request_it;
594
595 // If this request was still waiting for a ledger secret, and this is
596 // that secret
597 if (
598 request.ledger_secret_recovery_info != nullptr &&
599 request.ledger_secret_recovery_info->target_seqno == seqno)
600 {
601 // Handle it, hopefully extending earliest_known_ledger_secret to
602 // cover earlier entries
603 const auto valid_secret = handle_encrypted_past_ledger_secret(
604 store, std::move(request.ledger_secret_recovery_info));
605 if (!valid_secret)
606 {
607 // Invalid! Erase this request: host gave us junk, need to start
608 // over
609 request_it = requests.erase(request_it);
610 continue;
611 }
612
613 auto new_secret_fetch =
614 fetch_supporting_secret_if_needed(request.first_requested_seqno());
615 if (new_secret_fetch != nullptr)
616 {
617 request.ledger_secret_recovery_info = std::move(new_secret_fetch);
618 }
619 else
620 {
621 // Newly have all required secrets - begin fetching the actual
622 // entries. Note this is adding them to `all_stores`, from where
623 // they'll be requested on the next tick.
624 auto my_stores_it = request.my_stores.begin();
625 while (my_stores_it != request.my_stores.end())
626 {
627 auto [seqno, _] = *my_stores_it;
628 auto it = all_stores.find(seqno);
629 auto details =
630 it == all_stores.end() ? nullptr : it->second.lock();
631
632 if (details == nullptr)
633 {
634 details = std::make_shared<StoreDetails>();
635 all_stores.insert_or_assign(it, seqno, details);
636 }
637
638 my_stores_it->second = details;
639 ++my_stores_it;
640 }
641 }
642
643 // In either case, done with this request, try the next
644 ++request_it;
645 continue;
646 }
647
648 if (request.include_receipts)
649 {
650 const bool seqno_in_this_request =
651 (request.my_stores.find(seqno) != request.my_stores.end() ||
652 request.supporting_signatures.find(seqno) !=
653 request.supporting_signatures.end());
654 if (seqno_in_this_request)
655 {
656 request.populate_receipts(seqno);
657 }
658 }
659
660 ++request_it;
661 }
662 }

664 bool handle_encrypted_past_ledger_secret(
665 const kv::StorePtr& store,
666 std::unique_ptr<LedgerSecretRecoveryInfo> ledger_secret_recovery_info)
667 {
668 // Read encrypted secrets from store
669 auto tx = store->create_read_only_tx();
670 auto encrypted_past_ledger_secret =
671 tx.ro<ccf::EncryptedLedgerSecretsInfo>(
672 ccf::Tables::ENCRYPTED_PAST_LEDGER_SECRET);
673 if (!encrypted_past_ledger_secret)
674 {
675 return false;
676 }
677
678 // Construct description and decrypted secret
679 auto previous_ledger_secret =
680 encrypted_past_ledger_secret->get()->previous_ledger_secret;
681
682 auto recovered_ledger_secret = std::make_shared<LedgerSecret>(
683 ccf::decrypt_previous_ledger_secret_raw(
684 ledger_secret_recovery_info->last_ledger_secret,
685 std::move(previous_ledger_secret->encrypted_data)),
686 previous_ledger_secret->previous_secret_stored_version);
687
688 // Add recovered secret to historical secrets
689 historical_ledger_secrets->set_secret(
690 previous_ledger_secret->version, std::move(recovered_ledger_secret));
691
692 return true;
693 }

710 std::vector<StatePtr> get_states_internal(
711 const CompoundHandle& handle,
712 const SeqNoCollection& seqnos,
713 ExpiryDuration seconds_until_expiry,
714 bool include_receipts)
715 {
716 if (seqnos.empty())
717 {
718 throw std::logic_error(
719 "Invalid range for historical query: Cannot request empty range");
720 }
721
722 std::lock_guard<ccf::pal::Mutex> guard(requests_lock);
723
724 const auto ms_until_expiry =
725 std::chrono::duration_cast<std::chrono::milliseconds>(
726 seconds_until_expiry);
727
728 auto it = requests.find(handle);
729 if (it == requests.end())
730 {
731 // This is a new handle - insert a newly created Request for it
732 it = requests.emplace_hint(it, handle, Request(all_stores));
733 HISTORICAL_LOG("First time I've seen handle {}", handle);
734 }
735
736 Request& request = it->second;
737
738 auto [earliest_ledger_secret_seqno, _] =
739 get_earliest_known_ledger_secret();
740
741 // Update this Request to represent the currently requested ranges
742 HISTORICAL_LOG(
743 "Adjusting handle {} to cover {} seqnos starting at {} "
744 "(include_receipts={})",
745 handle,
746 seqnos.size(),
747 *seqnos.begin(),
748 include_receipts);
749 request.adjust_ranges(
750 seqnos, include_receipts, earliest_ledger_secret_seqno);
751
752 // If the earliest target entry cannot be deserialised with the earliest
753 // known ledger secret, record the target seqno and begin fetching the
754 // previous historical ledger secret.
755 auto secret_fetch =
756 fetch_supporting_secret_if_needed(request.first_requested_seqno());
757 if (secret_fetch != nullptr)
758 {
759 if (
760 request.ledger_secret_recovery_info == nullptr ||
761 request.ledger_secret_recovery_info->target_seqno !=
762 secret_fetch->target_seqno)

764 request.ledger_secret_recovery_info = std::move(secret_fetch);
765 }
766 }

769 request.time_to_expiry = ms_until_expiry;

772

774 {
775 auto target_details = request.get_store_details(seqno);
776
777 if (
778 target_details != nullptr &&
779 target_details->current_stage == RequestStage::Trusted &&
780 (!request.include_receipts || target_details->receipt != nullptr))
781 {
782 // Have this store, associated txid and receipt and trust it - add
783 // it to return list
784 StatePtr state = std::make_shared<State>(
785 target_details->store,
786 target_details->receipt,
787 target_details->transaction_id);
788 trusted_states.push_back(state);
789 }
790 else
791 {
792 // Still fetching this store or don't trust it yet, so range is
793 // incomplete - return empty vector
794 return {};
795 }
796 }

798 return trusted_states;

977 bool handle_ledger_entry(ccf::SeqNo seqno, const uint8_t* data, size_t size)
978 {
979 std::lock_guard<ccf::pal::Mutex> guard(requests_lock);
980 const auto it = all_stores.find(seqno);
981 auto details = it == all_stores.end() ? nullptr : it->second.lock();
982 if (
983 details == nullptr || details->current_stage != RequestStage::Fetching)
984 {
985 // Unexpected entry, we already have it or weren't asking for it -
986 // ignore this resubmission
987 return false;
988 }
989
990 kv::ApplyResult deserialise_result;
991 ccf::ClaimsDigest claims_digest;
992 bool has_commit_evidence;
993 auto store = deserialise_ledger_entry(
994 seqno,
995 data,
996 size,
997 deserialise_result,
998 claims_digest,
999 has_commit_evidence);
1000
1001 if (deserialise_result == kv::ApplyResult::FAIL)
1002 {
1003 return false;
1004 }
1005
1006 {
1007 // Confirm this entry is from a precursor of the current state, and not
1008 // a fork
1009 const auto tx_id = store->current_txid();
1010 if (tx_id.version != seqno)
1011 {
1012 LOG_FAIL_FMT(
1013 "Corrupt ledger entry received - claims to be {} but is actually "
1014 "{}.{}",
1015 seqno,
1016 tx_id.term,
1017 tx_id.version);
1018 return false;
1019 }
1020
1021 auto consensus = source_store.get_consensus();
1022 if (consensus == nullptr)
1023 {
1024 LOG_FAIL_FMT("No consensus on source store");
1025 return false;
1026 }
1027
1028 const auto actual_view = consensus->get_view(seqno);
1029 if (actual_view != tx_id.term)
1030 {
1031 LOG_FAIL_FMT(
1032 "Ledger entry comes from fork - contains {}.{} but this service "
1033 "expected {}.{}",
1034 tx_id.term,
1035 tx_id.version,
1036 actual_view,
1037 seqno);
1038 return false;
1039 }
1040 }
1041
1042 const auto is_signature =
1043 deserialise_result == kv::ApplyResult::PASS_SIGNATURE;
1044
1045 HISTORICAL_LOG(
1046 "Processing historical store at {} ({})",
1047 seqno,
1048 (size_t)deserialise_result);
1049 const auto entry_digest = crypto::Sha256Hash({data, size});
1050 process_deserialised_store(
1051 details,
1052 store,
1053 entry_digest,
1054 seqno,
1055 is_signature,
1056 std::move(claims_digest),
1057 has_commit_evidence);
1058
1059 return true;
1060 }

1130 kv::StorePtr deserialise_ledger_entry(
1131 ccf::SeqNo seqno,
1132 const uint8_t* data,
1133 size_t size,
1134 kv::ApplyResult& result,
1135 ccf::ClaimsDigest& claims_digest,
1136 bool& has_commit_evidence)
1137 {
1138 // Create a new store and try to deserialise this entry into it
1139 kv::StorePtr store = std::make_shared<kv::Store>(
1140 false /* Do not start from very first seqno */,
1141 true /* Make use of historical secrets */);
1142
1143 // If this is older than the node's currently known ledger secrets, use
1144 // the historical encryptor (which should have older secrets)
1145 if (seqno < source_ledger_secrets->get_first().first)
1146 {
1147 store->set_encryptor(historical_encryptor);
1148 }
1149 else
1150 {
1151 store->set_encryptor(source_store.get_encryptor());
1152 }
1153
1154 try
1155 {
1156 // Encrypted ledger secrets are deserialised in public-only mode. Their
1157 // Merkle tree integrity is not verified: even if the recovered ledger
1158 // secret was bogus, the deserialisation of subsequent ledger entries
1159 // would fail.
1160 bool public_only = false;
1161 for (const auto& [_, request] : requests)
1162 {
1163 if (
1164 request.ledger_secret_recovery_info != nullptr &&
1165 request.ledger_secret_recovery_info->target_seqno == seqno)
1166 {
1167 public_only = true;
1168 break;
1169 }
1170 }
1171
1172 auto exec = store->deserialize(
1173 {data, data + size}, ConsensusType::CFT, public_only);
1174 if (exec == nullptr)
1175 {
1176 result = kv::ApplyResult::FAIL;
1177 return nullptr;
1178 }
1179
1180 result = exec->apply();
1181 claims_digest = std::move(exec->consume_claims_digest());
1182
1183 auto commit_evidence_digest =
1184 std::move(exec->consume_commit_evidence_digest());
1185 has_commit_evidence = commit_evidence_digest.has_value();
1186 }
1187 catch (const std::exception& e)
1188 {
1189 LOG_FAIL_FMT(
1190 "Exception while attempting to deserialise entry {}: {}",
1191 seqno,
1192 e.what());
1193 result = kv::ApplyResult::FAIL;
1194 }
1195
1196 return store;
1197 }

1199 void tick(const std::chrono::milliseconds& elapsed_ms)
1200 {
1201 std::lock_guard<ccf::pal::Mutex> guard(requests_lock);
1202
1203 {
1204 auto it = requests.begin();
1205 while (it != requests.end())
1206 {
1207 auto& request = it->second;
1208 if (elapsed_ms >= request.time_to_expiry)
1209 {
1210 LOG_DEBUG_FMT(
1211 "Dropping expired historical query with handle {}", it->first);
1212 it = requests.erase(it);
1213 }
1214 else
1215 {
1216 request.time_to_expiry -= elapsed_ms;
1217 ++it;
1218 }
1219 }
1220 }
1221
1222 {
1223 auto it = all_stores.begin();
1224 std::optional<std::pair<ccf::SeqNo, ccf::SeqNo>> range_to_request =
1225 std::nullopt;
1226 while (it != all_stores.end())
1227 {
1228 auto details = it->second.lock();
1229 if (details == nullptr)
1230 {
1231 it = all_stores.erase(it);
1232 }
1233 else
1234 {
1235 if (details->current_stage == RequestStage::Fetching)
1236 {
1237 details->time_until_fetch -= elapsed_ms;
1238 if (details->time_until_fetch.count() <= 0)
1239 {
1240 details->time_until_fetch = slow_fetch_threshold;
1241
1242 const auto seqno = it->first;
1243 if (
1244 range_to_request.has_value() &&
1245 range_to_request->second + 1 == seqno)
1246 {
1247 range_to_request->second = seqno;
1248 }
1249 else
1250 {
1251 if (range_to_request.has_value())
1252 {
1253 // Submit fetch for previously tracked range
1254 fetch_entries_range(
1255 range_to_request->first, range_to_request->second);
1256 }
1257
1258 // Track new range
1259 range_to_request = std::make_pair(seqno, seqno);
1260 }
1261 }
1262 }
1263
1264 ++it;
1265 }
1266 }
1267
1268 if (range_to_request.has_value())
1269 {
1270 // Submit fetch for final tracked range
1271 fetch_entries_range(
1272 range_to_request->first, range_to_request->second);
1273 }
1274 }
1275 }
