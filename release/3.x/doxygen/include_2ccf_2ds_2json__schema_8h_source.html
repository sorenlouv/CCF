193 template <typename T>
194 inline void fill_schema(nlohmann::json& schema)
195 {
196 if constexpr (nonstd::is_specialization<T, std::optional>::value)
197 {
198 fill_schema<typename T::value_type>(schema);
199 }
200 else if constexpr (nonstd::is_specialization<T, std::vector>::value)
201 {
202 if constexpr (std::is_same<T, std::vector<uint8_t>>::value)
203 {
204 // Byte vectors are always base64 encoded
205 schema["type"] = "string";
206 schema["format"] = "base64";
207 }
208 else
209 {
210 schema["type"] = "array";
211 schema["items"] = schema_element<typename T::value_type>();
212 }
213 }
214 else if constexpr (
215 nonstd::is_specialization<T, std::map>::value ||
216 nonstd::is_specialization<T, std::unordered_map>::value)
217 {
218 // Nlohmann JSON serialises some maps as objects, if the keys can be
219 // converted to strings. This should detect those cases. The others are
220 // serialised as list-of-pairs
221 if constexpr (nlohmann::detail::
222 is_compatible_object_type<nlohmann::json, T>::value)
223 {
224 schema["type"] = "object";
225 schema["additionalProperties"] =
226 schema_element<typename T::mapped_type>();
227 }
228 else
229 {
230 schema["type"] = "array";
231 auto items = nlohmann::json::object();
232 {
233 items["type"] = "array";
234
235 auto sub_items = nlohmann::json::array();
236 sub_items.push_back(schema_element<typename T::key_type>());
237 sub_items.push_back(schema_element<typename T::mapped_type>());
238 items["items"] = sub_items;
239 }
240 schema["items"] = items;
241 }
242 }
243 else if constexpr (nonstd::is_specialization<T, std::pair>::value)
244 {
245 schema["type"] = "array";
246 auto items = nlohmann::json::array();
247 items.push_back(schema_element<typename T::first_type>());
248 items.push_back(schema_element<typename T::second_type>());
249 schema["items"] = items;
250 }
251 else if constexpr (std::is_same<T, std::string>::value)
252 {
253 schema["type"] = "string";
254 }
255 else if constexpr (std::is_same<T, bool>::value)
256 {
257 schema["type"] = "boolean";
258 }
259 else if constexpr (std::is_same<T, nlohmann::json>::value)
260 {
261 // Any field that contains more json is completely unconstrained, so we
262 // do not add a type or any other fields
263 schema = nlohmann::json::object();
264 }
265 else if constexpr (std::is_integral<T>::value)
266 {
267 fill_number_schema<T>(schema);
268 }
269 else if constexpr (std::is_floating_point<T>::value)
270 {
271 schema["type"] = "number";
272 }
273 else if constexpr (std::is_same<T, JsonSchema>::value)
274 {
275 schema["type"] = "object";
276 }
277 else
278 {
279 adl::fill_schema<T>(schema);
280 }
281 }

283 template <typename T>
284 inline nlohmann::json build_schema(const std::string& title = "")
285 {
286 auto schema = nlohmann::json::object();
287
288 if (!title.empty())
289 {
290 schema["title"] = title;
291 }
292
293 fill_schema<T>(schema);

295 return schema;
