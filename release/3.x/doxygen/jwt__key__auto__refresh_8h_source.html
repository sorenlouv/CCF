54 void start()
55 {
56 auto refresh_msg = std::make_unique<threading::Tmsg<RefreshTimeMsg>>(
57 [](std::unique_ptr<threading::Tmsg<RefreshTimeMsg>> msg) {
58 if (!msg->data.self.consensus->can_replicate())
59 {
60 LOG_DEBUG_FMT(
61 "JWT key auto-refresh: Node is not primary, skipping");
62 }
63 else
64 {
65 msg->data.self.refresh_jwt_keys();
66 }
67 LOG_DEBUG_FMT(
68 "JWT key auto-refresh: Scheduling in {}s",
69 msg->data.self.refresh_interval_s);
70 auto delay = std::chrono::seconds(msg->data.self.refresh_interval_s);
71 threading::ThreadMessaging::thread_messaging.add_task_after(
72 std::move(msg), delay);
73 },
74 *this);
75
76 LOG_DEBUG_FMT(
77 "JWT key auto-refresh: Scheduling in {}s", refresh_interval_s);
78 auto delay = std::chrono::seconds(refresh_interval_s);
79 threading::ThreadMessaging::thread_messaging.add_task_after(
80 std::move(refresh_msg), delay);
81 }

105 template <typename T>
106 void send_refresh_jwt_keys(T msg)
107 {
108 auto body = serdes::pack(msg, serdes::Pack::Text);
109
110 http::Request request(fmt::format(
111 "/{}/{}",
112 ccf::get_actor_prefix(ccf::ActorsType::nodes),
113 "jwt_keys/refresh"));
114 request.set_header(
115 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
116 request.set_body(&body);
117
118 auto packed = request.build_request();
119
120 auto node_session = std::make_shared<ccf::SessionContext>(
121 ccf::InvalidSessionId, node_cert.raw());
122 auto ctx = ccf::make_rpc_context(node_session, packed);
123
124 const auto actor_opt = http::extract_actor(*ctx);
125 if (!actor_opt.has_value())
126 {
127 throw std::logic_error("Unable to get actor");
128 }
129
130 const auto actor = rpc_map->resolve(actor_opt.value());
131 auto frontend_opt = this->rpc_map->find(actor);
132 if (!frontend_opt.has_value())
133 {
134 throw std::logic_error(
135 "RpcMap::find returned invalid (empty) frontend");
136 }
137 auto frontend = frontend_opt.value();
138 frontend->process(ctx);
139 }

149 void handle_jwt_jwks_response(
150 const std::string& issuer,
151 http_status status,
152 std::vector<uint8_t>&& data)
153 {
154 if (status != HTTP_STATUS_OK)
155 {
156 LOG_FAIL_FMT(
157 "JWT key auto-refresh: Error while requesting JWKS: {} {}{}",
158 status,
159 http_status_str(status),
160 data.empty() ?
161 "" :
162 fmt::format(" '{}'", std::string(data.begin(), data.end())));
163 send_refresh_jwt_keys_error();
164 return;
165 }
166
167 LOG_DEBUG_FMT(
168 "JWT key auto-refresh: Received JWKS for issuer '{}'", issuer);
169
170 JsonWebKeySet jwks;
171 try
172 {
173 jwks = nlohmann::json::parse(data).get<JsonWebKeySet>();
174 }
175 catch (const std::exception& e)
176 {
177 LOG_FAIL_FMT(
178 "JWT key auto-refresh: Cannot parse JWKS for issuer '{}': {}",
179 issuer,
180 e.what());
181 send_refresh_jwt_keys_error();
182 return;
183 }
184
185 // call internal endpoint to update keys
186 auto msg = SetJwtPublicSigningKeys{issuer, jwks};
187 send_refresh_jwt_keys(msg);
188 }

190 void handle_jwt_metadata_response(
191 const std::string& issuer,
192 std::shared_ptr<tls::CA> ca,
193 http_status status,
194 std::vector<uint8_t>&& data)
195 {
196 if (status != HTTP_STATUS_OK)
197 {
198 LOG_FAIL_FMT(
199 "JWT key auto-refresh: Error while requesting OpenID metadata: {} "
200 "{}{}",
201 status,
202 http_status_str(status),
203 data.empty() ?
204 "" :
205 fmt::format(" '{}'", std::string(data.begin(), data.end())));
206 send_refresh_jwt_keys_error();
207 return;
208 }
209
210 LOG_DEBUG_FMT(
211 "JWT key auto-refresh: Received OpenID metadata for issuer '{}'",
212 issuer);
213
214 std::string jwks_url_str;
215 try
216 {
217 auto metadata = nlohmann::json::parse(data);
218 jwks_url_str = metadata.at("jwks_uri").get<std::string>();
219 }
220 catch (const std::exception& e)
221 {
222 LOG_FAIL_FMT(
223 "JWT key auto-refresh: Cannot parse OpenID metadata for issuer '{}': "
224 "{}",
225 issuer,
226 e.what());
227 send_refresh_jwt_keys_error();
228 return;
229 }
230 http::URL jwks_url;
231 try
232 {
233 jwks_url = http::parse_url_full(jwks_url_str);
234 }
235 catch (const std::invalid_argument& e)
236 {
237 LOG_FAIL_FMT(
238 "JWT key auto-refresh: Cannot parse jwks_uri for issuer '{}': {}",
239 issuer,
240 jwks_url_str);
241 send_refresh_jwt_keys_error();
242 return;
243 }
244 auto jwks_url_port = !jwks_url.port.empty() ? jwks_url.port : "443";
245
246 auto ca_cert = std::make_shared<tls::Cert>(
247 ca, std::nullopt, std::nullopt, jwks_url.host);
248
249 LOG_DEBUG_FMT(
250 "JWT key auto-refresh: Requesting JWKS at https://{}:{}{}",
251 jwks_url.host,
252 jwks_url_port,
253 jwks_url.path);
254 auto http_client = rpcsessions->create_client(ca_cert);
255 // Note: Connection errors are not signalled and hence not tracked in
256 // endpoint metrics currently.
257 http_client->connect(
258 std::string(jwks_url.host),
259 std::string(jwks_url_port),
260 [this, issuer](
261 http_status status, http::HeaderMap&&, std::vector<uint8_t>&& data) {
262 handle_jwt_jwks_response(issuer, status, std::move(data));
263 return true;
264 });
265 http::Request r(jwks_url.path, HTTP_GET);
266 r.set_header(http::headers::HOST, std::string(jwks_url.host));
267 http_client->send_request(std::move(r));
268 }

270 void refresh_jwt_keys()
271 {
272 auto tx = network.tables->create_read_only_tx();
273 auto jwt_issuers = tx.ro(network.jwt_issuers);
274 auto ca_cert_bundles = tx.ro(network.ca_cert_bundles);
275 jwt_issuers->foreach([this, &ca_cert_bundles](
276 const JwtIssuer& issuer,
277 const JwtIssuerMetadata& metadata) {
278 if (!metadata.auto_refresh)
279 {
280 LOG_DEBUG_FMT(
281 "JWT key auto-refresh: Skipping issuer '{}', auto-refresh is "
282 "disabled",
283 issuer);
284 return true;
285 }
286
287 // Increment attempts, only when auto-refresh is enabled.
288 attempts++;
289
290 LOG_DEBUG_FMT(
291 "JWT key auto-refresh: Refreshing keys for issuer '{}'", issuer);
292 auto& ca_cert_bundle_name = metadata.ca_cert_bundle_name.value();
293 auto ca_cert_bundle_pem = ca_cert_bundles->get(ca_cert_bundle_name);
294 if (!ca_cert_bundle_pem.has_value())
295 {
296 LOG_FAIL_FMT(
297 "JWT key auto-refresh: CA cert bundle with name '{}' for issuer "
298 "'{}' not "
299 "found",
300 ca_cert_bundle_name,
301 issuer);
302 send_refresh_jwt_keys_error();
303 return true;
304 }
305
306 auto metadata_url_str = issuer + "/.well-known/openid-configuration";
307 auto metadata_url = http::parse_url_full(metadata_url_str);
308 auto metadata_url_port =
309 !metadata_url.port.empty() ? metadata_url.port : "443";
310
311 auto ca = std::make_shared<tls::CA>(ca_cert_bundle_pem.value());
312 auto ca_cert = std::make_shared<tls::Cert>(
313 ca, std::nullopt, std::nullopt, metadata_url.host);
314
315 LOG_DEBUG_FMT(
316 "JWT key auto-refresh: Requesting OpenID metadata at https://{}:{}{}",
317 metadata_url.host,
318 metadata_url_port,
319 metadata_url.path);
320 auto http_client = rpcsessions->create_client(ca_cert);
321 // Note: Connection errors are not signalled and hence not tracked in
322 // endpoint metrics currently.
323 http_client->connect(
324 std::string(metadata_url.host),
325 std::string(metadata_url_port),
326 [this, issuer, ca](
327 http_status status,
328 http::HeaderMap&&,
329 std::vector<uint8_t>&& data) {
330 handle_jwt_metadata_response(issuer, ca, status, std::move(data));
331 return true;
332 });
333 http::Request r(metadata_url.path, HTTP_GET);
334 r.set_header(http::headers::HOST, std::string(metadata_url.host));
335 http_client->send_request(std::move(r));
336 return true;
337 });
338 }

340 // Returns a copy of the current attempts
341 size_t get_attempts() const
342 {
343 return attempts.load();
344 }
