31 inline std::string error_string(int ec)
32 {
33 // ERR_error_string doesn't really expect the code could actually be zero
34 // and uses the `static char buf[256]` which is NOT cleaned nor checked
35 // if it has changed. So we use ERR_error_string_n directly.
36 if (ec)
37 {
38 std::string err(256, '\0');
39 ERR_error_string_n((unsigned long)ec, err.data(), err.size());
40 // Remove any trailing NULs before returning
41 err.resize(std::strlen(err.c_str()));
42 return err;
43 }
44 else
45 {
46 return "unknown error";
47 }
48 }

51 inline void CHECK1(int rc)
52 {
53 unsigned long ec = ERR_get_error();
54 if (rc != 1 && ec != 0)
55 {
56 throw std::runtime_error(
57 fmt::format("OpenSSL error: {}", error_string(ec)));
58 }
59 }

62 inline void CHECK0(int rc)
63 {
64 unsigned long ec = ERR_get_error();
65 if (rc == 0 && ec != 0)
66 {
67 throw std::runtime_error(
68 fmt::format("OpenSSL error: {}", error_string(ec)));
69 }
70 }

73 inline void CHECKNULL(void* ptr)
74 {
75 if (ptr == NULL)

77 throw std::runtime_error("OpenSSL error: missing object");

101 /*
102 * Unique pointer wrappers for SSL objects, with SSL' specific constructors
103 * and destructors. Some objects need special functionality, others are just
104 * wrappers around the same template interface Unique_SSL_OBJECT.
105 */
106
111 template <class T, T* (*CTOR)(), void (*DTOR)(T*)>
112 class Unique_SSL_OBJECT
113 {
114 protected:
116 std::unique_ptr<T, void (*)(T*)> p;
117
118 public:
120 Unique_SSL_OBJECT() : p(CTOR(), DTOR)
121 {
122 CHECKNULL(p.get());
123 }
125 Unique_SSL_OBJECT(T* ptr, void (*dtor)(T*), bool check_null = true) :
126 p(ptr, dtor)
127 {
128 if (check_null)
129 CHECKNULL(p.get());
130 }
132 operator T*()
133 {
134 return p.get();
135 }
137 operator T*() const
138 {
139 return p.get();
140 }
142 void reset(T* other)
143 {
144 p.reset(other);
145 }
147 T* release()
148 {
149 return p.release();
150 }
151 };

153 struct Unique_BIO : public Unique_SSL_OBJECT<BIO, nullptr, nullptr>
154 {
155 Unique_BIO() :
156 Unique_SSL_OBJECT(BIO_new(BIO_s_mem()), [](auto x) { BIO_free(x); })
157 {}
158 Unique_BIO(const void* buf, int len) :
159 Unique_SSL_OBJECT(
160 BIO_new_mem_buf(buf, len), [](auto x) { BIO_free(x); })
161 {}
162 Unique_BIO(const std::vector<uint8_t>& d) :
163 Unique_SSL_OBJECT(
164 BIO_new_mem_buf(d.data(), d.size()), [](auto x) { BIO_free(x); })
165 {}
166 Unique_BIO(const Pem& pem) :
167 Unique_SSL_OBJECT(
168 BIO_new_mem_buf(pem.data(), -1), [](auto x) { BIO_free(x); })
169 {}
170 Unique_BIO(SSL_CTX* ctx) :
171 Unique_SSL_OBJECT(
172 BIO_new_ssl_connect(ctx), [](auto x) { BIO_free_all(x); })
173 {}
174 };
