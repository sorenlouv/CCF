466 void add_configuration(
467 Index idx,
468 const kv::Configuration::Nodes& conf,
469 const std::unordered_set<ccf::NodeId>& new_learner_nodes = {},
470 const std::unordered_set<ccf::NodeId>& new_retired_nodes = {}) override
471 {
472 LOG_DEBUG_FMT(
473 "Configurations: add new configuration at {}: {{{}}}", idx, conf);
474
475 if (reconfiguration_type == ReconfigurationType::ONE_TRANSACTION)
476 {
477 assert(new_learner_nodes.empty());
478
479 // Detect when we are retired by observing a configuration
480 // from which we are absent following a configuration in which
481 // we were included. Note that this relies on retirement being
482 // a final state, and node identities never being re-used.
483 if (
484 !configurations.empty() &&
485 configurations.back().nodes.find(state->my_node_id) !=
486 configurations.back().nodes.end() &&
487 conf.find(state->my_node_id) == conf.end())
488 {
489 become_retired(idx, kv::RetirementPhase::Ordered);
490 }
491 }
492 else
493 {
494 if (
495 !configurations.empty() &&
496 configurations.back().nodes.find(state->my_node_id) ==
497 configurations.back().nodes.end() &&
498 conf.find(state->my_node_id) == conf.end() &&
499 (is_retiring() ||
500 (is_retired() &&
501 retirement_phase == kv::RetirementPhase::Committed)))
502 {
503 become_retired(idx, kv::RetirementPhase::Ordered);
504 }
505
506 if (!new_learner_nodes.empty())
507 {
508 LOG_DEBUG_FMT(
509 "Configurations: new learners: {{{}}}",
510 fmt::join(new_learner_nodes, ", "));
511 for (auto& id : new_learner_nodes)
512 {
513 if (learner_nodes.find(id) == learner_nodes.end())
514 {
515 learner_nodes[id] = idx;
516 }
517 }
518 }
519
520 if (!new_retired_nodes.empty())
521 {
522 LOG_DEBUG_FMT(
523 "Configurations: newly retired nodes: {{{}}}",
524 fmt::join(new_retired_nodes, ", "));
525 for (auto& id : new_retired_nodes)
526 {
527 if (retired_nodes.find(id) == retired_nodes.end())
528 {
529 retired_nodes[id] = idx;
530 }
531 }
532 }
533
534 if (!configurations.empty())
535 {
536 for (const auto& [nid, _] : conf)
537 {
538 if (
539 all_other_nodes.find(nid) != all_other_nodes.end() &&
540 learner_nodes.find(nid) != learner_nodes.end() &&
541 new_learner_nodes.find(nid) == new_learner_nodes.end())
542 {
543 // Promotion of known learner
544 learner_nodes.erase(nid);
545 }
546 if (is_learner() && nid == state->my_node_id)
547 {
548 LOG_DEBUG_FMT(
549 "Configurations: observing own promotion, becoming an active "
550 "follower");
551 leadership_state = kv::LeadershipState::Follower;
552 membership_state = kv::MembershipState::Active;
553 }
554 }
555 }
556
557 if (orc_sets.find(idx) == orc_sets.end())
558 {
559 orc_sets[idx] = {};
560 }
561 }
562
563 if (conf != configurations.back().nodes)
564 {
565 uint32_t offset = get_bft_offset(conf);
566 Configuration new_config = {idx, std::move(conf), offset, idx};
567 configurations.push_back(new_config);
568
569 if (
570 reconfiguration_type == ReconfigurationType::TWO_TRANSACTION &&
571 resharing_tracker)
572 {
573 resharing_tracker->add_network_configuration(new_config);
574 if (is_primary())
575 {
576 resharing_tracker->reshare(new_config);
577 }
578 }
579
580 create_and_remove_node_state();
581 }

623 // For more info about Observed Reconfiguration Commits see
624 // https://microsoft.github.io/CCF/main/architecture/consensus/2tx-reconfig.html
625 //
626 // Note that this call is not `const` and that it modifies `orc_sets`. This
627 // is safe, despite the fact that the primary may change or a
628 // reconfiguration may be (partially) rolled back, because the `orc_sets`
629 // are cleared upon entering/exiting the leader/follower replica states.
630 // This means that we never record spurious ORCs, while it is still
631 // guaranteed that we will eventually receive all of them, since all
632 // nodes keep re-submitting ORCs until they are able to switch to the next
633 // pending configuration.
634
635 std::optional<kv::Configuration::Nodes> orc(
636 kv::ReconfigurationId rid, const ccf::NodeId& node_id) override
637 {
638 std::lock_guard<ccf::pal::Mutex> guard(state->lock);
639
640 LOG_DEBUG_FMT(
641 "Configurations: ORC for configuration #{} from {}", rid, node_id);
642
643 const auto oit = orc_sets.find(rid);
644 if (oit == orc_sets.end())
645 {
646 throw std::logic_error(
647 fmt::format("Missing ORC set for configuration #{}", rid));
648 }
649
650 for (auto const& conf : configurations)
651 {
652 if (conf.rid == rid)
653 {
654 const auto& ncnodes = conf.nodes;
655 if (ncnodes.find(node_id) == ncnodes.end())
656 {
657 LOG_DEBUG_FMT("Node not in the configuration {}: {}", rid, node_id);
658 return std::nullopt;
659 }
660 else
661 {
662 oit->second.insert(node_id);
663 LOG_DEBUG_FMT(
664 "Configurations: have {} ORCs out of {} for configuration #{}",
665 oit->second.size(),
666 ncnodes.size(),
667 rid);
668
669 // Note: Learners in the next configuration become trusted when
670 // there is quorum in the next configuration, i.e. they may become
671 // trusted in the nodes table before they are fully caught up and
672 // have submitted their own ORC.
673
674 if (oit->second.size() >= get_quorum(ncnodes.size()))
675 {
676 return ncnodes;
677 }
678 else
679 {
680 return std::nullopt;
681 }
682 }
683 break;
684 }
685 }
686
687 return std::nullopt;
688 }

736 bool replicate(const kv::BatchVector& entries, Term term) override
737 {
738 std::lock_guard<ccf::pal::Mutex> guard(state->lock);
739
740 if (leadership_state != kv::LeadershipState::Leader)
741 {
742 LOG_FAIL_FMT(
743 "Failed to replicate {} items: not leader", entries.size());
744 rollback(state->last_idx);
745 return false;
746 }
747
748 if (term != state->current_view)
749 {
750 LOG_FAIL_FMT(
751 "Failed to replicate {} items at term {}, current term is {}",
752 entries.size(),
753 term,
754 state->current_view);
755 return false;
756 }
757
758 LOG_DEBUG_FMT("Replicating {} entries", entries.size());
759
760 for (auto& [index, data, is_globally_committable, hooks] : entries)
761 {
762 bool globally_committable = is_globally_committable;
763
764 if (index != state->last_idx + 1)
765 return false;
766
767 if (retirement_committable_idx.has_value())
768 {
769 CCF_ASSERT_FMT(
770 index > retirement_committable_idx.value(),
771 "Index {} unexpectedly lower than retirement_committable_idx {}",
772 index,
773 retirement_committable_idx.value());
774 return false;
775 }
776
777 LOG_DEBUG_FMT(
778 "Replicated on leader {}: {}{} ({} hooks)",
779 state->my_node_id,
780 index,
781 (globally_committable ? " committable" : ""),
782 hooks->size());
783
784 for (auto& hook : *hooks)
785 {
786 hook->call(this);
787 }
788
789 if (globally_committable)
790 {
791 LOG_DEBUG_FMT(
792 "membership: {} leadership: {}",
793 membership_state,
794 leadership_state_string());
795 if (
796 membership_state == kv::MembershipState::Retired &&
797 retirement_phase == kv::RetirementPhase::Ordered)
798 {
799 become_retired(index, kv::RetirementPhase::Signed);
800 }
801 committable_indices.push_back(index);
802 start_ticking_if_necessary();
803 }
804
805 state->last_idx = index;
806 ledger->put_entry(
807 *data, globally_committable, state->current_view, index);
808 entry_size_not_limited += data->size();
809 entry_count++;
810
811 state->view_history.update(index, state->current_view);
812 if (entry_size_not_limited >= append_entries_size_limit)
813 {
814 update_batch_size();
815 entry_count = 0;
816 entry_size_not_limited = 0;
817 for (const auto& it : all_other_nodes)
818 {
819 LOG_DEBUG_FMT("Sending updates to follower {}", it.first);
820 send_append_entries(it.first, it.second.sent_idx + 1);
821 }
822 }
823 }
824
825 // If we are the only node, attempt to commit immediately.
826 if (all_other_nodes.size() == 0)
827 {
828 update_commit();
829 }
830
831 return true;
832 }

896 void periodic(std::chrono::milliseconds elapsed) override
897 {
898 std::unique_lock<ccf::pal::Mutex> guard(state->lock);
899 timeout_elapsed += elapsed;
900
901 if (leadership_state == kv::LeadershipState::Leader)
902 {
903 if (timeout_elapsed >= request_timeout)
904 {
905 using namespace std::chrono_literals;
906 timeout_elapsed = 0ms;
907
908 update_batch_size();
909 // Send newly available entries to all other nodes.
910 for (const auto& node : all_other_nodes)
911 {
912 send_append_entries(node.first, node.second.sent_idx + 1);
913 }
914 }
915
916 for (auto& node : all_other_nodes)
917 {
918 node.second.last_ack_timeout += elapsed;
919 }
920
921 bool has_quorum_of_backups = false;
922 for (auto const& conf : configurations)
923 {
924 size_t backup_ack_timeout_count = 0;
925 for (auto const& node : conf.nodes)
926 {
927 auto search = all_other_nodes.find(node.first);
928 if (search == all_other_nodes.end())
929 {
930 // Ignore ourselves as primary
931 continue;
932 }
933 if (search->second.last_ack_timeout >= election_timeout)
934 {
935 LOG_DEBUG_FMT(
936 "No ack received from {} in last {}",
937 node.first,
938 election_timeout);
939 backup_ack_timeout_count++;
940 }
941 }
942
943 if (backup_ack_timeout_count < get_quorum(conf.nodes.size() - 1))
944 {
945 // If primary has quorum of active backups in _any_ configuration,
946 // it should remain primary
947 has_quorum_of_backups = true;
948 break;
949 }
950 }
951
952 if (!has_quorum_of_backups)
953 {
954 // CheckQuorum: The primary automatically steps down if there are no
955 // active configuration in which it has heard back from a majority of
956 // backups within an election timeout.
957 LOG_INFO_FMT(
958 "Stepping down as follower {}: No ack received from a majority of "
959 "backups in last {}",
960 state->my_node_id,
961 election_timeout);
962 become_follower();
963 }
964 }
965 else if (consensus_type != ConsensusType::BFT)
966 {
967 if (
968 can_endorse_primary() && ticking &&
969 timeout_elapsed >= election_timeout)
970 {
971 // Start an election.
972 become_candidate();
973 }
974 }

978 Index find_highest_possible_match(const ccf::TxID& tx_id)
979 {
980 // Find the highest TxID this node thinks exists, which is still
981 // compatible with the given tx_id. That is, given T.n, find largest n'
982 // such that n' <= n && term_of(n') == T' && T' <= T. This may be T.n
983 // itself, if this node holds that index. Otherwise, examine the final
984 // entry in each term, counting backwards, until we find one which is
985 // still possible.
986 Index probe_index = std::min(tx_id.seqno, state->last_idx);
987 Term term_of_probe = state->view_history.view_at(probe_index);
988 while (term_of_probe > tx_id.view)
989 {
990 // Next possible match is the end of the previous term, which is
991 // 1-before the start of the currently considered term. Anything after
992 // that must have a term which is still too high.
993 probe_index = state->view_history.start_of_view(term_of_probe);
994 if (probe_index > 0)
995 {
996 --probe_index;
997 }
998 term_of_probe = state->view_history.view_at(probe_index);
999 }
1000
1001 LOG_TRACE_FMT(
1002 "Looking for match with {}.{}, from {}.{}, best answer is {}",
1003 tx_id.view,
1004 tx_id.seqno,
1005 state->view_history.view_at(state->last_idx),
1006 state->last_idx,
1007 probe_index);
1008 return probe_index;
1009 }

1011 inline void update_batch_size()
1012 {
1013 auto avg_entry_size = (entry_count == 0) ?
1014 append_entries_size_limit :
1015 entry_size_not_limited / entry_count;
1016
1017 auto batch_size = (avg_entry_size == 0) ?
1018 append_entries_size_limit / 2 :
1019 append_entries_size_limit / avg_entry_size;
1020
1021 auto batch_avg = batch_window_sum / batch_window_size;
1022 // balance out total batch size across batch window
1023 batch_window_sum += (batch_size - batch_avg);
1024 entries_batch_size = std::max((batch_window_sum / batch_window_size), 1);
1025 }

1054 void send_append_entries(const ccf::NodeId& to, Index start_idx)
1055 {
1056 LOG_TRACE_FMT(
1057 "Sending append entries to node {} in batches of {}, covering the "
1058 "range {} -> {}",
1059 to,
1060 entries_batch_size,
1061 start_idx,
1062 state->last_idx);
1063
1064 auto calculate_end_index = [this](Index start) {
1065 // Cap the end index in 2 ways:
1066 // - Must contain no more than entries_batch_size entries
1067 // - Must contain entries from a single term
1068 auto max_idx = state->last_idx;
1069 const auto term_of_ae = state->view_history.view_at(start);
1070 const auto index_at_end_of_term =
1071 state->view_history.end_of_view(term_of_ae);
1072 if (index_at_end_of_term != kv::NoVersion)
1073 {
1074 max_idx = index_at_end_of_term;
1075 }
1076 return std::min(start + entries_batch_size - 1, max_idx);
1077 };
1078
1079 Index end_idx;
1080
1081 // We break _after_ sending, so that in the case where this is called
1082 // with start==last, we send a single empty heartbeat
1083 do
1084 {
1085 end_idx = calculate_end_index(start_idx);
1086 LOG_TRACE_FMT("Sending sub range {} -> {}", start_idx, end_idx);
1087 send_append_entries_range(to, start_idx, end_idx);
1088 start_idx = std::min(end_idx + 1, state->last_idx);
1089 } while (end_idx != state->last_idx);
1090 }
1091
1092 void send_append_entries_range(
1093 const ccf::NodeId& to, Index start_idx, Index end_idx)
1094 {
1095 const auto prev_idx = start_idx - 1;
1096
1097 if (
1098 is_retired() && retirement_phase > kv::RetirementPhase::Signed &&
1099 start_idx >= end_idx)
1100 {
1101 // Continue to replicate, but do not send heartbeats if we are retired
1102 return;
1103 }
1104
1105 const auto prev_term = get_term_internal(prev_idx);
1106 const auto term_of_idx = get_term_internal(end_idx);
1107 const bool contains_new_view =
1108 (state->new_view_idx > prev_idx) && (state->new_view_idx <= end_idx);
1109
1110 LOG_DEBUG_FMT(
1111 "Send append entries from {} to {}: ({}.{}, {}.{}] ({})",
1112 state->my_node_id,
1113 to,
1114 prev_term,
1115 prev_idx,
1116 term_of_idx,
1117 end_idx,
1118 state->commit_idx);
1119
1120 AppendEntries ae = {
1121 {raft_append_entries},
1122 {end_idx, prev_idx},
1123 state->current_view,
1124 prev_term,
1125 state->commit_idx,
1126 term_of_idx,
1127 contains_new_view};
1128
1129 auto& node = all_other_nodes.at(to);
1130
1131 // The host will append log entries to this message when it is
1132 // sent to the destination node.
1133 if (!channels->send_authenticated(
1134 to, ccf::NodeMsgType::consensus_msg, ae))
1135 {
1136 return;
1137 }
1138
1139 // Record the most recent index we have sent to this node.
1140 node.sent_idx = end_idx;
1141 }
1142
1143 void recv_append_entries(
1144 const ccf::NodeId& from,
1145 AppendEntries r,
1146 const uint8_t* data,
1147 size_t size)
1148 {
1149 std::unique_lock<ccf::pal::Mutex> guard(state->lock);
1150
1151 LOG_DEBUG_FMT(
1152 "Received append entries: {}.{} to {}.{} (from {} in term {})",
1153 r.prev_term,
1154 r.prev_idx,
1155 r.term_of_idx,
1156 r.idx,
1157 from,
1158 r.term);
1159
1160 // Don't check that the sender node ID is valid. Accept anything that
1161 // passes the integrity check. This way, entries containing dynamic
1162 // topology changes that include adding this new leader can be accepted.
1163
1164 // First, check append entries term against our own term, becoming
1165 // follower if necessary
1166 if (
1167 state->current_view == r.term &&
1168 leadership_state == kv::LeadershipState::Candidate)
1169 {
1170 become_aware_of_new_term(r.term);
1171 }
1172 else if (state->current_view < r.term)
1173 {
1174 become_aware_of_new_term(r.term);
1175 }
1176 else if (state->current_view > r.term)
1177 {
1178 // Reply false, since our term is later than the received term.
1179 LOG_INFO_FMT(
1180 "Recv append entries to {} from {} but our term is later ({} > {})",
1181 state->my_node_id,
1182 from,
1183 state->current_view,
1184 r.term);
1185 send_append_entries_response(from, AppendEntriesResponseType::FAIL);
1186 return;
1187 }
1188
1189 // Second, check term consistency with the entries we have so far
1190 const auto prev_term = get_term_internal(r.prev_idx);
1191 if (prev_term != r.prev_term)
1192 {
1193 LOG_DEBUG_FMT(
1194 "Previous term for {} should be {}", r.prev_idx, prev_term);
1195
1196 // Reply false if the log doesn't contain an entry at r.prev_idx
1197 // whose term is r.prev_term.
1198 if (prev_term == 0)
1199 {
1200 LOG_DEBUG_FMT(
1201 "Recv append entries to {} from {} but our log does not yet "
1202 "contain index {}",
1203 state->my_node_id,
1204 from,
1205 r.prev_idx);
1206 send_append_entries_response(from, AppendEntriesResponseType::FAIL);
1207 }
1208 else
1209 {
1210 LOG_DEBUG_FMT(
1211 "Recv append entries to {} from {} but our log at {} has the wrong "
1212 "previous term (ours: {}, theirs: {})",
1213 state->my_node_id,
1214 from,
1215 r.prev_idx,
1216 prev_term,
1217 r.prev_term);
1218 const ccf::TxID rejected_tx{r.prev_term, r.prev_idx};
1219 send_append_entries_response(
1220 from, AppendEntriesResponseType::FAIL, rejected_tx);
1221 }
1222 return;
1223 }
1224
1225 // Then check if those append entries extend past our retirement
1226 if (is_retired() && retirement_phase >= kv::RetirementPhase::Completed)
1227 {
1228 assert(retirement_committable_idx.has_value());
1229 if (r.idx > retirement_committable_idx)
1230 {
1231 send_append_entries_response(from, AppendEntriesResponseType::FAIL);
1232 return;
1233 }
1234 }
1235
1236 // If the terms match up, it is sufficient to convince us that the sender
1237 // is leader in our term
1238 restart_election_timeout();
1239 if (!leader_id.has_value() || leader_id.value() != from)
1240 {
1241 leader_id = from;
1242 LOG_DEBUG_FMT(
1243 "Node {} thinks leader is {}", state->my_node_id, leader_id.value());
1244 }
1245
1246 // Third, check index consistency, making sure entries are not in the past
1247 // or in the future
1248 if (
1249 consensus_type == ConsensusType::CFT && r.prev_idx < state->commit_idx)
1250 {
1251 LOG_DEBUG_FMT(
1252 "Recv append entries to {} from {} but prev_idx ({}) < commit_idx "
1253 "({})",
1254 state->my_node_id,
1255 from,
1256 r.prev_idx,
1257 state->commit_idx);
1258 return;
1259 }
1260 else if (r.prev_idx > state->last_idx)
1261 {
1262 LOG_DEBUG_FMT(
1263 "Recv append entries to {} from {} but prev_idx ({}) > last_idx ({})",
1264 state->my_node_id,
1265 from,
1266 r.prev_idx,
1267 state->last_idx);
1268 return;
1269 }
1270
1271 LOG_DEBUG_FMT(
1272 "Recv append entries to {} from {} for index {} and previous index {}",
1273 state->my_node_id,
1274 from,
1275 r.idx,
1276 r.prev_idx);
1277
1278 if (is_new_follower)
1279 {
1280 if (state->last_idx > r.prev_idx)
1281 {
1282 LOG_DEBUG_FMT(
1283 "New follower received first append entries with mismatch - "
1284 "rolling back from {} to {}",
1285 state->last_idx,
1286 r.prev_idx);
1287 auto rollback_level = r.prev_idx;
1288 rollback(rollback_level);
1289 }
1290 else
1291 {
1292 LOG_DEBUG_FMT(
1293 "New follower has no conflict with prev_idx {}", r.prev_idx);
1294 }
1295 is_new_follower = false;
1296 }
1297
1298 std::vector<
1299 std::tuple<std::unique_ptr<kv::AbstractExecutionWrapper>, kv::Version>>
1300 append_entries;
1301 // Finally, deserialise each entry in the batch
1302 for (Index i = r.prev_idx + 1; i <= r.idx; i++)
1303 {
1304 if (i <= state->last_idx)
1305 {
1306 // If the current entry has already been deserialised, skip the
1307 // payload for that entry
1308 ledger->skip_entry(data, size);
1309 continue;
1310 }
1311
1312 std::vector<uint8_t> entry;
1313 try
1314 {
1315 entry = LedgerProxy::get_entry(data, size);
1316 }
1317 catch (const std::logic_error& e)
1318 {
1319 // This should only fail if there is malformed data.
1320 LOG_FAIL_FMT(
1321 "Recv append entries to {} from {} but the data is malformed: {}",
1322 state->my_node_id,
1323 from,
1324 e.what());
1325 send_append_entries_response(from, AppendEntriesResponseType::FAIL);
1326 return;
1327 }
1328
1329 kv::TxID expected{r.term_of_idx, i};
1330 auto ds = store->apply(entry, consensus_type, public_only, expected);
1331 if (ds == nullptr)
1332 {
1333 LOG_FAIL_FMT(
1334 "Recv append entries to {} from {} but the entry could not be "
1335 "deserialised",
1336 state->my_node_id,
1337 from);
1338 send_append_entries_response(from, AppendEntriesResponseType::FAIL);
1339 return;
1340 }
1341 append_entries.push_back(std::make_tuple(std::move(ds), i));
1342 }
1343
1344 execute_append_entries_sync(
1345 std::move(append_entries), from, std::move(r));
1346 }
1347
1348 void execute_append_entries_sync(
1349 std::vector<std::tuple<
1350 std::unique_ptr<kv::AbstractExecutionWrapper>,
1351 kv::Version>>&& append_entries,
1352 const ccf::NodeId& from,
1353 AppendEntries&& r)
1354 {
1355 for (auto& ae : append_entries)
1356 {
1357 auto& [ds, i] = ae;
1358 LOG_DEBUG_FMT("Replicating on follower {}: {}", state->my_node_id, i);
1359
1360 kv::ApplyResult apply_success = ds->apply();
1361 if (apply_success == kv::ApplyResult::FAIL)
1362 {
1363 ledger->truncate(i - 1);
1364 send_append_entries_response(from, AppendEntriesResponseType::FAIL);
1365 return;
1366 }
1367 state->last_idx = i;
1368
1369 for (auto& hook : ds->get_hooks())
1370 {
1371 hook->call(this);
1372 }
1373
1374 bool globally_committable =
1375 (apply_success == kv::ApplyResult::PASS_SIGNATURE);
1376 if (globally_committable)
1377 {
1378 start_ticking_if_necessary();
1379 }
1380
1381 const auto& entry = ds->get_entry();
1382
1383 ledger->put_entry(
1384 entry, globally_committable, ds->get_term(), ds->get_index());
1385
1386 switch (apply_success)
1387 {
1388 case kv::ApplyResult::FAIL:
1389 {
1390 LOG_FAIL_FMT("Follower failed to apply log entry: {}", i);
1391 state->last_idx--;
1392 ledger->truncate(state->last_idx);
1393 send_append_entries_response(from, AppendEntriesResponseType::FAIL);
1394 break;
1395 }
1396
1397 case kv::ApplyResult::PASS_SIGNATURE:
1398 {
1399 LOG_DEBUG_FMT("Deserialising signature at {}", i);
1400 if (
1401 membership_state == kv::MembershipState::Retired &&
1402 retirement_phase == kv::RetirementPhase::Ordered)
1403 {
1404 become_retired(i, kv::RetirementPhase::Signed);
1405 }
1406 committable_indices.push_back(i);
1407
1408 if (ds->get_term())
1409 {
1410 // A signature for sig_term tells us that all transactions from
1411 // the previous signature onwards (at least, if not further back)
1412 // happened in sig_term. We reflect this in the history.
1413 if (r.term_of_idx == aft::ViewHistory::InvalidView)
1414 {
1415 state->view_history.update(1, r.term);
1416 }
1417 else
1418 {
1419 // NB: This is only safe as long as AppendEntries only contain a
1420 // single term. If they cover multiple terms, then we need to
1421 // know our previous signature locally.
1422 state->view_history.update(r.prev_idx + 1, ds->get_term());
1423 }
1424 commit_if_possible(r.leader_commit_idx);
1425 }
1426 break;
1427 }
1428
1429 case kv::ApplyResult::PASS:
1430 {
1431 break;
1432 }
1433
1434 case kv::ApplyResult::PASS_ENCRYPTED_PAST_LEDGER_SECRET:
1435 {
1436 break;
1437 }
1438
1439 default:
1440 {
1441 throw std::logic_error("Unknown ApplyResult value");
1442 }
1443 }
1444 }
1445
1446 execute_append_entries_finish(r, from);
1447 }
1448
1449 void execute_append_entries_finish(
1450 AppendEntries& r, const ccf::NodeId& from)
1451 {
1452 // After entries have been deserialised, try to commit the leader's
1453 // commit index and update our term history accordingly
1454 commit_if_possible(r.leader_commit_idx);
1455
1456 // The term may have changed, and we have not have seen a signature yet.
1457 auto lci = last_committable_index();
1458 if (r.term_of_idx == aft::ViewHistory::InvalidView)
1459 {
1460 // If we don't yet have a term history, then this must be happening in
1461 // the current term. This can only happen before _any_ transactions have
1462 // occurred, when processing a heartbeat at index 0, which does not
1463 // happen in a real node (due to the genesis transaction executing
1464 // before ticks start), but may happen in tests.
1465 state->view_history.update(1, r.term);
1466 }
1467 else
1468 {
1469 // The end of this append entries (r.idx) was not a signature, but may
1470 // be in a new term. If it's a new term, this term started immediately
1471 // after the previous signature we saw (lci, last committable index).
1472 if (r.idx > lci)
1473 {
1474 state->view_history.update(lci + 1, r.term_of_idx);
1475 }
1476 }
1477
1478 send_append_entries_response(from, AppendEntriesResponseType::OK);
1479 }

1481 void send_append_entries_response(
1482 ccf::NodeId to,
1483 AppendEntriesResponseType answer,
1484 const std::optional<ccf::TxID>& rejected = std::nullopt)
1485 {
1486 if (answer == AppendEntriesResponseType::REQUIRE_EVIDENCE)
1487 {
1488 state->requested_evidence_from = to;
1489 }
1490
1491 aft::Index response_idx = state->last_idx;
1492 aft::Term response_term = state->current_view;
1493
1494 // This matching-index-detection logic doesn't work on BFT, so is disabled
1495 // and still uses the original behaviour:
1496 // https://github.com/microsoft/CCF/issues/2853
1497 if (consensus_type != ConsensusType::BFT)
1498 {
1499 if (answer == AppendEntriesResponseType::FAIL && rejected.has_value())
1500 {
1501 response_idx = find_highest_possible_match(rejected.value());
1502 response_term = get_term_internal(response_idx);
1503 }
1504 }
1505
1506 LOG_DEBUG_FMT(
1507 "Send append entries response from {} to {} for index {}: {}",
1508 state->my_node_id,
1509 to,
1510 response_idx,
1511 answer);
1512
1513 AppendEntriesResponse response = {
1514 {raft_append_entries_response}, response_term, response_idx, answer};
1515
1516 channels->send_authenticated(
1517 to, ccf::NodeMsgType::consensus_msg, response);
1518 }
1519
1520 void recv_append_entries_response(
1521 const ccf::NodeId& from, AppendEntriesResponse r)
1522 {
1523 std::lock_guard<ccf::pal::Mutex> guard(state->lock);
1524 // Ignore if we're not the leader.
1525
1526 if (leadership_state != kv::LeadershipState::Leader)
1527 {
1528 LOG_FAIL_FMT(
1529 "Recv append entries response to {} from {}: no longer leader",
1530 state->my_node_id,
1531 from);
1532 return;
1533 }
1534
1535 auto node = all_other_nodes.find(from);
1536 if (node == all_other_nodes.end())
1537 {
1538 // Ignore if we don't recognise the node.
1539 LOG_FAIL_FMT(
1540 "Recv append entries response to {} from {}: unknown node",
1541 state->my_node_id,
1542 from);
1543 return;
1544 }
1545
1546 using namespace std::chrono_literals;
1547 node->second.last_ack_timeout = 0ms;
1548
1549 if (state->current_view < r.term)
1550 {
1551 // We are behind, update our state.
1552 LOG_DEBUG_FMT(
1553 "Recv append entries response to {} from {}: more recent term ({} "
1554 "> {})",
1555 state->my_node_id,
1556 from,
1557 r.term,
1558 state->current_view);
1559 become_aware_of_new_term(r.term);
1560 return;
1561 }
1562 else if (state->current_view != r.term)
1563 {
1564 // Stale response, discard if success.
1565 // Otherwise reset sent_idx and try again.
1566 // NB: In NACKs the term may be that of an estimated matching index
1567 // in the log, rather than the current term, so it is correct for it to
1568 // be older in this case.
1569 if (r.success == AppendEntriesResponseType::OK)
1570 {
1571 LOG_DEBUG_FMT(
1572 "Recv append entries response to {} from {}: stale term ({} != {})",
1573 state->my_node_id,
1574 from,
1575 r.term,
1576 state->current_view);
1577 return;
1578 }
1579 }
1580 else if (r.last_log_idx < node->second.match_idx)
1581 {
1582 // Response about past indices, discard if success.
1583 // Otherwise reset sent_idx and try again.
1584 // NB: It is correct for this index to move backwards during NACKs
1585 // which iteratively discover the last matching index of divergent logs
1586 // after an election.
1587 if (r.success == AppendEntriesResponseType::OK)
1588 {
1589 LOG_DEBUG_FMT(
1590 "Recv append entries response to {} from {}: stale idx",
1591 state->my_node_id,
1592 from);
1593 return;
1594 }
1595 }
1596
1597 // Update next and match for the responding node.
1598 auto& match_idx = node->second.match_idx;
1599 if (r.success == AppendEntriesResponseType::FAIL)
1600 {
1601 const auto this_match =
1602 find_highest_possible_match({r.term, r.last_log_idx});
1603 if (match_idx == 0)
1604 {
1605 match_idx = this_match;
1606 }
1607 else
1608 {
1609 match_idx = std::min(match_idx, this_match);
1610 }
1611 }
1612 else
1613 {
1614 match_idx = std::min(r.last_log_idx, state->last_idx);
1615 }
1616
1617 if (r.success != AppendEntriesResponseType::OK)
1618 {
1619 // Failed due to log inconsistency. Reset sent_idx, and try again soon.
1620 LOG_DEBUG_FMT(
1621 "Recv append entries response to {} from {}: failed",
1622 state->my_node_id,
1623 from);
1624 node->second.sent_idx = node->second.match_idx;
1625 return;
1626 }
1627
1628 LOG_DEBUG_FMT(
1629 "Recv append entries response to {} from {} for index {}: success",
1630 state->my_node_id,
1631 from,
1632 r.last_log_idx);
1633 update_commit();
1634 }

1655 void recv_request_vote(const ccf::NodeId& from, RequestVote r)
1656 {
1657 std::lock_guard<ccf::pal::Mutex> guard(state->lock);
1658
1659 // Do not check that from is a known node. It is possible to receive
1660 // RequestVotes from nodes that this node doesn't yet know, just as it
1661 // receives AppendEntries from those nodes. These should be obeyed just
1662 // like any other RequestVote - it is possible that this node is needed to
1663 // produce a primary in the new term, who will then help this node catch
1664 // up.
1665
1666 if (state->current_view > r.term)
1667 {
1668 // Reply false, since our term is later than the received term.
1669 LOG_DEBUG_FMT(
1670 "Recv request vote to {} from {}: our term is later ({} > {})",
1671 state->my_node_id,
1672 from,
1673 state->current_view,
1674 r.term);
1675 send_request_vote_response(from, false);
1676 return;
1677 }
1678 else if (state->current_view < r.term)
1679 {
1680 LOG_DEBUG_FMT(
1681 "Recv request vote to {} from {}: their term is later ({} < {})",
1682 state->my_node_id,
1683 from,
1684 state->current_view,
1685 r.term);
1686 become_aware_of_new_term(r.term);
1687 }
1688
1689 if (leader_id.has_value())
1690 {
1691 // Reply false, since we already know the leader in the current term.
1692 LOG_DEBUG_FMT(
1693 "Recv request vote to {} from {}: leader {} already known in term {}",
1694 state->my_node_id,
1695 from,
1696 leader_id.value(),
1697 state->current_view);
1698 send_request_vote_response(from, false);
1699 return;
1700 }
1701
1702 if ((voted_for.has_value()) && (voted_for.value() != from))
1703 {
1704 // Reply false, since we already voted for someone else.
1705 LOG_DEBUG_FMT(
1706 "Recv request vote to {} from {}: already voted for {}",
1707 state->my_node_id,
1708 from,
1709 voted_for.value());
1710 send_request_vote_response(from, false);
1711 return;
1712 }
1713
1714 // If the candidate's committable log is at least as up-to-date as ours,
1715 // vote yes
1716
1717 const auto last_committable_idx = last_committable_index();
1718 const auto term_of_last_committable_idx =
1719 get_term_internal(last_committable_idx);
1720
1721 const auto answer =
1722 (r.term_of_last_committable_idx > term_of_last_committable_idx) ||
1723 ((r.term_of_last_committable_idx == term_of_last_committable_idx) &&
1724 (r.last_committable_idx >= last_committable_idx));
1725
1726 if (answer)
1727 {
1728 // If we grant our vote, we also acknowledge that an election is in
1729 // progress.
1730 restart_election_timeout();
1731 leader_id.reset();
1732 voted_for = from;
1733 }
1734 else
1735 {
1736 LOG_INFO_FMT(
1737 "Voting against candidate at {}.{} because local state is at {}.{}",
1738 r.term_of_last_committable_idx,
1739 r.last_committable_idx,
1740 term_of_last_committable_idx,
1741 last_committable_idx);
1742 }
1743
1744 send_request_vote_response(from, answer);
1745 }

1872 void become_leader(bool force_become_leader = false)
1873 {
1874 if (is_retired())
1875 {
1876 return;
1877 }
1878
1879 // When we force to become the primary we are going around the
1880 // consensus protocol. This only happens when a node starts a new network
1881 // and has a genesis or recovery tx as the last transaction
1882 election_index = last_committable_index();
1883
1884 // A newly elected leader must not advance the commit index until a
1885 // transaction in the new term commits. We achieve this by clearing our
1886 // list committable indices - so nothing from a previous term is now
1887 // considered committable. Instead this new primary will shortly produce
1888 // their own signature, which _will_ be considered committable.
1889 committable_indices.clear();
1890
1891 LOG_DEBUG_FMT(
1892 "Election index is {} in term {}", election_index, state->current_view);
1893 // Discard any un-committable updates we may hold,
1894 // since we have no signature for them. Except at startup,
1895 // where we do not want to roll back the genesis transaction.
1896 if (state->commit_idx > 0)
1897 {
1898 rollback(election_index);
1899 }
1900 else
1901 {
1902 // but we still want the KV to know which term we're in
1903 store->initialise_term(state->current_view);
1904 }
1905
1906 leadership_state = kv::LeadershipState::Leader;
1907 leader_id = state->my_node_id;
1908 should_sign = true;
1909
1910 using namespace std::chrono_literals;
1911 timeout_elapsed = 0ms;
1912
1913 reset_last_ack_timeouts();
1914
1915 LOG_INFO_FMT(
1916 "Becoming leader {}: {}", state->my_node_id, state->current_view);
1917
1918 // Immediately commit if there are no other nodes.
1919 if (all_other_nodes.size() == 0)
1920 {
1921 commit(state->last_idx);
1922 return;
1923 }
1924
1925 // Reset next, match, and sent indices for all nodes.
1926 auto next = state->last_idx + 1;
1927
1928 for (auto& node : all_other_nodes)
1929 {
1930 node.second.match_idx = 0;
1931 node.second.sent_idx = next - 1;
1932
1933 // Send an empty append_entries to all nodes.
1934 send_append_entries(node.first, next);
1935 }
1936 }

2095 {

2097 // idx > commit_idx and a majority of nodes have replicated it,
2098 // commit to that idx.

2100
2101 // Obtain CFT watermarks
2102 for (auto const& c : configurations)
2103 {
2104 // The majority must be checked separately for each active
2105 // configuration.
2106 std::vector<Index> match;
2107 match.reserve(c.nodes.size() + 1);
2108
2109 for (auto node : c.nodes)
2110 {
2111 if (node.first == state->my_node_id)
2112 {
2113 match.push_back(state->last_idx);
2114 }
2115 else
2116 {
2117 match.push_back(all_other_nodes.at(node.first).match_idx);
2118 }
2119 }
2120
2121 sort(match.begin(), match.end());
2122 auto confirmed = match.at((match.size() - 1) / 2);
2123
2124 if (confirmed < new_commit_cft_idx)
2125 {
2126 new_commit_cft_idx = confirmed;
2127 }
2128 }
2129 LOG_DEBUG_FMT(
2130 "In update_commit, new_commit_cft_idx: {}, "
2131 "last_idx: {}",
2132 new_commit_cft_idx,
2133 state->last_idx);
2134
2135 if (new_commit_cft_idx != std::numeric_limits<Index>::max())
2136 {
2137 state->cft_watermark_idx = new_commit_cft_idx;
2138 }
2139
2140 if (get_commit_watermark_idx() > state->last_idx)
2141 {
2142 throw std::logic_error(
2143 "Followers appear to have later match indices than leader");
2144 }

2147 }

2246 void commit(Index idx)
2247 {
2248 if (idx > state->last_idx)
2249 {
2250 throw std::logic_error(fmt::format(
2251 "Tried to commit {} but last_idx is {}", idx, state->last_idx));
2252 }
2253
2254 LOG_DEBUG_FMT("Starting commit");
2255
2256 // This could happen if a follower becomes the leader when it
2257 // has committed fewer log entries, although it has them available.
2258 if (idx <= state->commit_idx)
2259 return;
2260
2261 state->commit_idx = idx;
2262 if (
2263 is_retired() && retirement_phase == kv::RetirementPhase::Signed &&
2264 retirement_committable_idx.has_value() &&
2265 idx >= retirement_committable_idx.value())
2266 {
2267 become_retired(idx, kv::RetirementPhase::Completed);
2268 }
2269
2270 LOG_DEBUG_FMT("Compacting...");
2271 store->compact(idx);
2272 ledger->commit(idx);
2273
2274 LOG_DEBUG_FMT("Commit on {}: {}", state->my_node_id, idx);
2275
2276 // Examine each configuration that is followed by a globally committed
2277 // configuration.
2278 bool changed = false;
2279
2280 while (true)
2281 {
2282 auto conf = configurations.begin();
2283 if (conf == configurations.end())
2284 {
2285 break;
2286 }
2287
2288 auto next = std::next(conf);
2289 if (next == configurations.end())
2290 {
2291 break;
2292 }
2293
2294 if (idx < next->idx)
2295 {
2296 break;
2297 }
2298
2299 if (require_identity_for_reconfig)
2300 {
2301 assert(resharing_tracker);
2302 auto rr = resharing_tracker->find_reconfiguration(next->nodes);
