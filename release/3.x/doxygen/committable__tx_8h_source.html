40 std::vector<uint8_t> serialise(
41 crypto::Sha256Hash& commit_evidence_digest,
42 std::string& commit_evidence,
43 const ccf::ClaimsDigest& claims_digest_,
44 bool include_reads = false)
45 {
46 if (!committed)
47 throw std::logic_error("Transaction not yet committed");
48
49 if (!success)
50 throw std::logic_error("Transaction aborted");
51
52 if (claims_digest_.empty())
53 throw std::logic_error("Missing claims");
54
55 // If no transactions made changes, return a zero length vector.
56 const bool any_changes =
57 std::any_of(all_changes.begin(), all_changes.end(), [](const auto& it) {
58 return it.second.changeset->has_writes();
59 });
60
61 if (!any_changes)
62 {
63 return {};
64 }
65
66 auto e = pimpl->store->get_encryptor();
67 if (e == nullptr)
68 {
69 throw KvSerialiserException("No encryptor set");
70 }
71
72 auto commit_nonce = e->get_commit_nonce({pimpl->commit_view, version});
73 commit_evidence = fmt::format(
74 "ce:{}.{}:{}", pimpl->commit_view, version, ds::to_hex(commit_nonce));
75 LOG_TRACE_FMT("Commit evidence: {}", commit_evidence);
76 crypto::Sha256Hash tx_commit_evidence_digest(commit_evidence);
77 commit_evidence_digest = tx_commit_evidence_digest;
78 auto entry_type = EntryType::WriteSetWithCommitEvidenceAndClaims;
79
80 if (flag_enabled(Flag::LEDGER_CHUNK_BEFORE_THIS_TX))
81 {
82 entry_flags |= EntryFlags::FORCE_LEDGER_CHUNK_BEFORE;
83 }
84
85 KvStoreSerialiser replicated_serialiser(
86 e,
87 {pimpl->commit_view, version},
88 entry_type,
89 entry_flags,
90 tx_commit_evidence_digest,
91 claims_digest_);
92
93 // Process in security domain order
94 for (auto domain : {SecurityDomain::PUBLIC, SecurityDomain::PRIVATE})
95 {
96 for (const auto& it : all_changes)
97 {
98 const auto& map = it.second.map;
99 const auto& changeset = it.second.changeset;
100 if (
101 map->get_security_domain() == domain && map->is_replicated() &&
102 changeset->has_writes())
103 {
104 map->serialise_changes(
105 changeset.get(), replicated_serialiser, include_reads);
106 }
107 }
108 }
109
110 // Return serialised Tx.
111 return replicated_serialiser.get_raw_data();
112 }

129 CommitResult commit(
130 const ccf::ClaimsDigest& claims = ccf::empty_claims(),
131 bool track_read_versions = false,
132 std::function<std::tuple<Version, Version>(bool has_new_map)>
133 version_resolver = nullptr,
134 std::function<void(
135 const std::vector<uint8_t>& write_set,
136 const std::string& commit_evidence)> write_set_observer = nullptr)
137 {
138 if (committed)
139 throw std::logic_error("Transaction already committed");
140
141 if (all_changes.empty())
142 {
143 committed = true;
144 success = true;
145 return CommitResult::SUCCESS;
146 }
147
148 // If this transaction creates any maps, ensure that commit gets a
149 // consistent snapshot of the existing maps
150 if (!pimpl->created_maps.empty())
151 this->pimpl->store->lock();
152
153 kv::ConsensusHookPtrs hooks;
154
155 std::optional<Version> new_maps_conflict_version = std::nullopt;
156
157 auto c = apply_changes(
158 all_changes,
159 version_resolver == nullptr ?
160 [&](bool has_new_map) {
161 return pimpl->store->next_version(has_new_map);
162 } :
163 version_resolver,
164 hooks,
165 pimpl->created_maps,
166 new_maps_conflict_version,
167 track_read_versions);
168
169 if (!pimpl->created_maps.empty())
170 this->pimpl->store->unlock();
171
172 success = c.has_value();
173
174 if (!success)
175 {
176 // This Tx is now in a dead state. Caller should create a new Tx and try
177 // again.
178 LOG_TRACE_FMT("Could not commit transaction due to conflict");
179 return CommitResult::FAIL_CONFLICT;
180 }
181 else
182 {
183 committed = true;
184 version = c.value();
185
186 if (flag_enabled(Flag::LEDGER_CHUNK_AT_NEXT_SIGNATURE))
187 {
188 pimpl->store->set_flag(
189 AbstractStore::Flag::LEDGER_CHUNK_AT_NEXT_SIGNATURE);
190 // This transaction indicates to the store that the next signature
191 // should trigger a new ledger chunk, but *this* transaction does not
192 // create a new ledger chunk
193 unset_flag(CommittableTx::Flag::LEDGER_CHUNK_AT_NEXT_SIGNATURE);
194 }
195
196 if (flag_enabled(Flag::SNAPSHOT_AT_NEXT_SIGNATURE))
197 {
198 pimpl->store->set_flag(
199 AbstractStore::Flag::SNAPSHOT_AT_NEXT_SIGNATURE);
200 unset_flag(CommittableTx::Flag::SNAPSHOT_AT_NEXT_SIGNATURE);
201 }
202
203 if (version == NoVersion)
204 {
205 // Read-only transaction
206 return CommitResult::SUCCESS;
207 }
208
209 // From here, we have received a unique commit version and made
210 // modifications to our local kv. If we fail in any way, we cannot
211 // recover.
212 try
213 {
214 crypto::Sha256Hash commit_evidence_digest;
215 std::string commit_evidence;
216 auto data =
217 serialise(commit_evidence_digest, commit_evidence, claims);
218
219 if (data.empty())
220 {
221 return CommitResult::SUCCESS;
222 }
223
224 if (write_set_observer != nullptr)
225 {
226 write_set_observer(data, commit_evidence);
227 }
228
229 auto claims_ = claims;
230
231 return pimpl->store->commit(
232 {pimpl->commit_view, version},
233 std::make_unique<MovePendingTx>(
234 std::move(data),
235 std::move(claims_),
236 std::move(commit_evidence_digest),
237 std::move(hooks)),
238 false);
239 }
240 catch (const std::exception& e)
241 {
242 committed = false;
243
244 LOG_FAIL_FMT("Error during serialisation");
245 LOG_DEBUG_FMT("Error during serialisation: {}", e.what());
246
247 // Discard original exception type, throw as now fatal
248 // KvSerialiserException
249 throw KvSerialiserException(e.what());
250 }
251 }
252 }

380 // Used by frontend for reserved transactions. These are constructed with a
381 // pre-reserved Version, and _must succeed_ to fulfil this version. Otherwise
382 // they create a hole in the transaction order, and no future transactions can
383 // complete. These transactions are used internally by CCF for the sole
384 // purpose of recording node signatures and are safe in this particular
385 // situation because they never perform any reads and therefore can
386 // never conflict.
387 class ReservedTx : public CommittableTx
388 {
389 public:
390 ReservedTx(
391 AbstractStore* _store, Term read_term, const TxID& reserved_tx_id) :
392 CommittableTx(_store)
393 {
394 version = reserved_tx_id.version;
395 pimpl->commit_view = reserved_tx_id.term;
396 pimpl->read_txid = TxID(read_term, reserved_tx_id.version - 1);
397 }
398
399 // Used by frontend to commit reserved transactions
400 PendingTxInfo commit_reserved()
401 {
402 if (committed)
403 throw std::logic_error("Transaction already committed");
404
405 if (all_changes.empty())
406 throw std::logic_error("Reserved transaction cannot be empty");
407
408 std::vector<ConsensusHookPtr> hooks;
409 auto c = apply_changes(
410 all_changes,
411 [this](bool) { return std::make_tuple(version, version - 1); },
412 hooks,
413 pimpl->created_maps,
414 version);
415 success = c.has_value();
416
417 if (!success)
418 throw std::logic_error("Failed to commit reserved transaction");
419
420 crypto::Sha256Hash commit_evidence_digest;
421 std::string commit_evidence;
422
423 // This is a signature and, if the ledger chunking or snapshot flags are
424 // enabled, we want the host to create a chunk when it sees this entry.
425 // version_lock held by Store::commit
426 if (pimpl->store->must_force_ledger_chunk_unsafe(version))
427 {
428 entry_flags |= EntryFlags::FORCE_LEDGER_CHUNK_AFTER;
429 LOG_DEBUG_FMT(
430 "Forcing ledger chunk for signature at {}.{}",
431 pimpl->commit_view,
432 version);
433 }
434
435 committed = true;
436 auto claims = ccf::empty_claims();
437 auto data = serialise(commit_evidence_digest, commit_evidence, claims);
438
439 // Reset ledger chunk flag in the store
440 pimpl->store->unset_flag_unsafe(
441 AbstractStore::Flag::LEDGER_CHUNK_AT_NEXT_SIGNATURE);
442
443 return {
444 CommitResult::SUCCESS,
445 std::move(data),
446 ccf::empty_claims(),
447 std::move(commit_evidence_digest),
448 std::move(hooks)};
449 }
450 };
