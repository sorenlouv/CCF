22// `It is recommended that applications attempt 10 retries in a tight loop in
23// the unlikely event that the RDRAND instruction does not return a random
24// number. This number is based on a binomial probability argument: given
25// the design margins of the DRNG, the odds of ten failures in a row are
26// astronomically small and would in fact be an indication of a larger CPU
27// issue.`
28#define RDRAND_RETRIES 10

58 static int get_drng_support()
59 {
60 static int drng_features = -1;
61
62 /* So we don't call cpuid multiple times for the same information */
63 if (drng_features == -1)
64 {
65 drng_features = DRNG_NO_SUPPORT;
66
67 if (ccf::pal::is_intel_cpu())
68 {
69 ccf::pal::CpuidInfo info;
70
71 ccf::pal::cpuid(&info, 1, 0);
72
73 if ((info.ecx & 0x40000000) == 0x40000000)
74 drng_features |= DRNG_HAS_RDRAND;
75
76 cpuid(&info, 7, 0);
77
78 if ((info.ebx & 0x40000) == 0x40000)
79 drng_features |= DRNG_HAS_RDSEED;
80 }
81 }
82
83 return drng_features;
84 }

151 static unsigned int rdrand_get_bytes(unsigned int n, unsigned char* dest)
152 {
153 unsigned char *headstart, *tailstart = nullptr;
154 uint64_t* blockstart;
155 unsigned int count, ltail, lhead, lblock;
156 uint64_t i, temprand;
157
158 /* Get the address of the first 64-bit aligned block in the
159 * destination buffer. */
160 headstart = dest;
161 if (((uint64_t)headstart % (uint64_t)8) == 0)
162 {
163 blockstart = (uint64_t*)headstart;
164 lblock = n;
165 lhead = 0;
166 }
167 else
168 {
169 blockstart =
170 (uint64_t*)(((uint64_t)headstart & ~(uint64_t)7) + (uint64_t)8);
171 lhead = (unsigned int)((uint64_t)blockstart - (uint64_t)headstart);
172 lblock =
173 ((n - lhead) & ~(unsigned int)7); // cwinter: this bit is/as buggy in
174 // the Intel examples.
175 }
176
177 /* Compute the number of 64-bit blocks and the remaining number
178 * of bytes (the tail) */
179 ltail = n - lblock - lhead;
180 count = lblock / 8; /* The number 64-bit rands needed */
181
182 assert(lhead < 8);
183 assert(lblock <= n);
184 assert(ltail < 8);
185
186 if (ltail)
187 tailstart = (unsigned char*)((uint64_t)blockstart + (uint64_t)lblock);
188
189 /* Populate the starting, mis-aligned section (the head) */
190 if (lhead)
191 {
192 if (!rdrand64_retry(RDRAND_RETRIES, &temprand))
193 return 0;
194 memcpy(headstart, &temprand, lhead);
195 }
196
197 /* Populate the central, aligned block */
198 for (i = 0; i < count; ++i, ++blockstart)
199 {
200 if (!rdrand64_retry(RDRAND_RETRIES, blockstart))
201 return i * 8 + lhead;
202 }
203
204 /* Populate the tail */
205 if (ltail)
206 {
207 if (!rdrand64_retry(RDRAND_RETRIES, &temprand))
208 return count * 8 + lhead;
209 memcpy(tailstart, &temprand, ltail);
210 }
211
212 return n;
213 }

249 std::vector<uint8_t> random(size_t len) override
250 {
251 std::vector<uint8_t> buf(len);
252
253 if (rdrand_get_bytes(buf.size(), buf.data()) < buf.size())
254 throw std::logic_error("Couldn't create random data");

262 uint64_t random64() override
263 {
264 uint64_t rnd;

266
267 if (rdrand_get_bytes(len, reinterpret_cast<unsigned char*>(&rnd)) < len)
268 {
269 throw std::logic_error("Couldn't create random data");
270 }

273 }
