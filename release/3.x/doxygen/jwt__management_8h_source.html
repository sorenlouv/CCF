19 static void remove_jwt_public_signing_keys(kv::Tx& tx, std::string issuer)

36 static oe_result_t oe_verify_attestation_certificate_with_evidence_cb(
37 oe_claim_t* claims, size_t claims_length, void* arg)
38 {
39 auto claims_map = (std::map<std::string, std::vector<uint8_t>>*)arg;
40 for (size_t i = 0; i < claims_length; i++)
41 {
42 std::string claim_name(claims[i].name);
43 std::vector<uint8_t> claim_value(
44 claims[i].value, claims[i].value + claims[i].value_size);
45 claims_map->emplace(std::move(claim_name), std::move(claim_value));
46 }
47 return OE_OK;
48 }

50 static bool set_jwt_public_signing_keys(
51 kv::Tx& tx,
52 const ProposalId& proposal_id,
53 std::string issuer,
54 const JwtIssuerMetadata& issuer_metadata,
55 const JsonWebKeySet& jwks)
56 {
57 auto keys = tx.rw<JwtPublicSigningKeys>(Tables::JWT_PUBLIC_SIGNING_KEYS);
58 auto key_issuer =
59 tx.rw<JwtPublicSigningKeyIssuer>(Tables::JWT_PUBLIC_SIGNING_KEY_ISSUER);
60
61 auto log_prefix = proposal_id.empty() ?
62 "JWT key auto-refresh" :
63 fmt::format("Proposal {}", proposal_id);
64
65 // add keys
66 if (jwks.keys.empty())
67 {
68 LOG_FAIL_FMT("{}: JWKS has no keys", log_prefix, proposal_id);
69 return false;
70 }
71 std::map<std::string, std::vector<uint8_t>> new_keys;
72 for (auto& jwk : jwks.keys)
73 {
74 if (!jwk.kid.has_value())
75 {
76 LOG_FAIL_FMT("No kid for JWT signing key");
77 return false;
78 }
79 auto const& kid = jwk.kid.value();
80
81 if (keys->has(kid) && key_issuer->get(kid).value() != issuer)
82 {
83 LOG_FAIL_FMT(
84 "{}: key id {} already added for different issuer", log_prefix, kid);
85 return false;
86 }
87 if (!jwk.x5c.has_value() && jwk.x5c->empty())
88 {
89 LOG_FAIL_FMT("{}: JWKS is invalid (empty x5c)", log_prefix);
90 return false;
91 }
92
93 auto& der_base64 = jwk.x5c.value()[0];
94 ccf::Cert der;
95 try
96 {
97 der = crypto::raw_from_b64(der_base64);
98 }
99 catch (const std::invalid_argument& e)
100 {
101 LOG_FAIL_FMT(
102 "{}: Could not parse x5c of key id {}: {}",
103 log_prefix,
104 kid,
105 e.what());
106 return false;
107 }
108
109 std::map<std::string, std::vector<uint8_t>> claims;
110 bool has_key_policy_sgx_claims = issuer_metadata.key_policy.has_value() &&
111 issuer_metadata.key_policy.value().sgx_claims.has_value() &&
112 !issuer_metadata.key_policy.value().sgx_claims.value().empty();
113 if (
114 issuer_metadata.key_filter == JwtIssuerKeyFilter::SGX ||
115 has_key_policy_sgx_claims)
116 {
117 oe_verify_attestation_certificate_with_evidence(
118 der.data(),
119 der.size(),
120 oe_verify_attestation_certificate_with_evidence_cb,
121 &claims);
122 }
123
124 if (
125 issuer_metadata.key_filter == JwtIssuerKeyFilter::SGX && claims.empty())
126 {
127 LOG_INFO_FMT(
128 "{}: Skipping JWT signing key with kid {} (not OE "
129 "attested)",
130 log_prefix,
131 kid);
132 continue;
133 }
134
135 if (has_key_policy_sgx_claims)
136 {
137 for (auto& [claim_name, expected_claim_val_hex] :
138 issuer_metadata.key_policy.value().sgx_claims.value())
139 {
140 if (claims.find(claim_name) == claims.end())
141 {
142 LOG_FAIL_FMT(
143 "{}: JWKS kid {} is missing the {} SGX claim",
144 log_prefix,
145 kid,
146 claim_name);
147 return false;
148 }
149 auto& actual_claim_val = claims[claim_name];
150 auto actual_claim_val_hex = ds::to_hex(actual_claim_val);
151 if (expected_claim_val_hex != actual_claim_val_hex)
152 {
153 LOG_FAIL_FMT(
154 "{}: JWKS kid {} has a mismatching {} SGX claim: {} != {}",
155 log_prefix,
156 kid,
157 claim_name,
158 expected_claim_val_hex,
159 actual_claim_val_hex);
160 return false;
161 }
162 }
163 }
164 else
165 {
166 try
167 {
168 crypto::make_unique_verifier(
169 (std::vector<uint8_t>)der); // throws on error
170 }
171 catch (std::invalid_argument& exc)
172 {
173 LOG_FAIL_FMT(
174 "{}: JWKS kid {} has an invalid X.509 certificate: {}",
175 log_prefix,
176 kid,
177 exc.what());
178 return false;
179 }
180 }
181 LOG_INFO_FMT("{}: Storing JWT signing key with kid {}", log_prefix, kid);
182 new_keys.emplace(kid, der);
183 }
184 if (new_keys.empty())
185 {
186 LOG_FAIL_FMT("{}: no keys left after applying filter", log_prefix);
187 return false;
188 }
189
190 std::set<std::string> existing_kids;
191 key_issuer->foreach(
192 [&existing_kids, &issuer](const auto& kid, const auto& issuer_) {
193 if (issuer_ == issuer)
194 {
195 existing_kids.insert(kid);
196 }
197 return true;
198 });
199
200 for (auto& [kid, der] : new_keys)
201 {
202 if (!existing_kids.contains(kid))
203 {
204 keys->put(kid, der);
205 key_issuer->put(kid, issuer);
206 }
207 }
208
209 for (auto& kid : existing_kids)
210 {
211 if (!new_keys.contains(kid))
212 {
213 keys->remove(kid);
214 key_issuer->remove(kid);
215 }
216 }
217
218 return true;
219 }
