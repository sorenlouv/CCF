73 std::shared_ptr<ClientSession> create_unauthenticated_client()
74 {
75 // Note: server CA is not checked here as this client is not sending
76 // private data. If the server was malicious and the certificate chain was
77 // bogus, the verification of the endorsement of the quote would fail
78 // anyway.
79 return rpcsessions->create_client(std::make_shared<tls::Cert>(
80 nullptr, std::nullopt, std::nullopt, std::nullopt, false));

83 void send_request(
84 const std::shared_ptr<ClientSession>& client,
85 const EndpointInfo& endpoint)
86 {
87 {
88 http::Request r(endpoint.uri, HTTP_GET);
89 for (auto const& [k, v] : endpoint.params)
90 {
91 r.set_query_param(k, v);
92 }
93 r.set_header(http::headers::HOST, endpoint.host);
94
95 LOG_INFO_FMT(
96 "Fetching endorsements for attestation report at https://{}{}{}",
97 endpoint.host,
98 r.get_path(),
99 r.get_formatted_query());
100 client->send_request(std::move(r));
101 }
102
103 // Start watchdog to send request on new server if it is unresponsive
104 auto msg = std::make_unique<
105 threading::Tmsg<QuoteEndorsementsClientTimeoutMsg>>(
106 [](std::unique_ptr<threading::Tmsg<QuoteEndorsementsClientTimeoutMsg>>
107 msg) {
108 if (msg->data.self->has_completed)
109 {
110 return;
111 }
112 if (msg->data.request_id >= msg->data.self->last_received_request_id)
113 {
114 LOG_FAIL_FMT(
115 "Timed out reaching endorsement server {}",
116 msg->data.endpoint.host);
117
118 auto& servers = msg->data.self->config.servers;
119 msg->data.self->server_retries_count++;
120 if (
121 msg->data.self->server_retries_count >= max_server_retries_count)
122 {
123 if (servers.size() > 1)
124 {
125 // Move on to next server if we have passed max retries count
126 servers.pop_front();
127 }
128 else
129 {
130 auto& server = servers.front();
131 LOG_FAIL_FMT(
132 "Giving up retrying fetching attestation endorsements from "
133 "{} after {} attempts ",
134 server.front().host,
135 max_server_retries_count);
136 return;
137 }
138 }
139
140 msg->data.self->fetch(servers.front());
141 }
142 },
143 shared_from_this(),
144 endpoint,
145 last_received_request_id);
146
147 threading::ThreadMessaging::thread_messaging.add_task_after(
148 std::move(msg),
149 std::chrono::milliseconds(server_connection_timeout_s * 1000));
150 }

152 void handle_success_response(std::vector<uint8_t>&& data, bool is_der)
153 {
154 if (has_completed)
155 {
156 // We may receive a response to an in-flight request after having
157 // fetched all endorsements
158 return;
159 }
160
161 if (is_der)
162 {
163 auto raw = crypto::cert_der_to_pem(data).raw();
164 endorsements_pem.insert(endorsements_pem.end(), raw.begin(), raw.end());
165 }
166 else
167 {
168 endorsements_pem.insert(
169 endorsements_pem.end(), data.begin(), data.end());
170 }
171
172 auto& server = config.servers.front();
173 server.pop_front();
174 if (server.empty())
175 {
176 LOG_INFO_FMT("Complete endorsement chain successfully retrieved");
177 has_completed = true;
178 done_cb(std::move(endorsements_pem));
179 }
180 else
181 {
182 fetch(server);
183 }
184 }

186 void fetch(const Server& server)
187 {
188 auto& endpoint = server.front();
189
190 auto c = create_unauthenticated_client();
191 c->connect(
192 endpoint.host,
193 endpoint.port,
194 [this, server](
195 http_status status,
196 http::HeaderMap&& headers,
197 std::vector<uint8_t>&& data) {
198 last_received_request_id++;
199 auto& endpoint = server.front();
200
201 if (status == HTTP_STATUS_OK)
202 {
203 LOG_INFO_FMT(
204 "Successfully retrieved endorsements for attestation report: "
205 "{} bytes",
206 data.size());
207
208 handle_success_response(std::move(data), endpoint.response_is_der);
209 return;
210 }
211
212 LOG_DEBUG_FMT(
213 "Error fetching endorsements for attestation report: {}", status);
214 if (status == HTTP_STATUS_TOO_MANY_REQUESTS)
215 {
216 constexpr size_t default_retry_after_s = 3;
217 size_t retry_after_s = default_retry_after_s;
218 auto h = headers.find(http::headers::RETRY_AFTER);
219 if (h != headers.end())
220 {
221 const auto& retry_after_value = h->second;
222 // If value is invalid, retry_after_s is unchanged
223 std::from_chars(
224 retry_after_value.data(),
225 retry_after_value.data() + retry_after_value.size(),
226 retry_after_s);
227 }
228
229 auto msg =
230 std::make_unique<threading::Tmsg<QuoteEndorsementsClientMsg>>(
231 [](std::unique_ptr<threading::Tmsg<QuoteEndorsementsClientMsg>>
232 msg) { msg->data.self->fetch(msg->data.server); },
233 shared_from_this(),
234 server);
235
236 LOG_INFO_FMT(
237 "{} endorsements endpoint had too many requests. Retrying "
238 "in {}s",
239 endpoint.host,
240 retry_after_s);
241
242 threading::ThreadMessaging::thread_messaging.add_task_after(
243 std::move(msg), std::chrono::milliseconds(retry_after_s * 1000));
244 }
245 return;
246 },
247 [host = endpoint.host](const std::string& error_msg) {
248 LOG_FAIL_FMT(
249 "TLS error when connecting to quote endorsements endpoint {}: {}",
250 host,
251 error_msg);
252 });
253 send_request(c, endpoint);
254 }
