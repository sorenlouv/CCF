44 // Set the execution dependency for the snapshot to be the version
45 // previous to said snapshot to ensure that the correct snapshot is
46 // serialized.
47 // Notes:
48 // - Snapshots are always taken at compacted state so version only is
49 // unique enough to prevent IV reuse
50 // - Because snapshot generation and ledger rekey can be interleaved,
51 // consider historical ledger secrets when encrypting snapshot (see
52 // https://github.com/microsoft/CCF/issues/3796).
53 KvStoreSerialiser serialiser(
54 encryptor,
55 {0, version},
56 kv::EntryType::Snapshot,
57 0,
58 {},
59 ccf::no_claims(),
60 true /* historical_hint */);
61
62 if (hash_at_snapshot.has_value())
63 {
64 serialiser.serialise_raw(hash_at_snapshot.value());
65 }
66
67 if (view_history.has_value())
68 {
69 serialiser.serialise_view_history(view_history.value());
70 }
71
72 for (auto domain : {SecurityDomain::PUBLIC, SecurityDomain::PRIVATE})
73 {
74 for (const auto& it : snapshots)
75 {
76 if (it->get_security_domain() == domain)
77 {
78 it->serialise(serialiser);
79 }
80 }
81 }
82
83 return serialiser.get_raw_data();
84 }
