13 struct ProposalSubmissionResult
14 {
15 enum class Status
16 {
17 Acceptable,
18 DuplicateInWindow,
19 TooOld
20 } status;
21
22 // May be empty, a colliding proposal ID, or a min_created_at value,
23 // depending on status
24 std::string info = "";

27 ProposalSubmissionResult validate_proposal_submission_time(
28 kv::Tx& tx,
29 const std::string& created_at,
30 const std::vector<uint8_t>& request_digest,
31 const ccf::ProposalId& proposal_id)
32 {
33 auto cose_recent_proposals =
34 tx.rw<ccf::COSERecentProposals>(ccf::Tables::COSE_RECENT_PROPOSALS);
35 auto key = fmt::format("{}:{}", created_at, ds::to_hex(request_digest));
36
37 if (cose_recent_proposals->has(key))
38 {
39 auto colliding_proposal_id = cose_recent_proposals->get(key).value();
40 return {
41 ProposalSubmissionResult::Status::DuplicateInWindow,
42 colliding_proposal_id};
43 }
44
45 std::vector<std::string> replay_keys;
46 cose_recent_proposals->foreach_key(
47 [&replay_keys](const std::string& replay_key) {
48 replay_keys.push_back(replay_key);
49 return true;
50 });
51
52 std::sort(replay_keys.begin(), replay_keys.end());
53
54 // New proposal must be more recent than median proposal kept
55 if (!replay_keys.empty())
56 {
57 const auto [min_created_at, _] =
58 nonstd::split_1(replay_keys[replay_keys.size() / 2], ":");
59 auto [key_ts, __] = nonstd::split_1(key, ":");
60 if (key_ts < min_created_at)
61 {
62 return {
63 ProposalSubmissionResult::Status::TooOld,
64 std::string(min_created_at)};
65 }
66 }
67
68 size_t window_size = ccf::default_recent_cose_proposals_window_size;
69 auto config_handle =
70 tx.ro<ccf::Configuration>(ccf::Tables::CONFIGURATION);
71 auto config = config_handle->get();
72 if (
73 config.has_value() &&
74 config->recent_cose_proposals_window_size.has_value())
75 {
76 window_size = config->recent_cose_proposals_window_size.value();
77 }
78 cose_recent_proposals->put(key, proposal_id);
79 // Only keep the most recent window_size proposals, to avoid
80 // unbounded memory usage
81 if (replay_keys.size() >= (window_size - 1) /* We just added one */)
82 {
83 for (size_t i = 0; i < (replay_keys.size() - (window_size - 1)); i++)
84 {
85 cose_recent_proposals->remove(replay_keys[i]);
86 }
87 }
88 return {ProposalSubmissionResult::Status::Acceptable};
89 }

102 void remove_all_other_non_open_proposals(
103 kv::Tx& tx, const ProposalId& proposal_id)
104 {
105 auto p = tx.rw<ccf::jsgov::ProposalMap>(jsgov::Tables::PROPOSALS);
106 auto pi =
107 tx.rw<ccf::jsgov::ProposalInfoMap>(jsgov::Tables::PROPOSALS_INFO);
108 std::vector<ProposalId> to_be_removed;
109 pi->foreach(
110 [&to_be_removed, &proposal_id](
111 const ProposalId& pid, const ccf::jsgov::ProposalInfo& pinfo) {
112 if (pid != proposal_id && pinfo.state != ProposalState::OPEN)
113 {
114 to_be_removed.push_back(pid);
115 }
116 return true;
117 });
118 for (const auto& pr : to_be_removed)
119 {
120 p->remove(pr);
121 pi->remove(pr);
122 }
123 }

125 ccf::jsgov::ProposalInfoSummary resolve_proposal(
126 ccfapp::AbstractNodeContext& context,
127 ccf::NetworkState& network,
128 kv::Tx& tx,
129 const ProposalId& proposal_id,
130 const std::span<const uint8_t>& proposal,
131 ccf::jsgov::ProposalInfo& proposal_info,
132 const std::string& constitution)
133 {
134 std::vector<std::pair<MemberId, bool>> votes;
135 std::optional<jsgov::Failure> failure = std::nullopt;
136 std::optional<ccf::jsgov::Votes> final_votes = std::nullopt;
137 std::optional<ccf::jsgov::VoteFailures> vote_failures = std::nullopt;
138
139 // Evaluate ballots
140 for (const auto& [mid, mb] : proposal_info.ballots)
141 {
142 js::Context js_context(js::TxAccess::GOV_RO);
143 js::TxContext txctx{&tx};
144 js::populate_global_ccf_kv(&txctx, js_context);
145 auto ballot_func = js_context.function(
146 mb,
147 "vote",
148 fmt::format(
149 "public:ccf.gov.proposal_info[{}].ballots[{}]", proposal_id, mid));
150
151 std::vector<js::JSWrappedValue> argv = {
152 js_context.new_string_len(
153 (const char*)proposal.data(), proposal.size()),
154 js_context.new_string_len(
155 proposal_info.proposer_id.data(),
156 proposal_info.proposer_id.size())};
157
158 auto val = js_context.call_with_rt_options(
159 ballot_func,
160 argv,
161 &tx,
162 js::RuntimeLimitsPolicy::NO_LOWER_THAN_DEFAULTS);
163
164 if (!val.is_exception())
165 {
166 votes.emplace_back(mid, val.is_true());
167 }
168 else
169 {
170 if (!vote_failures.has_value())
171 {
172 vote_failures = ccf::jsgov::VoteFailures();
173 }
174
175 auto [reason, trace] = js::js_error_message(js_context);
176
177 if (js_context.interrupt_data.request_timed_out)
178 {
179 reason = "Operation took too long to complete.";
180 }
181 vote_failures.value()[mid] = ccf::jsgov::Failure{reason, trace};
182 }
183 }
184
185 // Evaluate resolve function
186 {
187 {
188 js::Context js_context(js::TxAccess::GOV_RO);
189 js::TxContext txctx{&tx};
190 js::populate_global_ccf_kv(&txctx, js_context);
191 auto resolve_func = js_context.function(
192 constitution, "resolve", "public:ccf.gov.constitution[0]");
193
194 std::vector<js::JSWrappedValue> argv;
195 argv.push_back(js_context.new_string_len(
196 (const char*)proposal.data(), proposal.size()));
197
198 argv.push_back(js_context.new_string_len(
199 proposal_info.proposer_id.data(),
200 proposal_info.proposer_id.size()));
201
202 auto vs = js_context.new_array();
203 size_t index = 0;
204 for (auto& [mid, vote] : votes)
205 {
206 auto v = JS_NewObject(js_context);
207 auto member_id =
208 JS_NewStringLen(js_context, mid.data(), mid.size());
209 JS_DefinePropertyValueStr(
210 js_context, v, "member_id", member_id, JS_PROP_C_W_E);
211 auto vote_status = JS_NewBool(js_context, vote);
212 JS_DefinePropertyValueStr(
213 js_context, v, "vote", vote_status, JS_PROP_C_W_E);
214 JS_DefinePropertyValueUint32(
215 js_context, vs, index++, v, JS_PROP_C_W_E);
216 }
217 argv.push_back(vs);
218
219 auto val = js_context.call_with_rt_options(
220 resolve_func,
221 argv,
222 &tx,
223 js::RuntimeLimitsPolicy::NO_LOWER_THAN_DEFAULTS);
224
225 if (JS_IsException(val))
226 {
227 proposal_info.state = ProposalState::FAILED;
228 auto [reason, trace] = js::js_error_message(js_context);
229 if (js_context.interrupt_data.request_timed_out)
230 {
231 reason = "Operation took too long to complete.";
232 }
233 failure = ccf::jsgov::Failure{
234 fmt::format("Failed to resolve(): {}", reason), trace};
235 }
236 else
237 {
238 auto status = js_context.to_str(val).value_or("");
239 // NB: It is not possible to produce every possible ProposalState
240 // here! WITHDRAWN and DROPPED are states that we transition to
241 // elsewhere, but not valid return values from resolve()
242 const std::unordered_map<std::string, ProposalState>
243 js_str_to_status = {
244 {"Open", ProposalState::OPEN},
245 {"Accepted", ProposalState::ACCEPTED},
246 {"Rejected", ProposalState::REJECTED}};
247 const auto it = js_str_to_status.find(status);
248 if (it != js_str_to_status.end())
249 {
250 proposal_info.state = it->second;
251 }
252 else
253 {
254 proposal_info.state = ProposalState::FAILED;
255 failure = ccf::jsgov::Failure{
256 fmt::format(
257 "resolve() returned invalid status value: \"{}\"", status),
258 std::nullopt // No trace
259 };
260 }
261 }
262 }
263
264 if (proposal_info.state != ProposalState::OPEN)
265 {
266 remove_all_other_non_open_proposals(tx, proposal_id);
267 final_votes = std::unordered_map<ccf::MemberId, bool>();
268 for (auto& [mid, vote] : votes)
269 {
270 final_votes.value()[mid] = vote;
271 }
272
273 if (proposal_info.state == ProposalState::ACCEPTED)
274 {
275 // Evaluate apply function
276 js::Context js_context(js::TxAccess::GOV_RW);
277 js::TxContext txctx{&tx};
278
279 auto gov_effects =
280 context.get_subsystem<AbstractGovernanceEffects>();
281 if (gov_effects == nullptr)
282 {
283 throw std::logic_error(
284 "Unexpected: Could not access GovEffects subsytem");
285 }
286
287 js::populate_global_ccf_kv(&txctx, js_context);
288 js::populate_global_ccf_node(gov_effects.get(), js_context);
289 js::populate_global_ccf_network(&network, js_context);
290 js::populate_global_ccf_gov_actions(js_context);
291
292 auto apply_func = js_context.function(
293 constitution, "apply", "public:ccf.gov.constitution[0]");
294
295 std::vector<js::JSWrappedValue> argv = {
296 js_context.new_string_len(
297 (const char*)proposal.data(), proposal.size()),
298 js_context.new_string_len(
299 proposal_id.c_str(), proposal_id.size())};
300
301 auto val = js_context.call_with_rt_options(
302 apply_func,
303 argv,
304 &tx,
305 js::RuntimeLimitsPolicy::NO_LOWER_THAN_DEFAULTS);
306
307 if (JS_IsException(val))
308 {
309 proposal_info.state = ProposalState::FAILED;
310 auto [reason, trace] = js::js_error_message(js_context);
311 if (js_context.interrupt_data.request_timed_out)
312 {
313 reason = "Operation took too long to complete.";
314 }
315 failure = ccf::jsgov::Failure{
316 fmt::format("Failed to apply(): {}", reason), trace};
317 }
318 }
319 }
320
321 return jsgov::ProposalInfoSummary{
322 proposal_id,
323 proposal_info.proposer_id,
324 proposal_info.state,
325 proposal_info.ballots.size(),
326 final_votes,
327 vote_failures,
328 failure};
329 }
330 }

332 // We have several structurally-similar types for representing proposals (in
333 // KV vs post-exec vs fully expanded). This aims to take any, and produce
334 // the same API-compatible description.
335 template <typename TProposal>
336 nlohmann::json convert_proposal_to_api_format(const TProposal& summary)
337 {
338 auto response_body = nlohmann::json::object();
339
340 response_body["proposerId"] = summary.proposer_id;
341 response_body["proposalState"] = summary.state;
342
343 std::optional<ccf::jsgov::Votes> votes;
344
345 if constexpr (std::is_same_v<TProposal, ccf::jsgov::ProposalInfoSummary>)
346 {
347 response_body["proposalId"] = summary.proposal_id;
348 response_body["ballotCount"] = summary.ballot_count;
349
350 votes = summary.votes;
351 }
352 else if constexpr (std::is_same_v<TProposal, ccf::jsgov::ProposalInfo>)
353 {
354 response_body["ballotCount"] = summary.ballots.size();
355
356 votes = summary.final_votes;
357 }
358
359 if (votes.has_value())
360 {
361 auto final_votes = nlohmann::json::object();
362 for (const auto& [voter_id, vote_result] : *votes)
363 {
364 final_votes[voter_id.value()] = vote_result;
365 }
366 response_body["finalVotes"] = final_votes;
367 }
368
369 if (summary.vote_failures.has_value())
370 {
371 auto vote_failures = nlohmann::json::object();
372 for (const auto& [failer_id, failure] : *summary.vote_failures)
373 {
374 vote_failures[failer_id.value()] = failure;
375 }
376 response_body["voteFailures"] = vote_failures;
377 }
378
379 if (summary.failure.has_value())
380 {
381 auto failure = nlohmann::json::object();
382 response_body["failure"] = *summary.failure;
383 }
384
385 return response_body;
386 }

395 auto create_proposal = [&](auto& ctx, ApiVersion api_version) {
396 switch (api_version)
397 {
398 case ApiVersion::preview_v1:
399 default:
400 {
401 std::span<const uint8_t> proposal_body;
402 ccf::jsgov::ProposalInfo proposal_info;
403 std::optional<std::string> constitution;
404
405 const auto& cose_ident =
406 ctx.template get_caller<ccf::MemberCOSESign1AuthnIdentity>();
407
408 // Construct proposal_id, as digest of request and root
409 ProposalId proposal_id;
410 std::vector<uint8_t> request_digest;
411 {
412 auto root_at_read = ctx.tx.get_root_at_read_version();
413 if (!root_at_read.has_value())
414 {
415 detail::set_gov_error(
416 ctx.rpc_ctx,
417 HTTP_STATUS_INTERNAL_SERVER_ERROR,
418 ccf::errors::InternalError,
419 "Proposal failed to bind to state.");
420 return;
421 }
422
423 auto hasher = crypto::make_incremental_sha256();
424 hasher->update_hash(root_at_read.value().h);
425
426 request_digest = crypto::sha256(
427 cose_ident.signature.data(), cose_ident.signature.size());
428
429 hasher->update_hash(request_digest);
430
431 const crypto::Sha256Hash proposal_hash = hasher->finalise();
432 proposal_id = proposal_hash.hex_str();
433 }
434
435 // Validate proposal, by calling into JS constitution
436 {
437 constitution =
438 ctx.tx.template ro<ccf::Constitution>(ccf::Tables::CONSTITUTION)
439 ->get();
440 if (!constitution.has_value())
441 {
442 detail::set_gov_error(
443 ctx.rpc_ctx,
444 HTTP_STATUS_INTERNAL_SERVER_ERROR,
445 ccf::errors::InternalError,
446 "No constitution is set - proposals cannot be evaluated");
447 return;
448 }
449
450 js::Context context(js::TxAccess::GOV_RO);
451 js::TxContext txctx{&ctx.tx};
452 js::populate_global_ccf_kv(&txctx, context);
453
454 auto validate_func = context.function(
455 constitution.value(),
456 "validate",
457 "public:ccf.gov.constitution[0]");
458
459 proposal_body = cose_ident.content;
460 auto proposal_arg = context.new_string_len(
461 (const char*)proposal_body.data(), proposal_body.size());
462 auto validate_result = context.call_with_rt_options(
463 validate_func,
464 {proposal_arg},
465 &ctx.tx,
466 js::RuntimeLimitsPolicy::NO_LOWER_THAN_DEFAULTS);
467
468 // Handle error cases of validation
469 {
470 if (validate_result.is_exception())
471 {
472 auto [reason, trace] = js_error_message(context);
473 if (context.interrupt_data.request_timed_out)
474 {
475 reason = "Operation took too long to complete.";
476 }
477 detail::set_gov_error(
478 ctx.rpc_ctx,
479 HTTP_STATUS_INTERNAL_SERVER_ERROR,
480 ccf::errors::InternalError,
481 fmt::format(
482 "Failed to execute validation: {} {}",
483 reason,
484 trace.value_or("")));
485 return;
486 }
487
488 if (!validate_result.is_obj())
489 {
490 detail::set_gov_error(
491 ctx.rpc_ctx,
492 HTTP_STATUS_INTERNAL_SERVER_ERROR,
493 ccf::errors::InternalError,
494 "Validation failed to return an object");
495 return;
496 }
497
498 std::string description;
499 auto desc = validate_result["description"];
500 if (desc.is_str())
501 {
502 description = context.to_str(desc).value_or("");
503 }
504
505 auto valid = validate_result["valid"];
506 if (!valid.is_true())
507 {
508 detail::set_gov_error(
509 ctx.rpc_ctx,
510 HTTP_STATUS_BAD_REQUEST,
511 ccf::errors::ProposalFailedToValidate,
512 fmt::format("Proposal failed to validate: {}", description));
513 return;
514 }
515 }
516
517 // Write proposal to KV
518 {
519 auto proposals_handle =
520 ctx.tx.template rw<ccf::jsgov::ProposalMap>(
521 jsgov::Tables::PROPOSALS);
522 // Introduce a read dependency, so that if identical proposal
523 // creations are in-flight and reading at the same version, all
524 // except the first conflict and are re-executed. If we ever
525 // produce a proposal ID which already exists, we must have a
526 // hash collision.
527 if (proposals_handle->has(proposal_id))
528 {
529 detail::set_gov_error(
530 ctx.rpc_ctx,
531 HTTP_STATUS_INTERNAL_SERVER_ERROR,
532 ccf::errors::InternalError,
533 "Proposal ID collision.");
534 return;
535 }
536 proposals_handle->put(
537 proposal_id, {proposal_body.begin(), proposal_body.end()});
538
539 auto proposal_info_handle =
540 ctx.tx.template wo<ccf::jsgov::ProposalInfoMap>(
541 jsgov::Tables::PROPOSALS_INFO);
542
543 proposal_info.proposer_id = cose_ident.member_id;
544 proposal_info.state = ccf::ProposalState::OPEN;
545
546 proposal_info_handle->put(proposal_id, proposal_info);
547
548 detail::record_cose_governance_history(
549 ctx.tx, cose_ident.member_id, cose_ident.envelope);
550 }
551 }
552
553 // Validate proposal's created_at time
554 {
555 // created_at, submitted as a binary integer number of seconds
556 // since epoch in the COSE Sign1 envelope, is converted to a
557 // decimal representation in ASCII, stored as a string, and
558 // compared alphanumerically. This is partly to keep governance as
559 // text-based as possible, to faciliate audit, but also to be able
560 // to benefit from future planned ordering support in the KV. To
561 // compare correctly, the string representation needs to be padded
562 // with leading zeroes, and must therefore not exceed a fixed
563 // digit width. 10 digits is enough to last until November 2286,
564 // ie. long enough.
565 if (cose_ident.protected_header.gov_msg_created_at > 9'999'999'999)
566 {
567 detail::set_gov_error(
568 ctx.rpc_ctx,
569 HTTP_STATUS_BAD_REQUEST,
570 ccf::errors::InvalidCreatedAt,
571 "Header parameter created_at value is too large");
572 return;
573 }
574
575 const auto created_at_str = fmt::format(
576 "{:0>10}", cose_ident.protected_header.gov_msg_created_at);
577
578 ccf::ProposalId colliding_proposal_id;
579 std::string min_created_at;
580
581 const auto subtime_result =
582 detail::validate_proposal_submission_time(
583 ctx.tx, created_at_str, request_digest, proposal_id);
584 switch (subtime_result.status)
585 {
586 case detail::ProposalSubmissionResult::Status::TooOld:
587 {
588 detail::set_gov_error(
589 ctx.rpc_ctx,
590 HTTP_STATUS_BAD_REQUEST,
591 ccf::errors::ProposalCreatedTooLongAgo,
592 fmt::format(
593 "Proposal created too long ago, created_at must be greater "
594 "than {}",
595 subtime_result.info));
596 return;
597 }
598
599 case detail::ProposalSubmissionResult::Status::DuplicateInWindow:
600 {
601 detail::set_gov_error(
602 ctx.rpc_ctx,
603 HTTP_STATUS_BAD_REQUEST,
604 ccf::errors::ProposalReplay,
605 fmt::format(
606 "Proposal submission replay, already exists as proposal {}",
607 subtime_result.info));
608 return;
609 }
610
611 case detail::ProposalSubmissionResult::Status::Acceptable:
612 {
613 break;
614 }
615
616 default:
617 {
618 throw std::runtime_error(
619 "Invalid ProposalSubmissionStatus value");
620 }
621 }
622 }
623
624 // Resolve proposal (may pass immediately)
625 {
626 const auto resolve_result = detail::resolve_proposal(
627 node_context,
628 network,
629 ctx.tx,
630 proposal_id,
631 proposal_body,
632 proposal_info,
633 constitution.value());
634
635 if (resolve_result.state == ProposalState::FAILED)
636 {
637 // If the proposal failed to apply, we want to discard the tx and
638 // not apply its side-effects to the KV state, because it may have
639 // failed mid-execution (eg - thrown an exception), in which case
640 // we do not want to apply partial writes. Note this differs from
641 // a failure that happens after a vote, in that this proposal is
642 // not recorded in the KV at all.
643 detail::set_gov_error(
644 ctx.rpc_ctx,
645 HTTP_STATUS_INTERNAL_SERVER_ERROR,
646 ccf::errors::InternalError,
647 fmt::format("{}", resolve_result.failure));
648 return;
649 }
650
651 // Write updated proposal info
652 {
653 auto proposal_info_handle =
654 ctx.tx.template wo<ccf::jsgov::ProposalInfoMap>(
655 jsgov::Tables::PROPOSALS_INFO);
656
657 proposal_info.state = resolve_result.state;
658 proposal_info.failure = resolve_result.failure;
659
660 proposal_info_handle->put(proposal_id, proposal_info);
661 }
662
663 const auto response_body =
664 detail::convert_proposal_to_api_format(resolve_result);
665
666 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
667 return;
668 }
669 }
670 }
671 };

673 .make_endpoint(

675 HTTP_POST,
676 api_version_adapter(create_proposal),

678 .set_openapi_hidden(true)

681 auto withdraw_proposal = [&](auto& ctx, ApiVersion api_version) {
682 switch (api_version)
683 {
684 case ApiVersion::preview_v1:
685 default:
686 {
687 const auto& cose_ident =
688 ctx.template get_caller<ccf::MemberCOSESign1AuthnIdentity>();
689 ccf::ProposalId proposal_id;
690
691 if (!detail::try_parse_signed_proposal_id(
692 cose_ident, ctx.rpc_ctx, proposal_id))
693 {
694 return;
695 }
696
697 auto proposal_info_handle =
698 ctx.tx.template rw<ccf::jsgov::ProposalInfoMap>(
699 jsgov::Tables::PROPOSALS_INFO);
700
701 // Check proposal exists
702 auto proposal_info = proposal_info_handle->get(proposal_id);
703 if (!proposal_info.has_value())
704 {
705 // If it doesn't, then withdrawal is idempotent - we don't know if
706 // this previously existed or not, was withdrawn or accepted, but
707 // return a 204
708 ctx.rpc_ctx->set_response_status(HTTP_STATUS_NO_CONTENT);
709 return;
710 }
711
712 // Check caller is proposer
713 const auto member_id = cose_ident.member_id;
714 if (member_id != proposal_info->proposer_id)
715 {
716 detail::set_gov_error(
717 ctx.rpc_ctx,
718 HTTP_STATUS_FORBIDDEN,
719 ccf::errors::AuthorizationFailed,
720 fmt::format(
721 "Proposal {} can only be withdrawn by proposer {}, not caller "
722 "{}.",
723 proposal_id,
724 proposal_info->proposer_id,
725 member_id));
726 return;
727 }
728
729 // If proposal is still known, and state is neither OPEN nor
730 // WITHDRAWN, return an error - caller has done something wrong
731 if (
732 proposal_info->state != ProposalState::OPEN &&
733 proposal_info->state != ProposalState::WITHDRAWN)
734 {
735 detail::set_gov_error(
736 ctx.rpc_ctx,
737 HTTP_STATUS_BAD_REQUEST,
738 ccf::errors::ProposalNotOpen,
739 fmt::format(
740 "Proposal {} is currently in state {} and cannot be withdrawn.",
741 proposal_id,
742 proposal_info->state));
743 return;
744 }
745

901 auto submit_ballot =
902 [&](ccf::endpoints::EndpointContext& ctx, ApiVersion api_version) {
903 switch (api_version)
904 {
905 case ApiVersion::preview_v1:
906 default:
907 {
908 const auto& cose_ident =
909 ctx.template get_caller<ccf::MemberCOSESign1AuthnIdentity>();
910
911 ccf::ProposalId proposal_id;
912 if (!detail::try_parse_signed_proposal_id(
913 cose_ident, ctx.rpc_ctx, proposal_id))
914 {
915 return;
916 }
917
918 ccf::MemberId member_id;
919 if (!detail::try_parse_signed_member_id(
920 cose_ident, ctx.rpc_ctx, member_id))
921 {
922 return;
923 }
924
925 // Look up proposal info and check expected state
926 auto proposal_info_handle =
927 ctx.tx.template rw<ccf::jsgov::ProposalInfoMap>(
928 jsgov::Tables::PROPOSALS_INFO);
929 auto proposal_info = proposal_info_handle->get(proposal_id);
930 if (!proposal_info.has_value())
931 {
932 detail::set_gov_error(
933 ctx.rpc_ctx,
934 HTTP_STATUS_NOT_FOUND,
935 ccf::errors::ProposalNotFound,
936 fmt::format("Could not find proposal {}.", proposal_id));
937 return;
938 }
939
940 if (proposal_info->state != ccf::ProposalState::OPEN)
941 {
942 detail::set_gov_error(
943 ctx.rpc_ctx,
944 HTTP_STATUS_BAD_REQUEST,
945 ccf::errors::ProposalNotOpen,
946 fmt::format(
947 "Proposal {} is currently in state {} - only {} proposals "
948 "can receive votes",
949 proposal_id,
950 proposal_info->state,
951 ProposalState::OPEN));
952 return;
953 }
954
955 // Look up proposal contents
956 auto proposals_handle = ctx.tx.template ro<ccf::jsgov::ProposalMap>(
957 ccf::jsgov::Tables::PROPOSALS);
958 const auto proposal = proposals_handle->get(proposal_id);
959 if (!proposal.has_value())
960 {
961 detail::set_gov_error(
962 ctx.rpc_ctx,
963 HTTP_STATUS_NOT_FOUND,
964 ccf::errors::ProposalNotFound,
965 fmt::format("Could not find proposal {}.", proposal_id));
966 return;
967 }
968
969 // Parse and validate incoming ballot
970 const auto params = nlohmann::json::parse(cose_ident.content);
971 const auto ballot_it = params.find("ballot");
972 if (ballot_it == params.end() || !ballot_it.value().is_string())
973 {
974 detail::set_gov_error(
975 ctx.rpc_ctx,
976 HTTP_STATUS_BAD_REQUEST,
977 ccf::errors::InvalidInput,
978 "Signed request body is not a JSON object containing required "
979 "string field \"ballot\"");
980 return;
981 }
982
983 // Access constitution to evaluate ballots
984 const auto constitution =
985 ctx.tx.template ro<ccf::Constitution>(ccf::Tables::CONSTITUTION)
986 ->get();
987 if (!constitution.has_value())
988 {
989 detail::set_gov_error(
990 ctx.rpc_ctx,
991 HTTP_STATUS_INTERNAL_SERVER_ERROR,
992 ccf::errors::InternalError,
993 "No constitution is set - ballots cannot be evaluated");
994 return;
995 }
996
997 const auto ballot = ballot_it.value().get<std::string>();
998
999 const auto info_ballot_it = proposal_info->ballots.find(member_id);
1000 if (info_ballot_it != proposal_info->ballots.end())
1001 {
1002 // If ballot matches previously submitted, aim for idempotent
1003 // matching response
1004 if (info_ballot_it->second == ballot)
1005 {
1006 const auto response_body =
1007 detail::convert_proposal_to_api_format(proposal_info.value());
1008
1009 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
1010 return;
1011 }
1012 else
1013 {
1014 detail::set_gov_error(
1015 ctx.rpc_ctx,
1016 HTTP_STATUS_BAD_REQUEST,
1017 ccf::errors::VoteAlreadyExists,
1018 fmt::format(
1019 "Different ballot already submitted by {} for {}.",
1020 member_id,
1021 proposal_id));
1022 return;
1023 }
1024 }
1025
1026 // Store newly provided ballot
1027 proposal_info->ballots.insert_or_assign(
1028 info_ballot_it, member_id, ballot_it.value().get<std::string>());
1029
1030 detail::record_cose_governance_history(
1031 ctx.tx, cose_ident.member_id, cose_ident.envelope);
1032
1033 const auto resolve_result = detail::resolve_proposal(
1034 node_context,
1035 network,
1036 ctx.tx,
1037 proposal_id,
1038 proposal.value(),
1039 proposal_info.value(),
1040 constitution.value());
1041
1042 if (resolve_result.state == ProposalState::FAILED)
1043 {
1044 // If the proposal failed to apply, we want to discard the tx and
1045 // not apply its side-effects to the KV state, because it may have
1046 // failed mid-execution (eg - thrown an exception), in which case
1047 // we do not want to apply partial writes
1048 detail::set_gov_error(
1049 ctx.rpc_ctx,
1050 HTTP_STATUS_INTERNAL_SERVER_ERROR,
1051 ccf::errors::InternalError,
1052 fmt::format("{}", resolve_result.failure));
1053 return;
1054 }
1055
1056 // Write updated proposal info
1057 {
1058 proposal_info->state = resolve_result.state;
1059 proposal_info->failure = resolve_result.failure;
1060
1061 proposal_info_handle->put(proposal_id, proposal_info.value());
1062 }
1063
1064 const auto response_body =
1065 detail::convert_proposal_to_api_format(resolve_result);
1066
1067 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
1068 return;
1069 }
1070 }
1071 };
