212 CreateNodeStatus create_new_node(
213 StartType start_type_,
214 StartupConfig&& ccf_config_,
215 std::vector<uint8_t>&& startup_snapshot,
216 uint8_t* node_cert,
217 size_t node_cert_size,
218 size_t* node_cert_len,
219 uint8_t* service_cert,
220 size_t service_cert_size,
221 size_t* service_cert_len)
222 {
223 // node_cert_size and service_cert_size are ignored here, but we pass them
224 // in because it allows us to set EDL an annotation so that node_cert_len
225 // <= node_cert_size is checked by the EDL-generated wrapper
226
227 start_type = start_type_;
228
229 rpcsessions->update_listening_interface_options(ccf_config_.network);
230
231 node->set_n2n_message_limit(ccf_config_.node_to_node_message_limit);
232
233 // If we haven't heard from a node for multiple elections, then cleanup
234 // their node-to-node channel
235 const auto idle_timeout =
236 std::chrono::milliseconds(ccf_config_.consensus.election_timeout) * 4;
237 node->set_n2n_idle_timeout(idle_timeout);
238
239 ccf::NodeCreateInfo r;
240 try
241 {
242 LOG_TRACE_FMT(
243 "Creating node with start_type {}", start_type_to_str(start_type));
244 r = node->create(
245 start_type, std::move(ccf_config_), std::move(startup_snapshot));
246 }
247 catch (const std::exception& e)
248 {
249 LOG_FAIL_FMT("Error starting node: {}", e.what());
250 return CreateNodeStatus::InternalError;
251 }
252
253 // Copy node and service certs out
254 if (r.self_signed_node_cert.size() > node_cert_size)
255 {
256 LOG_FAIL_FMT(
257 "Insufficient space ({}) to copy node_cert out ({})",
258 node_cert_size,
259 r.self_signed_node_cert.size());
260 return CreateNodeStatus::InternalError;
261 }
262 pal::safe_memcpy(
263 node_cert,
264 r.self_signed_node_cert.data(),
265 r.self_signed_node_cert.size());
266 *node_cert_len = r.self_signed_node_cert.size();
267
268 if (start_type == StartType::Start || start_type == StartType::Recover)
269 {
270 // When starting a node in start or recover modes, fresh network secrets
271 // are created and the associated certificate can be passed to the host
272 if (r.service_cert.size() > service_cert_size)
273 {
274 LOG_FAIL_FMT(
275 "Insufficient space ({}) to copy service_cert out ({})",
276 service_cert_size,
277 r.service_cert.size());
278 return CreateNodeStatus::InternalError;
279 }
280 pal::safe_memcpy(
281 service_cert, r.service_cert.data(), r.service_cert.size());
282 *service_cert_len = r.service_cert.size();
283 }
284
285 return CreateNodeStatus::OK;
286 }

288 bool run_main()
289 {
290 crypto::openssl_sha256_init();
291 LOG_DEBUG_FMT("Running main thread");
292#ifndef VIRTUAL_ENCLAVE
293 try
294#endif
295 {
296 messaging::BufferProcessor bp("Enclave");
297
298 // reconstruct oversized messages sent to the enclave
299 oversized::FragmentReconstructor fr(bp.get_dispatcher());
300
301 lfs_access->register_message_handlers(bp.get_dispatcher());
302
303 DISPATCHER_SET_MESSAGE_HANDLER(
304 bp, AdminMessage::stop, [&bp](const uint8_t*, size_t) {
305 bp.set_finished();
306 threading::ThreadMessaging::instance().set_finished();
307 });
308
309 DISPATCHER_SET_MESSAGE_HANDLER(
310 bp, AdminMessage::stop_notice, [this](const uint8_t*, size_t) {
311 node->stop_notice();
312 });
313
314 last_tick_time = ccf::get_enclave_time();
315
316 DISPATCHER_SET_MESSAGE_HANDLER(
317 bp,
318 AdminMessage::tick,
319 [this, &disp = bp.get_dispatcher()](const uint8_t*, size_t) {
320 const auto message_counts = disp.retrieve_message_counts();
321 const auto j = disp.convert_message_counts(message_counts);
322 RINGBUFFER_WRITE_MESSAGE(
323 AdminMessage::work_stats, to_host, j.dump());
324
325 const auto time_now = ccf::get_enclave_time();
326 ringbuffer_logger->set_time(time_now);
327
328 const auto elapsed_ms =
329 std::chrono::duration_cast<std::chrono::milliseconds>(
330 time_now - last_tick_time);
331 if (elapsed_ms.count() > 0)
332 {
333 last_tick_time += elapsed_ms;
334
335 node->tick(elapsed_ms);
336 historical_state_cache->tick(elapsed_ms);
337 threading::ThreadMessaging::instance().tick(elapsed_ms);
338 // When recovering, no signature should be emitted while the
339 // public ledger is being read
340 if (!node->is_reading_public_ledger())
341 {
342 for (auto& [actor, frontend] : rpc_map->frontends())
343 {
344 frontend->tick(elapsed_ms);
345 }
346 }
347 node->tick_end();
348 }
349 });
350
351 DISPATCHER_SET_MESSAGE_HANDLER(
352 bp, ccf::node_inbound, [this](const uint8_t* data, size_t size) {
353 try
354 {
355 node->recv_node_inbound(data, size);
356 }
357 catch (const std::exception& e)
358 {
359 LOG_DEBUG_FMT(
360 "Ignoring node_inbound message due to exception: {}", e.what());
361 }
362 });
363
364 DISPATCHER_SET_MESSAGE_HANDLER(
365 bp,
366 consensus::ledger_entry_range,
367 [this](const uint8_t* data, size_t size) {
368 const auto [from_seqno, to_seqno, purpose, body] =
369 ringbuffer::read_message<consensus::ledger_entry_range>(
370 data, size);
371 switch (purpose)
372 {
373 case consensus::LedgerRequestPurpose::Recovery:
374 {
375 if (node->is_reading_public_ledger())
376 {
377 node->recover_public_ledger_entries(body);
378 }
379 else if (node->is_reading_private_ledger())
380 {
381 node->recover_private_ledger_entries(body);
382 }
383 else
384 {
385 auto [s, _, __] = node->state();
386 LOG_FAIL_FMT(
387 "Cannot recover ledger entry: Unexpected node state {}", s);
388 }
389 break;
390 }
391 case consensus::LedgerRequestPurpose::HistoricalQuery:
392 {
393 historical_state_cache->handle_ledger_entries(
394 from_seqno, to_seqno, body);
395 break;
396 }
397 default:
398 {
399 LOG_FAIL_FMT("Unhandled purpose: {}", purpose);
400 }
401 }
402 });
403
404 DISPATCHER_SET_MESSAGE_HANDLER(
405 bp,
406 consensus::ledger_no_entry_range,
407 [this](const uint8_t* data, size_t size) {
408 const auto [from_seqno, to_seqno, purpose] =
409 ringbuffer::read_message<consensus::ledger_no_entry_range>(
410 data, size);
411 switch (purpose)
412 {
413 case consensus::LedgerRequestPurpose::Recovery:
414 {
415 node->recover_ledger_end();
416 break;
417 }
418 case consensus::LedgerRequestPurpose::HistoricalQuery:
419 {
420 historical_state_cache->handle_no_entry_range(
421 from_seqno, to_seqno);
422 break;
423 }
424 default:
425 {
426 LOG_FAIL_FMT("Unhandled purpose: {}", purpose);
427 }
428 }
429 });
430
431 DISPATCHER_SET_MESSAGE_HANDLER(
432 bp,
433 consensus::snapshot_allocated,
434 [this](const uint8_t* data, size_t size) {
435 const auto [snapshot_span, generation_count] =
436 ringbuffer::read_message<consensus::snapshot_allocated>(
437 data, size);
438
439 node->write_snapshot(snapshot_span, generation_count);
440 });
441
442 rpcsessions->register_message_handlers(bp.get_dispatcher());
443
444 // Maximum number of inbound ringbuffer messages which will be
445 // processed in a single iteration
446 static constexpr size_t max_messages = 256;
447
448 size_t consecutive_idles = 0u;
449 while (!bp.get_finished())
450 {
451 // First, read some messages from the ringbuffer
452 auto read = bp.read_n(max_messages, circuit->read_from_outside());
453
454 // Then, execute some thread messages
455 size_t thread_msg = 0;
456 while (thread_msg < max_messages &&
457 threading::ThreadMessaging::instance().run_one())
458 {
459 thread_msg++;
460 }
461
462 // If no messages were read from the ringbuffer and no thread
463 // messages were executed, idle
464 if (read == 0 && thread_msg == 0)
465 {
466 const auto time_now = ccf::get_enclave_time();
467 static std::chrono::microseconds idling_start_time;
468
469 if (consecutive_idles == 0)
470 {
471 idling_start_time = time_now;
472 }
473
474 // Handle initial idles by pausing, eventually sleep (in host)
475 constexpr std::chrono::milliseconds timeout(5);
476 if ((time_now - idling_start_time) > timeout)
477 {
478 std::this_thread::sleep_for(timeout * 10);
479 }
480 else
481 {
482 CCF_PAUSE();
483 }
484
485 consecutive_idles++;
486 }
487 else
488 {
489 // If some messages were read, reset consecutive idles count
490 consecutive_idles = 0;
491 }
492 }
493
494 LOG_INFO_FMT("Enclave stopped successfully. Stopping host...");
495 RINGBUFFER_WRITE_MESSAGE(AdminMessage::stopped, to_host);
496
497 crypto::openssl_sha256_shutdown();
498
499 return true;
500 }
501#ifndef VIRTUAL_ENCLAVE
502 catch (const std::exception& e)
503 {
504 // It is expected that all enclave modules consuming ring buffer
505 // messages catch any thrown exception they can recover from. Uncaught
506 // exceptions bubble up to here and cause the node to shutdown.
507 RINGBUFFER_WRITE_MESSAGE(
508 AdminMessage::fatal_error_msg, to_host, std::string(e.what()));
509 crypto::openssl_sha256_shutdown();
510 return false;
511 }
512#endif
513 }
