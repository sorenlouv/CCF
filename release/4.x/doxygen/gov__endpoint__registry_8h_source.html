53 void api_endpoint(ccf::endpoints::ReadOnlyEndpointContext& ctx) override
54 {
55 using namespace ccf::gov::endpoints;
56 const auto api_version = get_api_version(ctx);
57 if (api_version.has_value())
58 {
59 switch (api_version.value())
60 {
61 case ApiVersion::preview_v1:
62 {
63 ctx.rpc_ctx->set_response_body(schema::v2023_06_01_preview);
64 ctx.rpc_ctx->set_response_header(
65 http::headers::CONTENT_TYPE,
66 http::headervalues::contenttype::JSON);
67 ctx.rpc_ctx->set_response_status(HTTP_STATUS_OK);
68 break;
69 }
70 }
71 }
72 else
73 {
74 CommonEndpointRegistry::api_endpoint(ctx);
75 }
76 }
