41 }
42
43 void close_session() override
44 {
45 tls_io->close();
46 }
47
48 void handle_incoming_data_thread(std::vector<uint8_t>&& data) override
49 {
50 tls_io->recv_buffered(data.data(), data.size());
51
52 LOG_TRACE_FMT("recv called with {} bytes", data.size());
53
54 // Try to parse all incoming data, reusing the vector we were just passed
55 // for storage. Increase the size if the received vector was too small
56 // (for the case where this chunk is very small, but we had some previous
57 // data to continue reading).
58 constexpr auto min_read_block_size = 4096;
59 if (data.size() < min_read_block_size)
60 {
61 data.resize(min_read_block_size);
62 }
63
64 auto n_read = tls_io->read(data.data(), data.size(), false);
65
66 while (true)
67 {
68 if (n_read == 0)
69 {
70 return;
71 }
72
73 LOG_TRACE_FMT("Going to parse {} bytes", n_read);
74
75 bool cont = parse({data.data(), n_read});
76 if (!cont)
77 {
78 return;
79 }
80
81 // Used all provided bytes - check if more are available
82 n_read = tls_io->read(data.data(), data.size(), false);
83 }
84 }

337 bool parse(std::span<const uint8_t> data) override
338 {
339 try
340 {
341 if (!server_parser->execute(data.data(), data.size()))
342 {
343 // Close session gracefully
344 tls_io->close();
345 return false;
346 }
347 return true;
348 }
349 catch (http::RequestPayloadTooLargeException& e)
350 {
351 if (error_reporter)
352 {
353 error_reporter->report_request_payload_too_large_error(interface_id);
354 }
355
356 LOG_DEBUG_FMT("Request is too large: {}", e.what());
357
358 auto error = ccf::ErrorDetails{
359 HTTP_STATUS_PAYLOAD_TOO_LARGE,
360 ccf::errors::RequestBodyTooLarge,
361 e.what()};
362
363 respond_with_error(e.get_stream_id(), error);
364
365 tls_io->close();
366 }
367 catch (http::RequestHeaderTooLargeException& e)
368 {
369 if (error_reporter)
370 {
371 error_reporter->report_request_header_too_large_error(interface_id);
372 }
373
374 LOG_DEBUG_FMT("Request header is too large: {}", e.what());
375
376 auto error = ccf::ErrorDetails{
377 HTTP_STATUS_REQUEST_HEADER_FIELDS_TOO_LARGE,
378 ccf::errors::RequestHeaderTooLarge,
379 e.what()};
380
381 respond_with_error(e.get_stream_id(), error);
382
383 tls_io->close();
384 }
385 catch (const std::exception& e)
386 {
387 if (error_reporter)
388 {
389 error_reporter->report_parsing_error(interface_id);
390 }
391
392 LOG_DEBUG_FMT("Error parsing HTTP request: {}", e.what());
393
394 // For generic parsing errors, as it is not trivial to construct a valid
395 // HTTP/2 response to send back to the default stream (0), the session
396 // is simply closed.
397
398 tls_io->close();
399 }
400 return false;
401 }

403 void handle_request(
404 llhttp_method verb,
405 const std::string_view& url,
406 http::HeaderMap&& headers,
407 std::vector<uint8_t>&& body,
408 int32_t stream_id) override
409 {
410 LOG_TRACE_FMT(
411 "Processing msg({}, {} [{} bytes])",
412 llhttp_method_name(verb),
413 url,
414 body.size());
415
416 auto responder = get_stream_responder(stream_id);
417 auto session_ctx = get_session_ctx(stream_id);
418
419 try
420 {
421 std::shared_ptr<http::HttpRpcContext> rpc_ctx = nullptr;
422 try
423 {
424 rpc_ctx = std::make_shared<HttpRpcContext>(
425 session_ctx,
426 ccf::HttpVersion::HTTP2,
427 verb,
428 url,
429 std::move(headers),
430 std::move(body),
431 responder);
432 }
433 catch (std::exception& e)
434 {
435 send_odata_error_response(ccf::ErrorDetails{
436 HTTP_STATUS_INTERNAL_SERVER_ERROR,
437 ccf::errors::InternalError,
438 fmt::format("Error constructing RpcContext: {}", e.what())});
439 }
440 std::shared_ptr<ccf::RpcHandler> search =
441 http::fetch_rpc_handler(rpc_ctx, rpc_map);
442
443 search->process(rpc_ctx);
444
445 if (rpc_ctx->response_is_pending)
446 {
447 // If the RPC is pending, hold the connection.
448 LOG_TRACE_FMT("Pending");
449 return;
450 }
451 else
452 {
453 responder->send_response(
454 rpc_ctx->get_response_http_status(),
455 rpc_ctx->get_response_headers(),
456 rpc_ctx->get_response_trailers(),
457 std::move(rpc_ctx->get_response_body()));
458 }
459 }
460 catch (const std::exception& e)
461 {
462 responder->send_odata_error_response(ccf::ErrorDetails{
463 HTTP_STATUS_INTERNAL_SERVER_ERROR,
464 ccf::errors::InternalError,
465 fmt::format("Exception: {}", e.what())});
466
467 // On any exception, close the connection.
468 LOG_FAIL_FMT("Closing connection");
469 LOG_DEBUG_FMT("Closing connection due to exception: {}", e.what());
470 tls_io->close();
471 throw;
472 }
473 }
