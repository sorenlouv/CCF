15 auto get_transaction_status = [&](auto& ctx, ApiVersion api_version) {
16 switch (api_version)
17 {
18 case ApiVersion::preview_v1:
19 default:
20 {
21 // Extract transaction ID from path parameter
22 std::string tx_id_str, error;
23 if (!ccf::endpoints::get_path_param(
24 ctx.rpc_ctx->get_request_path_params(),
25 "transactionId",
26 tx_id_str,
27 error))
28 {
29 detail::set_gov_error(
30 ctx.rpc_ctx,
31 HTTP_STATUS_BAD_REQUEST,
32 ccf::errors::InvalidResourceName,
33 std::move(error));
34 return;
35 }
36
37 // Parse transaction ID from string
38 const auto tx_id = ccf::TxID::from_str(tx_id_str);
39 if (!tx_id.has_value())
40 {
41 detail::set_gov_error(
42 ctx.rpc_ctx,
43 HTTP_STATUS_BAD_REQUEST,
44 ccf::errors::InvalidQueryParameterValue,
45 fmt::format(
46 "The value '{}' passed as parameter 'transactionId' could not "
47 "be converted to a valid Transaction ID.",
48 tx_id_str));
49 return;
50 }
51
52 // Lookup status
53 ccf::TxStatus status;
54 const auto result =
55 registry.get_status_for_txid_v1(tx_id->view, tx_id->seqno, status);
56 if (result != ccf::ApiResult::OK)
57 {
58 detail::set_gov_error(
59 ctx.rpc_ctx,
60 HTTP_STATUS_INTERNAL_SERVER_ERROR,
61 ccf::errors::InternalError,
62 fmt::format(
63 "get_status_for_txid_v1 returned error: {}",
64 ccf::api_result_to_str(result)));
65 return;
66 }
67
68 // Build response
69 auto body = nlohmann::json::object();
70
71 body["status"] = status;
72 body["transactionId"] = tx_id->to_str();
73
74 ctx.rpc_ctx->set_response_json(body, HTTP_STATUS_OK);
75 return;
76 }
77 }
78 };
79 registry
80 .make_command_endpoint(
81 "/service/transactions/{transactionId}",
82 HTTP_GET,
83 api_version_adapter(get_transaction_status),
84 no_auth_required)
