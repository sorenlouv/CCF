12 void init_recovery_handlers(
13 ccf::BaseEndpointRegistry& registry,
14 ShareManager& share_manager,
15 ccfapp::AbstractNodeContext& node_context)
16 {
17 auto get_encrypted_share_for_member =
18 [&](auto& ctx, ApiVersion api_version) {
19 switch (api_version)
20 {
21 case ApiVersion::preview_v1:
22 default:
23 {
24 ccf::MemberId member_id;
25 if (!detail::try_parse_member_id(ctx.rpc_ctx, member_id))
26 {
27 return;
28 }
29
30 auto encrypted_share =
31 ShareManager::get_encrypted_share(ctx.tx, member_id);
32
33 if (!encrypted_share.has_value())
34 {
35 detail::set_gov_error(
36 ctx.rpc_ctx,
37 HTTP_STATUS_NOT_FOUND,
38 ccf::errors::ResourceNotFound,
39 fmt::format(
40 "Recovery share not found for member {}.", member_id));
41 return;
42 }
43
44 auto response_body = nlohmann::json::object();
45 response_body["memberId"] = member_id;
46 response_body["encryptedShare"] =
47 crypto::b64_from_raw(encrypted_share.value());
48
49 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
50 return;
51 }
52 }
53 };
54 registry
55 .make_read_only_endpoint(
56 "/recovery/encrypted-shares/{memberId}",
57 HTTP_GET,
58 api_version_adapter(get_encrypted_share_for_member),
59 ccf::no_auth_required)
60 .set_openapi_hidden(true)
61 .install();
62
63 auto submit_recovery_share = [&](auto& ctx, ApiVersion api_version) {
64 switch (api_version)
65 {
66 case ApiVersion::preview_v1:
67 default:
68 {
69 if (
70 InternalTablesAccess::get_service_status(ctx.tx) !=
71 ServiceStatus::WAITING_FOR_RECOVERY_SHARES)
72 {
73 detail::set_gov_error(
74 ctx.rpc_ctx,
75 HTTP_STATUS_FORBIDDEN,
76 errors::ServiceNotWaitingForRecoveryShares,
77 "Service is not waiting for recovery shares.");
78 return;
79 }
80
81 auto node_operation =
82 node_context.get_subsystem<AbstractNodeOperation>();
83 if (node_operation == nullptr)
84 {
85 detail::set_gov_error(
86 ctx.rpc_ctx,
87 HTTP_STATUS_INTERNAL_SERVER_ERROR,
88 ccf::errors::InternalError,
89 "Could not access NodeOperation subsystem.");
90 return;
91 }
92
93 if (node_operation->is_reading_private_ledger())
94 {
95 detail::set_gov_error(
96 ctx.rpc_ctx,
97 HTTP_STATUS_FORBIDDEN,
98 errors::NodeAlreadyRecovering,
99 "Node is already recovering private ledger.");
100 return;
101 }
102
103 ccf::MemberId member_id;
104 if (!detail::try_parse_member_id(ctx.rpc_ctx, member_id))
105 {
106 return;
107 }
108
109 const auto& raw_body = ctx.rpc_ctx->get_request_body();
110 const nlohmann::json params = nlohmann::json::parse(raw_body);
111
112 auto raw_recovery_share =
113 crypto::raw_from_b64(params["share"].get<std::string>());
114
115 // Cleanse other copies of secret where possible. Note that this
116 // leaves a JSON-parsed copy, and potentially others in the TLS/HTTP
117 // stack.
118 OPENSSL_cleanse((char*)raw_body.data(), raw_body.size());
119
120 size_t submitted_shares_count = 0;
121 try
122 {
123 submitted_shares_count = share_manager.submit_recovery_share(
124 ctx.tx, member_id, raw_recovery_share);
125
126 OPENSSL_cleanse(
127 raw_recovery_share.data(), raw_recovery_share.size());
128 }
129 catch (const std::exception& e)
130 {
131 OPENSSL_cleanse(
132 raw_recovery_share.data(), raw_recovery_share.size());
133
134 constexpr auto error_msg = "Error submitting recovery shares.";
135 GOV_FAIL_FMT(error_msg);
136 GOV_DEBUG_FMT("Error: {}", e.what());
137 detail::set_gov_error(
138 ctx.rpc_ctx,
139 HTTP_STATUS_INTERNAL_SERVER_ERROR,
140 errors::InternalError,
141 error_msg);
142 return;
143 }
144
145 const auto threshold =
146 InternalTablesAccess::get_recovery_threshold(ctx.tx);
147
148 // Same format of message, whether this is sufficient to trigger
149 // recovery or not
150 std::string message = fmt::format(
151 "{}/{} recovery shares successfully submitted",
152 submitted_shares_count,
153 threshold);
154
155 if (submitted_shares_count >= threshold)
156 {
157 message += "\nEnd of recovery procedure initiated";
158 GOV_INFO_FMT("{} - initiating recovery", message);
159
160 // Initiate recovery
161 try
162 {
163 node_operation->initiate_private_recovery(ctx.tx);
164 }
165 catch (const std::exception& e)
166 {
167 // Clear the submitted shares if combination fails so that members
168 // can start over.
169 constexpr auto error_msg = "Failed to initiate private recovery.";
170 GOV_FAIL_FMT(error_msg);
171 GOV_DEBUG_FMT("Error: {}", e.what());
172 ShareManager::clear_submitted_recovery_shares(ctx.tx);
173 ctx.rpc_ctx->set_apply_writes(true);
174 detail::set_gov_error(
175 ctx.rpc_ctx,
176 HTTP_STATUS_INTERNAL_SERVER_ERROR,
177 errors::InternalError,
178 error_msg);
179 return;
180 }
181 }
182
183 auto response_body = nlohmann::json::object();
184 response_body["message"] = message;
185 response_body["submittedCount"] = submitted_shares_count;
186 response_body["recoveryThreshold"] = threshold;
187
188 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
189 return;
190 }
191 }
192 };
193 registry
194 .make_endpoint(
195 "/recovery/members/{memberId}:recover",
196 HTTP_POST,
197 api_version_adapter(submit_recovery_share),
198 ccf::no_auth_required)
199 .set_openapi_hidden(true)
200 .install();
201 }
