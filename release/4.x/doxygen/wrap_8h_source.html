369 // The interpreter can cache loaded modules so they do not need to be loaded
370 // from the KV for every execution, which is particularly useful when
371 // re-using interpreters. A module can only be loaded once per interpreter,
372 // and the entire interpreter should be thrown away if _any_ of its modules
373 // needs to be refreshed.

697 class JSWrappedPropertyEnum
698 {
699 public:
700 JSWrappedPropertyEnum(JSContext* ctx_, const JSWrappedValue& value) :
701 ctx(ctx_)
702 {
703 if (!value.is_obj())
704 {
705 throw std::logic_error(
706 fmt::format("object value required for property enum"));
707 }
708
709 if (
710 JS_GetOwnPropertyNames(
711 ctx,
712 &prop_enum,
713 &prop_count,
714 value,
715 JS_GPN_STRING_MASK | JS_GPN_ENUM_ONLY) == -1)
716 {
717 throw std::logic_error(
718 fmt::format("Could not extract property names of enum"));
719 }
720 }
721
722 ~JSWrappedPropertyEnum()
723 {
724 for (uint32_t i = 0; i < prop_count; i++)
725 {
726 JS_FreeAtom(ctx, prop_enum[i].atom);
727 }
728 js_free(ctx, prop_enum);
729 };
730
731 JSAtom& operator[](size_t i) const
732 {
733 return prop_enum[i].atom;
734 }
735
736 size_t size() const
737 {
738 return prop_count;
739 }
740
741 JSPropertyEnum* prop_enum = nullptr;
742 uint32_t prop_count = 0;
743 JSContext* ctx = nullptr;
744 };
