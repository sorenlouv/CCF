97 std::vector<uint8_t> wrap(const LedgerSecretPtr& ledger_secret)
98 {
99 if (has_wrapped)
100 {
101 throw std::logic_error(
102 "Ledger secret wrapping key has already wrapped once");
103 }
104
105 crypto::GcmCipher encrypted_ls(ledger_secret->raw_key.size());
106
107 crypto::make_key_aes_gcm(data)->encrypt(
108 encrypted_ls.hdr.get_iv(), // iv is always 0 here as the share wrapping
109 // key is never re-used for encryption
110 ledger_secret->raw_key,
111 {},
112 encrypted_ls.cipher,
113 encrypted_ls.hdr.tag);
114
115 has_wrapped = true;
116
117 return encrypted_ls.serialise();
118 }

120 LedgerSecretPtr unwrap(
121 const std::vector<uint8_t>& wrapped_latest_ledger_secret)
122 {
123 crypto::GcmCipher encrypted_ls;
124 encrypted_ls.deserialise(wrapped_latest_ledger_secret);
125 std::vector<uint8_t> decrypted_ls;
126
127 if (!crypto::make_key_aes_gcm(data)->decrypt(
128 encrypted_ls.hdr.get_iv(),
129 encrypted_ls.hdr.tag,
130 encrypted_ls.cipher,
131 {},
132 decrypted_ls))
133 {
134 throw std::logic_error("Unwrapping latest ledger secret failed");
135 }
136
137 return std::make_shared<LedgerSecret>(std::move(decrypted_ls));
138 }

145 // The ShareManager class provides the interface between the ledger secrets
146 // object and the shares, ledger secrets and submitted shares KV tables. In
147 // particular, it is used to:
148 // - Issue new recovery shares whenever required (e.g. on startup, rekey and
149 // membership updates)
150 // - Re-assemble the ledger secrets on recovery, once a threshold of members

157 EncryptedSharesMap compute_encrypted_shares(
158 kv::Tx& tx, const LedgerSecretWrappingKey& ls_wrapping_key)
159 {
160 EncryptedSharesMap encrypted_shares;
161 auto shares = ls_wrapping_key.get_shares();
162
163 auto active_recovery_members_info =
164 InternalTablesAccess::get_active_recovery_members(tx);
165
166 size_t share_index = 0;
167 for (auto const& [member_id, enc_pub_key] : active_recovery_members_info)
168 {
169 auto member_enc_pubk = crypto::make_rsa_public_key(enc_pub_key);
170 auto raw_share = std::vector<uint8_t>(
171 shares[share_index].begin(), shares[share_index].end());
172 encrypted_shares[member_id] = member_enc_pubk->rsa_oaep_wrap(raw_share);
173 OPENSSL_cleanse(raw_share.data(), raw_share.size());
174 OPENSSL_cleanse(shares[share_index].data(), shares[share_index].size());
175 share_index++;
176 }
177
178 return encrypted_shares;
179 }

181 void shuffle_recovery_shares(
182 kv::Tx& tx, const LedgerSecretPtr& latest_ledger_secret)
183 {
184 auto active_recovery_members_info =
185 InternalTablesAccess::get_active_recovery_members(tx);
186 size_t recovery_threshold =
187 InternalTablesAccess::get_recovery_threshold(tx);
188
189 if (active_recovery_members_info.empty())
190 {
191 throw std::logic_error(
192 "There should be at least one active recovery member to issue "
193 "recovery shares");
194 }
195
196 if (recovery_threshold == 0)
197 {
198 throw std::logic_error(
199 "Recovery threshold should be set before recovery "
200 "shares are computed");
201 }
202
203 if (recovery_threshold > active_recovery_members_info.size())
204 {
205 throw std::logic_error(fmt::format(
206 "Recovery threshold {} should be equal to or less than the number of "
207 "active recovery members {}",
208 recovery_threshold,
209 active_recovery_members_info.size()));
210 }
211
212 const auto num_shares = active_recovery_members_info.size();
213 auto ls_wrapping_key =
214 LedgerSecretWrappingKey(num_shares, recovery_threshold);
215
216 auto wrapped_latest_ls = ls_wrapping_key.wrap(latest_ledger_secret);
217 auto recovery_shares = tx.rw<ccf::RecoveryShares>(Tables::SHARES);
218 recovery_shares->put(
219 {wrapped_latest_ls,
220 compute_encrypted_shares(tx, ls_wrapping_key),
221 latest_ledger_secret->previous_secret_stored_version});
222 }

224 void set_recovery_shares_info(
225 kv::Tx& tx,
226 const LedgerSecretPtr& latest_ledger_secret,
227 const std::optional<VersionedLedgerSecret>& previous_ledger_secret =
228 std::nullopt,
229 std::optional<kv::Version> latest_ls_version = std::nullopt)
230 {
231 // First, generate a fresh ledger secrets wrapping key and wrap the
232 // latest ledger secret with it. Then, encrypt the penultimate ledger
233 // secret with the latest ledger secret and split the ledger secret
234 // wrapping key, allocating a new share for each active recovery member.
235 // Finally, encrypt each share with the public key of each member and
236 // record it in the shares table.
237
238 shuffle_recovery_shares(tx, latest_ledger_secret);
239
240 auto encrypted_ls = tx.rw<ccf::EncryptedLedgerSecretsInfo>(
241 Tables::ENCRYPTED_PAST_LEDGER_SECRET);
242
243 std::vector<uint8_t> encrypted_previous_secret = {};
244 kv::Version version_previous_secret = kv::NoVersion;
245 if (previous_ledger_secret.has_value())
246 {
247 version_previous_secret = previous_ledger_secret->first;
248
249 crypto::GcmCipher encrypted_previous_ls(
250 previous_ledger_secret->second->raw_key.size());
251 encrypted_previous_ls.hdr.set_random_iv();
252
253 latest_ledger_secret->key->encrypt(
254 encrypted_previous_ls.hdr.get_iv(),
255 previous_ledger_secret->second->raw_key,
256 {},
257 encrypted_previous_ls.cipher,
258 encrypted_previous_ls.hdr.tag);
259
260 encrypted_previous_secret = encrypted_previous_ls.serialise();
261 encrypted_ls->put(
262 {PreviousLedgerSecretInfo(
263 std::move(encrypted_previous_secret),
264 version_previous_secret,
265 encrypted_ls->get_version_of_previous_write()),
266 latest_ls_version});
267 }
268 else
269 {
270 encrypted_ls->put({std::nullopt, latest_ls_version});
271 }
272 }

311 LedgerSecretWrappingKey combine_from_encrypted_submitted_shares(kv::Tx& tx)
312 {
313 auto encrypted_submitted_shares = tx.rw<ccf::EncryptedSubmittedShares>(
314 Tables::ENCRYPTED_SUBMITTED_SHARES);
315 auto config = tx.rw<ccf::Configuration>(Tables::CONFIGURATION);
316
317 std::vector<crypto::Share> new_shares = {};
318 std::vector<SecretSharing::Share> old_shares = {};
319 // Defensively allow shares in both formats for the time being, even if we
320 // get a mix, and so long as we have enough of one or the other, attempt
321 // to reassemble the secret. We only try with the most numerous kind of
322 // share, we won't try with the minority even if it meets the threshold
323 // too.
324 encrypted_submitted_shares->foreach(
325 [&new_shares, &old_shares, &tx, this](
326 const MemberId, const EncryptedSubmittedShare& encrypted_share) {
327 auto decrypted_share = decrypt_submitted_share(
328 encrypted_share, ledger_secrets->get_latest(tx).second);
329 switch (decrypted_share.size())
330 {
331 case crypto::Share::serialised_size:
332 {
333 new_shares.emplace_back(decrypted_share);
334 break;
335 }
336 case SecretSharing::SHARE_LENGTH:
337 {
338 SecretSharing::Share share;
339 std::copy_n(
340 decrypted_share.begin(),
341 SecretSharing::SHARE_LENGTH,
342 share.begin());
343 old_shares.emplace_back(std::move(share));
344 break;
345 }
346 default:
347 {
348 OPENSSL_cleanse(decrypted_share.data(), decrypted_share.size());
349 throw std::logic_error(fmt::format(
350 "Error combining recovery shares: decrypted share of {} bytes "
351 "is neither a new-style share of {} bytes nor an old-style "
352 "share of {} bytes",
353 decrypted_share.size(),
354 crypto::Share::serialised_size,
355 SecretSharing::SHARE_LENGTH));
356 }
357 }
358 OPENSSL_cleanse(decrypted_share.data(), decrypted_share.size());
359 return true;
360 });
361
362 auto num_shares = std::max(old_shares.size(), new_shares.size());
363
364 auto recovery_threshold = config->get()->recovery_threshold;
365 if (recovery_threshold > num_shares)
366 {
367 throw std::logic_error(fmt::format(
368 "Error combining recovery shares: only {} recovery shares were "
369 "submitted but recovery threshold is {}",
370 num_shares,
371 recovery_threshold));
372 }
373
374 if (new_shares.size() > old_shares.size())
375 {
376 return LedgerSecretWrappingKey(
377 std::move(new_shares), recovery_threshold);
378 }
379 else
380 {
381 return LedgerSecretWrappingKey(
382 std::move(old_shares), recovery_threshold);
383 }
384 }

453 LedgerSecretsMap restore_recovery_shares_info(
454 kv::Tx& tx,
455 const RecoveredEncryptedLedgerSecrets& recovery_ledger_secrets)
456 {
457 // First, re-assemble the ledger secret wrapping key from the submitted
458 // encrypted shares. Then, unwrap the latest ledger secret and use it to
459 // decrypt the sequence of recovered ledger secrets, from the last one.
460
461 if (recovery_ledger_secrets.empty())
462 {
463 throw std::logic_error("No recovery ledger secrets");
464 }
465
466 auto recovery_shares_info =
467 tx.ro<ccf::RecoveryShares>(Tables::SHARES)->get();
468 if (!recovery_shares_info.has_value())
469 {
470 throw std::logic_error(
471 "Failed to retrieve current recovery shares info");
472 }
473
474 auto restored_ls = combine_from_encrypted_submitted_shares(tx).unwrap(
475 recovery_shares_info->wrapped_latest_ledger_secret);
476
477 LOG_DEBUG_FMT(
478 "Recovering {} encrypted ledger secrets",
479 recovery_ledger_secrets.size());
480
481 auto& current_ledger_secret_version =
482 recovery_ledger_secrets.back().next_version;
483 if (!current_ledger_secret_version.has_value())
484 {
485 // This should always be set by the recovery hook, which sets this to
486 // the version at which it is called if unset in the store
487 throw std::logic_error("Current ledger secret version should be set");
488 }
489
490 auto encrypted_previous_ledger_secret =
491 tx.ro<ccf::EncryptedLedgerSecretsInfo>(
492 Tables::ENCRYPTED_PAST_LEDGER_SECRET);
493
494 LedgerSecretsMap restored_ledger_secrets = {};
495 auto s = restored_ledger_secrets.emplace(
496 current_ledger_secret_version.value(),
497 std::make_shared<LedgerSecret>(
498 std::move(restored_ls->raw_key),
499 encrypted_previous_ledger_secret->get_version_of_previous_write()));
500 auto latest_ls = s.first->second;
501
502 for (auto it = recovery_ledger_secrets.rbegin();
503 it != recovery_ledger_secrets.rend();
504 it++)
505 {
506 if (!it->previous_ledger_secret.has_value())
507 {
508 // Very first entry does not encrypt any other ledger secret
509 break;
510 }
511
512 auto decrypted_ls_raw = decrypt_previous_ledger_secret_raw(
513 latest_ls, it->previous_ledger_secret->encrypted_data);
514
515 auto secret = restored_ledger_secrets.emplace(
516 it->previous_ledger_secret->version,
517 std::make_shared<LedgerSecret>(
518 std::move(decrypted_ls_raw),
519 it->previous_ledger_secret->previous_secret_stored_version));
520 latest_ls = secret.first->second;
521 }
522
523 return restored_ledger_secrets;
524 }
