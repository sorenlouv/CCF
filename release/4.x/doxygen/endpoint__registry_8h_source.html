45 template <typename T>
46 inline bool get_path_param(
47 const ccf::PathParams& params,
48 const std::string& param_name,
49 T& value,
50 std::string& error)
51 {
52 const auto it = params.find(param_name);
53 if (it == params.end())
54 {
55 error = fmt::format("No parameter named '{}' in path", param_name);
56 return false;
57 }
58
59 const auto param_s = it->second;
60 const auto [p, ec] =
61 std::from_chars(param_s.data(), param_s.data() + param_s.size(), value);
62 if (ec != std::errc())
63 {
64 error = fmt::format(
65 "Unable to parse path parameter '{}' as a {}", param_s, param_name);
66 return false;
67 }
68
69 return true;
70 }
