217 static EndorsementEndpointsConfiguration
218 make_endorsement_endpoint_configuration(
219 const Attestation& quote,
220 const snp::EndorsementsServers& endorsements_servers = {})
221 {
222 EndorsementEndpointsConfiguration config;
223
224 auto chip_id_hex = fmt::format("{:02x}", fmt::join(quote.chip_id, ""));
225 auto reported_tcb = fmt::format("{:0x}", *(uint64_t*)(&quote.reported_tcb));
226
227 if (endorsements_servers.empty())
228 {
229 // Default to Azure server if no servers are specified
230 config.servers.emplace_back(make_azure_endorsements_server(
231 default_azure_endorsements_endpoint_host, chip_id_hex, reported_tcb));
232 return config;
233 }
234
235 for (auto const& server : endorsements_servers)
236 {
237 switch (server.type)
238 {
239 case EndorsementsEndpointType::Azure:
240 {
241 auto url =
242 server.url.value_or(default_azure_endorsements_endpoint_host);
243 config.servers.emplace_back(
244 make_azure_endorsements_server(url, chip_id_hex, reported_tcb));
245 break;
246 }
247 case EndorsementsEndpointType::AMD:
248 {
249 auto boot_loader = fmt::format("{}", quote.reported_tcb.boot_loader);
250 auto tee = fmt::format("{}", quote.reported_tcb.tee);
251 auto snp = fmt::format("{}", quote.reported_tcb.snp);
252 auto microcode = fmt::format("{}", quote.reported_tcb.microcode);
253
254 auto url =
255 server.url.value_or(default_azure_endorsements_endpoint_host);
256 config.servers.emplace_back(make_amd_endorsements_server(
257 url, chip_id_hex, boot_loader, tee, snp, microcode));
258 break;
259 }
260 default:
261 {
262 throw std::logic_error(fmt::format(
263 "Unsupported endorsements server type: {}", server.type));
264 }
265 }
266 }
267
268 return config;
269 }
