157 static std::pair<http_status, std::string> quote_verification_error(
158 QuoteVerificationResult result)
159 {
160 switch (result)
161 {
162 case QuoteVerificationResult::Failed:
163 return std::make_pair(
164 HTTP_STATUS_UNAUTHORIZED, "Quote could not be verified");
165 case QuoteVerificationResult::FailedMeasurementNotFound:
166 return std::make_pair(
167 HTTP_STATUS_UNAUTHORIZED,
168 "Quote does not contain known enclave measurement");
169 case QuoteVerificationResult::FailedInvalidQuotedPublicKey:
170 return std::make_pair(
171 HTTP_STATUS_UNAUTHORIZED,
172 "Quote report data does not contain node's public key hash");
173 case QuoteVerificationResult::FailedHostDataDigestNotFound:
174 return std::make_pair(
175 HTTP_STATUS_UNAUTHORIZED,
176 "Quote does not contain trusted host data");
177 case QuoteVerificationResult::FailedInvalidHostData:
178 return std::make_pair(
179 HTTP_STATUS_UNAUTHORIZED, "Quote host data is not authorised");
180 case ccf::QuoteVerificationResult::FailedUVMEndorsementsNotFound:
181 return std::make_pair(
182 HTTP_STATUS_UNAUTHORIZED, "UVM endorsements are not authorised");
183 default:
184 return std::make_pair(
185 HTTP_STATUS_INTERNAL_SERVER_ERROR,
186 "Unknown quote verification error");
187 }

197 std::optional<ExistingNodeInfo> check_node_exists(
198 kv::Tx& tx,
199 const std::vector<uint8_t>& self_signed_node_der,
200 std::optional<NodeStatus> node_status = std::nullopt)
201 {
202 // Check that a node exists by looking up its public key in the nodes
203 // table.
204 auto nodes = tx.ro(network.nodes);
205 auto endorsed_node_certificates =
206 tx.ro(network.node_endorsed_certificates);
207
208 LOG_DEBUG_FMT(
209 "Check node exists with certificate [{}]", self_signed_node_der);
210 auto pk_pem = crypto::public_key_pem_from_cert(self_signed_node_der);
211
212 std::optional<ExistingNodeInfo> existing_node_info = std::nullopt;
213 nodes->foreach([&existing_node_info,
214 &pk_pem,
215 &node_status,
216 &endorsed_node_certificates](
217 const NodeId& nid, const NodeInfo& ni) {
218 if (
219 ni.public_key == pk_pem &&
220 (!node_status.has_value() || ni.status == node_status.value()))
221 {
222 existing_node_info = {
223 nid, ni.ledger_secret_seqno, endorsed_node_certificates->get(nid)};
224 return false;
225 }
226 return true;
227 });
228
229 return existing_node_info;
230 }
231
232 std::optional<NodeId> check_conflicting_node_network(
233 kv::Tx& tx, const NodeInfoNetwork& node_info_network)
234 {
235 auto nodes = tx.rw(network.nodes);
236
237 std::optional<NodeId> duplicate_node_id = std::nullopt;
238 nodes->foreach([&node_info_network, &duplicate_node_id](
239 const NodeId& nid, const NodeInfo& ni) {
240 if (
241 node_info_network.node_to_node_interface.published_address ==
242 ni.node_to_node_interface.published_address &&
243 ni.status != NodeStatus::RETIRED)
244 {
245 duplicate_node_id = nid;
246 return false;
247 }
248 return true;
249 });
250
251 return duplicate_node_id;
252 }

259 auto add_node(
260 kv::Tx& tx,
261 const std::vector<uint8_t>& node_der,
262 const JoinNetworkNodeToNode::In& in,
263 NodeStatus node_status,
264 ServiceStatus service_status,
265 ReconfigurationType reconfiguration_type)
266 {
267 auto nodes = tx.rw(network.nodes);
268 auto node_endorsed_certificates =
269 tx.rw(network.node_endorsed_certificates);
270
271 auto conflicting_node_id =
272 check_conflicting_node_network(tx, in.node_info_network);
273 if (conflicting_node_id.has_value())
274 {
275 return make_error(
276 HTTP_STATUS_BAD_REQUEST,
277 ccf::errors::NodeAlreadyExists,
278 fmt::format(
279 "A node with the same published node address {} already exists "
280 "(node id: {}).",
281 in.node_info_network.node_to_node_interface.published_address,
282 conflicting_node_id.value()));
283 }
284
285 auto pubk_der = crypto::public_key_der_from_cert(node_der);
286 NodeId joining_node_id = compute_node_id_from_pubk_der(pubk_der);
287
288 pal::PlatformAttestationMeasurement measurement;
289
290 QuoteVerificationResult verify_result = this->node_operation.verify_quote(
291 tx, in.quote_info, pubk_der, measurement);
292 if (verify_result != QuoteVerificationResult::Verified)
293 {
294 const auto [code, message] = quote_verification_error(verify_result);
295 return make_error(code, ccf::errors::InvalidQuote, message);
296 }
297
298 std::optional<kv::Version> ledger_secret_seqno = std::nullopt;
299 if (node_status == NodeStatus::TRUSTED)
300 {
301 ledger_secret_seqno =
302 this->network.ledger_secrets->get_latest(tx).first;
303 }
304
305 // Note: All new nodes should specify a CSR from 2.x
306 auto client_public_key_pem = crypto::public_key_pem_from_cert(node_der);
307 if (in.certificate_signing_request.has_value())
308 {
309 // Verify that client's public key matches the one specified in the CSR
310 auto csr_public_key_pem = crypto::public_key_pem_from_csr(
311 in.certificate_signing_request.value());
312 if (client_public_key_pem != csr_public_key_pem)
313 {
314 return make_error(
315 HTTP_STATUS_BAD_REQUEST,
316 ccf::errors::CSRPublicKeyInvalid,
317 "Public key in CSR does not match TLS client identity.");
318 }
319 }
320
321 NodeInfo node_info = {
322 in.node_info_network,
323 in.quote_info,
324 in.public_encryption_key,
325 node_status,
326 ledger_secret_seqno,
327 measurement.hex_str(),
328 in.certificate_signing_request,
329 client_public_key_pem,
330 in.node_data};
331
332 nodes->put(joining_node_id, node_info);
333
334 LOG_INFO_FMT("Node {} added as {}", joining_node_id, node_status);
335
336 JoinNetworkNodeToNode::Out rep;
337 rep.node_status = node_status;
338 rep.node_id = joining_node_id;
339
340 if (node_status == NodeStatus::TRUSTED)
341 {
342 // Joining node only submit a CSR from 2.x
343 std::optional<crypto::Pem> endorsed_certificate = std::nullopt;
344 if (in.certificate_signing_request.has_value())
345 {
346 // For a pre-open service, extract the validity period of self-signed
347 // node certificate and use it verbatim in endorsed certificate
348 auto [valid_from, valid_to] =
349 crypto::make_verifier(node_der)->validity_period();
350 endorsed_certificate = crypto::create_endorsed_cert(
351 in.certificate_signing_request.value(),
352 valid_from,
353 valid_to,
354 this->network.identity->priv_key,
355 this->network.identity->cert);
356
357 node_endorsed_certificates->put(
358 joining_node_id, {endorsed_certificate.value()});
359 }
360
361 rep.network_info = JoinNetworkNodeToNode::Out::NetworkInfo{
362 node_operation.is_part_of_public_network(),
363 node_operation.get_last_recovered_signed_idx(),
364 ReconfigurationType::ONE_TRANSACTION,
365 this->network.ledger_secrets->get(tx),
366 *this->network.identity.get(),
367 service_status,
368 endorsed_certificate};
369 }
370 return make_success(rep);
371 }

387 void init_handlers() override
388 {
389 CommonEndpointRegistry::init_handlers();
390
391 auto accept = [this](auto& args, const nlohmann::json& params) {
392 const auto in = params.get<JoinNetworkNodeToNode::In>();
393
394 if (
395 !this->node_operation.is_part_of_network() &&
396 !this->node_operation.is_part_of_public_network() &&
397 !this->node_operation.is_reading_private_ledger())
398 {
399 return make_error(
400 HTTP_STATUS_INTERNAL_SERVER_ERROR,
401 ccf::errors::InternalError,
402 "Target node should be part of network to accept new nodes.");
403 }
404
405 // Make sure that the joiner's snapshot is more recent than this node's
406 // snapshot. Otherwise, the joiner may not be given all the ledger
407 // secrets required to replay historical transactions.
408 auto this_startup_seqno =
409 this->node_operation.get_startup_snapshot_seqno();
410 if (
411 in.startup_seqno.has_value() &&
412 this_startup_seqno > in.startup_seqno.value())
413 {
414 return make_error(
415 HTTP_STATUS_BAD_REQUEST,
416 ccf::errors::StartupSeqnoIsOld,
417 fmt::format(
418 "Node requested to join from seqno {} which is older than this "
419 "node startup seqno {}. A snapshot at least as recent as {} must "
420 "be used instead.",
421 in.startup_seqno.value(),
422 this_startup_seqno,
423 this_startup_seqno));
424 }
425
426 auto nodes = args.tx.rw(this->network.nodes);
427 auto service = args.tx.rw(this->network.service);
428
429 auto active_service = service->get();
430 if (!active_service.has_value())
431 {
432 return make_error(
433 HTTP_STATUS_INTERNAL_SERVER_ERROR,
434 ccf::errors::InternalError,
435 "No service is available to accept new node.");
436 }
437
438 auto config = args.tx.ro(network.config);
439 auto service_config = config->get();
440
441 if (
442 active_service->status == ServiceStatus::OPENING ||
443 active_service->status == ServiceStatus::RECOVERING)
444 {
445 // If the service is opening, new nodes are trusted straight away
446 NodeStatus joining_node_status = NodeStatus::TRUSTED;
447
448 // If the node is already trusted, return network secrets
449 auto existing_node_info = check_node_exists(
450 args.tx,
451 args.rpc_ctx->get_session_context()->caller_cert,
452 joining_node_status);
453 if (existing_node_info.has_value())
454 {
455 JoinNetworkNodeToNode::Out rep;
456 rep.node_status = joining_node_status;
457 rep.network_info = JoinNetworkNodeToNode::Out::NetworkInfo(
458 node_operation.is_part_of_public_network(),
459 node_operation.get_last_recovered_signed_idx(),
460 ReconfigurationType::ONE_TRANSACTION,
461 this->network.ledger_secrets->get(
462 args.tx, existing_node_info->ledger_secret_seqno),
463 *this->network.identity.get(),
464 active_service->status,
465 existing_node_info->endorsed_certificate);
466
467 return make_success(rep);
468 }
469
470 if (consensus != nullptr && !this->node_operation.can_replicate())
471 {
472 auto primary_id = consensus->primary();
473 if (primary_id.has_value())
474 {
475 auto info = nodes->get(primary_id.value());
476 if (info)
477 {
478 auto& interface_id =
479 args.rpc_ctx->get_session_context()->interface_id;
480 if (!interface_id.has_value())
481 {
482 return make_error(
483 HTTP_STATUS_INTERNAL_SERVER_ERROR,
484 ccf::errors::InternalError,
485 "Cannot redirect non-RPC request.");
486 }
487 const auto& address =
488 info->rpc_interfaces[interface_id.value()].published_address;
489 args.rpc_ctx->set_response_header(
490 http::headers::LOCATION,
491 fmt::format("https://{}/node/join", address));
492
493 return make_error(
494 HTTP_STATUS_PERMANENT_REDIRECT,
495 ccf::errors::NodeCannotHandleRequest,
496 "Node is not primary; cannot handle write");
497 }
498 }
499
500 return make_error(
501 HTTP_STATUS_INTERNAL_SERVER_ERROR,
502 ccf::errors::InternalError,
503 "Primary unknown");
504 }
505
506 return add_node(
507 args.tx,
508 args.rpc_ctx->get_session_context()->caller_cert,
509 in,
510 joining_node_status,
511 active_service->status,
512 ReconfigurationType::ONE_TRANSACTION);
513 }
514
515 // If the service is open, new nodes are first added as pending and
516 // then only trusted via member governance. It is expected that a new
517 // node polls the network to retrieve the network secrets until it is
518 // trusted
519
520 auto existing_node_info = check_node_exists(
521 args.tx, args.rpc_ctx->get_session_context()->caller_cert);
522 if (existing_node_info.has_value())
523 {
524 JoinNetworkNodeToNode::Out rep;
525
526 // If the node already exists, return network secrets if is already
527 // trusted. Otherwise, only return its status
528 auto node_info = nodes->get(existing_node_info->node_id);
529 auto node_status = node_info->status;
530 rep.node_status = node_status;
531 if (is_taking_part_in_acking(node_status))
532 {
533 rep.network_info = JoinNetworkNodeToNode::Out::NetworkInfo(
534 node_operation.is_part_of_public_network(),
535 node_operation.get_last_recovered_signed_idx(),
536 ReconfigurationType::ONE_TRANSACTION,
537 this->network.ledger_secrets->get(
538 args.tx, existing_node_info->ledger_secret_seqno),
539 *this->network.identity.get(),
540 active_service->status,
541 existing_node_info->endorsed_certificate);
542
543 return make_success(rep);
544 }
545 else if (node_status == NodeStatus::PENDING)
546 {
547 // Only return node status and ID
548 return make_success(rep);
549 }
550 else
551 {
552 return make_error(
553 HTTP_STATUS_BAD_REQUEST,
554 ccf::errors::InvalidNodeState,
555 fmt::format(
556 "Joining node is not in expected state ({}).", node_status));
557 }
558 }
559 else
560 {
561 if (consensus != nullptr && !this->node_operation.can_replicate())
562 {
563 auto primary_id = consensus->primary();
564 if (primary_id.has_value())
565 {
566 auto info = nodes->get(primary_id.value());
567 if (info)
568 {
569 auto& interface_id =
570 args.rpc_ctx->get_session_context()->interface_id;
571 if (!interface_id.has_value())
572 {
573 return make_error(
574 HTTP_STATUS_INTERNAL_SERVER_ERROR,
575 ccf::errors::InternalError,
576 "Cannot redirect non-RPC request.");
577 }
578 const auto& address =
579 info->rpc_interfaces[interface_id.value()].published_address;
580 args.rpc_ctx->set_response_header(
581 http::headers::LOCATION,
582 fmt::format("https://{}/node/join", address));
583
584 return make_error(
585 HTTP_STATUS_PERMANENT_REDIRECT,
586 ccf::errors::NodeCannotHandleRequest,
587 "Node is not primary; cannot handle write");
588 }
589 }
590
591 return make_error(
592 HTTP_STATUS_INTERNAL_SERVER_ERROR,
593 ccf::errors::InternalError,
594 "Primary unknown");
595 }
596
597 // If the node does not exist, add it to the KV in state pending
598 return add_node(
599 args.tx,
600 args.rpc_ctx->get_session_context()->caller_cert,
601 in,
602 NodeStatus::PENDING,
603 active_service->status,
604 ReconfigurationType::ONE_TRANSACTION);
605 }

684 auto get_quote = [this](auto& args, nlohmann::json&&) {
685 QuoteInfo node_quote_info;
686 const auto result =
687 get_quote_for_this_node_v1(args.tx, node_quote_info);
688 if (result == ApiResult::OK)
689 {
690 Quote q;
691 q.node_id = context.get_node_id();
692 q.raw = node_quote_info.quote;
693 q.endorsements = node_quote_info.endorsements;
694 q.format = node_quote_info.format;
695 q.uvm_endorsements = node_quote_info.uvm_endorsements;
696
697 // get_measurement attempts to re-validate the quote to extract
698 // mrenclave and the Open Enclave is insufficiently flexible to allow
699 // quotes with expired collateral to be parsed at all. Recent nodes
700 // therefore cache their code digest on startup, and this code
701 // attempts to fetch that value when possible and only call the
702 // unreliable get_measurement otherwise.
703 auto nodes = args.tx.ro(network.nodes);
704 auto node_info = nodes->get(context.get_node_id());
705 if (node_info.has_value() && node_info->code_digest.has_value())
706 {
707 q.mrenclave = node_info->code_digest.value();
708 }
709 else
710 {
711 auto measurement =
712 AttestationProvider::get_measurement(node_quote_info);
713 if (measurement.has_value())
714 {
715 q.mrenclave = measurement.value().hex_str();
716 }
717 else
718 {
719 return make_error(
720 HTTP_STATUS_INTERNAL_SERVER_ERROR,
721 ccf::errors::InvalidQuote,
722 "Failed to extract code id from node quote.");
723 }
724 }
725
726 return make_success(q);
727 }
728 else if (result == ApiResult::NotFound)
729 {
730 return make_error(

873 parsed_query, "status", error_string);
874
875 std::optional<NodeStatus> status;
876 if (status_str.has_value())
877 {
878 // Convert the query argument to a JSON string, try to parse it as
879 // a NodeStatus, return an error if this doesn't work
880 try
881 {
882 status = nlohmann::json(status_str.value()).get<NodeStatus>();
883 }
884 catch (const JsonParseError& e)
885 {
886 return ccf::make_error(
887 HTTP_STATUS_BAD_REQUEST,
888 ccf::errors::InvalidQueryParameterValue,
889 fmt::format(
890 "Query parameter '{}' is not a valid node status",
891 status_str.value()));

895 GetNodes::Out out;
896
897 auto nodes = args.tx.ro(this->network.nodes);
898 nodes->foreach([this, host, port, status, &out, nodes](
899 const NodeId& nid, const NodeInfo& ni) {
900 if (status.has_value() && status.value() != ni.status)
901 {
902 return true;
903 }
904
905 // Match on any interface
906 bool is_matched = false;
907 for (auto const& interface : ni.rpc_interfaces)
908 {
909 const auto& [pub_host, pub_port] =
910 split_net_address(interface.second.published_address);
911
912 if (
913 (!host.has_value() || host.value() == pub_host) &&
914 (!port.has_value() || port.value() == pub_port))
915 {
916 is_matched = true;
917 break;
918 }
919 }
920
921 if (!is_matched)
922 {
923 return true;
924 }
925
926 bool is_primary = false;
927 if (consensus != nullptr)
928 {
929 is_primary = consensus->primary() == nid;
930 }
931
932 out.nodes.push_back(
933 {nid,
934 ni.status,
935 is_primary,
936 ni.rpc_interfaces,
937 ni.node_data,
938 nodes->get_version_of_previous_write(nid).value_or(0)});
939 return true;
940 });
941
942 return make_success(out);
943 };

997 auto delete_retired_committed_node =
998 [this](auto& args, nlohmann::json&&) {
999 GetNodes::Out out;
1000
1001 std::string node_id;
1002 std::string error;
1003 if (!get_path_param(
1004 args.rpc_ctx->get_request_path_params(),
1005 "node_id",
1006 node_id,
1007 error))
1008 {
1009 return make_error(
1010 HTTP_STATUS_BAD_REQUEST, ccf::errors::InvalidResourceName, error);
1011 }
1012
1013 auto nodes = args.tx.rw(this->network.nodes);
1014 if (!nodes->has(node_id))
1015 {
1016 return make_error(
1017 HTTP_STATUS_NOT_FOUND,
1018 ccf::errors::ResourceNotFound,
1019 "No such node");
1020 }
1021
1022 auto node_endorsed_certificates =
1023 args.tx.rw(network.node_endorsed_certificates);
1024
1025 // A node's retirement is only complete when the
1026 // transition of retired_committed is itself committed,
1027 // i.e. when the next eligible primary is guaranteed to
1028 // be aware the retirement is committed.
1029 // As a result, the handler must check node info at the
1030 // current committed level, rather than at the end of the
1031 // local suffix.
1032 // While this transaction does execute a write, it specifically
1033 // deletes the value it reads from. It is therefore safe to
1034 // execute on the basis of a potentially stale read-set,
1035 // which get_globally_committed() typically produces.
1036 auto node = nodes->get_globally_committed(node_id);
1037 if (
1038 node.has_value() && node->status == ccf::NodeStatus::RETIRED &&
1039 node->retired_committed)
1040 {
1041 nodes->remove(node_id);
1042 node_endorsed_certificates->remove(node_id);
1043 }
1044 else
1045 {
1046 return make_error(
1047 HTTP_STATUS_BAD_REQUEST,
1048 ccf::errors::NodeNotRetiredCommitted,
1049 "Node is not completely retired");
1050 }
1051
1052 return make_success(true);
1053 };
