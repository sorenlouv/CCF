14 inline std::vector<uint8_t> error(ccf::ErrorDetails&& error)
15 {
16 nlohmann::json body = ccf::ODataErrorResponse{
17 ccf::ODataError{std::move(error.code), std::move(error.msg)}};
18 const auto s = body.dump();
19
20 std::vector<uint8_t> data(s.begin(), s.end());
21 auto response = http::Response(error.status);
22
23 response.set_header(
24 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
25 response.set_body(&data);
26
27 return response.build_response();
28 }

64 void serialise()
65 {
66 if (!serialised)
67 {
68 const auto request_prefix = fmt::format(
69 "{} {} HTTP/1.1\r\n"
70 "{}"
71 "\r\n",
72 verb.c_str(),
73 url,
74 http::get_header_string(request_headers));
75
76 serialised_request.resize(request_prefix.size() + request_body.size());
77 ::memcpy(
78 serialised_request.data(),
79 request_prefix.data(),
80 request_prefix.size());
81 if (!request_body.empty())
82 {
83 ::memcpy(
84 serialised_request.data() + request_prefix.size(),
85 request_body.data(),
86 request_body.size());
87 }
88 }
89
90 serialised = true;
91 }

301 inline static std::optional<std::string> extract_actor(HttpRpcContext& ctx)
302 {
303 const auto path = ctx.get_method();
304 const auto first_slash = path.find_first_of('/');
305 const auto second_slash = path.find_first_of('/', first_slash + 1);
306
307 if (first_slash != 0 || second_slash == std::string::npos)
308 {
309 return std::nullopt;
310 }
311
312 auto actor = path.substr(first_slash + 1, second_slash - first_slash - 1);
313 auto remaining_path = path.substr(second_slash);
314
315 // The "actor" is generally just a single path component, eg. `gov` or
316 // `node`. We make an exception for .well-known paths however, which use
317 // two components, ie. `.well-known/acme-challenge`. This restricts the
318 // ACME frontend to just that particular sub-directory, and allows the
319 // application to handle other .well-known paths.
320 if (actor == ".well-known")
321 {
322 const auto third_slash = path.find_first_of('/', second_slash + 1);
323 if (third_slash == std::string::npos)
324 {
325 return std::nullopt;
326 }
327
328 actor = path.substr(first_slash + 1, third_slash - first_slash - 1);
329 remaining_path = path.substr(third_slash);
330 }
331
332 if (actor.empty() || remaining_path.empty())
333 {
334 return std::nullopt;
335 }
336
337 // if the extracted actor is a known type, set the remaining path
338 if (ccf::is_valid_actor(actor))
339 {
340 ctx.set_method(remaining_path);
341 }
342 return actor;
343 }

345 inline static std::shared_ptr<ccf::RpcHandler> fetch_rpc_handler(
346 std::shared_ptr<http::HttpRpcContext>& ctx,
347 std::shared_ptr<ccf::RPCMap>& rpc_map)
348 {
349 const auto actor_opt = http::extract_actor(*ctx);
350 std::optional<std::shared_ptr<ccf::RpcHandler>> search;
351 ccf::ActorsType actor = ccf::ActorsType::unknown;
352
353 if (actor_opt.has_value())
354 {
355 const auto& actor_s = actor_opt.value();
356 actor = rpc_map->resolve(actor_s);
357 search = rpc_map->find(actor);
358 }
359 if (
360 !actor_opt.has_value() || actor == ccf::ActorsType::unknown ||
361 !search.has_value())
362 {
363 // if there is no actor, proceed with the "app" as the ActorType and
364 // process the request
365 search = rpc_map->find(ccf::ActorsType::users);
366 }
367 return search.value();
368 }

373 inline std::shared_ptr<http::HttpRpcContext> make_rpc_context(
374 std::shared_ptr<ccf::SessionContext> s, const std::vector<uint8_t>& packed)
375 {
376 http::SimpleRequestProcessor processor;
377 http::RequestParser parser(processor, http::permissive_configuration());
378 parser.execute(packed.data(), packed.size());
379
380 if (processor.received.size() != 1)
381 {
382 throw std::logic_error(fmt::format(
383 "Expected packed to contain a single complete HTTP message. Actually "
384 "parsed {} messages",
385 processor.received.size()));
386 }
387
388 const auto& msg = processor.received.front();
389
390 return std::make_shared<http::HttpRpcContext>(
391 s,
392 ccf::HttpVersion::HTTP1,
393 msg.method,
394 msg.url,
395 msg.headers,
396 msg.body,
397 nullptr,
398 packed);
399 }
