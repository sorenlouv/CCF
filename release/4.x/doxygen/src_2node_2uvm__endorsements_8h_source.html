59 static std::vector<std::vector<uint8_t>> decode_x5chain(
60 QCBORDecodeContext& ctx, const QCBORItem& x5chain)
61 {
62 std::vector<std::vector<uint8_t>> parsed;
63
64 if (x5chain.uDataType == QCBOR_TYPE_ARRAY)
65 {
66 QCBORDecode_EnterArrayFromMapN(&ctx, headers::PARAM_X5CHAIN);
67 while (true)
68 {
69 QCBORItem item;
70 auto result = QCBORDecode_GetNext(&ctx, &item);
71 if (result == QCBOR_ERR_NO_MORE_ITEMS)
72 {
73 break;
74 }
75 if (result != QCBOR_SUCCESS)
76 {
77 throw COSEDecodeError("Item in x5chain is not well-formed");
78 }
79 if (item.uDataType == QCBOR_TYPE_BYTE_STRING)
80 {
81 parsed.push_back(qcbor_buf_to_byte_vector(item.val.string));
82 }
83 else
84 {
85 throw COSEDecodeError(
86 "Next item in x5chain was not of type byte string");
87 }
88 }
89 QCBORDecode_ExitArray(&ctx);
90 if (parsed.empty())
91 {
92 throw COSEDecodeError("x5chain array length was 0 in COSE header");
93 }
94 }
95 else if (x5chain.uDataType == QCBOR_TYPE_BYTE_STRING)
96 {
97 parsed.push_back(qcbor_buf_to_byte_vector(x5chain.val.string));
98 }
99 else
100 {
101 throw COSEDecodeError(fmt::format(
102 "Value type {} of x5chain in COSE header is not array or byte string",
103 x5chain.uDataType));
104 }
105
106 return parsed;
107 }

109 static UvmEndorsementsProtectedHeader decode_protected_header(
110 const std::vector<uint8_t>& uvm_endorsements_raw)
111 {
112 UsefulBufC msg{uvm_endorsements_raw.data(), uvm_endorsements_raw.size()};
113
114 QCBORError qcbor_result;
115
116 QCBORDecodeContext ctx;
117 QCBORDecode_Init(&ctx, msg, QCBOR_DECODE_MODE_NORMAL);
118
119 QCBORDecode_EnterArray(&ctx, nullptr);
120 qcbor_result = QCBORDecode_GetError(&ctx);
121 if (qcbor_result != QCBOR_SUCCESS)
122 {
123 throw COSEDecodeError("Failed to parse COSE_Sign1 outer array");
124 }
125
126 uint64_t tag = QCBORDecode_GetNthTagOfLast(&ctx, 0);
127 if (tag != CBOR_TAG_COSE_SIGN1)
128 {
129 throw COSEDecodeError("Failed to parse COSE_Sign1 tag");
130 }
131
132 struct q_useful_buf_c protected_parameters;
133 QCBORDecode_EnterBstrWrapped(
134 &ctx, QCBOR_TAG_REQUIREMENT_NOT_A_TAG, &protected_parameters);
135 QCBORDecode_EnterMap(&ctx, NULL);
136
137 enum
138 {
139 ALG_INDEX,
140 CONTENT_TYPE_INDEX,
141 X5_CHAIN_INDEX,
142 ISS_INDEX,
143 FEED_INDEX,
144 END_INDEX
145 };
146 QCBORItem header_items[END_INDEX + 1];
147
148 header_items[ALG_INDEX].label.int64 = headers::PARAM_ALG;
149 header_items[ALG_INDEX].uLabelType = QCBOR_TYPE_INT64;
150 header_items[ALG_INDEX].uDataType = QCBOR_TYPE_INT64;
151
152 header_items[CONTENT_TYPE_INDEX].label.int64 =
153 headers::PARAM_CONTENT_TYPE;
154 header_items[CONTENT_TYPE_INDEX].uLabelType = QCBOR_TYPE_INT64;
155 header_items[CONTENT_TYPE_INDEX].uDataType = QCBOR_TYPE_TEXT_STRING;
156
157 header_items[X5_CHAIN_INDEX].label.int64 = headers::PARAM_X5CHAIN;
158 header_items[X5_CHAIN_INDEX].uLabelType = QCBOR_TYPE_INT64;
159 header_items[X5_CHAIN_INDEX].uDataType = QCBOR_TYPE_ANY;
160
161 header_items[ISS_INDEX].label.string =
162 UsefulBuf_FromSZ(HEADER_PARAM_ISSUER);
163 header_items[ISS_INDEX].uLabelType = QCBOR_TYPE_TEXT_STRING;
164 header_items[ISS_INDEX].uDataType = QCBOR_TYPE_TEXT_STRING;
165
166 header_items[FEED_INDEX].label.string =
167 UsefulBuf_FromSZ(HEADER_PARAM_FEED);
168 header_items[FEED_INDEX].uLabelType = QCBOR_TYPE_TEXT_STRING;
169 header_items[FEED_INDEX].uDataType = QCBOR_TYPE_TEXT_STRING;
170
171 header_items[END_INDEX].uLabelType = QCBOR_TYPE_NONE;
172
173 QCBORDecode_GetItemsInMap(&ctx, header_items);
174 qcbor_result = QCBORDecode_GetError(&ctx);
175 if (qcbor_result != QCBOR_SUCCESS)
176 {
177 throw COSEDecodeError("Failed to decode protected header");
178 }
179
180 UvmEndorsementsProtectedHeader phdr = {};
181
182 if (header_items[ALG_INDEX].uDataType != QCBOR_TYPE_NONE)
183 {
184 phdr.alg = header_items[ALG_INDEX].val.int64;
185 }
186
187 if (header_items[CONTENT_TYPE_INDEX].uDataType != QCBOR_TYPE_NONE)
188 {
189 phdr.content_type =
190 qcbor_buf_to_string(header_items[CONTENT_TYPE_INDEX].val.string);
191 }
192
193 if (header_items[X5_CHAIN_INDEX].uDataType != QCBOR_TYPE_NONE)
194 {
195 phdr.x5_chain = decode_x5chain(ctx, header_items[X5_CHAIN_INDEX]);
196 }
197
198 if (header_items[ISS_INDEX].uDataType != QCBOR_TYPE_NONE)
199 {
200 phdr.iss = qcbor_buf_to_string(header_items[ISS_INDEX].val.string);
201 }
202
203 if (header_items[FEED_INDEX].uDataType != QCBOR_TYPE_NONE)
204 {
205 phdr.feed = qcbor_buf_to_string(header_items[FEED_INDEX].val.string);
206 }
207
208 QCBORDecode_ExitMap(&ctx);
209 QCBORDecode_ExitBstrWrapped(&ctx);
210
211 qcbor_result = QCBORDecode_GetError(&ctx);
212 if (qcbor_result != QCBOR_SUCCESS)
213 {
214 throw COSEDecodeError(
215 fmt::format("Failed to decode protected header: {}", qcbor_result));
216 }
217
218 return phdr;
219 }

222 static std::span<const uint8_t> verify_uvm_endorsements_signature(
223 const crypto::RSAPublicKeyPtr& leef_cert_pub_key,
224 const std::vector<uint8_t>& uvm_endorsements_raw)
225 {
226 auto verifier = crypto::make_cose_verifier(leef_cert_pub_key);
227
228 std::span<uint8_t> payload;
229 if (!verifier->verify(uvm_endorsements_raw, payload))
230 {
231 throw cose::COSESignatureValidationError("Signature verification failed");
232 }
233
234 return payload;

237 static UVMEndorsements verify_uvm_endorsements(
238 const std::vector<uint8_t>& uvm_endorsements_raw,
239 const pal::PlatformAttestationMeasurement& uvm_measurement)
240 {
241 auto phdr = cose::decode_protected_header(uvm_endorsements_raw);
242
243 if (!cose::is_rsa_alg(phdr.alg))
244 {
245 throw std::logic_error(
246 fmt::format("Signature algorithm {} is not expected RSA", phdr.alg));
247 }
248
249 std::string pem_chain;
250 for (auto const& c : phdr.x5_chain)
251 {
252 pem_chain += crypto::cert_der_to_pem(c).str();
253 }
254
255 const auto& did = phdr.iss;
256
257 auto did_document_str = didx509::resolve(pem_chain, did);
258 did::DIDDocument did_document = nlohmann::json::parse(did_document_str);
259
260 if (did_document.verification_method.empty())
261 {
262 throw std::logic_error(fmt::format(
263 "Could not find verification method for DID document: {}",
264 did_document_str));
265 }
266
267 crypto::RSAPublicKeyPtr pubk = nullptr;
268 for (auto const& vm : did_document.verification_method)
269 {
270 if (vm.controller == did && vm.public_key_jwk.has_value())
271 {
272 pubk = crypto::make_rsa_public_key(vm.public_key_jwk.value());
273 break;
274 }
275 }
276
277 if (pubk == nullptr)
278 {
279 throw std::logic_error(fmt::format(
280 "Could not find matching public key for DID {} in {}",
281 did,
282 did_document_str));
283 }
284
285 auto raw_payload =
286 verify_uvm_endorsements_signature(pubk, uvm_endorsements_raw);
287
288 if (phdr.content_type != cose::headers::CONTENT_TYPE_APPLICATION_JSON_VALUE)
289 {
290 throw std::logic_error(fmt::format(
291 "Unexpected payload content type {}, expected {}",
292 phdr.content_type,
293 cose::headers::CONTENT_TYPE_APPLICATION_JSON_VALUE));
294 }
295
296 UVMEndorsementsPayload payload = nlohmann::json::parse(raw_payload);
297 if (payload.sevsnpvm_launch_measurement != uvm_measurement.hex_str())
298 {
299 throw std::logic_error(fmt::format(
300 "Launch measurement in UVM endorsements payload {} is not equal "
301 "to UVM attestation measurement {}",
302 payload.sevsnpvm_launch_measurement,
303 uvm_measurement.hex_str()));
304 }
305
306 LOG_INFO_FMT(
307 "Successfully verified endorsements for attested measurement {} against "
308 "{}, feed {}, svn {}",
309 payload.sevsnpvm_launch_measurement,
310 did,
311 phdr.feed,
312 payload.sevsnpvm_guest_svn);
313
314 return {did, phdr.feed, payload.sevsnpvm_guest_svn};
315 }
