54 const std::shared_ptr<ccf::RpcContext>& ctx,
55 std::string& error_reason) override
56 {
57 const auto& executor_cert = ctx->get_session_context()->caller_cert;
58 if (executor_cert.empty())
59 {
60 error_reason = "No Executor certificate";
61 return nullptr;
62 }
63
64 auto pubk_der = crypto::public_key_der_from_cert(executor_cert);
65 auto executor_id = crypto::Sha256Hash(pubk_der).hex_str();
66
67 if (executor_certs_map.find(executor_id) != executor_certs_map.end())
68 {
69 auto executor_identity = std::make_unique<ExecutorIdentity>();
70 executor_identity->executor_id = executor_id;
71 return executor_identity;
72 }
73 error_reason = "Could not find matching Executor certificate";
74 return nullptr;
75 }
