26 inline std::pair<grpc_status, std::string> verification_error(
27 ccf::QuoteVerificationResult result)
28 {
29 switch (result)
30 {
31 case ccf::QuoteVerificationResult::Failed:
32 return std::make_pair(
33 GRPC_STATUS_UNAUTHENTICATED, "Quote could not be verified");
34 case ccf::QuoteVerificationResult::FailedMeasurementNotFound:
35 return std::make_pair(
36 GRPC_STATUS_UNAUTHENTICATED,
37 "Quote does not contain known enclave measurement");
38 case ccf::QuoteVerificationResult::FailedInvalidQuotedPublicKey:
39 return std::make_pair(
40 GRPC_STATUS_UNAUTHENTICATED,
41 "Quote report data does not contain node's public key hash");
42 case ccf::QuoteVerificationResult::FailedHostDataDigestNotFound:
43 return std::make_pair(
44 GRPC_STATUS_UNAUTHENTICATED, "Quote does not contain host data");
45 case ccf::QuoteVerificationResult::FailedInvalidHostData:
46 return std::make_pair(
47 GRPC_STATUS_UNAUTHENTICATED, "Quote host data is not authorised");
48 case ccf::QuoteVerificationResult::FailedUVMEndorsementsNotFound:
49 return std::make_pair(
50 GRPC_STATUS_UNAUTHENTICATED, "UVM endorsements are not authorised");
51 default:
52 return std::make_pair(
53 GRPC_STATUS_INTERNAL, "Unknown quote verification error");
54 }
55 }
