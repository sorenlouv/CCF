265 template <typename In, typename Out>
266 Endpoint& set_auto_schema(std::optional<http_status> status = std::nullopt)
267 {
268 if constexpr (!std::is_same_v<In, void>)
269 {
270 params_schema = ds::json::build_schema<In>();
271
272 schema_builders.push_back(
273 [](nlohmann::json& document, const Endpoint& endpoint) {
274 const auto http_verb = endpoint.dispatch.verb.get_http_method();
275 if (!http_verb.has_value())
276 {
277 // Non-HTTP endpoints are not documented
278 return;
279 }
280
281 ds::openapi::add_request_body_schema<In>(
282 document, endpoint.full_uri_path, http_verb.value());
283 });
284 }
285 else
286 {
287 params_schema = nullptr;
288 }
289
290 if constexpr (!std::is_same_v<Out, void>)
291 {
292 success_status = status.value_or(HTTP_STATUS_OK);
293
294 result_schema = ds::json::build_schema<Out>();
295
296 schema_builders.push_back(
297 [](nlohmann::json& document, const Endpoint& endpoint) {
298 const auto http_verb = endpoint.dispatch.verb.get_http_method();
299 if (!http_verb.has_value())
300 {
301 return;
302 }
303
304 ds::openapi::add_response_schema<Out>(
305 document,
306 endpoint.full_uri_path,
307 http_verb.value(),
308 endpoint.success_status);
309 });
310 }
311 else
312 {
313 success_status = status.value_or(HTTP_STATUS_NO_CONTENT);
314 result_schema = nullptr;
315 }
316
317 return *this;
318 }

351 template <typename T>
352 Endpoint& add_query_parameter(
353 const std::string& param_name,
354 QueryParamPresence presence = RequiredParameter)
355 {
356 schema_builders.push_back(
357 [param_name,
358 presence](nlohmann::json& document, const Endpoint& endpoint) {
359 const auto http_verb = endpoint.dispatch.verb.get_http_method();
360 if (!http_verb.has_value())
361 {
362 // Non-HTTP endpoints are not documented
363 return;
364 }
365
366 const auto schema_name = ds::json::schema_name<T>();
367 const auto query_schema = ds::json::build_schema<T>();
368
369 auto parameter = nlohmann::json::object();
370 parameter["name"] = param_name;
371 parameter["in"] = "query";
372 parameter["required"] = presence == RequiredParameter;
373 parameter["schema"] = ds::openapi::add_schema_to_components(
374 document, schema_name, query_schema);
375 ds::openapi::add_request_parameter_schema(
376 document, endpoint.full_uri_path, http_verb.value(), parameter);
377 });
378
379 return *this;
380 }
