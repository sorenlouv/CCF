20 std::optional<ApiVersion> get_api_version(
21 ccf::endpoints::CommandEndpointContext& ctx)
22 {
23 const auto param_name = "api-version";
24 const auto parsed_query =
25 http::parse_query(ctx.rpc_ctx->get_request_query());
26 const auto qit = parsed_query.find(param_name);
27 if (qit == parsed_query.end())
28 {
29 ctx.rpc_ctx->set_error(
30 HTTP_STATUS_BAD_REQUEST,
31 ccf::errors::MissingApiVersionParameter,
32 fmt::format(
33 "The api-version query parameter (?{}=) is required for all "
34 "requests.",
35 param_name));
36 return std::nullopt;
37 }
38
39 const auto it = std::find_if(
40 std::begin(api_version_strings),
41 std::end(api_version_strings),
42 [&qit](const auto& p) { return p.second == qit->second; });
43 if (it == std::end(api_version_strings))
44 {
45 auto message = fmt::format(
46 "Unsupported api-version '{}'. The supported api-versions are: ",
47 qit->second);
48 auto first = true;
49 for (const auto& p : api_version_strings)
50 {
51 if (first)
52 {
53 message += p.second;
54 first = false;
55 }
56 else
57 {
58 message += fmt::format(", {}", p.second);
59 }
60 }
61 ctx.rpc_ctx->set_error(
62 HTTP_STATUS_BAD_REQUEST,
63 ccf::errors::UnsupportedApiVersionValue,
64 std::move(message));
65 return std::nullopt;
66 }
67
68 return it->first;
69 }

71 // Extracts api-version from query parameter, and passes this to the given
72 // functor. Will return error responses for missing and unknown api-versions.
73 // This means handler functors can safely provide a default implementation
74 // without validating the given API version, so long as the behaviour is the
75 // same for *all* accepted versions.
76 template <typename Fn>
77 auto api_version_adapter(Fn&& f)
78 {
79 return [f](auto& ctx) {
80 const auto api_version = get_api_version(ctx);
81 if (api_version.has_value())
82 {
83 f(ctx, api_version.value());
84 }
85
86 return;
87 };
88 }
