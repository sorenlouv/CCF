113 static EndorsementEndpointsConfiguration::Server make_amd_endorsements_server(
114 const HostPort& endpoint,
115 const std::string& chip_id_hex,
116 const std::string& boot_loader,
117 const std::string& tee,
118 const std::string& snp,
119 const std::string& microcode)
120 {
121 std::map<std::string, std::string> params;
122 params["blSPL"] = boot_loader;
123 params["teeSPL"] = tee;
124 params["snpSPL"] = snp;
125 params["ucodeSPL"] = microcode;
126
127 EndorsementEndpointsConfiguration::Server server;
128 server.push_back({
129 endpoint.host,
130 endpoint.port,
131 fmt::format("/vcek/v1/{}/{}", product_name, chip_id_hex),
132 params,
133 true // DER
134 });
135 server.push_back(
136 {endpoint.host,
137 endpoint.port,
138 fmt::format("/vcek/v1/{}/cert_chain", product_name),
139 {}});
140
141 return server;
142 }
