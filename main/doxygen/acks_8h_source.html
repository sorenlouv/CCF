18 auto get_state_digest = [&](auto& ctx, ApiVersion api_version) {
19 switch (api_version)
20 {
21 case ApiVersion::preview_v1:
22 default:
23 {
24 // Get memberId from path parameter
25 std::string error;
26 std::string member_id_str;
27 if (!ccf::endpoints::get_path_param(
28 ctx.rpc_ctx->get_request_path_params(),
29 "memberId",
30 member_id_str,
31 error))
32 {
33 detail::set_gov_error(
34 ctx.rpc_ctx,
35 HTTP_STATUS_BAD_REQUEST,
36 ccf::errors::InvalidResourceName,
37 std::move(error));
38 return;
39 }
40
41 // Read member's ack from KV
42 ccf::MemberId member_id(member_id_str);
43 auto acks_handle =
44 ctx.tx.template ro<ccf::MemberAcks>(Tables::MEMBER_ACKS);
45 auto ack = acks_handle->get(member_id);
46 if (!ack.has_value())
47 {
48 detail::set_gov_error(
49 ctx.rpc_ctx,
50 HTTP_STATUS_NOT_FOUND,
51 ccf::errors::ResourceNotFound,
52 fmt::format("No ACK record exists for member {}.", member_id));
53 return;
54 }
55
56 auto response_body = nlohmann::json::object();
57 response_body["memberId"] = member_id_str;
58 response_body["stateDigest"] = ack->state_digest;
59 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
60 return;
61 }
62 }
63 };
64 registry
65 .make_read_only_endpoint(
66 "/members/state-digests/{memberId}",
67 HTTP_GET,
68 api_version_adapter(get_state_digest),
69 no_auth_required)
70 .set_openapi_hidden(true)
71 .install();

73 auto update_state_digest = [&](auto& ctx, ApiVersion api_version) {
74 switch (api_version)
75 {
76 case ApiVersion::preview_v1:
77 default:
78 {
79 // Get memberId from path parameter
80 std::string error;
81 std::string member_id_str;
82 if (!ccf::endpoints::get_path_param(
83 ctx.rpc_ctx->get_request_path_params(),
84 "memberId",
85 member_id_str,
86 error))
87 {
88 detail::set_gov_error(
89 ctx.rpc_ctx,
90 HTTP_STATUS_BAD_REQUEST,
91 ccf::errors::InvalidResourceName,
92 std::move(error));
93 return;
94 }
95
96 // Confirm this matches memberId from signature
97 ccf::MemberId member_id(member_id_str);
98 const auto& cose_ident =
99 ctx.template get_caller<ccf::MemberCOSESign1AuthnIdentity>();
100 if (cose_ident.member_id != member_id)
101 {
102 detail::set_gov_error(
103 ctx.rpc_ctx,
104 HTTP_STATUS_BAD_REQUEST,
105 ccf::errors::InvalidAuthenticationInfo,
106 fmt::format(
107 "Member ID from path parameter ({}) does not match "
108 "member ID from body signature ({}).",
109 member_id,
110 cose_ident.member_id));
111 return;
112 }
113
114 ccf::MemberAck ack;
115
116 // Get previous ack, if it exists
117 auto acks_handle =
118 ctx.tx.template rw<ccf::MemberAcks>(Tables::MEMBER_ACKS);
119 auto ack_opt = acks_handle->get(member_id);
120 if (ack_opt.has_value())
121 {
122 ack = ack_opt.value();
123 }
124
125 // Get signature, containing merkle root state digest
126 auto sigs_handle =
127 ctx.tx.template ro<ccf::Signatures>(Tables::SIGNATURES);
128 auto sig = sigs_handle->get();
129 if (!sig.has_value())
130 {
131 detail::set_gov_error(
132 ctx.rpc_ctx,
133 HTTP_STATUS_INTERNAL_SERVER_ERROR,
134 ccf::errors::InternalError,
135 "Service has no signatures to ack yet - try again soon.");
136 return;
137 }
138
139 // Write ack back to the KV
140 ack.state_digest = sig->root.hex_str();
141 acks_handle->put(member_id, ack);
142
143 auto body = nlohmann::json::object();
144 body["memberId"] = member_id_str;
145 body["stateDigest"] = ack.state_digest;
146 ctx.rpc_ctx->set_response_json(body, HTTP_STATUS_OK);
147 return;
148 }
149 }
150 };

160 auto ack_state_digest = [&](auto& ctx, ApiVersion api_version) {
161 switch (api_version)
162 {
163 case ApiVersion::preview_v1:
164 default:
165 {
166 // Get memberId from path parameter
167 std::string error;
168 std::string member_id_str;
169 if (!ccf::endpoints::get_path_param(
170 ctx.rpc_ctx->get_request_path_params(),
171 "memberId",
172 member_id_str,
173 error))
174 {
175 detail::set_gov_error(
176 ctx.rpc_ctx,
177 HTTP_STATUS_BAD_REQUEST,
178 ccf::errors::InvalidResourceName,
179 std::move(error));
180 return;
181 }
182
183 // Confirm this matches memberId from signature
184 ccf::MemberId member_id(member_id_str);
185 const auto& cose_ident =
186 ctx.template get_caller<ccf::MemberCOSESign1AuthnIdentity>();
187 if (cose_ident.member_id != member_id)
188 {
189 detail::set_gov_error(
190 ctx.rpc_ctx,
191 HTTP_STATUS_BAD_REQUEST,
192 ccf::errors::InvalidAuthenticationInfo,
193 fmt::format(
194 "Member ID from path parameter ({}) does not match "
195 "member ID from body signature ({}).",
196 member_id,
197 cose_ident.member_id));
198 return;
199 }
200
201 // Check an expected digest for this member is in the KV
202 auto acks_handle =
203 ctx.tx.template rw<ccf::MemberAcks>(Tables::MEMBER_ACKS);
204 auto ack = acks_handle->get(member_id);
205 if (!ack.has_value())
206 {
207 detail::set_gov_error(
208 ctx.rpc_ctx,
209 HTTP_STATUS_FORBIDDEN,
210 ccf::errors::AuthorizationFailed,
211 fmt::format("No ACK record exists for member {}.", member_id));
212 return;
213 }
214
215 // Check signed digest matches expected digest in KV
216 const auto expected_digest = ack->state_digest;
217 const auto signed_body = nlohmann::json::parse(cose_ident.content);
218 const auto actual_digest =
219 signed_body["stateDigest"].template get<std::string>();
220 if (expected_digest != actual_digest)
221 {
222 detail::set_gov_error(
223 ctx.rpc_ctx,
224 HTTP_STATUS_BAD_REQUEST,
225 ccf::errors::StateDigestMismatch,
226 fmt::format(
227 "Submitted state digest is not valid.\n"
228 "Expected\n"
229 " {}\n"
230 "Received\n"
231 " {}",
232 expected_digest,
233 actual_digest));
234 return;
235 }
236
237 // Ensure old HTTP signed req is nulled
238 ack->signed_req = std::nullopt;
239
240 // Insert new signature
241 ack->cose_sign1_req = std::vector<uint8_t>(
242 cose_ident.envelope.begin(), cose_ident.envelope.end());
243
244 // Store signed ACK in KV
245 acks_handle->put(member_id, ack.value());
246
247 // Update member details
248 {
249 // Update member status to ACTIVE
250 bool newly_active = false;
251 try
252 {
253 newly_active =
254 InternalTablesAccess::activate_member(ctx.tx, member_id);
255 }
256 catch (const std::logic_error& e)
257 {
258 detail::set_gov_error(
259 ctx.rpc_ctx,
260 HTTP_STATUS_INTERNAL_SERVER_ERROR,
261 ccf::errors::InternalError,
262 fmt::format("Error activating member: {}", e.what()));
263 return;
264 }
265
266 // If this is a newly-active recovery member in an open service,
267 // allocate them a recovery share immediately
268 if (
269 newly_active &&
270 InternalTablesAccess::is_recovery_member(ctx.tx, member_id))
271 {
272 auto service_status =
273 InternalTablesAccess::get_service_status(ctx.tx);
274 if (!service_status.has_value())
275 {
276 detail::set_gov_error(
277 ctx.rpc_ctx,
278 HTTP_STATUS_INTERNAL_SERVER_ERROR,
279 ccf::errors::InternalError,
280 "No service currently available.");
281 return;
282 }
283
284 if (service_status.value() == ServiceStatus::OPEN)
285 {
286 try
287 {
288 share_manager.shuffle_recovery_shares(ctx.tx);
289 }
290 catch (const std::logic_error& e)
291 {
292 detail::set_gov_error(
293 ctx.rpc_ctx,
294 HTTP_STATUS_INTERNAL_SERVER_ERROR,
295 ccf::errors::InternalError,
296 fmt::format(
297 "Error issuing new recovery shares: {}", e.what()));
298 return;
299 }
300 }
301 }
302 }
303
304 ctx.rpc_ctx->set_response_status(HTTP_STATUS_NO_CONTENT);
305 return;
306 break;
307 }
308 }
309 };
