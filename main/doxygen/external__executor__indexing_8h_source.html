59 class LRUIndexCache
60 {
61 public:
62 using Entry = std::pair<const std::string, std::string>;
63 using List = std::list<Entry>;
64 using Map = std::map<std::string, typename List::iterator>;
65
66 private:
67 ccf::pal::Mutex results_access;
68 // Entries are ordered by when they were most recently accessed, with most
69 // recent at the front
70 List entries_list;
71
72 // Maps from keys to iterators from entries_list, which must remain valid
73 // even when entries_list is modified
74 Map iter_map;
75
76 std::shared_ptr<ccf::indexing::AbstractLFSAccess> lfs_access;
77 size_t max_size;
78 const std::string map_name;
79
80 void flush_to_disk()
81 {
82 while (entries_list.size() > max_size)
83 {
84 const auto& least_recent_entry = entries_list.back();
85 auto key = least_recent_entry.first;
86 auto value = least_recent_entry.second;
87 std::string blob_name = get_blob_name(map_name, key);
88 ccf::indexing::LFSContents contents(value.begin(), value.end());
89 lfs_access->store(blob_name, std::move(contents));
90 iter_map.erase(least_recent_entry.first);
91 entries_list.pop_back();
92 }
93 }
94
95 public:
96 LRUIndexCache() {}
97 LRUIndexCache(
98 std::shared_ptr<ccf::indexing::AbstractLFSAccess> lfs_ptr,
99 size_t max_size,
100 const std::string& map) :
101 lfs_access(lfs_ptr),
102 max_size(max_size),
103 map_name(map)
104 {}
105
106 size_t size() const
107 {
108 return iter_map.size();
109 }
110
111 void set_max_size(size_t ms)
112 {
113 max_size = ms;
114 }
115
116 size_t get_max_size() const
117 {
118 return max_size;
119 }

121 std::optional<std::string> find(const std::string& k)
122 {
123 const auto it = iter_map.find(k);
124 if (it != iter_map.end())
125 {
126 // move it to the front and return
127 auto& list_it = it->second;
128 entries_list.splice(entries_list.begin(), entries_list, list_it);
129 return list_it->second;
130 }
131
132 return std::nullopt;
133 }
134
135 void insert(const std::string& k, const std::string& v)
136 {
137 std::lock_guard<ccf::pal::Mutex> guard(results_access);
138 auto it = iter_map.find(k);
139 if (it != iter_map.end())
140 {
141 // If it already exists, move to the front
142 auto& list_it = it->second;
143 list_it->second = v;
144 entries_list.splice(entries_list.begin(), entries_list, list_it);
145 }
146 else
147 {
148 // Else add a new entry to both containers, and flush if necessary
149 entries_list.push_front(std::make_pair(k, v));
150 const auto list_it = entries_list.begin();
151 iter_map.emplace_hint(it, k, list_it);
152 flush_to_disk();
153 }
154 }
155
156 void clear()
157 {
158 entries_list.clear();
159 iter_map.clear();
160 }
161 };
