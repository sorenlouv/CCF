48 void handle_incoming_data_thread(std::vector<uint8_t>&& data) override
49 {
50 tls_io->recv_buffered(data.data(), data.size());
51
52 LOG_TRACE_FMT("recv called with {} bytes", data.size());
53
54 // Try to parse all incoming data, reusing the vector we were just passed
55 // for storage. Increase the size if the received vector was too small
56 // (for the case where this chunk is very small, but we had some previous
57 // data to continue reading).
58 constexpr auto min_read_block_size = 4096;
59 if (data.size() < min_read_block_size)
60 {
61 data.resize(min_read_block_size);
62 }
63
64 auto n_read = tls_io->read(data.data(), data.size(), false);
65
66 while (true)
67 {
68 if (n_read == 0)
69 {
70 return;
71 }
72
73 LOG_TRACE_FMT("Going to parse {} bytes", n_read);
74
75 bool cont = parse({data.data(), n_read});
76 if (!cont)
77 {
78 return;
79 }
80
81 // Used all provided bytes - check if more are available
82 n_read = tls_io->read(data.data(), data.size(), false);
83 }
84 }

114 bool parse(std::span<const uint8_t> data) override
115 {
116 // Catch request parsing errors and convert them to error responses
117 try
118 {
119 request_parser.execute(data.data(), data.size());
120
121 return true;
122 }
123 catch (RequestPayloadTooLargeException& e)
124 {
125 if (error_reporter)
126 {
127 error_reporter->report_request_payload_too_large_error(interface_id);
128 }
129
130 LOG_DEBUG_FMT("Request is too large: {}", e.what());
131
132 send_odata_error_response(ccf::ErrorDetails{
133 HTTP_STATUS_PAYLOAD_TOO_LARGE,
134 ccf::errors::RequestBodyTooLarge,
135 e.what()});
136
137 tls_io->close();
138 }
139 catch (RequestHeaderTooLargeException& e)
140 {
141 if (error_reporter)
142 {
143 error_reporter->report_request_header_too_large_error(interface_id);
144 }
145
146 LOG_DEBUG_FMT("Request header is too large: {}", e.what());
147
148 send_odata_error_response(ccf::ErrorDetails{
149 HTTP_STATUS_REQUEST_HEADER_FIELDS_TOO_LARGE,
150 ccf::errors::RequestHeaderTooLarge,
151 e.what()});
152
153 tls_io->close();
154 }
155 catch (const std::exception& e)
156 {
157 if (error_reporter)
158 {
159 error_reporter->report_parsing_error(interface_id);
160 }
161 LOG_DEBUG_FMT("Error parsing HTTP request: {}", e.what());
162
163 http::HeaderMap headers;
164 headers[http::headers::CONTENT_TYPE] =
165 http::headervalues::contenttype::TEXT;
166
167 // NB: Avoid formatting input data a string, as it may contain null
168 // bytes. Instead insert it at the end of this message, verbatim
169 auto body_s = fmt::format(
170 "Unable to parse data as a HTTP request. Error message is: {}\n"
171 "Error occurred while parsing fragment:\n",
172 e.what());
173 std::vector<uint8_t> response_body(
174 std::begin(body_s), std::end(body_s));
175 response_body.insert(response_body.end(), data.begin(), data.end());
176
177 send_response(
178 HTTP_STATUS_BAD_REQUEST,
179 std::move(headers),
180 {},
181 std::move(response_body));
182
183 tls_io->close();
184 }
185
186 return false;
187 }

189 void handle_request(
190 llhttp_method verb,
191 const std::string_view& url,
192 http::HeaderMap&& headers,
193 std::vector<uint8_t>&& body,
194 int32_t) override
195 {
196 LOG_TRACE_FMT(
197 "Processing msg({}, {} [{} bytes])",
198 llhttp_method_name(verb),
199 url,
200 body.size());
201
202 try
203 {
204 if (session_ctx == nullptr)
205 {
206 session_ctx = std::make_shared<ccf::SessionContext>(
207 session_id, tls_io->peer_cert(), interface_id);
208 }
209
210 std::shared_ptr<http::HttpRpcContext> rpc_ctx = nullptr;
211 try
212 {
213 rpc_ctx = std::make_shared<HttpRpcContext>(
214 session_ctx,
215 ccf::HttpVersion::HTTP1,
216 verb,
217 url,
218 std::move(headers),
219 std::move(body));
220 }
221 catch (std::exception& e)
222 {
223 send_odata_error_response(ccf::ErrorDetails{
224 HTTP_STATUS_INTERNAL_SERVER_ERROR,
225 ccf::errors::InternalError,
226 fmt::format("Error constructing RpcContext: {}", e.what())});
227 tls_io->close();
228 }
229
230 std::shared_ptr<ccf::RpcHandler> search =
231 http::fetch_rpc_handler(rpc_ctx, rpc_map);
232
233 search->process(rpc_ctx);
234
235 if (rpc_ctx->response_is_pending)
236 {
237 // If the RPC is pending, hold the connection.
238 LOG_TRACE_FMT("Pending");
239 return;
240 }
241 else
242 {
243 send_response(
244 rpc_ctx->get_response_http_status(),
245 rpc_ctx->get_response_headers(),
246 rpc_ctx->get_response_trailers(),
247 std::move(rpc_ctx->get_response_body()));
248
249 if (rpc_ctx->terminate_session)
250 {
251 tls_io->close();
252 }
253 }
254 }
255 catch (const std::exception& e)
256 {
257 send_odata_error_response(ccf::ErrorDetails{
258 HTTP_STATUS_INTERNAL_SERVER_ERROR,
259 ccf::errors::InternalError,
260 fmt::format("Exception: {}", e.what())});
261
262 // On any exception, close the connection.
263 LOG_FAIL_FMT("Closing connection");
264 LOG_DEBUG_FMT("Closing connection due to exception: {}", e.what());
265 tls_io->close();
266 throw;
267 }
268 }

270 bool send_response(
271 http_status status_code,
272 http::HeaderMap&& headers,
273 http::HeaderMap&& trailers,
274 std::span<const uint8_t> body) override
275 {
276 if (!trailers.empty())
277 {
278 throw std::logic_error("Cannot return trailers over HTTP/1");
279 }
280
281 auto response = http::Response(status_code);
282 for (const auto& [k, v] : headers)
283 {
284 response.set_header(k, v);
285 }
286 response.set_body(body.data(), body.size());
287
288 auto data = response.build_response();
289 tls_io->send_raw(data.data(), data.size());
290 return true;
291 }
