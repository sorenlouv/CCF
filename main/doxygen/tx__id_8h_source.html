41 // The combination of View and SeqNo produce a unique TxID for each
42 // transaction executed by CCF.
43 struct TxID
44 {

46 SeqNo seqno;

48 std::string to_str() const
49 {

51 }

53 static std::optional<TxID> from_str(const std::string_view& sv)
54 {
55 const auto separator_idx = sv.find(".");
56 if (separator_idx == std::string_view::npos)
57 {
58 return std::nullopt;
59 }
60
61 TxID tx_id;
62
63 {
64 const auto view_sv = sv.substr(0, separator_idx);
65 const auto [p, ec] =
66 std::from_chars(view_sv.begin(), view_sv.end(), tx_id.view);
67 if (ec != std::errc() || p != view_sv.end())
68 {
69 return std::nullopt;
70 }
71 }
72
73 {
74 const auto seqno_sv = sv.substr(separator_idx + 1);
75 const auto [p, ec] =
76 std::from_chars(seqno_sv.begin(), seqno_sv.end(), tx_id.seqno);
77 if (ec != std::errc() || p != seqno_sv.end())
78 {
79 return std::nullopt;
80 }
81 }
82
83 return tx_id;
84 }
85
86 bool operator==(const TxID& other) const
87 {
88 return view == other.view && seqno == other.seqno;
89 }
90 };
