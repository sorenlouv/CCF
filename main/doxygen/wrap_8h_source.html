160 int set_getter(const char* prop, JSWrappedValue&& getter) const
161 {
162 JSAtom size_atom = JS_NewAtom(ctx, prop);
163 if (size_atom == JS_ATOM_NULL)
164 {
165 getter.val = ccf::js::constants::Null;
166 return -1;
167 }
168
169 // NB: Where other calls check the return code to determine whether they
170 // are responsible for freeing, this call unconditionally frees the getter
171 // arg, so we call .take() to always drop our local owning reference
172 int rc = JS_DefinePropertyGetSet(
173 ctx, val, size_atom, getter.take(), ccf::js::constants::Undefined, 0);
174
175 JS_FreeAtom(ctx, size_atom);
176
177 return rc;
178 }
