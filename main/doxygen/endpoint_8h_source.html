275 template <typename In, typename Out>
276 Endpoint& set_auto_schema(std::optional<http_status> status = std::nullopt)
277 {
278 if constexpr (!std::is_same_v<In, void>)
279 {
280 params_schema = ds::json::build_schema<In>();
281
282 schema_builders.push_back(
283 [](nlohmann::json& document, const Endpoint& endpoint) {
284 const auto http_verb = endpoint.dispatch.verb.get_http_method();
285 if (!http_verb.has_value())
286 {
287 // Non-HTTP endpoints are not documented
288 return;
289 }
290
291 ds::openapi::add_request_body_schema<In>(
292 document, endpoint.full_uri_path, http_verb.value());
293 });
294 }
295 else
296 {
297 params_schema = nullptr;
298 }
299
300 if constexpr (!std::is_same_v<Out, void>)
301 {
302 success_status = status.value_or(HTTP_STATUS_OK);
303
304 result_schema = ds::json::build_schema<Out>();
305
306 schema_builders.push_back(
307 [](nlohmann::json& document, const Endpoint& endpoint) {
308 const auto http_verb = endpoint.dispatch.verb.get_http_method();
309 if (!http_verb.has_value())
310 {
311 return;
312 }
313
314 ds::openapi::add_response_schema<Out>(
315 document,
316 endpoint.full_uri_path,
317 http_verb.value(),
318 endpoint.success_status);
319 });
320 }
321 else
322 {
323 success_status = status.value_or(HTTP_STATUS_NO_CONTENT);
324 result_schema = nullptr;
325 }
326
327 return *this;
328 }

361 template <typename T>
362 Endpoint& add_query_parameter(
363 const std::string& param_name,
364 QueryParamPresence presence = RequiredParameter)
365 {
366 schema_builders.push_back(
367 [param_name,
368 presence](nlohmann::json& document, const Endpoint& endpoint) {
369 const auto http_verb = endpoint.dispatch.verb.get_http_method();
370 if (!http_verb.has_value())
371 {
372 // Non-HTTP endpoints are not documented
373 return;
374 }
375
376 const auto schema_name = ds::json::schema_name<T>();
377 const auto query_schema = ds::json::build_schema<T>();
378
379 auto parameter = nlohmann::json::object();
380 parameter["name"] = param_name;
381 parameter["in"] = "query";
382 parameter["required"] = presence == RequiredParameter;
383 parameter["schema"] = ds::openapi::add_schema_to_components(
384 document, schema_name, query_schema);
385 ds::openapi::add_request_parameter_schema(
386 document, endpoint.full_uri_path, http_verb.value(), parameter);
387 });
388
389 return *this;
390 }
