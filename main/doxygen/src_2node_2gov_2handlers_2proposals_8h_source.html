13 struct ProposalSubmissionResult
14 {
15 enum class Status
16 {
17 Acceptable,
18 DuplicateInWindow,
19 TooOld
20 } status;
21
22 // May be empty, a colliding proposal ID, or a min_created_at value,
23 // depending on status
24 std::string info = "";

27 ProposalSubmissionResult validate_proposal_submission_time(
28 kv::Tx& tx,
29 const std::string& created_at,
30 const std::vector<uint8_t>& request_digest,
31 const ccf::ProposalId& proposal_id)
32 {
33 auto cose_recent_proposals =
34 tx.rw<ccf::COSERecentProposals>(ccf::Tables::COSE_RECENT_PROPOSALS);
35 auto key = fmt::format("{}:{}", created_at, ds::to_hex(request_digest));
36
37 if (cose_recent_proposals->has(key))
38 {
39 auto colliding_proposal_id = cose_recent_proposals->get(key).value();
40 return {
41 ProposalSubmissionResult::Status::DuplicateInWindow,
42 colliding_proposal_id};
43 }
44
45 std::vector<std::string> replay_keys;
46 cose_recent_proposals->foreach_key(
47 [&replay_keys](const std::string& replay_key) {
48 replay_keys.push_back(replay_key);
49 return true;
50 });
51
52 std::sort(replay_keys.begin(), replay_keys.end());
53
54 // New proposal must be more recent than median proposal kept
55 if (!replay_keys.empty())
56 {
57 const auto [min_created_at, _] =
58 nonstd::split_1(replay_keys[replay_keys.size() / 2], ":");
59 auto [key_ts, __] = nonstd::split_1(key, ":");
60 if (key_ts < min_created_at)
61 {
62 return {
63 ProposalSubmissionResult::Status::TooOld,
64 std::string(min_created_at)};
65 }
66 }
67
68 size_t window_size = ccf::default_recent_cose_proposals_window_size;
69 auto config_handle =
70 tx.ro<ccf::Configuration>(ccf::Tables::CONFIGURATION);
71 auto config = config_handle->get();
72 if (
73 config.has_value() &&
74 config->recent_cose_proposals_window_size.has_value())
75 {
76 window_size = config->recent_cose_proposals_window_size.value();
77 }
78 cose_recent_proposals->put(key, proposal_id);
79 // Only keep the most recent window_size proposals, to avoid
80 // unbounded memory usage
81 if (replay_keys.size() >= (window_size - 1) /* We just added one */)
82 {
83 for (size_t i = 0; i < (replay_keys.size() - (window_size - 1)); i++)
84 {
85 cose_recent_proposals->remove(replay_keys[i]);
86 }
87 }
88 return {ProposalSubmissionResult::Status::Acceptable};
89 }

102 void remove_all_other_non_open_proposals(
103 kv::Tx& tx, const ProposalId& proposal_id)
104 {
105 auto p = tx.rw<ccf::jsgov::ProposalMap>(jsgov::Tables::PROPOSALS);
106 auto pi =
107 tx.rw<ccf::jsgov::ProposalInfoMap>(jsgov::Tables::PROPOSALS_INFO);
108 std::vector<ProposalId> to_be_removed;
109 pi->foreach(
110 [&to_be_removed, &proposal_id](
111 const ProposalId& pid, const ccf::jsgov::ProposalInfo& pinfo) {
112 if (pid != proposal_id && pinfo.state != ProposalState::OPEN)
113 {
114 to_be_removed.push_back(pid);
115 }
116 return true;
117 });
118 for (const auto& pr : to_be_removed)
119 {
120 p->remove(pr);
121 pi->remove(pr);
122 }
123 }

125 ccf::jsgov::ProposalInfoSummary resolve_proposal(
126 ccfapp::AbstractNodeContext& context,
127 ccf::NetworkState& network,
128 kv::Tx& tx,
129 const ProposalId& proposal_id,
130 const std::span<const uint8_t>& proposal,
131 ccf::jsgov::ProposalInfo& proposal_info,
132 const std::string& constitution)
133 {
134 std::vector<std::pair<MemberId, bool>> votes;
135 std::optional<jsgov::Failure> failure = std::nullopt;
136 std::optional<ccf::jsgov::Votes> final_votes = std::nullopt;
137 std::optional<ccf::jsgov::VoteFailures> vote_failures = std::nullopt;
138
139 // Evaluate ballots
140 for (const auto& [mid, mb] : proposal_info.ballots)
141 {
142 js::Context js_context(js::TxAccess::GOV_RO);
143 js::populate_global_ccf_kv(tx, js_context);
144 auto ballot_func = js_context.function(
145 mb,
146 "vote",
147 fmt::format(
148 "public:ccf.gov.proposal_info[{}].ballots[{}]", proposal_id, mid));
149
150 std::vector<js::JSWrappedValue> argv = {
151 js_context.new_string_len(
152 (const char*)proposal.data(), proposal.size()),
153 js_context.new_string_len(
154 proposal_info.proposer_id.data(),
155 proposal_info.proposer_id.size())};
156
157 auto val = js_context.call_with_rt_options(
158 ballot_func,
159 argv,
160 &tx,
161 js::RuntimeLimitsPolicy::NO_LOWER_THAN_DEFAULTS);
162
163 if (!val.is_exception())
164 {
165 votes.emplace_back(mid, val.is_true());
166 }
167 else
168 {
169 if (!vote_failures.has_value())
170 {
171 vote_failures = ccf::jsgov::VoteFailures();
172 }
173
174 auto [reason, trace] = js::js_error_message(js_context);
175
176 if (js_context.interrupt_data.request_timed_out)
177 {
178 reason = "Operation took too long to complete.";
179 }
180 vote_failures.value()[mid] = ccf::jsgov::Failure{reason, trace};
181 }
182 }
183
184 // Evaluate resolve function
185 {
186 {
187 js::Context js_context(js::TxAccess::GOV_RO);
188 js::populate_global_ccf_kv(tx, js_context);
189 auto resolve_func = js_context.function(
190 constitution, "resolve", "public:ccf.gov.constitution[0]");
191
192 std::vector<js::JSWrappedValue> argv;
193 argv.push_back(js_context.new_string_len(
194 (const char*)proposal.data(), proposal.size()));
195
196 argv.push_back(js_context.new_string_len(
197 proposal_info.proposer_id.data(),
198 proposal_info.proposer_id.size()));
199
200 auto vs = js_context.new_array();
201 size_t index = 0;
202 for (auto& [mid, vote] : votes)
203 {
204 auto v = JS_NewObject(js_context);
205 auto member_id =
206 JS_NewStringLen(js_context, mid.data(), mid.size());
207 JS_DefinePropertyValueStr(
208 js_context, v, "member_id", member_id, JS_PROP_C_W_E);
209 auto vote_status = JS_NewBool(js_context, vote);
210 JS_DefinePropertyValueStr(
211 js_context, v, "vote", vote_status, JS_PROP_C_W_E);
212 JS_DefinePropertyValueUint32(
213 js_context, vs.val, index++, v, JS_PROP_C_W_E);
214 }
215 argv.push_back(vs);
216
217 auto val = js_context.call_with_rt_options(
218 resolve_func,
219 argv,
220 &tx,
221 js::RuntimeLimitsPolicy::NO_LOWER_THAN_DEFAULTS);
222
223 if (val.is_exception())
224 {
225 proposal_info.state = ProposalState::FAILED;
226 auto [reason, trace] = js::js_error_message(js_context);
227 if (js_context.interrupt_data.request_timed_out)
228 {
229 reason = "Operation took too long to complete.";
230 }
231 failure = ccf::jsgov::Failure{
232 fmt::format("Failed to resolve(): {}", reason), trace};
233 }
234 else
235 {
236 auto status = js_context.to_str(val).value_or("");
237 // NB: It is not possible to produce every possible ProposalState
238 // here! WITHDRAWN and DROPPED are states that we transition to
239 // elsewhere, but not valid return values from resolve()
240 const std::unordered_map<std::string, ProposalState>
241 js_str_to_status = {
242 {"Open", ProposalState::OPEN},
243 {"Accepted", ProposalState::ACCEPTED},
244 {"Rejected", ProposalState::REJECTED}};
245 const auto it = js_str_to_status.find(status);
246 if (it != js_str_to_status.end())
247 {
248 proposal_info.state = it->second;
249 }
250 else
251 {
252 proposal_info.state = ProposalState::FAILED;
253 failure = ccf::jsgov::Failure{
254 fmt::format(
255 "resolve() returned invalid status value: \"{}\"", status),
256 std::nullopt // No trace
257 };
258 }
259 }
260 }
261
262 if (proposal_info.state != ProposalState::OPEN)
263 {
264 remove_all_other_non_open_proposals(tx, proposal_id);
265 final_votes = std::unordered_map<ccf::MemberId, bool>();
266 for (auto& [mid, vote] : votes)
267 {
268 final_votes.value()[mid] = vote;
269 }
270
271 if (proposal_info.state == ProposalState::ACCEPTED)
272 {
273 // Evaluate apply function
274 js::Context js_context(js::TxAccess::GOV_RW);
275
276 auto gov_effects =
277 context.get_subsystem<AbstractGovernanceEffects>();
278 if (gov_effects == nullptr)
279 {
280 throw std::logic_error(
281 "Unexpected: Could not access GovEffects subsytem");
282 }
283
284 js::populate_global_ccf_kv(tx, js_context);
285 js::populate_global_ccf_node(gov_effects.get(), js_context);
286 js::populate_global_ccf_network(&network, js_context);
287 js::populate_global_ccf_gov_actions(js_context);
288
289 auto apply_func = js_context.function(
290 constitution, "apply", "public:ccf.gov.constitution[0]");
291
292 std::vector<js::JSWrappedValue> argv = {
293 js_context.new_string_len(
294 (const char*)proposal.data(), proposal.size()),
295 js_context.new_string_len(
296 proposal_id.c_str(), proposal_id.size())};
297
298 auto val = js_context.call_with_rt_options(
299 apply_func,
300 argv,
301 &tx,
302 js::RuntimeLimitsPolicy::NO_LOWER_THAN_DEFAULTS);
303
304 if (val.is_exception())
305 {
306 proposal_info.state = ProposalState::FAILED;
307 auto [reason, trace] = js::js_error_message(js_context);
308 if (js_context.interrupt_data.request_timed_out)
309 {
310 reason = "Operation took too long to complete.";
311 }
312 failure = ccf::jsgov::Failure{
313 fmt::format("Failed to apply(): {}", reason), trace};
314 }
315 }
316 }
317
318 return jsgov::ProposalInfoSummary{
319 proposal_id,
320 proposal_info.proposer_id,
321 proposal_info.state,
322 proposal_info.ballots.size(),
323 final_votes,
324 vote_failures,
325 failure};
326 }
327 }

329 // We have several structurally-similar types for representing proposals (in
330 // KV vs post-exec vs fully expanded). This aims to take any, and produce
331 // the same API-compatible description.
332 template <typename TProposal>
333 nlohmann::json convert_proposal_to_api_format(const TProposal& summary)
334 {
335 auto response_body = nlohmann::json::object();
336
337 response_body["proposerId"] = summary.proposer_id;
338 response_body["proposalState"] = summary.state;
339
340 std::optional<ccf::jsgov::Votes> votes;
341
342 if constexpr (std::is_same_v<TProposal, ccf::jsgov::ProposalInfoSummary>)
343 {
344 response_body["proposalId"] = summary.proposal_id;
345 response_body["ballotCount"] = summary.ballot_count;
346
347 votes = summary.votes;
348 }
349 else if constexpr (std::is_same_v<TProposal, ccf::jsgov::ProposalInfo>)
350 {
351 response_body["ballotCount"] = summary.ballots.size();
352
353 votes = summary.final_votes;
354 }
355
356 if (votes.has_value())
357 {
358 auto final_votes = nlohmann::json::object();
359 for (const auto& [voter_id, vote_result] : *votes)
360 {
361 final_votes[voter_id.value()] = vote_result;
362 }
363 response_body["finalVotes"] = final_votes;
364 }
365
366 if (summary.vote_failures.has_value())
367 {
368 auto vote_failures = nlohmann::json::object();
369 for (const auto& [failer_id, failure] : *summary.vote_failures)
370 {
371 vote_failures[failer_id.value()] = failure;
372 }
373 response_body["voteFailures"] = vote_failures;
374 }
375
376 if (summary.failure.has_value())
377 {
378 auto failure = nlohmann::json::object();
379 response_body["failure"] = *summary.failure;
380 }
381
382 return response_body;
383 }

386 void init_proposals_handlers(
387 ccf::BaseEndpointRegistry& registry,
388 NetworkState& network,
389 ccfapp::AbstractNodeContext& node_context)
390 {
392 auto create_proposal = [&](auto& ctx, ApiVersion api_version) {
393 switch (api_version)
394 {
395 case ApiVersion::preview_v1:
396 default:
397 {
398 std::span<const uint8_t> proposal_body;
399 ccf::jsgov::ProposalInfo proposal_info;
400 std::optional<std::string> constitution;
401
402 const auto& cose_ident =
403 ctx.template get_caller<ccf::MemberCOSESign1AuthnIdentity>();
404
405 // Construct proposal_id, as digest of request and root
406 ProposalId proposal_id;
407 std::vector<uint8_t> request_digest;
408 {
409 auto root_at_read = ctx.tx.get_root_at_read_version();
410 if (!root_at_read.has_value())
411 {
412 detail::set_gov_error(
413 ctx.rpc_ctx,
414 HTTP_STATUS_INTERNAL_SERVER_ERROR,
415 ccf::errors::InternalError,
416 "Proposal failed to bind to state.");
417 return;
418 }
419
420 auto hasher = crypto::make_incremental_sha256();
421 hasher->update_hash(root_at_read.value().h);
422
423 request_digest = crypto::sha256(
424 cose_ident.signature.data(), cose_ident.signature.size());
425
426 hasher->update_hash(request_digest);
427
428 const crypto::Sha256Hash proposal_hash = hasher->finalise();
429 proposal_id = proposal_hash.hex_str();
430 }
431
432 // Validate proposal, by calling into JS constitution
433 {
434 constitution =
435 ctx.tx.template ro<ccf::Constitution>(ccf::Tables::CONSTITUTION)
436 ->get();
437 if (!constitution.has_value())
438 {
439 detail::set_gov_error(
440 ctx.rpc_ctx,
441 HTTP_STATUS_INTERNAL_SERVER_ERROR,
442 ccf::errors::InternalError,
443 "No constitution is set - proposals cannot be evaluated");
444 return;
445 }
446
447 js::Context context(js::TxAccess::GOV_RO);
448 js::populate_global_ccf_kv(ctx.tx, context);
449
450 auto validate_func = context.function(
451 constitution.value(),
452 "validate",
453 "public:ccf.gov.constitution[0]");
454
455 proposal_body = cose_ident.content;
456 auto proposal_arg = context.new_string_len(
457 (const char*)proposal_body.data(), proposal_body.size());
458 auto validate_result = context.call_with_rt_options(
459 validate_func,
460 {proposal_arg},
461 &ctx.tx,
462 js::RuntimeLimitsPolicy::NO_LOWER_THAN_DEFAULTS);
463
464 // Handle error cases of validation
465 {
466 if (validate_result.is_exception())
467 {
468 auto [reason, trace] = js_error_message(context);
469 if (context.interrupt_data.request_timed_out)
470 {
471 reason = "Operation took too long to complete.";
472 }
473 detail::set_gov_error(
474 ctx.rpc_ctx,
475 HTTP_STATUS_INTERNAL_SERVER_ERROR,
476 ccf::errors::InternalError,
477 fmt::format(
478 "Failed to execute validation: {} {}",
479 reason,
480 trace.value_or("")));
481 return;
482 }
483
484 if (!validate_result.is_obj())
485 {
486 detail::set_gov_error(
487 ctx.rpc_ctx,
488 HTTP_STATUS_INTERNAL_SERVER_ERROR,
489 ccf::errors::InternalError,
490 "Validation failed to return an object");
491 return;
492 }
493
494 std::string description;
495 auto desc = validate_result["description"];
496 if (desc.is_str())
497 {
498 description = context.to_str(desc).value_or("");
499 }
500
501 auto valid = validate_result["valid"];
502 if (!valid.is_true())
503 {
504 detail::set_gov_error(
505 ctx.rpc_ctx,
506 HTTP_STATUS_BAD_REQUEST,
507 ccf::errors::ProposalFailedToValidate,
508 fmt::format("Proposal failed to validate: {}", description));
509 return;
510 }
511 }
512
513 // Write proposal to KV
514 {
515 auto proposals_handle =
516 ctx.tx.template rw<ccf::jsgov::ProposalMap>(
517 jsgov::Tables::PROPOSALS);
518 // Introduce a read dependency, so that if identical proposal
519 // creations are in-flight and reading at the same version, all
520 // except the first conflict and are re-executed. If we ever
521 // produce a proposal ID which already exists, we must have a
522 // hash collision.
523 if (proposals_handle->has(proposal_id))
524 {
525 detail::set_gov_error(
526 ctx.rpc_ctx,
527 HTTP_STATUS_INTERNAL_SERVER_ERROR,
528 ccf::errors::InternalError,
529 "Proposal ID collision.");
530 return;
531 }
532 proposals_handle->put(
533 proposal_id, {proposal_body.begin(), proposal_body.end()});
534
535 auto proposal_info_handle =
536 ctx.tx.template wo<ccf::jsgov::ProposalInfoMap>(
537 jsgov::Tables::PROPOSALS_INFO);
538
539 proposal_info.proposer_id = cose_ident.member_id;
540 proposal_info.state = ccf::ProposalState::OPEN;
541
542 proposal_info_handle->put(proposal_id, proposal_info);
543
544 detail::record_cose_governance_history(
545 ctx.tx, cose_ident.member_id, cose_ident.envelope);
546 }
547 }
548
549 // Validate proposal's created_at time
550 {
551 // created_at, submitted as a binary integer number of seconds
552 // since epoch in the COSE Sign1 envelope, is converted to a
553 // decimal representation in ASCII, stored as a string, and
554 // compared alphanumerically. This is partly to keep governance as
555 // text-based as possible, to faciliate audit, but also to be able
556 // to benefit from future planned ordering support in the KV. To
557 // compare correctly, the string representation needs to be padded
558 // with leading zeroes, and must therefore not exceed a fixed
559 // digit width. 10 digits is enough to last until November 2286,
560 // ie. long enough.
561 if (cose_ident.protected_header.gov_msg_created_at > 9'999'999'999)
562 {
563 detail::set_gov_error(
564 ctx.rpc_ctx,
565 HTTP_STATUS_BAD_REQUEST,
566 ccf::errors::InvalidCreatedAt,
567 "Header parameter created_at value is too large");
568 return;
569 }
570
571 const auto created_at_str = fmt::format(
572 "{:0>10}", cose_ident.protected_header.gov_msg_created_at);
573
574 ccf::ProposalId colliding_proposal_id;
575 std::string min_created_at;
576
577 const auto subtime_result =
578 detail::validate_proposal_submission_time(
579 ctx.tx, created_at_str, request_digest, proposal_id);
580 switch (subtime_result.status)
581 {
582 case detail::ProposalSubmissionResult::Status::TooOld:
583 {
584 detail::set_gov_error(
585 ctx.rpc_ctx,
586 HTTP_STATUS_BAD_REQUEST,
587 ccf::errors::ProposalCreatedTooLongAgo,
588 fmt::format(
589 "Proposal created too long ago, created_at must be greater "
590 "than {}",
591 subtime_result.info));
592 return;
593 }
594
595 case detail::ProposalSubmissionResult::Status::DuplicateInWindow:
596 {
597 detail::set_gov_error(
598 ctx.rpc_ctx,
599 HTTP_STATUS_BAD_REQUEST,
600 ccf::errors::ProposalReplay,
601 fmt::format(
602 "Proposal submission replay, already exists as proposal {}",
603 subtime_result.info));
604 return;
605 }
606
607 case detail::ProposalSubmissionResult::Status::Acceptable:
608 {
609 break;
610 }
611
612 default:
613 {
614 throw std::runtime_error(
615 "Invalid ProposalSubmissionStatus value");
616 }
617 }
618 }
619
620 // Resolve proposal (may pass immediately)
621 {
622 const auto resolve_result = detail::resolve_proposal(
623 node_context,
624 network,
625 ctx.tx,
626 proposal_id,
627 proposal_body,
628 proposal_info,
629 constitution.value());
630
631 if (resolve_result.state == ProposalState::FAILED)
632 {
633 // If the proposal failed to apply, we want to discard the tx and
634 // not apply its side-effects to the KV state, because it may have
635 // failed mid-execution (eg - thrown an exception), in which case
636 // we do not want to apply partial writes. Note this differs from
637 // a failure that happens after a vote, in that this proposal is
638 // not recorded in the KV at all.
639 detail::set_gov_error(
640 ctx.rpc_ctx,
641 HTTP_STATUS_INTERNAL_SERVER_ERROR,
642 ccf::errors::InternalError,
643 fmt::format("{}", resolve_result.failure));
644 return;
645 }
646
647 // Write updated proposal info
648 {
649 auto proposal_info_handle =
650 ctx.tx.template wo<ccf::jsgov::ProposalInfoMap>(
651 jsgov::Tables::PROPOSALS_INFO);
652
653 proposal_info.state = resolve_result.state;
654 proposal_info.failure = resolve_result.failure;
655
656 proposal_info_handle->put(proposal_id, proposal_info);
657 }
658
659 const auto response_body =
660 detail::convert_proposal_to_api_format(resolve_result);
661
662 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
663 return;
664 }
665 }
666 }
667 };
668 registry
669 .make_endpoint(
670 "/members/proposals:create",
671 HTTP_POST,
672 api_version_adapter(create_proposal),
673 detail::active_member_sig_only_policies("proposal"))
674 .set_openapi_hidden(true)
675 .install();

677 auto withdraw_proposal = [&](auto& ctx, ApiVersion api_version) {
678 switch (api_version)
679 {
680 case ApiVersion::preview_v1:
681 default:
682 {
683 const auto& cose_ident =
684 ctx.template get_caller<ccf::MemberCOSESign1AuthnIdentity>();
685 ccf::ProposalId proposal_id;
686
687 if (!detail::try_parse_signed_proposal_id(
688 cose_ident, ctx.rpc_ctx, proposal_id))
689 {
690 return;
691 }
692
693 auto proposal_info_handle =
694 ctx.tx.template rw<ccf::jsgov::ProposalInfoMap>(
695 jsgov::Tables::PROPOSALS_INFO);
696
697 // Check proposal exists
698 auto proposal_info = proposal_info_handle->get(proposal_id);
699 if (!proposal_info.has_value())
700 {
701 // If it doesn't, then withdrawal is idempotent - we don't know if
702 // this previously existed or not, was withdrawn or accepted, but
703 // return a 204
704 ctx.rpc_ctx->set_response_status(HTTP_STATUS_NO_CONTENT);
705 return;
706 }
707
708 // Check caller is proposer
709 const auto member_id = cose_ident.member_id;
710 if (member_id != proposal_info->proposer_id)
711 {
712 detail::set_gov_error(
713 ctx.rpc_ctx,
714 HTTP_STATUS_FORBIDDEN,
715 ccf::errors::AuthorizationFailed,
716 fmt::format(
717 "Proposal {} can only be withdrawn by proposer {}, not caller "
718 "{}.",
719 proposal_id,
720 proposal_info->proposer_id,
721 member_id));
722 return;
723 }
724
725 // If proposal is still known, and state is neither OPEN nor
726 // WITHDRAWN, return an error - caller has done something wrong
727 if (
728 proposal_info->state != ProposalState::OPEN &&
729 proposal_info->state != ProposalState::WITHDRAWN)
730 {
731 detail::set_gov_error(
732 ctx.rpc_ctx,
733 HTTP_STATUS_BAD_REQUEST,
734 ccf::errors::ProposalNotOpen,
735 fmt::format(
736 "Proposal {} is currently in state {} and cannot be withdrawn.",
737 proposal_id,
738 proposal_info->state));
739 return;
740 }
741
742 // Check proposal is open - only write withdrawal if currently
743 // open
744 if (proposal_info->state == ProposalState::OPEN)
745 {

897 auto submit_ballot =
898 [&](ccf::endpoints::EndpointContext& ctx, ApiVersion api_version) {
899 switch (api_version)
900 {
901 case ApiVersion::preview_v1:
902 default:
903 {
904 const auto& cose_ident =
905 ctx.template get_caller<ccf::MemberCOSESign1AuthnIdentity>();
906
907 ccf::ProposalId proposal_id;
908 if (!detail::try_parse_signed_proposal_id(
909 cose_ident, ctx.rpc_ctx, proposal_id))
910 {
911 return;
912 }
913
914 ccf::MemberId member_id;
915 if (!detail::try_parse_signed_member_id(
916 cose_ident, ctx.rpc_ctx, member_id))
917 {
918 return;
919 }
920
921 // Look up proposal info and check expected state
922 auto proposal_info_handle =
923 ctx.tx.template rw<ccf::jsgov::ProposalInfoMap>(
924 jsgov::Tables::PROPOSALS_INFO);
925 auto proposal_info = proposal_info_handle->get(proposal_id);
926 if (!proposal_info.has_value())
927 {
928 detail::set_gov_error(
929 ctx.rpc_ctx,
930 HTTP_STATUS_NOT_FOUND,
931 ccf::errors::ProposalNotFound,
932 fmt::format("Could not find proposal {}.", proposal_id));
933 return;
934 }
935
936 if (proposal_info->state != ccf::ProposalState::OPEN)
937 {
938 detail::set_gov_error(
939 ctx.rpc_ctx,
940 HTTP_STATUS_BAD_REQUEST,
941 ccf::errors::ProposalNotOpen,
942 fmt::format(
943 "Proposal {} is currently in state {} - only {} proposals "
944 "can receive votes",
945 proposal_id,
946 proposal_info->state,
947 ProposalState::OPEN));
948 return;
949 }
950
951 // Look up proposal contents
952 auto proposals_handle = ctx.tx.template ro<ccf::jsgov::ProposalMap>(
953 ccf::jsgov::Tables::PROPOSALS);
954 const auto proposal = proposals_handle->get(proposal_id);
955 if (!proposal.has_value())
956 {
957 detail::set_gov_error(
958 ctx.rpc_ctx,
959 HTTP_STATUS_NOT_FOUND,
960 ccf::errors::ProposalNotFound,
961 fmt::format("Could not find proposal {}.", proposal_id));
962 return;
963 }
964
965 // Parse and validate incoming ballot
966 const auto params = nlohmann::json::parse(cose_ident.content);
967 const auto ballot_it = params.find("ballot");
968 if (ballot_it == params.end() || !ballot_it.value().is_string())
969 {
970 detail::set_gov_error(
971 ctx.rpc_ctx,
972 HTTP_STATUS_BAD_REQUEST,
973 ccf::errors::InvalidInput,
974 "Signed request body is not a JSON object containing required "
975 "string field \"ballot\"");
976 return;
977 }
978
979 // Access constitution to evaluate ballots
980 const auto constitution =
981 ctx.tx.template ro<ccf::Constitution>(ccf::Tables::CONSTITUTION)
982 ->get();
983 if (!constitution.has_value())
984 {
985 detail::set_gov_error(
986 ctx.rpc_ctx,
987 HTTP_STATUS_INTERNAL_SERVER_ERROR,
988 ccf::errors::InternalError,
989 "No constitution is set - ballots cannot be evaluated");
990 return;
991 }
992
993 const auto ballot = ballot_it.value().get<std::string>();
994
995 const auto info_ballot_it = proposal_info->ballots.find(member_id);
996 if (info_ballot_it != proposal_info->ballots.end())
997 {
998 // If ballot matches previously submitted, aim for idempotent
999 // matching response
1000 if (info_ballot_it->second == ballot)
1001 {
1002 const auto response_body =
1003 detail::convert_proposal_to_api_format(proposal_info.value());
1004
1005 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
1006 return;
1007 }
1008 else
1009 {
1010 detail::set_gov_error(
1011 ctx.rpc_ctx,
1012 HTTP_STATUS_BAD_REQUEST,
1013 ccf::errors::VoteAlreadyExists,
1014 fmt::format(
1015 "Different ballot already submitted by {} for {}.",
1016 member_id,
1017 proposal_id));
1018 return;
1019 }
1020 }
1021
1022 // Store newly provided ballot
1023 proposal_info->ballots.insert_or_assign(
1024 info_ballot_it, member_id, ballot_it.value().get<std::string>());
1025
1026 detail::record_cose_governance_history(
1027 ctx.tx, cose_ident.member_id, cose_ident.envelope);
1028
1029 const auto resolve_result = detail::resolve_proposal(
1030 node_context,
1031 network,
1032 ctx.tx,
1033 proposal_id,
1034 proposal.value(),
1035 proposal_info.value(),
1036 constitution.value());
1037
1038 if (resolve_result.state == ProposalState::FAILED)
1039 {
1040 // If the proposal failed to apply, we want to discard the tx and
1041 // not apply its side-effects to the KV state, because it may have
1042 // failed mid-execution (eg - thrown an exception), in which case
1043 // we do not want to apply partial writes
1044 detail::set_gov_error(
1045 ctx.rpc_ctx,
1046 HTTP_STATUS_INTERNAL_SERVER_ERROR,
1047 ccf::errors::InternalError,
1048 fmt::format("{}", resolve_result.failure));
1049 return;
1050 }
1051
1052 // Write updated proposal info
1053 {
1054 proposal_info->state = resolve_result.state;
1055 proposal_info->failure = resolve_result.failure;
1056
1057 proposal_info_handle->put(proposal_id, proposal_info.value());
1058 }
1059
1060 const auto response_body =
1061 detail::convert_proposal_to_api_format(resolve_result);
1062
1063 ctx.rpc_ctx->set_response_json(response_body, HTTP_STATUS_OK);
1064 return;
1065 }
1066 }
1067 };
